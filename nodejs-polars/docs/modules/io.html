  • readJSON(pathOrBody: string | Buffer, options?: Partial<ReadJsonOptions>): DataFrame
    • Defined in io.ts:123

    Read a JSON file or string into a DataFrame.

    Note: Currently only newline delimited JSON is supported

    example
    const jsonString = `
    {"a", 1, "b", "foo", "c": 3}
    {"a": 2, "b": "bar", "c": 6}
    `
    > const df = pl.readJSON(jsonString)
    > console.log(df)
      shape: (2, 3)

    ├╌╌╌╌╌┼╌╌╌╌╌┼╌╌╌╌╌┤
    │ 2   ┆ bar ┆ 6   │
