6#[macro_export]
7macro_rules! format_pl_smallstr {
8    ($($arg:tt)*) => {{
9        use std::fmt::Write;
10
11        let mut string = $crate::pl_str::PlSmallStr::EMPTY;
12        write!(string, $($arg)*).unwrap();
13        string
14    }}
15}
