1//! This module supports mapping SQL datatypes to Polars datatypes.
2//!
3//! It also provides utility functions for working with SQL datatypes.
4use polars_core::datatypes::{DataType, TimeUnit};
5use polars_error::{PolarsResult, polars_bail};
6use polars_plan::dsl::{Expr, lit};
7use sqlparser::ast::{
8    ArrayElemTypeDef, DataType as SQLDataType, ExactNumberInfo, Ident, ObjectName, TimezoneInfo,
9};

11polars_utils::regex_cache::cached_regex! {
12    static DATETIME_LITERAL_RE = r"^\d{4}-[01]\d-[0-3]\d[ T](?:[01][0-9]|2[0-3]):[0-5][0-9](?::[0-5][0-9](\.\d{1,9})?)?$";
13    static DATE_LITERAL_RE = r"^\d{4}-[01]\d-[0-3]\d$";
14    static TIME_LITERAL_RE = r"^(?:[01][0-9]|2[0-3]):[0-5][0-9](?::[0-5][0-9](\.\d{1,9})?)?$";
15}

17pub fn bitstring_to_bytes_literal(b: &String) -> PolarsResult<Expr> {
18    let n_bits = b.len();
19    if !b.chars().all(|c| c == '0' || c == '1') || n_bits > 64 {
20        polars_bail!(
21            SQLSyntax:
22            "bit string literal should contain only 0s and 1s and have length <= 64; found '{}' with length {}", b, n_bits
23        )
24    }
25    let s = b.as_str();
26    Ok(lit(match n_bits {
27        0 => b"".to_vec(),
28        1..=8 => u8::from_str_radix(s, 2).unwrap().to_be_bytes().to_vec(),
29        9..=16 => u16::from_str_radix(s, 2).unwrap().to_be_bytes().to_vec(),
30        17..=32 => u32::from_str_radix(s, 2).unwrap().to_be_bytes().to_vec(),
31        _ => u64::from_str_radix(s, 2).unwrap().to_be_bytes().to_vec(),
32    }))
33}

35pub fn is_iso_datetime(value: &str) -> bool {
36    DATETIME_LITERAL_RE.is_match(value)
37}
38
39pub fn is_iso_date(value: &str) -> bool {
40    DATE_LITERAL_RE.is_match(value)
41}
42
43pub fn is_iso_time(value: &str) -> bool {
44    TIME_LITERAL_RE.is_match(value)
45}

47fn timeunit_from_precision(prec: &Option<u64>) -> PolarsResult<TimeUnit> {
48    Ok(match prec {
49        None => TimeUnit::Microseconds,
50        Some(n) if (1u64..=3u64).contains(n) => TimeUnit::Milliseconds,
51        Some(n) if (4u64..=6u64).contains(n) => TimeUnit::Microseconds,
52        Some(n) if (7u64..=9u64).contains(n) => TimeUnit::Nanoseconds,
53        Some(n) => {
54            polars_bail!(SQLSyntax: "invalid temporal type precision (expected 1-9, found {})", n)
55        },
56    })
57}

59pub(crate) fn map_sql_dtype_to_polars(dtype: &SQLDataType) -> PolarsResult<DataType> {
60    Ok(match dtype {
61        // ---------------------------------
62        // array/list
63        // ---------------------------------
64        SQLDataType::Array(ArrayElemTypeDef::AngleBracket(inner_type))
65        | SQLDataType::Array(ArrayElemTypeDef::SquareBracket(inner_type, _)) => {
66            DataType::List(Box::new(map_sql_dtype_to_polars(inner_type)?))
67        },
68
69        // ---------------------------------
70        // binary
71        // ---------------------------------
72        SQLDataType::Bytea
73        | SQLDataType::Bytes(_)
74        | SQLDataType::Binary(_)
75        | SQLDataType::Blob(_)
76        | SQLDataType::Varbinary(_) => DataType::Binary,
77
78        // ---------------------------------
79        // boolean
80        // ---------------------------------
81        SQLDataType::Boolean | SQLDataType::Bool => DataType::Boolean,
82
83        // ---------------------------------
84        // signed integer
85        // ---------------------------------
86        SQLDataType::Int16 | SQLDataType::Int2(_) => DataType::Int16,
87        SQLDataType::Int32 | SQLDataType::Int4(_) => DataType::Int32,
88        SQLDataType::Int64 | SQLDataType::Int8(_) => DataType::Int64,
89        SQLDataType::Int128 => DataType::Int128,
90        SQLDataType::Integer(_) | SQLDataType::Int(_) => DataType::Int32,
91        SQLDataType::SmallInt(_) => DataType::Int16,
92        SQLDataType::MediumInt(_) => DataType::Int32,
93        SQLDataType::BigInt(_) => DataType::Int64,
94        SQLDataType::TinyInt(_) => DataType::Int8,
95
96        // ---------------------------------
97        // unsigned integer
98        // ---------------------------------
99        SQLDataType::UInt16 => DataType::UInt16,
100        SQLDataType::UInt32 => DataType::UInt32,
101        SQLDataType::UInt64 => DataType::UInt64,
102        SQLDataType::UnsignedTinyInt(_) => DataType::UInt8,
103        SQLDataType::UnsignedInt(_) | SQLDataType::UnsignedInteger(_) => DataType::UInt32,
104        SQLDataType::UnsignedInt2(_) | SQLDataType::UnsignedSmallInt(_) => DataType::UInt16,
105        SQLDataType::UnsignedInt4(_) | SQLDataType::UnsignedMediumInt(_) => DataType::UInt32,
106        SQLDataType::UnsignedInt8(_) | SQLDataType::UnsignedBigInt(_) | SQLDataType::UInt8 => {
107            DataType::UInt64
108        },
109
110        // ---------------------------------
111        // float
112        // ---------------------------------
113        SQLDataType::Float4 | SQLDataType::Real => DataType::Float32,
114        SQLDataType::Double | SQLDataType::DoublePrecision | SQLDataType::Float8 => {
115            DataType::Float64
116        },
117        SQLDataType::Float(n_bytes) => match n_bytes {
118            Some(n) if (1u64..=24u64).contains(n) => DataType::Float32,
119            Some(n) if (25u64..=53u64).contains(n) => DataType::Float64,
120            Some(n) => {
121                polars_bail!(SQLSyntax: "unsupported `float` size (expected a value between 1 and 53, found {})", n)
122            },
123            None => DataType::Float64,
124        },
125
126        // ---------------------------------
127        // decimal
128        // ---------------------------------
129        #[cfg(feature = "dtype-decimal")]
130        SQLDataType::Dec(info)
131        | SQLDataType::Decimal(info)
132        | SQLDataType::BigDecimal(info)
133        | SQLDataType::Numeric(info) => match *info {
134            ExactNumberInfo::PrecisionAndScale(p, s) => DataType::Decimal(p as usize, s as usize),
135            ExactNumberInfo::Precision(p) => DataType::Decimal(p as usize, 0),
136            ExactNumberInfo::None => DataType::Decimal(38, 9),
137        },
138
139        // ---------------------------------
140        // temporal
141        // ---------------------------------
142        SQLDataType::Date => DataType::Date,
143        SQLDataType::Interval => DataType::Duration(TimeUnit::Microseconds),
144        SQLDataType::Time(_, tz) => match tz {
145            TimezoneInfo::None => DataType::Time,
146            _ => {
147                polars_bail!(SQLInterface: "`time` with timezone is not supported; found tz={}", tz)
148            },
149        },
150        SQLDataType::Datetime(prec) => DataType::Datetime(timeunit_from_precision(prec)?, None),
151        SQLDataType::Timestamp(prec, tz) => match tz {
152            TimezoneInfo::None => DataType::Datetime(timeunit_from_precision(prec)?, None),
153            _ => {
154                polars_bail!(SQLInterface: "`timestamp` with timezone is not (yet) supported")
155            },
156        },
157
158        // ---------------------------------
159        // string
160        // ---------------------------------
161        SQLDataType::Char(_)
162        | SQLDataType::CharVarying(_)
163        | SQLDataType::Character(_)
164        | SQLDataType::CharacterVarying(_)
165        | SQLDataType::Clob(_)
166        | SQLDataType::String(_)
167        | SQLDataType::Text
168        | SQLDataType::Uuid
169        | SQLDataType::Varchar(_) => DataType::String,
170
171        // ---------------------------------
172        // custom
173        // ---------------------------------
174        SQLDataType::Custom(ObjectName(idents), _) => match idents.as_slice() {
175            [Ident { value, .. }] => match value.to_lowercase().as_str() {
176                // these integer types are not supported by the PostgreSQL core distribution,
177                // but they ARE available via `pguint` (https://github.com/petere/pguint), an
178                // extension maintained by one of the PostgreSQL core developers, and/or DuckDB.
179                "int1" => DataType::Int8,
180                "uint1" | "utinyint" => DataType::UInt8,
181                "uint2" | "usmallint" => DataType::UInt16,
182                "uint4" | "uinteger" | "uint" => DataType::UInt32,
183                "uint8" | "ubigint" => DataType::UInt64,
184                "hugeint" => DataType::Int128,
185                "uhugeint" => DataType::UInt128,
186                _ => {
187                    polars_bail!(SQLInterface: "datatype {:?} is not currently supported", value)
188                },
189            },
190            _ => {
191                polars_bail!(SQLInterface: "datatype {:?} is not currently supported", idents)
192            },
193        },
194        _ => {
195            polars_bail!(SQLInterface: "datatype {:?} is not currently supported", dtype)
196        },
197    })
