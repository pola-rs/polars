77/// Disambiguate qualified wildcard columns that conflict with each other or other projections.
78fn disambiguate_projection_cols(
79    items: Vec<ProjectionItem>,
80    schema: &Schema,
81) -> PolarsResult<Vec<Expr>> {
82    // Establish qualified wildcard names (with counts), and other expression names
83    let mut qualified_wildcard_names: PlHashMap<PlSmallStr, usize> = PlHashMap::new();
84    let mut other_names: PlHashSet<PlSmallStr> = PlHashSet::new();
85    for item in &items {
86        match item {
87            ProjectionItem::QualifiedExprs(_, exprs) => {
88                for expr in exprs {
89                    if let Some(name) = expr_output_name(expr) {
90                        *qualified_wildcard_names.entry(name.clone()).or_insert(0) += 1;
91                    }
92                }
93            },
94            ProjectionItem::Exprs(exprs) => {
95                for expr in exprs {
96                    if let Some(name) = expr_output_name(expr) {
97                        other_names.insert(name.clone());
98                    }
99                }
100            },
101        }
102    }
103
104    // Names requiring disambiguation (duplicates across wildcards, eg: `tbl1.*`,`tbl2.*`)
105    let needs_suffix: PlHashSet<PlSmallStr> = qualified_wildcard_names
106        .into_iter()
107        .filter(|(name, count)| *count > 1 || other_names.contains(name))
108        .map(|(name, _)| name)
109        .collect();
110
111    // Output, applying suffixes where needed
112    let mut result: Vec<Expr> = Vec::new();
113    for item in items {
114        match item {
115            ProjectionItem::QualifiedExprs(tbl_name, exprs) if !needs_suffix.is_empty() => {
116                for expr in exprs {
117                    if let Some(name) = expr_output_name(&expr) {
118                        if needs_suffix.contains(name) {
119                            let suffixed = format_pl_smallstr!("{}:{}", name, tbl_name);
120                            if schema.contains(suffixed.as_str()) {
121                                result.push(col(suffixed));
122                                continue;
123                            }
124                            if other_names.contains(name) {
125                                polars_bail!(
126                                    SQLInterface:
127                                    "column '{}' is duplicated in the SELECT (explicitly, and via the `*` wildcard)", name
128                                );
129                            }
130                        }
131                    }
132                    result.push(expr);
133                }
134            },
135            ProjectionItem::QualifiedExprs(_, exprs) | ProjectionItem::Exprs(exprs) => {
136                result.extend(exprs);
137            },
138        }
139    }
140    Ok(result)
141}

215    /// Execute a SQL query, returning a [`LazyFrame`].
216    /// ```rust
217    /// # use polars_sql::SQLContext;
218    /// # use polars_core::prelude::*;
219    /// # use polars_lazy::prelude::*;
220    /// # fn main() {
221    ///
222    /// let mut ctx = SQLContext::new();
223    /// let df = df! {
224    ///    "a" =>  [1, 2, 3],
225    /// }
226    /// .unwrap();
227    ///
228    /// ctx.register("df", df.clone().lazy());
229    /// let sql_df = ctx.execute("SELECT * FROM df").unwrap().collect().unwrap();
230    /// assert!(sql_df.equals(&df));
231    /// # }
232    ///```
233    pub fn execute(&mut self, query: &str) -> PolarsResult<LazyFrame> {
234        let mut parser = Parser::new(&GenericDialect);
235        parser = parser.with_options(ParserOptions {
236            trailing_commas: true,
237            ..Default::default()
238        });
239
240        let ast = parser
241            .try_with_sql(query)
242            .map_err(to_sql_interface_err)?
243            .parse_statements()
244            .map_err(to_sql_interface_err)?;
245
246        polars_ensure!(ast.len() == 1, SQLInterface: "one (and only one) statement can be parsed at a time");
247        let res = self.execute_statement(ast.first().unwrap())?;
248
249        // Ensure the result uses the proper arenas.
250        // This will instantiate new arenas with a new version.
251        let lp_arena = std::mem::take(&mut self.lp_arena);
252        let expr_arena = std::mem::take(&mut self.expr_arena);
253        res.set_cached_arena(lp_arena, expr_arena);
254
255        // Every execution should clear the statement-level maps.
256        self.cte_map.clear();
257        self.table_aliases.clear();
258        self.joined_aliases.clear();
259        self.named_windows.clear();
260
261        Ok(res)
262    }

351    /// Execute a query in an isolated context. This prevents subqueries from mutating
352    /// arenas and other context state. Returns both the LazyFrame *and* its associated
353    /// Schema (so that the correct arenas are used when determining schema).
354    pub(crate) fn execute_isolated<F>(&mut self, query: F) -> PolarsResult<LazyFrame>
355    where
356        F: FnOnce(&mut Self) -> PolarsResult<LazyFrame>,
357    {
358        let mut ctx = self.isolated();

364        lf.set_cached_arena(ctx.lp_arena, ctx.expr_arena);

369    fn expr_or_ordinal(
370        &mut self,
371        e: &SQLExpr,
372        exprs: &[Expr],
373        selected: Option<&[Expr]>,
374        schema: Option<&Schema>,
375        clause: &str,
376    ) -> PolarsResult<Expr> {
377        match e {
378            SQLExpr::UnaryOp {
379                op: UnaryOperator::Minus,
380                expr,
381            } if matches!(
382                **expr,
383                SQLExpr::Value(ValueWithSpan {
384                    value: SQLValue::Number(_, _),
385                    ..
386                })
387            ) =>
388            {
389                if let SQLExpr::Value(ValueWithSpan {
390                    value: SQLValue::Number(ref idx, _),
391                    ..
392                }) = **expr
393                {
394                    Err(polars_err!(
395                    SQLSyntax:
396                    "negative ordinal values are invalid for {}; found -{}",
397                    clause,
398                    idx
399                    ))
400                } else {
401                    unreachable!()
402                }
403            },
404            SQLExpr::Value(ValueWithSpan {
405                value: SQLValue::Number(idx, _),
406                ..
407            }) => {
408                // note: sql queries are 1-indexed
409                let idx = idx.parse::<usize>().map_err(|_| {
410                    polars_err!(
411                        SQLSyntax:
412                        "negative ordinal values are invalid for {}; found {}",
413                        clause,
414                        idx
415                    )
416                })?;
417                // note: "selected" cols represent final projection order, so we use those for
418                // ordinal resolution. "exprs" may include cols that are subsequently dropped.
419                let cols = if let Some(cols) = selected {
420                    cols
421                } else {
422                    exprs
423                };
424                Ok(cols
425                    .get(idx - 1)
426                    .ok_or_else(|| {
427                        polars_err!(
428                            SQLInterface:
429                            "{} ordinal value must refer to a valid column; found {}",
430                            clause,
431                            idx
432                        )
433                    })?
434                    .clone())
435            },
436            SQLExpr::Value(v) => Err(polars_err!(
437                SQLSyntax:
438                "{} requires a valid expression or positive ordinal; found {}", clause, v,
439            )),
440            _ => {
441                // Handle qualified cross-aliasing in ORDER BY clauses
442                // (eg: `SELECT a AS b, -b AS a ... ORDER BY self.a`)
443                let mut expr = parse_sql_expr(e, self, schema)?;
444                if matches!(e, SQLExpr::CompoundIdentifier(_)) {
445                    if let Some(schema) = schema {
446                        expr = expr.map_expr(|ex| match &ex {
447                            Expr::Column(name) => {
448                                let prefixed = format!("__POLARS_ORIG_{}", name.as_str());
449                                if schema.contains(prefixed.as_str()) {
450                                    col(prefixed)
451                                } else {
452                                    ex
453                                }
454                            },
455                            _ => ex,
456                        });
457                    }
458                }
459                Ok(expr)
460            },
461        }
462    }

473    fn process_query(&mut self, expr: &SetExpr, query: &Query) -> PolarsResult<LazyFrame> {
474        match expr {
475            SetExpr::Select(select_stmt) => self.execute_select(select_stmt, query),
476            SetExpr::Query(nested_query) => {
477                let lf = self.execute_query_no_ctes(nested_query)?;
478                self.process_order_by(lf, &query.order_by, None)
479            },
480            SetExpr::SetOperation {
481                op: SetOperator::Union,
482                set_quantifier,
483                left,
484                right,
485            } => self.process_union(left, right, set_quantifier, query),
486
487            #[cfg(feature = "semi_anti_join")]
488            SetExpr::SetOperation {
489                op: SetOperator::Intersect | SetOperator::Except,
490                set_quantifier,
491                left,
492                right,
493            } => self.process_except_intersect(left, right, set_quantifier, query),
494
495            SetExpr::Values(Values {
496                explicit_row: _,
497                rows,
498                value_keyword: _,
499            }) => self.process_values(rows),
500
501            SetExpr::Table(tbl) => {
502                if let Some(table_name) = tbl.table_name.as_ref() {
503                    self.get_table_from_current_scope(table_name)
504                        .ok_or_else(|| {
505                            polars_err!(
506                                SQLInterface: "no table or alias named '{}' found",
507                                tbl
508                            )
509                        })
510                } else {
511                    polars_bail!(SQLInterface: "'TABLE' requires valid table name")
512                }
513            },
514            op => {
515                polars_bail!(SQLInterface: "'{}' operation is currently unsupported", op)
516            },
517        }
518    }

520    #[cfg(feature = "semi_anti_join")]
521    fn process_except_intersect(
522        &mut self,
523        left: &SetExpr,
524        right: &SetExpr,
525        quantifier: &SetQuantifier,
526        query: &Query,
527    ) -> PolarsResult<LazyFrame> {
528        let (join_type, op_name) = match *query.body {
529            SetExpr::SetOperation {
530                op: SetOperator::Except,
531                ..
532            } => (JoinType::Anti, "EXCEPT"),
533            _ => (JoinType::Semi, "INTERSECT"),
534        };
535
536        // Note: each side of the EXCEPT/INTERSECT operation should execute
537        // in isolation to prevent context state leakage between them
538        let mut lf = self.execute_isolated(|ctx| ctx.process_query(left, query))?;
539        let mut rf = self.execute_isolated(|ctx| ctx.process_query(right, query))?;
540        let lf_schema = self.get_frame_schema(&mut lf)?;
541
542        let lf_cols: Vec<_> = lf_schema.iter_names_cloned().map(col).collect();
543        let rf_cols = match quantifier {
544            SetQuantifier::ByName => None,
545            SetQuantifier::Distinct | SetQuantifier::None => {
546                let rf_schema = self.get_frame_schema(&mut rf)?;
547                let rf_cols: Vec<_> = rf_schema.iter_names_cloned().map(col).collect();
548                if lf_cols.len() != rf_cols.len() {
549                    polars_bail!(SQLInterface: "{} requires equal number of columns in each table (use '{} BY NAME' to combine mismatched tables)", op_name, op_name)
550                }
551                Some(rf_cols)
552            },
553            _ => {
554                polars_bail!(SQLInterface: "'{} {}' is not supported", op_name, quantifier.to_string())
555            },
556        };
557        let join = lf.join_builder().with(rf).how(join_type).join_nulls(true);
558        let joined_tbl = match rf_cols {
559            Some(rf_cols) => join.left_on(lf_cols).right_on(rf_cols).finish(),
560            None => join.on(lf_cols).finish(),
561        };
562        let lf = joined_tbl.unique(None, UniqueKeepStrategy::Any);
563        self.process_order_by(lf, &query.order_by, None)
564    }

566    fn process_union(
567        &mut self,
568        left: &SetExpr,
569        right: &SetExpr,
570        quantifier: &SetQuantifier,
571        query: &Query,
572    ) -> PolarsResult<LazyFrame> {
573        let quantifier = *quantifier;
574
575        // Note: each side of the UNION operation should execute
576        // in isolation to prevent context state leakage between them
577        let mut lf = self.execute_isolated(|ctx| ctx.process_query(left, query))?;
578        let mut rf = self.execute_isolated(|ctx| ctx.process_query(right, query))?;
579
580        let opts = UnionArgs {
581            parallel: true,
582            to_supertypes: true,
583            maintain_order: false,
584            ..Default::default()
585        };
586        let lf = match quantifier {
587            // UNION [ALL | DISTINCT]
588            SetQuantifier::All | SetQuantifier::Distinct | SetQuantifier::None => {
589                let lf_schema = self.get_frame_schema(&mut lf)?;
590                let rf_schema = self.get_frame_schema(&mut rf)?;
591                if lf_schema.len() != rf_schema.len() {
592                    polars_bail!(SQLInterface: "UNION requires equal number of columns in each table (use 'UNION BY NAME' to combine mismatched tables)")
593                }
594                // rename `rf` columns to match `lf` if they differ; SQL behaves
595                // positionally on UNION ops (unless using the "BY NAME" qualifier)
596                if lf_schema.iter_names().ne(rf_schema.iter_names()) {
597                    rf = rf.rename(rf_schema.iter_names(), lf_schema.iter_names(), true);
598                }
599                let concatenated = concat(vec![lf, rf], opts);
600                match quantifier {
601                    SetQuantifier::Distinct | SetQuantifier::None => {
602                        concatenated.map(|lf| lf.unique(None, UniqueKeepStrategy::Any))
603                    },
604                    _ => concatenated,
605                }
606            },
607            // UNION ALL BY NAME
608            #[cfg(feature = "diagonal_concat")]
609            SetQuantifier::AllByName => concat_lf_diagonal(vec![lf, rf], opts),
610            // UNION [DISTINCT] BY NAME
611            #[cfg(feature = "diagonal_concat")]
612            SetQuantifier::ByName | SetQuantifier::DistinctByName => {
613                let concatenated = concat_lf_diagonal(vec![lf, rf], opts);
614                concatenated.map(|lf| lf.unique(None, UniqueKeepStrategy::Any))
615            },
616            #[allow(unreachable_patterns)]
617            _ => {
618                polars_bail!(SQLInterface: "'UNION {}' is not currently supported", quantifier)
619            },
620        }?;
621
622        self.process_order_by(lf, &query.order_by, None)
623    }

625    /// Process UNNEST as a lateral operation when it contains column references
626    /// (handles `CROSS JOIN UNNEST(col) AS name` by exploding the referenced col).
627    fn process_unnest_lateral(
628        &self,
629        lf: LazyFrame,
630        alias: &Option<TableAlias>,
631        array_exprs: &[SQLExpr],
632        with_offset: bool,
633    ) -> PolarsResult<LazyFrame> {
634        let alias = alias
635            .as_ref()
636            .ok_or_else(|| polars_err!(SQLSyntax: "UNNEST table must have an alias"))?;
637        polars_ensure!(!with_offset, SQLInterface: "UNNEST tables do not (yet) support WITH ORDINALITY|OFFSET");
638
639        let (mut explode_cols, mut rename_from, mut rename_to) = (
640            Vec::with_capacity(array_exprs.len()),
641            Vec::with_capacity(array_exprs.len()),
642            Vec::with_capacity(array_exprs.len()),
643        );
644        let is_single_col = array_exprs.len() == 1;
645
646        for (i, arr_expr) in array_exprs.iter().enumerate() {
647            let col_name = match arr_expr {
648                SQLExpr::Identifier(ident) => PlSmallStr::from_str(&ident.value),
649                SQLExpr::CompoundIdentifier(parts) => {
650                    PlSmallStr::from_str(&parts.last().unwrap().value)
651                },
652                SQLExpr::Array(_) => polars_bail!(
653                    SQLInterface: "CROSS JOIN UNNEST with both literal arrays and column references is not supported"
654                ),
655                other => polars_bail!(
656                    SQLSyntax: "UNNEST expects column references or array literals, found {:?}", other
657                ),
658            };
659            // alias: column name from "AS t(col)", or table alias
660            if let Some(name) = alias
661                .columns
662                .get(i)
663                .map(|c| c.name.value.as_str())
664                .or_else(|| is_single_col.then_some(alias.name.value.as_str()))
665                .filter(|name| !name.is_empty() && *name != col_name.as_str())
666            {
667                rename_from.push(col_name.clone());
668                rename_to.push(PlSmallStr::from_str(name));
669            }
670            explode_cols.push(col_name);
671        }
672
673        let mut lf = lf.explode(
674            Selector::ByName {
675                names: Arc::from(explode_cols),
676                strict: true,
677            },
678            ExplodeOptions {
679                empty_as_null: true,
680                keep_nulls: true,
681            },
682        );
683        if !rename_from.is_empty() {
684            lf = lf.rename(rename_from, rename_to, true);
685        }
686        Ok(lf)
687    }

746    // DELETE FROM <tbl> [WHERE ...]
747    fn execute_delete_from_table(&mut self, stmt: &Statement) -> PolarsResult<LazyFrame> {
748        if let Statement::Delete(Delete {
749            tables,
750            from,
751            using,
752            selection,
753            returning,
754            order_by,
755            limit,
756            delete_token: _,
757        }) = stmt
758        {
759            if !tables.is_empty()
760                || using.is_some()
761                || returning.is_some()
762                || limit.is_some()
763                || !order_by.is_empty()
764            {
765                let error_message = match () {
766                    _ if !tables.is_empty() => "DELETE expects exactly one table name",
767                    _ if using.is_some() => "DELETE does not support the USING clause",
768                    _ if returning.is_some() => "DELETE does not support the RETURNING clause",
769                    _ if limit.is_some() => "DELETE does not support the LIMIT clause",
770                    _ if !order_by.is_empty() => "DELETE does not support the ORDER BY clause",
771                    _ => unreachable!(),
772                };
773                polars_bail!(SQLInterface: error_message);
774            }
775            let from_tables = match &from {
776                FromTable::WithFromKeyword(from) => from,
777                FromTable::WithoutKeyword(from) => from,
778            };
779            if from_tables.len() > 1 {
780                polars_bail!(SQLInterface: "cannot have multiple tables in DELETE FROM (found {})", from_tables.len())
781            }
782            let tbl_expr = from_tables.first().unwrap();
783            if !tbl_expr.joins.is_empty() {
784                polars_bail!(SQLInterface: "DELETE does not support table JOINs")
785            }
786            let (_, lf) = self.get_table(&tbl_expr.relation)?;
787            if selection.is_none() {
788                // no WHERE clause; equivalent to TRUNCATE (drop all rows)
789                Ok(lf.clear())
790            } else {
791                // apply constraint as inverted filter (drops rows matching the selection)
792                Ok(self.process_where(lf.clone(), selection, true, None)?)
793            }
794        } else {
795            polars_bail!(SQLInterface: "unexpected statement type; expected DELETE")
796        }
797    }

872    /// execute the 'FROM' part of the query
873    fn execute_from_statement(&mut self, tbl_expr: &TableWithJoins) -> PolarsResult<LazyFrame> {
874        let (l_name, mut lf) = self.get_table(&tbl_expr.relation)?;
875        if !tbl_expr.joins.is_empty() {
876            for join in &tbl_expr.joins {
877                // Handle "CROSS JOIN UNNEST(col)" as a lateral join op
878                if let (
879                    JoinOperator::CrossJoin(JoinConstraint::None),
880                    TableFactor::UNNEST {
881                        alias,
882                        array_exprs,
883                        with_offset,
884                        ..
885                    },
886                ) = (&join.join_operator, &join.relation)
887                {
888                    if array_exprs.iter().any(|e| !matches!(e, SQLExpr::Array(_))) {
889                        lf = self.process_unnest_lateral(lf, alias, array_exprs, *with_offset)?;
890                        continue;
891                    }
892                }
893
894                let (r_name, mut rf) = self.get_table(&join.relation)?;
895                if r_name.is_empty() {
896                    // Require non-empty to avoid duplicate column errors from nested self-joins.
897                    polars_bail!(
898                        SQLInterface:
899                        "cannot JOIN on unnamed relation; please provide an alias"
900                    )
901                }
902                let left_schema = self.get_frame_schema(&mut lf)?;
903                let right_schema = self.get_frame_schema(&mut rf)?;
904
905                lf = match &join.join_operator {
906                    op @ (JoinOperator::Join(constraint)  // note: bare "join" is inner
907                    | JoinOperator::FullOuter(constraint)
908                    | JoinOperator::Left(constraint)
909                    | JoinOperator::LeftOuter(constraint)
910                    | JoinOperator::Right(constraint)
911                    | JoinOperator::RightOuter(constraint)
912                    | JoinOperator::Inner(constraint)
913                    | JoinOperator::Anti(constraint)
914                    | JoinOperator::Semi(constraint)
915                    | JoinOperator::LeftAnti(constraint)
916                    | JoinOperator::LeftSemi(constraint)
917                    | JoinOperator::RightAnti(constraint)
918                    | JoinOperator::RightSemi(constraint)) => {
919                        let (lf, rf) = match op {
920                            JoinOperator::RightAnti(_) | JoinOperator::RightSemi(_) => (rf, lf),
921                            _ => (lf, rf),
922                        };
923                        self.process_join(
924                            &TableInfo {
925                                frame: lf,
926                                name: (&l_name).into(),
927                                schema: left_schema.clone(),
928                            },
929                            &TableInfo {
930                                frame: rf,
931                                name: (&r_name).into(),
932                                schema: right_schema.clone(),
933                            },
934                            constraint,
935                            match op {
936                                JoinOperator::Join(_) | JoinOperator::Inner(_) => JoinType::Inner,
937                                JoinOperator::Left(_) | JoinOperator::LeftOuter(_) => {
938                                    JoinType::Left
939                                },
940                                JoinOperator::Right(_) | JoinOperator::RightOuter(_) => {
941                                    JoinType::Right
942                                },
943                                JoinOperator::FullOuter(_) => JoinType::Full,
944                                #[cfg(feature = "semi_anti_join")]
945                                JoinOperator::Anti(_)
946                                | JoinOperator::LeftAnti(_)
947                                | JoinOperator::RightAnti(_) => JoinType::Anti,
948                                #[cfg(feature = "semi_anti_join")]
949                                JoinOperator::Semi(_)
950                                | JoinOperator::LeftSemi(_)
951                                | JoinOperator::RightSemi(_) => JoinType::Semi,
952                                join_type => polars_bail!(
953                                    SQLInterface:
954                                    "join type '{:?}' not currently supported",
955                                    join_type
956                                ),
957                            },
958                        )?
959                    },
960                    JoinOperator::CrossJoin(JoinConstraint::None) => {
961                        lf.cross_join(rf, Some(format_pl_smallstr!(":{}", r_name)))
962                    },
963                    JoinOperator::CrossJoin(constraint) => {
964                        polars_bail!(
965                            SQLInterface:
966                            "CROSS JOIN does not support {:?} constraint; consider INNER JOIN instead",
967                            constraint
968                        )
969                    },
970                    join_type => {
971                        polars_bail!(SQLInterface: "join type '{:?}' not currently supported", join_type)
972                    },
973                };
974
975                // track join-aliased columns so we can resolve/check them later
976                let joined_schema = self.get_frame_schema(&mut lf)?;
977
978                self.joined_aliases.insert(
979                    r_name.clone(),
980                    right_schema
981                        .iter_names()
982                        .filter_map(|name| {
983                            // col exists in both tables and is aliased in the joined result
984                            let aliased_name = format!("{name}:{r_name}");
985                            if left_schema.contains(name)
986                                && joined_schema.contains(aliased_name.as_str())
987                            {
988                                Some((name.to_string(), aliased_name))
989                            } else {
990                                None
991                            }
992                        })
993                        .collect::<PlHashMap<String, String>>(),
994                );
995            }
996        };
997        Ok(lf)
998    }

1088    /// Execute the 'SELECT' part of the query.
1089    fn execute_select(&mut self, select_stmt: &Select, query: &Query) -> PolarsResult<LazyFrame> {
1090        // Check that the statement doesn't contain unsupported SELECT clauses
1091        self.validate_select(select_stmt)?;
1092
1093        // Parse named windows first, as they may be referenced in the SELECT clause
1094        self.register_named_windows(&select_stmt.named_window)?;
1095
1096        // Get `FROM` table/data
1097        let (mut lf, base_table_name) = if select_stmt.from.is_empty() {
1098            (DataFrame::empty().lazy(), None)
1099        } else {
1100            // Note: implicit joins need more work to support properly,
1101            // explicit joins are preferred for now (ref: #16662)
1102            let from = select_stmt.clone().from;
1103            if from.len() > 1 {
1104                polars_bail!(SQLInterface: "multiple tables in FROM clause are not currently supported (found {}); use explicit JOIN syntax instead", from.len())
1105            }
1106            let tbl_expr = from.first().unwrap();
1107            let lf = self.execute_from_statement(tbl_expr)?;
1108            let base_name = get_table_name(&tbl_expr.relation);
1109            (lf, base_name)
1110        };
1111
1112        // Check for ambiguous column references in SELECT and WHERE (if there were joins)
1113        if let Some(ref base_name) = base_table_name {
1114            if !self.joined_aliases.is_empty() {
1115                // Extract USING columns from joins (these are coalesced and not ambiguous)
1116                let using_cols: PlHashSet<String> = select_stmt
1117                    .from
1118                    .first()
1119                    .into_iter()
1120                    .flat_map(|t| t.joins.iter())
1121                    .filter_map(|join| get_using_cols(&join.join_operator))
1122                    .flatten()
1123                    .collect();
1124
1125                // Check SELECT and WHERE expressions for ambiguous column references
1126                let check_expr = |e| {
1127                    check_for_ambiguous_column_refs(e, &self.joined_aliases, base_name, &using_cols)
1128                };
1129                for item in &select_stmt.projection {
1130                    match item {
1131                        SelectItem::UnnamedExpr(e) | SelectItem::ExprWithAlias { expr: e, .. } => {
1132                            check_expr(e)?
1133                        },
1134                        _ => {},
1135                    }
1136                }
1137                if let Some(ref where_expr) = select_stmt.selection {
1138                    check_expr(where_expr)?;
1139                }
1140            }
1141        }
1142
1143        // Apply `WHERE` constraint
1144        let mut schema = self.get_frame_schema(&mut lf)?;
1145        lf = self.process_where(lf, &select_stmt.selection, false, Some(schema.clone()))?;
1146
1147        // Determine projections
1148        let mut select_modifiers = SelectModifiers {
1149            ilike: None,
1150            exclude: PlHashSet::new(),
1151            rename: PlHashMap::new(),
1152            replace: vec![],
1153        };
1154
1155        // Collect window function cols if QUALIFY is present (we check at the
1156        // SQL level because empty OVER() clauses don't create Expr::Over)
1157        let window_fn_columns = if select_stmt.qualify.is_some() {
1158            select_stmt
1159                .projection
1160                .iter()
1161                .filter_map(|item| match item {
1162                    SelectItem::ExprWithAlias { expr, alias }
1163                        if expr_has_window_functions(expr) =>
1164                    {
1165                        Some(alias.value.clone())
1166                    },
1167                    _ => None,
1168                })
1169                .collect::<PlHashSet<_>>()
1170        } else {
1171            PlHashSet::new()
1172        };
1173
1174        let mut projections =
1175            self.column_projections(select_stmt, &schema, &mut select_modifiers)?;
1176
1177        // Apply `UNNEST` expressions
1178        let mut explode_names = Vec::new();
1179        let mut explode_exprs = Vec::new();
1180        let mut explode_lookup = PlHashMap::new();
1181
1182        for expr in &projections {
1183            for e in expr {
1184                if let Expr::Explode { input, .. } = e {
1185                    match input.as_ref() {
1186                        Expr::Column(name) => explode_names.push(name.clone()),
1187                        other_expr => {
1188                            // Note: skip aggregate expressions; those are handled in the GROUP BY phase
1189                            if !has_expr(other_expr, |e| matches!(e, Expr::Agg(_) | Expr::Len)) {
1190                                let temp_name =
1191                                    format_pl_smallstr!("__POLARS_UNNEST_{}", explode_exprs.len());
1192                                explode_exprs.push(other_expr.clone().alias(temp_name.as_str()));
1193                                explode_lookup.insert(other_expr.clone(), temp_name.clone());
1194                                explode_names.push(temp_name);
1195                            }
1196                        },
1197                    }
1198                }
1199            }
1200        }
1201        if !explode_names.is_empty() {
1202            if !explode_exprs.is_empty() {
1203                lf = lf.with_columns(explode_exprs);
1204            }
1205            lf = lf.explode(
1206                Selector::ByName {
1207                    names: Arc::from(explode_names),
1208                    strict: true,
1209                },
1210                ExplodeOptions {
1211                    empty_as_null: true,
1212                    keep_nulls: true,
1213                },
1214            );
1215            projections = projections
1216                .into_iter()
1217                .map(|p| {
1218                    // Update "projections" with column refs to the now-exploded expressions
1219                    p.map_expr(|e| match e {
1220                        Expr::Explode { input, .. } => explode_lookup
1221                            .get(input.as_ref())
1222                            .map(|name| Expr::Column(name.clone()))
1223                            .unwrap_or_else(|| input.as_ref().clone()),
1224                        _ => e,
1225                    })
1226                })
1227                .collect();
1228
1229            schema = self.get_frame_schema(&mut lf)?;
1230        }
1231
1232        // Check for "GROUP BY ..." (after determining projections)
1233        let mut group_by_keys: Vec<Expr> = Vec::new();
1234        match &select_stmt.group_by {
1235            // Standard "GROUP BY x, y, z" syntax (also recognising ordinal values)
1236            GroupByExpr::Expressions(group_by_exprs, modifiers) => {
1237                if !modifiers.is_empty() {
1238                    polars_bail!(SQLInterface: "GROUP BY does not support CUBE, ROLLUP, or TOTALS modifiers")
1239                }
1240                // Translate the group expressions, resolving ordinal values and SELECT aliases
1241                group_by_keys = group_by_exprs
1242                    .iter()
1243                    .map(|e| match e {
1244                        SQLExpr::Identifier(ident) => {
1245                            resolve_select_alias(&ident.value, &projections, &schema).map_or_else(
1246                                || {
1247                                    self.expr_or_ordinal(
1248                                        e,
1249                                        &projections,
1250                                        None,
1251                                        Some(&schema),
1252                                        "GROUP BY",
1253                                    )
1254                                },
1255                                Ok,
1256                            )
1257                        },
1258                        _ => self.expr_or_ordinal(e, &projections, None, Some(&schema), "GROUP BY"),
1259                    })
1260                    .collect::<PolarsResult<_>>()?
1261            },
1262            // "GROUP BY ALL" syntax; automatically adds expressions that do not contain
1263            // nested agg/window funcs to the group key (also ignores literals).
1264            GroupByExpr::All(modifiers) => {
1265                if !modifiers.is_empty() {
1266                    polars_bail!(SQLInterface: "GROUP BY does not support CUBE, ROLLUP, or TOTALS modifiers")
1267                }
1268                projections.iter().for_each(|expr| match expr {
1269                    // immediately match the most common cases (col|agg|len|lit, optionally aliased).
1270                    Expr::Agg(_) | Expr::Len | Expr::Literal(_) => (),
1271                    Expr::Column(_) => group_by_keys.push(expr.clone()),
1272                    Expr::Alias(e, _)
1273                        if matches!(&**e, Expr::Agg(_) | Expr::Len | Expr::Literal(_)) => {},
1274                    Expr::Alias(e, _) if matches!(&**e, Expr::Column(_)) => {
1275                        if let Expr::Column(name) = &**e {
1276                            group_by_keys.push(col(name.clone()));
1277                        }
1278                    },
1279                    _ => {
1280                        // If not quick-matched, add if no nested agg/window expressions
1281                        if !has_expr(expr, |e| {
1282                            matches!(e, Expr::Agg(_))
1283                                || matches!(e, Expr::Len)
1284                                || matches!(e, Expr::Over { .. })
1285                                || {
1286                                    #[cfg(feature = "dynamic_group_by")]
1287                                    {
1288                                        matches!(e, Expr::Rolling { .. })
1289                                    }
1290                                    #[cfg(not(feature = "dynamic_group_by"))]
1291                                    {
1292                                        false
1293                                    }
1294                                }
1295                        }) {
1296                            group_by_keys.push(expr.clone())
1297                        }
1298                    },
1299                });
1300            },
1301        };
1302
1303        lf = if group_by_keys.is_empty() {
1304            // The 'having' clause is only valid inside 'group by'
1305            if select_stmt.having.is_some() {
1306                polars_bail!(SQLSyntax: "HAVING clause not valid outside of GROUP BY; found:\n{:?}", select_stmt.having);
1307            };
1308
1309            // Final/selected cols, accounting for 'SELECT *' modifiers
1310            let mut retained_cols = Vec::with_capacity(projections.len());
1311            let mut retained_names = Vec::with_capacity(projections.len());
1312            let have_order_by = query.order_by.is_some();
1313
1314            // Initialize containing InheritsContext to handle empty projection case.
1315            let mut projection_heights = ExprSqlProjectionHeightBehavior::InheritsContext;
1316
1317            // Note: if there is an 'order by' then we project everything (original cols
1318            // and new projections) and *then* select the final cols; the retained cols
1319            // are used to ensure a correct final projection. If there's no 'order by',
1320            // clause then we can project the final column *expressions* directly.
1321            for p in projections.iter() {
1322                let name = p.to_field(schema.deref())?.name.to_string();
1323                if select_modifiers.matches_ilike(&name)
1324                    && !select_modifiers.exclude.contains(&name)
1325                {
1326                    projection_heights |= ExprSqlProjectionHeightBehavior::identify_from_expr(p);
1327
1328                    retained_cols.push(if have_order_by {
1329                        col(name.as_str())
1330                    } else {
1331                        p.clone()
1332                    });
1333                    retained_names.push(col(name));
1334                }
1335            }
1336
1337            // Apply the remaining modifiers and establish the final projection
1338            if have_order_by {
1339                // We can safely use `with_columns()` and avoid a join if:
1340                // * There is already a projection that projects to the table height.
1341                // * All projection heights inherit from context (e.g. all scalar literals that
1342                //   are to be broadcasted to table height).
1343                if projection_heights.contains(ExprSqlProjectionHeightBehavior::MaintainsColumn)
1344                    || projection_heights == ExprSqlProjectionHeightBehavior::InheritsContext
1345                {
1346                    lf = lf.with_columns(projections);
1347                } else {
1348                    // We hit this branch if the output height is not guaranteed to match the table
1349                    // height. E.g.:
1350                    //
1351                    // * SELECT COUNT(*) FROM df ORDER BY sort_key;
1352                    //
1353                    // For these cases we truncate / extend the sorting columns with NULLs to match
1354                    // the output height. We do this by projecting independently and then joining
1355                    // back the original frame on the row index.
1356                    const NAME: PlSmallStr = PlSmallStr::from_static("__PL_INDEX");
1357                    lf = lf
1358                        .clone()
1359                        .select(projections)
1360                        .with_row_index(NAME, None)
1361                        .join(
1362                            lf.with_row_index(NAME, None),
1363                            [col(NAME)],
1364                            [col(NAME)],
1365                            JoinArgs {
1366                                how: JoinType::Left,
1367                                validation: Default::default(),
1368                                suffix: None,
1369                                slice: None,
1370                                nulls_equal: false,
1371                                coalesce: Default::default(),
1372                                maintain_order: polars_ops::frame::MaintainOrderJoin::Left,
1373                                build_side: None,
1374                            },
1375                        );
1376                }
1377            }
1378            if !select_modifiers.replace.is_empty() {
1379                lf = lf.with_columns(&select_modifiers.replace);
1380            }
1381            if !select_modifiers.rename.is_empty() {
1382                lf = lf.with_columns(select_modifiers.renamed_cols());
1383            }
1384            lf = self.process_order_by(lf, &query.order_by, Some(&retained_cols))?;
1385
1386            // Note: If `have_order_by`, with_columns is already done above.
1387            if projection_heights == ExprSqlProjectionHeightBehavior::InheritsContext
1388                && !have_order_by
1389            {
1390                // All projections need to be broadcasted to table height, so evaluate in `with_columns()`
1391                lf = lf.with_columns(retained_cols).select(retained_names);
1392            } else {
1393                lf = lf.select(retained_cols);
1394            }
1395            if !select_modifiers.rename.is_empty() {
1396                lf = lf.rename(
1397                    select_modifiers.rename.keys(),
1398                    select_modifiers.rename.values(),
1399                    true,
1400                );
1401            };
1402            lf
1403        } else {
1404            let having = select_stmt
1405                .having
1406                .as_ref()
1407                .map(|expr| parse_sql_expr(expr, self, Some(&schema)))
1408                .transpose()?;
1409            lf = self.process_group_by(lf, &group_by_keys, &projections, having)?;
1410            lf = self.process_order_by(lf, &query.order_by, None)?;
1411
1412            // Drop any extra columns (eg: added to maintain ORDER BY access to original cols)
1413            let output_cols: Vec<_> = projections
1414                .iter()
1415                .map(|p| p.to_field(&schema))
1416                .collect::<PolarsResult<Vec<_>>>()?
1417                .into_iter()
1418                .map(|f| col(f.name))
1419                .collect();
1420
1421            lf.select(&output_cols)
1422        };
1423
1424        // Apply optional QUALIFY clause (filters on window functions).
1425        lf = self.process_qualify(lf, &select_stmt.qualify, &window_fn_columns)?;
1426
1427        // Apply optional DISTINCT clause.
1428        lf = match &select_stmt.distinct {
1429            Some(Distinct::Distinct) => lf.unique_stable(None, UniqueKeepStrategy::Any),
1430            Some(Distinct::On(exprs)) => {
1431                // TODO: support exprs in `unique` see https://github.com/pola-rs/polars/issues/5760
1432                let schema = Some(self.get_frame_schema(&mut lf)?);
1433                let cols = exprs
1434                    .iter()
1435                    .map(|e| {
1436                        let expr = parse_sql_expr(e, self, schema.as_deref())?;
1437                        if let Expr::Column(name) = expr {
1438                            Ok(name)
1439                        } else {
1440                            Err(polars_err!(SQLSyntax:"DISTINCT ON only supports column names"))
1441                        }
1442                    })
1443                    .collect::<PolarsResult<Vec<_>>>()?;
1444
1445                // DISTINCT ON has to apply the ORDER BY before the operation.
1446                lf = self.process_order_by(lf, &query.order_by, None)?;
1447                return Ok(lf.unique_stable(
1448                    Some(Selector::ByName {
1449                        names: cols.into(),
1450                        strict: true,
1451                    }),
1452                    UniqueKeepStrategy::First,
1453                ));
1454            },
1455            None => lf,
1456        };
1457        Ok(lf)
1458    }

1460    fn column_projections(
1461        &mut self,
1462        select_stmt: &Select,
1463        schema: &SchemaRef,
1464        select_modifiers: &mut SelectModifiers,
1465    ) -> PolarsResult<Vec<Expr>> {
1466        let mut items: Vec<ProjectionItem> = Vec::with_capacity(select_stmt.projection.len());
1467        let mut has_qualified_wildcard = false;
1468
1469        for select_item in &select_stmt.projection {
1470            match select_item {
1471                SelectItem::UnnamedExpr(expr) => {
1472                    items.push(ProjectionItem::Exprs(vec![parse_sql_expr(
1473                        expr,
1474                        self,
1475                        Some(schema),
1476                    )?]));
1477                },
1478                SelectItem::ExprWithAlias { expr, alias } => {
1479                    let expr = parse_sql_expr(expr, self, Some(schema))?;
1480                    items.push(ProjectionItem::Exprs(vec![
1481                        expr.alias(PlSmallStr::from_str(alias.value.as_str())),
1482                    ]));
1483                },
1484                SelectItem::QualifiedWildcard(kind, wildcard_options) => match kind {
1485                    SelectItemQualifiedWildcardKind::ObjectName(obj_name) => {
1486                        let tbl_name = obj_name
1487                            .0
1488                            .last()
1489                            .and_then(|p| p.as_ident())
1490                            .map(|i| PlSmallStr::from_str(&i.value))
1491                            .unwrap_or_default();
1492                        let exprs = self.process_qualified_wildcard(
1493                            obj_name,
1494                            wildcard_options,
1495                            select_modifiers,
1496                            Some(schema),
1497                        )?;
1498                        items.push(ProjectionItem::QualifiedExprs(tbl_name, exprs));
1499                        has_qualified_wildcard = true;
1500                    },
1501                    SelectItemQualifiedWildcardKind::Expr(_) => {
1502                        polars_bail!(SQLSyntax: "qualified wildcard on expressions not yet supported: {:?}", select_item)
1503                    },
1504                },
1505                SelectItem::Wildcard(wildcard_options) => {
1506                    let cols = schema
1507                        .iter_names()
1508                        .map(|name| col(name.clone()))
1509                        .collect::<Vec<_>>();
1510
1511                    items.push(ProjectionItem::Exprs(
1512                        self.process_wildcard_additional_options(
1513                            cols,
1514                            wildcard_options,
1515                            select_modifiers,
1516                            Some(schema),
1517                        )?,
1518                    ));
1519                },
1520            }
1521        }
1522
1523        // Disambiguate qualified wildcards (if any) and flatten expressions
1524        let exprs = if has_qualified_wildcard {
1525            disambiguate_projection_cols(items, schema)?
1526        } else {
1527            items
1528                .into_iter()
1529                .flat_map(|item| match item {
1530                    ProjectionItem::Exprs(exprs) | ProjectionItem::QualifiedExprs(_, exprs) => {
1531                        exprs
1532                    },
1533                })
1534                .collect()
1535        };
1536        let flattened_exprs = exprs
1537            .into_iter()
1538            .flat_map(|expr| expand_exprs(expr, schema))
1539            .collect();
1540
1541        Ok(flattened_exprs)
1542    }

1544    fn process_where(
1545        &mut self,
1546        mut lf: LazyFrame,
1547        expr: &Option<SQLExpr>,
1548        invert_filter: bool,
1549        schema: Option<SchemaRef>,
1550    ) -> PolarsResult<LazyFrame> {
1551        if let Some(expr) = expr {
1552            let schema = match schema {
1553                None => self.get_frame_schema(&mut lf)?,
1554                Some(s) => s,
1555            };
1556
1557            // shortcut filter evaluation if given expression is just TRUE or FALSE
1558            let (all_true, all_false) = match expr {
1559                SQLExpr::Value(ValueWithSpan {
1560                    value: SQLValue::Boolean(b),
1561                    ..
1562                }) => (*b, !*b),
1563                SQLExpr::BinaryOp { left, op, right } => match (&**left, &**right, op) {
1564                    (SQLExpr::Value(a), SQLExpr::Value(b), BinaryOperator::Eq) => {
1565                        (a.value == b.value, a.value != b.value)
1566                    },
1567                    (SQLExpr::Value(a), SQLExpr::Value(b), BinaryOperator::NotEq) => {
1568                        (a.value != b.value, a.value == b.value)
1569                    },
1570                    _ => (false, false),
1571                },
1572                _ => (false, false),
1573            };
1574            if (all_true && !invert_filter) || (all_false && invert_filter) {
1575                return Ok(lf);
1576            } else if (all_false && !invert_filter) || (all_true && invert_filter) {
1577                return Ok(lf.clear());
1578            }
1579
1580            // ...otherwise parse and apply the filter as normal
1581            let mut filter_expression = parse_sql_expr(expr, self, Some(schema).as_deref())?;
1582            if filter_expression.clone().meta().has_multiple_outputs() {
1583                filter_expression = all_horizontal([filter_expression])?;
1584            }
1585            lf = self.process_subqueries(lf, vec![&mut filter_expression]);
1586            lf = if invert_filter {
1587                lf.remove(filter_expression)
1588            } else {
1589                lf.filter(filter_expression)
1590            };
1591        }
1592        Ok(lf)
1593    }

1595    pub(super) fn process_join(
1596        &mut self,
1597        tbl_left: &TableInfo,
1598        tbl_right: &TableInfo,
1599        constraint: &JoinConstraint,
1600        join_type: JoinType,
1601    ) -> PolarsResult<LazyFrame> {
1602        let (left_on, right_on) = process_join_constraint(constraint, tbl_left, tbl_right, self)?;
1603        let coalesce_type = match constraint {
1604            // "NATURAL" joins should coalesce; otherwise we disambiguate
1605            JoinConstraint::Natural => JoinCoalesce::CoalesceColumns,
1606            _ => JoinCoalesce::KeepColumns,
1607        };
1608        let joined = tbl_left
1609            .frame
1610            .clone()
1611            .join_builder()
1612            .with(tbl_right.frame.clone())
1613            .left_on(left_on)
1614            .right_on(right_on)
1615            .how(join_type)
1616            .suffix(format!(":{}", tbl_right.name))
1617            .coalesce(coalesce_type)
1618            .finish();
1619
1620        Ok(joined)
1621    }

1623    fn process_qualify(
1624        &mut self,
1625        mut lf: LazyFrame,
1626        qualify_expr: &Option<SQLExpr>,
1627        window_fn_columns: &PlHashSet<String>,
1628    ) -> PolarsResult<LazyFrame> {
1629        if let Some(expr) = qualify_expr {
1630            // Check the QUALIFY expression to identify window functions
1631            // and collect column refs (for looking up aliases from SELECT)
1632            let (has_window_fns, column_refs) = QualifyExpression::analyze(expr);
1633            let references_window_alias = column_refs.iter().any(|c| window_fn_columns.contains(c));
1634            if !has_window_fns && !references_window_alias {
1635                polars_bail!(
1636                    SQLSyntax:
1637                    "QUALIFY clause must reference window functions either explicitly or via SELECT aliases"
1638                );
1639            }
1640            let schema = self.get_frame_schema(&mut lf)?;
1641            let mut filter_expression = parse_sql_expr(expr, self, Some(&schema))?;
1642            if filter_expression.clone().meta().has_multiple_outputs() {
1643                filter_expression = all_horizontal([filter_expression])?;
1644            }
1645            lf = self.process_subqueries(lf, vec![&mut filter_expression]);
1646            lf = lf.filter(filter_expression);
1647        }
1648        Ok(lf)
1649    }

1651    fn process_subqueries(&self, lf: LazyFrame, exprs: Vec<&mut Expr>) -> LazyFrame {
1652        let mut subplans = vec![];
1653
1654        for e in exprs {
1655            *e = e.clone().map_expr(|e| {
1656                if let Expr::SubPlan(lp, names) = e {
1657                    assert_eq!(
1658                        names.len(),
1659                        1,
1660                        "multiple columns in subqueries not yet supported"
1661                    );
1662
1663                    let select_expr = names[0].1.clone();
1664                    let cb =
1665                        PlanCallback::new(move |(plans, schemas): (Vec<DslPlan>, Vec<SchemaRef>)| {
1666                            let schema = &schemas[0];
1667                            polars_ensure!(schema.len() == 1,  SQLSyntax: "SQL subquery returns more than one column");
1668                            Ok(LazyFrame::from(plans.into_iter().next().unwrap()).select([select_expr.clone()]).logical_plan)
1669                        });
1670                    subplans.push(LazyFrame::from((**lp).clone()).pipe_with_schema(cb));
1671                    Expr::Column(names[0].0.clone()).first()
1672                } else {
1673                    e
1674                }
1675            });
1676        }
1677
1678        if subplans.is_empty() {
1679            lf
1680        } else {
1681            subplans.insert(0, lf);
1682            concat_lf_horizontal(
1683                subplans,
1684                HConcatOptions {
1685                    broadcast_unit_length: true,
1686                    ..Default::default()
1687                },
1688            )
1689            .unwrap()
1690        }
1691    }

1693    fn execute_create_table(&mut self, stmt: &Statement) -> PolarsResult<LazyFrame> {
1694        if let Statement::CreateTable(CreateTable {
1695            if_not_exists,
1696            name,
1697            query,
1698            columns,
1699            like,
1700            ..
1701        }) = stmt
1702        {
1703            let tbl_name = name.0.first().unwrap().as_ident().unwrap().value.as_str();
1704            if *if_not_exists && self.table_map.read().unwrap().contains_key(tbl_name) {
1705                polars_bail!(SQLInterface: "relation '{}' already exists", tbl_name);
1706            }
1707            let lf = match (query, columns.is_empty(), like) {
1708                (Some(query), true, None) => {
1709                    // ----------------------------------------------------
1710                    // CREATE TABLE [IF NOT EXISTS] <name> AS <query>
1711                    // ----------------------------------------------------
1712                    self.execute_query(query)?
1713                },
1714                (None, false, None) => {
1715                    // ----------------------------------------------------
1716                    // CREATE TABLE [IF NOT EXISTS] <name> (<coldef>, ...)
1717                    // ----------------------------------------------------
1718                    let mut schema = Schema::with_capacity(columns.len());
1719                    for col in columns {
1720                        let col_name = col.name.value.as_str();
1721                        let dtype = map_sql_dtype_to_polars(&col.data_type)?;
1722                        schema.insert_at_index(schema.len(), col_name.into(), dtype)?;
1723                    }
1724                    DataFrame::empty_with_schema(&schema).lazy()
1725                },
1726                (None, true, Some(like_kind)) => {
1727                    // ----------------------------------------------------
1728                    // CREATE TABLE [IF NOT EXISTS] <name> LIKE <table>
1729                    // ----------------------------------------------------
1730                    let like_name = match like_kind {
1731                        CreateTableLikeKind::Plain(like)
1732                        | CreateTableLikeKind::Parenthesized(like) => &like.name,
1733                    };
1734                    let like_table = like_name
1735                        .0
1736                        .first()
1737                        .unwrap()
1738                        .as_ident()
1739                        .unwrap()
1740                        .value
1741                        .as_str();
1742                    if let Some(table) = self.table_map.read().unwrap().get(like_table).cloned() {
1743                        table.clear()
1744                    } else {
1745                        polars_bail!(SQLInterface: "table given in LIKE does not exist: {}", like_table)
1746                    }
1747                },
1748                // No valid options provided
1749                (None, true, None) => {
1750                    polars_bail!(SQLInterface: "CREATE TABLE expected a query, column definitions, or LIKE clause")
1751                },
1752                // Mutually exclusive options
1753                _ => {
1754                    polars_bail!(
1755                        SQLInterface: "CREATE TABLE received mutually exclusive options:\nquery = {:?}\ncolumns = {:?}\nlike = {:?}",
1756                        query,
1757                        columns,
1758                        like,
1759                    )
1760                },
1761            };
1762            self.register(tbl_name, lf);
1763
1764            let df_created = df! { "Response" => [format!("CREATE TABLE {}", name.0.first().unwrap().as_ident().unwrap().value)] };
1765            Ok(df_created.unwrap().lazy())
1766        } else {
1767            unreachable!()
1768        }
1769    }

1771    fn get_table(&mut self, relation: &TableFactor) -> PolarsResult<(String, LazyFrame)> {
1772        match relation {
1773            TableFactor::Table {
1774                name, alias, args, ..
1775            } => {
1776                if let Some(args) = args {
1777                    return self.execute_table_function(name, alias, &args.args);
1778                }
1779                let tbl_name = name.0.first().unwrap().as_ident().unwrap().value.as_str();
1780                if let Some(lf) = self.get_table_from_current_scope(tbl_name) {
1781                    match alias {
1782                        Some(alias) => {
1783                            self.table_aliases
1784                                .insert(alias.name.value.clone(), tbl_name.to_string());
1785                            Ok((alias.name.value.clone(), lf))
1786                        },
1787                        None => Ok((tbl_name.to_string(), lf)),
1788                    }
1789                } else {
1790                    polars_bail!(SQLInterface: "relation '{}' was not found", tbl_name);
1791                }
1792            },
1793            TableFactor::Derived {
1794                lateral,
1795                subquery,
1796                alias,
1797            } => {
1798                polars_ensure!(!(*lateral), SQLInterface: "LATERAL not supported");
1799                if let Some(alias) = alias {
1800                    let mut lf = self.execute_query_no_ctes(subquery)?;
1801                    lf = self.rename_columns_from_table_alias(lf, alias)?;
1802                    self.table_map
1803                        .write()
1804                        .unwrap()
1805                        .insert(alias.name.value.clone(), lf.clone());
1806                    Ok((alias.name.value.clone(), lf))
1807                } else {
1808                    let lf = self.execute_query_no_ctes(subquery)?;
1809                    Ok(("".to_string(), lf))
1810                }
1811            },
1812            TableFactor::UNNEST {
1813                alias,
1814                array_exprs,
1815                with_offset,
1816                with_offset_alias: _,
1817                ..
1818            } => {
1819                if let Some(alias) = alias {
1820                    let column_names: Vec<Option<PlSmallStr>> = alias
1821                        .columns
1822                        .iter()
1823                        .map(|c| {
1824                            if c.name.value.is_empty() {
1825                                None
1826                            } else {
1827                                Some(PlSmallStr::from_str(c.name.value.as_str()))
1828                            }
1829                        })
1830                        .collect();
1831
1832                    let column_values: Vec<Series> = array_exprs
1833                        .iter()
1834                        .map(|arr| parse_sql_array(arr, self))
1835                        .collect::<Result<_, _>>()?;
1836
1837                    polars_ensure!(!column_names.is_empty(),
1838                        SQLSyntax:
1839                        "UNNEST table alias must also declare column names, eg: {} (a,b,c)", alias.name.to_string()
1840                    );
1841                    if column_names.len() != column_values.len() {
1842                        let plural = if column_values.len() > 1 { "s" } else { "" };
1843                        polars_bail!(
1844                            SQLSyntax:
1845                            "UNNEST table alias requires {} column name{}, found {}", column_values.len(), plural, column_names.len()
1846                        );
1847                    }
1848                    let column_series: Vec<Column> = column_values
1849                        .into_iter()
1850                        .zip(column_names)
1851                        .map(|(s, name)| {
1852                            if let Some(name) = name {
1853                                s.with_name(name)
1854                            } else {
1855                                s
1856                            }
1857                        })
1858                        .map(Column::from)
1859                        .collect();
1860
1861                    let lf = DataFrame::new_infer_height(column_series)?.lazy();
1862
1863                    if *with_offset {
1864                        // TODO: support 'WITH ORDINALITY|OFFSET' modifier.
1865                        polars_bail!(SQLInterface: "UNNEST tables do not (yet) support WITH ORDINALITY|OFFSET");
1866                    }
1867                    let table_name = alias.name.value.clone();
1868                    self.table_map
1869                        .write()
1870                        .unwrap()
1871                        .insert(table_name.clone(), lf.clone());
1872                    Ok((table_name, lf))
1873                } else {
1874                    polars_bail!(SQLSyntax: "UNNEST table must have an alias");
1875                }
1876            },
1877            TableFactor::NestedJoin {
1878                table_with_joins,
1879                alias,
1880            } => {
1881                let lf = self.execute_from_statement(table_with_joins)?;
1882                match alias {
1883                    Some(a) => Ok((a.name.value.clone(), lf)),
1884                    None => Ok(("".to_string(), lf)),
1885                }
1886            },
1887            // Support bare table, optionally with an alias, for now
1888            _ => polars_bail!(SQLInterface: "not yet implemented: {}", relation),
1889        }
1890    }

1914    fn process_order_by(
1915        &mut self,
1916        mut lf: LazyFrame,
1917        order_by: &Option<OrderBy>,
1918        selected: Option<&[Expr]>,
1919    ) -> PolarsResult<LazyFrame> {
1920        if order_by.as_ref().is_none_or(|ob| match &ob.kind {
1921            OrderByKind::Expressions(exprs) => exprs.is_empty(),
1922            OrderByKind::All(_) => false,
1923        }) {
1924            return Ok(lf);
1925        }
1926        let schema = self.get_frame_schema(&mut lf)?;
1927        let columns_iter = schema.iter_names().map(|e| col(e.clone()));
1928        let (order_by, order_by_all, n_order_cols) = match &order_by.as_ref().unwrap().kind {
1929            OrderByKind::Expressions(exprs) => {
1930                // TODO: will look at making an upstream PR that allows us to more easily
1931                //  create a GenericDialect variant supporting "OrderByKind::All" instead
1932                if exprs.len() == 1
1933                    && matches!(&exprs[0].expr, SQLExpr::Identifier(ident)
1934                        if ident.value.to_uppercase() == "ALL"
1935                        && !schema.iter_names().any(|name| name.to_uppercase() == "ALL"))
1936                {
1937                    // Treat as ORDER BY ALL
1938                    let n_cols = if let Some(selected) = selected {
1939                        selected.len()
1940                    } else {
1941                        schema.len()
1942                    };
1943                    (vec![], Some(&exprs[0].options), n_cols)
1944                } else {
1945                    (exprs.clone(), None, exprs.len())
1946                }
1947            },
1948            OrderByKind::All(opts) => {
1949                let n_cols = if let Some(selected) = selected {
1950                    selected.len()
1951                } else {
1952                    schema.len()
1953                };
1954                (vec![], Some(opts), n_cols)
1955            },
1956        };
1957        let mut descending = Vec::with_capacity(n_order_cols);
1958        let mut nulls_last = Vec::with_capacity(n_order_cols);
1959        let mut by: Vec<Expr> = Vec::with_capacity(n_order_cols);
1960
1961        if let Some(opts) = order_by_all {
1962            if let Some(selected) = selected {
1963                by.extend(selected.iter().cloned());
1964            } else {
1965                by.extend(columns_iter);
1966            };
1967            let desc_order = !opts.asc.unwrap_or(true);
1968            nulls_last.resize(by.len(), !opts.nulls_first.unwrap_or(desc_order));
1969            descending.resize(by.len(), desc_order);
1970        } else {
1971            let columns = &columns_iter.collect::<Vec<_>>();
1972            for ob in order_by {
1973                // note: if not specified 'NULLS FIRST' is default for DESC, 'NULLS LAST' otherwise
1974                // https://www.postgresql.org/docs/current/queries-order.html
1975                let desc_order = !ob.options.asc.unwrap_or(true);
1976                nulls_last.push(!ob.options.nulls_first.unwrap_or(desc_order));
1977                descending.push(desc_order);
1978
1979                // translate order expression, allowing ordinal values
1980                by.push(self.expr_or_ordinal(
1981                    &ob.expr,
1982                    columns,
1983                    selected,
1984                    Some(&schema),
1985                    "ORDER BY",
1986                )?)
1987            }
1988        }
1989        Ok(lf.sort_by_exprs(
1990            &by,
1991            SortMultipleOptions::default()
1992                .with_order_descending_multi(descending)
1993                .with_nulls_last_multi(nulls_last),
1994        ))
1995    }

1997    fn process_group_by(
1998        &mut self,
1999        mut lf: LazyFrame,
2000        group_by_keys: &[Expr],
2001        projections: &[Expr],
2002        having: Option<Expr>,
2003    ) -> PolarsResult<LazyFrame> {
2004        let schema_before = self.get_frame_schema(&mut lf)?;
2005        let group_by_keys_schema =
2006            expressions_to_schema(group_by_keys, &schema_before, |duplicate_name: &str| {
2007                format!("group_by keys contained duplicate output name '{duplicate_name}'")
2008            })?;
2009
2010        // Note: remove the `group_by` keys as Polars adds those implicitly.
2011        let mut aliased_aggregations: PlHashMap<PlSmallStr, PlSmallStr> = PlHashMap::new();
2012        let mut aggregation_projection = Vec::with_capacity(projections.len());
2013        let mut projection_overrides = PlHashMap::with_capacity(projections.len());
2014        let mut projection_aliases = PlHashSet::new();
2015        let mut group_key_aliases = PlHashSet::new();
2016
2017        // Pre-compute group key data (stripped expression + output name) to avoid repeated work.
2018        // We check both expression AND output name match to avoid cross-aliasing issues.
2019        let group_key_data: Vec<_> = group_by_keys
2020            .iter()
2021            .map(|gk| {
2022                (
2023                    strip_outer_alias(gk),
2024                    gk.to_field(&schema_before).ok().map(|f| f.name),
2025                )
2026            })
2027            .collect();
2028
2029        let projection_matches_group_key: Vec<bool> = projections
2030            .iter()
2031            .map(|p| {
2032                let p_stripped = strip_outer_alias(p);
2033                let p_name = p.to_field(&schema_before).ok().map(|f| f.name);
2034                group_key_data
2035                    .iter()
2036                    .any(|(gk_stripped, gk_name)| *gk_stripped == p_stripped && *gk_name == p_name)
2037            })
2038            .collect();
2039
2040        for (e, &matches_group_key) in projections.iter().zip(&projection_matches_group_key) {
2041            // `Len` represents COUNT(*) so we treat as an aggregation here.
2042            let is_non_group_key_expr = !matches_group_key
2043                && has_expr(e, |e| {
2044                    match e {
2045                        Expr::Agg(_) | Expr::Len | Expr::Over { .. } => true,
2046                        #[cfg(feature = "dynamic_group_by")]
2047                        Expr::Rolling { .. } => true,
2048                        Expr::Function { function: func, .. }
2049                            if !matches!(func, FunctionExpr::StructExpr(_)) =>
2050                        {
2051                            // If it's a function call containing a column NOT in the group by keys,
2052                            // we treat it as an aggregation.
2053                            has_expr(e, |e| match e {
2054                                Expr::Column(name) => !group_by_keys_schema.contains(name),
2055                                _ => false,
2056                            })
2057                        },
2058                        _ => false,
2059                    }
2060                });
2061
2062            // Note: if simple aliased expression we defer aliasing until after the group_by.
2063            // Use `e_inner` to track the potentially unwrapped expression for field lookup.
2064            let mut e_inner = e;
2065            if let Expr::Alias(expr, alias) = e {
2066                if e.clone().meta().is_simple_projection(Some(&schema_before)) {
2067                    group_key_aliases.insert(alias.as_ref());
2068                    e_inner = expr
2069                } else if let Expr::Function {
2070                    function: FunctionExpr::StructExpr(StructFunction::FieldByName(name)),
2071                    ..
2072                } = expr.deref()
2073                {
2074                    projection_overrides
2075                        .insert(alias.as_ref(), col(name.clone()).alias(alias.clone()));
2076                } else if !is_non_group_key_expr && !group_by_keys_schema.contains(alias) {
2077                    projection_aliases.insert(alias.as_ref());
2078                }
2079            }
2080            let field = e_inner.to_field(&schema_before)?;
2081            if is_non_group_key_expr {
2082                let mut e = e.clone();
2083                if let Expr::Agg(AggExpr::Implode(expr)) = &e {
2084                    e = (**expr).clone();
2085                } else if let Expr::Alias(expr, name) = &e {
2086                    if let Expr::Agg(AggExpr::Implode(expr)) = expr.as_ref() {
2087                        e = (**expr).clone().alias(name.clone());
2088                    }
2089                }
2090                // If aggregation colname conflicts with a group key,
2091                // alias it to avoid duplicate/mis-tracked columns
2092                if group_by_keys_schema.get(&field.name).is_some() {
2093                    let alias_name = format!("__POLARS_AGG_{}", field.name);
2094                    e = e.alias(alias_name.as_str());
2095                    aliased_aggregations.insert(field.name.clone(), alias_name.as_str().into());
2096                }
2097                aggregation_projection.push(e);
2098            } else if !matches_group_key {
2099                // Non-aggregated columns must be part of the GROUP BY clause
2100                if let Expr::Column(_)
2101                | Expr::Function {
2102                    function: FunctionExpr::StructExpr(StructFunction::FieldByName(_)),
2103                    ..
2104                } = e_inner
2105                {
2106                    if !group_by_keys_schema.contains(&field.name) {
2107                        polars_bail!(SQLSyntax: "'{}' should participate in the GROUP BY clause or an aggregate function", &field.name);
2108                    }
2109                }
2110            }
2111        }
2112
2113        // Process HAVING clause: identify aggregate expressions, reusing those already
2114        // in projections, or compute as temporary columns and then post-filter/discard
2115        let having_filter = if let Some(having_expr) = having {
2116            let mut agg_to_name: Vec<(Expr, PlSmallStr)> = aggregation_projection
2117                .iter()
2118                .filter_map(|p| match p {
2119                    Expr::Alias(inner, name) if matches!(**inner, Expr::Agg(_) | Expr::Len) => {
2120                        Some((inner.as_ref().clone(), name.clone()))
2121                    },
2122                    e @ (Expr::Agg(_) | Expr::Len) => Some((
2123                        e.clone(),
2124                        e.to_field(&schema_before)
2125                            .map(|f| f.name)
2126                            .unwrap_or_default(),
2127                    )),
2128                    _ => None,
2129                })
2130                .collect();
2131
2132            let mut n_having_aggs = 0;
2133            let updated_having = having_expr.map_expr(|e| {
2134                if !matches!(&e, Expr::Agg(_) | Expr::Len) {
2135                    return e;
2136                }
2137                let name = agg_to_name
2138                    .iter()
2139                    .find_map(|(expr, n)| (*expr == e).then(|| n.clone()))
2140                    .unwrap_or_else(|| {
2141                        let n = format_pl_smallstr!("__POLARS_HAVING_{n_having_aggs}");
2142                        aggregation_projection.push(e.clone().alias(n.clone()));
2143                        agg_to_name.push((e.clone(), n.clone()));
2144                        n_having_aggs += 1;
2145                        n
2146                    });
2147                col(name)
2148            });
2149            Some(updated_having)
2150        } else {
2151            None
2152        };
2153
2154        // Apply HAVING filter after aggregation
2155        let mut aggregated = lf.group_by(group_by_keys).agg(&aggregation_projection);
2156        if let Some(filter_expr) = having_filter {
2157            aggregated = aggregated.filter(filter_expr);
2158        }
2159
2160        let projection_schema =
2161            expressions_to_schema(projections, &schema_before, |duplicate_name: &str| {
2162                format!("group_by aggregations contained duplicate output name '{duplicate_name}'")
2163            })?;
2164
2165        // A final projection to get the proper order and any deferred transforms/aliases
2166        // (will also drop any temporary columns created for the HAVING post-filter).
2167        let final_projection = projection_schema
2168            .iter_names()
2169            .zip(projections.iter().zip(&projection_matches_group_key))
2170            .map(|(name, (projection_expr, &matches_group_key))| {
2171                if let Some(expr) = projection_overrides.get(name.as_str()) {
2172                    expr.clone()
2173                } else if let Some(aliased_name) = aliased_aggregations.get(name) {
2174                    col(aliased_name.clone()).alias(name.clone())
2175                } else if group_by_keys_schema.get(name).is_some() && matches_group_key {
2176                    col(name.clone())
2177                } else if group_by_keys_schema.get(name).is_some()
2178                    || projection_aliases.contains(name.as_str())
2179                    || group_key_aliases.contains(name.as_str())
2180                {
2181                    if has_expr(projection_expr, |e| {
2182                        matches!(e, Expr::Agg(_) | Expr::Len | Expr::Over { .. })
2183                    }) {
2184                        col(name.clone())
2185                    } else {
2186                        projection_expr.clone()
2187                    }
2188                } else {
2189                    col(name.clone())
2190                }
2191            })
2192            .collect::<Vec<_>>();
2193
2194        // Include original GROUP BY columns for ORDER BY access (if aliased).
2195        let mut output_projection = final_projection;
2196        for key_name in group_by_keys_schema.iter_names() {
2197            if !projection_schema.contains(key_name) {
2198                // Original col name not in output - add for ORDER BY access
2199                output_projection.push(col(key_name.clone()));
2200            } else if group_by_keys.iter().any(|k| is_simple_col_ref(k, key_name)) {
2201                // Original col name in output - check if cross-aliased
2202                let is_cross_aliased = projections.iter().any(|p| {
2203                    p.to_field(&schema_before).is_ok_and(|f| f.name == key_name)
2204                        && !is_simple_col_ref(p, key_name)
2205                });
2206                if is_cross_aliased {
2207                    // Add original name under a prefixed alias for subsequent ORDER BY resolution
2208                    let internal_name = format_pl_smallstr!("__POLARS_ORIG_{}", key_name);
2209                    output_projection.push(col(key_name.clone()).alias(internal_name));
2210                }
2211            }
2212        }
2213        Ok(aggregated.select(&output_projection))
2214    }

2217        &self,
2218        lf: LazyFrame,
2219        limit_clause: &Option<LimitClause>,
2220        fetch: &Option<Fetch>,
2221    ) -> PolarsResult<LazyFrame> {
2222        // Extract limit and offset from LimitClause
2223        let (limit, offset) = match limit_clause {
2224            Some(LimitClause::LimitOffset {
2225                limit,
2226                offset,
2227                limit_by,
2228            }) => {
2229                if !limit_by.is_empty() {
2230                    // TODO: might be able to support as an aggregate `top_k_by` operation?
2231                    //  (https://clickhouse.com/docs/sql-reference/statements/select/limit-by)
2232                    polars_bail!(SQLSyntax: "`LIMIT <n> BY <exprs>` clause is not supported");

2236            Some(LimitClause::OffsetCommaLimit { offset, limit }) => (Some(limit), Some(offset)),

2238        };
2239
2240        // Handle FETCH clause (alternative to LIMIT, mutually exclusive)
2241        let limit = match (fetch, limit) {
2242            (Some(fetch), None) => fetch.quantity.as_ref(),
2243            (Some(_), Some(_)) => {
2244                polars_bail!(SQLSyntax: "cannot use both `LIMIT` and `FETCH` in the same query")
2245            },
2246            (None, limit) => limit,
2247        };
2248
2249        // Apply limit and/or offset
2250        match (offset, limit) {
2251            (
2252                Some(SQLExpr::Value(ValueWithSpan {
2253                    value: SQLValue::Number(offset, _),
2254                    ..
2255                })),
2256                Some(SQLExpr::Value(ValueWithSpan {
2257                    value: SQLValue::Number(limit, _),
2258                    ..
2259                })),
2260            ) => Ok(lf.slice(
2261                offset
2262                    .parse()
2263                    .map_err(|e| polars_err!(SQLInterface: "OFFSET conversion error: {}", e))?,
2264                limit.parse().map_err(
2265                    |e| polars_err!(SQLInterface: "LIMIT/FETCH conversion error: {}", e),
2266                )?,
2267            )),
2268            (
2269                Some(SQLExpr::Value(ValueWithSpan {
2270                    value: SQLValue::Number(offset, _),
2271                    ..
2272                })),
2273                None,
2274            ) => Ok(lf.slice(
2275                offset
2276                    .parse()
2277                    .map_err(|e| polars_err!(SQLInterface: "OFFSET conversion error: {}", e))?,
2278                IdxSize::MAX,
2279            )),
2280            (
2281                None,
2282                Some(SQLExpr::Value(ValueWithSpan {
2283                    value: SQLValue::Number(limit, _),
2284                    ..
2285                })),
2286            ) => {
2287                Ok(lf.limit(limit.parse().map_err(
2288                    |e| polars_err!(SQLInterface: "LIMIT/FETCH conversion error: {}", e),
2289                )?))
2290            },
2291            (None, None) => Ok(lf),
2292            _ => polars_bail!(
2293                SQLSyntax: "non-numeric arguments for LIMIT/OFFSET/FETCH are not supported",
2294            ),
2295        }
2296    }

2315    fn process_wildcard_additional_options(
2316        &mut self,
2317        exprs: Vec<Expr>,
2318        options: &WildcardAdditionalOptions,
2319        modifiers: &mut SelectModifiers,
2320        schema: Option<&Schema>,
2321    ) -> PolarsResult<Vec<Expr>> {
2322        if options.opt_except.is_some() && options.opt_exclude.is_some() {
2323            polars_bail!(SQLInterface: "EXCLUDE and EXCEPT wildcard options cannot be used together (prefer EXCLUDE)")
2324        } else if options.opt_exclude.is_some() && options.opt_ilike.is_some() {
2325            polars_bail!(SQLInterface: "EXCLUDE and ILIKE wildcard options cannot be used together")
2326        }
2327
2328        // SELECT * EXCLUDE
2329        if let Some(items) = &options.opt_exclude {
2330            match items {
2331                ExcludeSelectItem::Single(ident) => {
2332                    modifiers.exclude.insert(ident.value.clone());
2333                },
2334                ExcludeSelectItem::Multiple(idents) => {
2335                    modifiers
2336                        .exclude
2337                        .extend(idents.iter().map(|i| i.value.clone()));
2338                },
2339            };
2340        }
2341
2342        // SELECT * EXCEPT
2343        if let Some(items) = &options.opt_except {
2344            modifiers.exclude.insert(items.first_element.value.clone());
2345            modifiers
2346                .exclude
2347                .extend(items.additional_elements.iter().map(|i| i.value.clone()));
2348        }
2349
2350        // SELECT * ILIKE
2351        if let Some(item) = &options.opt_ilike {
2352            let rx = regex::escape(item.pattern.as_str())
2353                .replace('%', ".*")
2354                .replace('_', ".");
2355
2356            modifiers.ilike = Some(
2357                polars_utils::regex_cache::compile_regex(format!("^(?is){rx}$").as_str()).unwrap(),
2358            );
2359        }
2360
2361        // SELECT * RENAME
2362        if let Some(items) = &options.opt_rename {
2363            let renames = match items {
2364                RenameSelectItem::Single(rename) => std::slice::from_ref(rename),
2365                RenameSelectItem::Multiple(renames) => renames.as_slice(),
2366            };
2367            for rn in renames {
2368                let before = PlSmallStr::from_str(rn.ident.value.as_str());
2369                let after = PlSmallStr::from_str(rn.alias.value.as_str());
2370                if before != after {
2371                    modifiers.rename.insert(before, after);
2372                }
2373            }
2374        }
2375
2376        // SELECT * REPLACE
2377        if let Some(replacements) = &options.opt_replace {
2378            for rp in &replacements.items {
2379                let replacement_expr = parse_sql_expr(&rp.expr, self, schema);
2380                modifiers
2381                    .replace
2382                    .push(replacement_expr?.alias(rp.column_name.value.as_str()));
2383            }
2384        }
2385        Ok(exprs)
2386    }

2507/// Resolve a SELECT alias to its underlying expression (for use in GROUP BY).
2508///
2509/// Returns the expression WITH alias if the name matches a projection alias and is NOT a column
2510/// that exists in the schema; otherwise returns `None` to use the default/standard resolution.
2511fn resolve_select_alias(name: &str, projections: &[Expr], schema: &Schema) -> Option<Expr> {
2512    // Original columns take precedence over SELECT aliases
2513    if schema.contains(name) {
2514        return None;
2515    }
2516    // Find a projection with this alias and return its expression (preserving the alias)
2517    projections.iter().find_map(|p| match p {
2518        Expr::Alias(inner, alias) if alias.as_str() == name => {
2519            Some(inner.as_ref().clone().alias(alias.clone()))
2520        },
2521        _ => None,
2522    })
2523}

2541/// Determine which parsed join expressions actually belong in `left_om` and which in `right_on`.
2542///
2543/// This needs to be handled carefully because in SQL joins you can write "join on" constraints
2544/// either way round, and in joins with more than two tables you can also join against an earlier
2545/// table (e.g.: you could be joining `df1` to `df2` to `df3`, but the final join condition where
2546/// we join `df2` to `df3` could refer to `df1.a = df3.b`; this takes a little more work to
2547/// resolve as our native `join` function operates on only two tables at a time.
2548fn determine_left_right_join_on(
2549    ctx: &mut SQLContext,
2550    expr_left: &SQLExpr,
2551    expr_right: &SQLExpr,
2552    tbl_left: &TableInfo,
2553    tbl_right: &TableInfo,
2554    join_schema: &Schema,
2555) -> PolarsResult<(Vec<Expr>, Vec<Expr>)> {
2556    // parse, removing any aliases that may have been added by `resolve_column`
2557    // (called inside `parse_sql_expr`) as we need the actual/underlying col
2558    let left_on = match parse_sql_expr(expr_left, ctx, Some(join_schema))? {
2559        Expr::Alias(inner, _) => Arc::unwrap_or_clone(inner),
2560        e => e,
2561    };
2562    let right_on = match parse_sql_expr(expr_right, ctx, Some(join_schema))? {
2563        Expr::Alias(inner, _) => Arc::unwrap_or_clone(inner),
2564        e => e,
2565    };
2566
2567    // ------------------------------------------------------------------
2568    // simple/typical case: can fully resolve SQL-level table references
2569    // ------------------------------------------------------------------
2570    let left_refs = (
2571        expr_refers_to_table(expr_left, &tbl_left.name),
2572        expr_refers_to_table(expr_left, &tbl_right.name),
2573    );
2574    let right_refs = (
2575        expr_refers_to_table(expr_right, &tbl_left.name),
2576        expr_refers_to_table(expr_right, &tbl_right.name),
2577    );
2578    // if the SQL-level references unambiguously indicate table ownership, we're done
2579    match (left_refs, right_refs) {
2580        // standard: left expr → left table, right expr → right table
2581        ((true, false), (false, true)) => return Ok((vec![left_on], vec![right_on])),
2582        // reversed: left expr → right table, right expr → left table
2583        ((false, true), (true, false)) => return Ok((vec![right_on], vec![left_on])),
2584        // unsupported: one side references *both* tables
2585        ((true, true), _) | (_, (true, true)) if tbl_left.name != tbl_right.name => {
2586            polars_bail!(
2587               SQLInterface: "unsupported join condition: {} side references both '{}' and '{}'",
2588               if left_refs.0 && left_refs.1 {
2589                    "left"
2590                } else {
2591                    "right"
2592                }, tbl_left.name, tbl_right.name
2593            )
2594        },
2595        // fall through to the more involved col/ref resolution
2596        _ => {},
2597    }
2598
2599    // ------------------------------------------------------------------
2600    // more involved: additionally employ schema-based column resolution
2601    // (applies to unqualified columns and/or chained joins)
2602    // ------------------------------------------------------------------
2603    let left_on_cols_in = (
2604        expr_cols_all_in_schema(&left_on, &tbl_left.schema),
2605        expr_cols_all_in_schema(&left_on, &tbl_right.schema),
2606    );
2607    let right_on_cols_in = (
2608        expr_cols_all_in_schema(&right_on, &tbl_left.schema),
2609        expr_cols_all_in_schema(&right_on, &tbl_right.schema),
2610    );
2611    match (left_on_cols_in, right_on_cols_in) {
2612        // each expression's columns exist in exactly one schema
2613        ((true, false), (false, true)) => Ok((vec![left_on], vec![right_on])),
2614        ((false, true), (true, false)) => Ok((vec![right_on], vec![left_on])),
2615        // one expression in both, other only in one; prefer the unique one
2616        ((true, true), (true, false)) => Ok((vec![right_on], vec![left_on])),
2617        ((true, true), (false, true)) => Ok((vec![left_on], vec![right_on])),
2618        ((true, false), (true, true)) => Ok((vec![left_on], vec![right_on])),
2619        ((false, true), (true, true)) => Ok((vec![right_on], vec![left_on])),
2620        // pass through as-is
2621        _ => Ok((vec![left_on], vec![right_on])),
2622    }
2623}

2625fn process_join_on(
2626    ctx: &mut SQLContext,
2627    sql_expr: &SQLExpr,
2628    tbl_left: &TableInfo,
2629    tbl_right: &TableInfo,
2630) -> PolarsResult<(Vec<Expr>, Vec<Expr>)> {
2631    match sql_expr {
2632        SQLExpr::BinaryOp { left, op, right } => match op {
2633            BinaryOperator::And => {
2634                let (mut left_i, mut right_i) = process_join_on(ctx, left, tbl_left, tbl_right)?;
2635                let (mut left_j, mut right_j) = process_join_on(ctx, right, tbl_left, tbl_right)?;
2636                left_i.append(&mut left_j);
2637                right_i.append(&mut right_j);
2638                Ok((left_i, right_i))
2639            },
2640            BinaryOperator::Eq => {
2641                // establish unified schema with cols from both tables; needed for multi/chained
2642                // joins where suffixed intermediary/joined cols aren't in an existing schema.
2643                let mut join_schema =
2644                    Schema::with_capacity(tbl_left.schema.len() + tbl_right.schema.len());
2645                for (name, dtype) in tbl_left.schema.iter() {
2646                    join_schema.insert_at_index(join_schema.len(), name.clone(), dtype.clone())?;
2647                }
2648                for (name, dtype) in tbl_right.schema.iter() {
2649                    if !join_schema.contains(name) {
2650                        join_schema.insert_at_index(
2651                            join_schema.len(),
2652                            name.clone(),
2653                            dtype.clone(),
2654                        )?;
2655                    }
2656                }
2657                determine_left_right_join_on(ctx, left, right, tbl_left, tbl_right, &join_schema)
2658            },
2659            _ => polars_bail!(
2660                // TODO: should be able to support more operators later (via `join_where`?)
2661                SQLInterface: "only equi-join constraints (combined with 'AND') are currently supported; found op = '{:?}'", op
2662            ),
2663        },
2664        SQLExpr::Nested(expr) => process_join_on(ctx, expr, tbl_left, tbl_right),
2665        _ => polars_bail!(
2666            SQLInterface: "only equi-join constraints are currently supported; found expression = {:?}", sql_expr
2667        ),
2668    }
2669}

2671fn process_join_constraint(
2672    constraint: &JoinConstraint,
2673    tbl_left: &TableInfo,
2674    tbl_right: &TableInfo,
2675    ctx: &mut SQLContext,
2676) -> PolarsResult<(Vec<Expr>, Vec<Expr>)> {
2677    match constraint {
2678        JoinConstraint::On(expr @ SQLExpr::BinaryOp { .. }) => {
2679            process_join_on(ctx, expr, tbl_left, tbl_right)
2680        },
2681        JoinConstraint::Using(idents) if !idents.is_empty() => {
2682            let using: Vec<Expr> = idents
2683                .iter()
2684                .map(|ObjectName(parts)| {
2685                    if parts.len() != 1 {
2686                        polars_bail!(SQLSyntax: "JOIN \"USING\" clause expects simple column names, not qualified names");
2687                    }
2688                    match parts[0].as_ident() {
2689                        Some(ident) => Ok(col(ident.value.as_str())),
2690                        None => polars_bail!(SQLSyntax: "JOIN \"USING\" clause expects identifiers, not functions"),
2691                    }
2692                })
2693                .collect::<PolarsResult<Vec<_>>>()?;
2694            Ok((using.clone(), using))
2695        },
2696        JoinConstraint::Natural => {
2697            let left_names = tbl_left.schema.iter_names().collect::<PlHashSet<_>>();
2698            let right_names = tbl_right.schema.iter_names().collect::<PlHashSet<_>>();
2699            let on: Vec<Expr> = left_names
2700                .intersection(&right_names)
2701                .map(|&name| col(name.clone()))
2702                .collect();
2703            if on.is_empty() {
2704                polars_bail!(SQLInterface: "no common columns found for NATURAL JOIN")
2705            }
2706            Ok((on.clone(), on))
2707        },
2708        _ => polars_bail!(SQLInterface: "unsupported SQL join constraint:\n{:?}", constraint),
2709    }
2710}

2712/// Extract table identifiers referenced in a SQL query; uses a visitor to
2713/// collect all table names that appear in FROM clauses, JOINs, TABLE refs
2714/// in set operations, and subqueries.
2715pub fn extract_table_identifiers(
2716    query: &str,
2717    include_schema: bool,
2718    unique: bool,

2721    parser = parser.with_options(ParserOptions {
2722        trailing_commas: true,
2723        ..Default::default()
2724    });
2725    let ast = parser
2726        .try_with_sql(query)
2727        .map_err(to_sql_interface_err)?
2728        .parse_statements()
2729        .map_err(to_sql_interface_err)?;
2730
2731    let mut collector = TableIdentifierCollector {
2732        include_schema,
2733        ..Default::default()
2734    };
2735    for stmt in &ast {
2736        let _ = stmt.visit(&mut collector);

2738    Ok(if unique {
2739        collector
2740            .tables
2741            .into_iter()
2742            .collect::<PlIndexSet<_>>()
2743            .into_iter()
2744            .collect()
2745    } else {

2747    })

2750bitflags::bitflags! {
2751    /// Bitfield indicating whether there exists a projection with the specified height behavior.
2752    ///
2753    /// Used to help determine whether to execute projections in `select()` or `with_columns()`
2754    /// context.
2755    #[derive(PartialEq)]
2756    struct ExprSqlProjectionHeightBehavior: u8 {
2757        /// Maintains the height of input column(s)
2758        const MaintainsColumn = 1 << 0;
2759        /// Height is independent of input, e.g.:
2760        /// * expressions that change length: e.g. slice, explode, filter, gather etc.
2761        /// * aggregations: count(*), first(), sum() etc.
2762        const Independent = 1 << 1;
2763        /// "Inherits" the height of the context, e.g.:
2764        /// * Scalar literals
2765        const InheritsContext = 1 << 2;
2766    }
2767}
2768
2769impl ExprSqlProjectionHeightBehavior {
2770    fn identify_from_expr(expr: &Expr) -> Self {
2771        let mut has_column = false;
2772        let mut has_independent = false;
2773
2774        for e in expr.into_iter() {
2775            use Expr::*;
2776            has_column |= matches!(e, Column(_) | Selector(_));
2777            has_independent |= match e {
2778                // @TODO: This is broken now with functions.
2779                AnonymousFunction { options, .. } => {
2780                    options.returns_scalar() || !options.is_length_preserving()
2781                },
2782                Literal(v) => !v.is_scalar(),
2783                Explode { .. } | Filter { .. } | Gather { .. } | Slice { .. } => true,
2784                Agg { .. } | Len => true,
2785                _ => false,
2786            }
2787        }
2788        if has_independent {
2789            Self::Independent
2790        } else if has_column {
2791            Self::MaintainsColumn
2792        } else {
2793            Self::InheritsContext
2794        }
2795    }
