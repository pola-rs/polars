1//! Rewrites of `[NOT] EXISTS` / `[NOT] IN (subquery)` predicates into semi /
2//! anti joins, decorrelating equi-correlation predicates into join keys.
3//! Subquery shapes the rewrites can't soundly express return `None` so the
4//! caller falls back to the generic filter path.

24impl SQLContext {
25    // Entry point: offer each WHERE conjunct to the rewrite, returning the
26    // (possibly join-extended) frame together with the conjuncts left for the
27    // ordinary filter path. In `KeepTrue` mode each top-level AND-conjunct is
28    // offered independently. In `RemoveTrue` mode conjuncts can't be split
29    // (`NOT (a AND b)` is a disjunction), so only a sole (possibly
30    // parenthesized) subquery predicate rewrites, and anything else is
31    // returned whole as the residual.
32    pub(crate) fn rewrite_subquery_conjuncts<'a>(
33        &mut self,
34        mut lf: LazyFrame,
35        expr: &'a SQLExpr,
36        filter_mode: FilterMode,
37        schema: &Schema,
38    ) -> PolarsResult<(LazyFrame, Vec<&'a SQLExpr>)> {
39        let residual = match filter_mode {
40            FilterMode::RemoveTrue => {
41                let mut unwrapped = expr;
42                while let SQLExpr::Nested(inner) = unwrapped {
43                    unwrapped = inner;
44                }
45                match self.try_rewrite_subquery_conjunct(&lf, unwrapped, filter_mode, schema)? {
46                    Some(new_lf) => {
47                        lf = new_lf;
48                        Vec::new()
49                    },
50                    None => vec![expr],
51                }
52            },
53            FilterMode::KeepTrue => {
54                let mut residual = Vec::new();
55                for conj in MintermIter::new(expr) {
56                    if let Some(new_lf) =
57                        self.try_rewrite_subquery_conjunct(&lf, conj, filter_mode, schema)?
58                    {
59                        lf = new_lf;
60                    } else {
61                        residual.push(conj);
62                    }
63                }
64                residual
65            },
66        };
67        Ok((lf, residual))
68    }
69
70    // Dispatch one conjunct to the matching rewrite. `RemoveTrue` mode (DELETE)
71    // flips the join polarity. Removing `IN` rows keeps
72    // rows whose membership is false or NULL, exactly what an anti-join
73    // produces; removing `NOT IN` rows would additionally keep NULL keys, which
74    // a semi join can't express, so it stays on the filter path.
75    fn try_rewrite_subquery_conjunct(
76        &mut self,
77        lf: &LazyFrame,
78        conj: &SQLExpr,
79        filter_mode: FilterMode,
80        schema: &Schema,
81    ) -> PolarsResult<Option<LazyFrame>> {
82        let removing = filter_mode == FilterMode::RemoveTrue;
83        match conj {
84            SQLExpr::Exists { subquery, negated } => {
85                self.try_rewrite_exists_as_join(lf, subquery, *negated != removing, schema)
86            },
87            SQLExpr::InSubquery {
88                expr: lhs,
89                subquery,
90                negated,
91            } if !(*negated && removing) => self.try_rewrite_in_subquery_as_join(
92                lf,
93                lhs,
94                subquery,
95                *negated != removing,
96                filter_mode,
97                schema,
98            ),
99            _ => Ok(None),
100        }
101    }
102
103    // Lower `[NOT] EXISTS (SELECT ... FROM rel WHERE rel.k = outer.k ...)` to a
104    // semi / anti join by decorrelating the equi-correlation predicate(s) into
105    // join keys. DISTINCT is ignored: existence is invariant under
106    // deduplication.
107    #[cfg(feature = "semi_anti_join")]
108    fn try_rewrite_exists_as_join(
109        &mut self,
110        lf: &LazyFrame,
111        subquery: &Query,
112        negated: bool,
113        outer_schema: &Schema,
114    ) -> PolarsResult<Option<LazyFrame>> {
115        let Some(select) = eligible_subquery_select(subquery) else {
116            return Ok(None);
117        };
118        let Some(selection) = &select.selection else {
119            return Ok(None);
120        };
121        // Resolve and parse the inner relation in an isolated context so its
122        // table/alias registrations don't leak into the outer query's scope.
123        let mut ctx = self.isolated();
124        let Some((inner_names, inner_lf, inner_schema)) =
125            ctx.resolve_subquery_from(&select.from[0])?
126        else {
127            return Ok(None);
128        };
129        let Some(SubqueryConjuncts {
130            left_on,
131            right_on,
132            local_filters,
133        }) = ctx.split_subquery_conjuncts(selection, &inner_names, &inner_schema, outer_schema)?
134        else {
135            return Ok(None);
136        };
137        // An uncorrelated EXISTS (no correlation key found) has no join key to
138        // build from, so leave it to the existing path.
139        if left_on.is_empty() {
140            return Ok(None);
141        }
142        Ok(Some(ctx.finish_decorrelated_join(
143            lf,
144            inner_lf,
145            left_on,
146            right_on,
147            local_filters,
148            negated,
149        )))
150    }

152    // Lower `lhs [NOT] IN (SELECT col FROM rel ...)` to a semi / anti join: the
153    // projected column is the membership key and any equi-correlations in the
154    // subquery WHERE become additional join keys.
155    #[cfg(feature = "semi_anti_join")]
156    fn try_rewrite_in_subquery_as_join(
157        &mut self,
158        lf: &LazyFrame,
159        lhs: &SQLExpr,
160        subquery: &Query,
161        anti: bool,
162        filter_mode: FilterMode,
163        outer_schema: &Schema,
164    ) -> PolarsResult<Option<LazyFrame>> {
165        let Some(select) = eligible_subquery_select(subquery) else {
166            return Ok(None);
167        };
168        // DISTINCT is membership-invariant, but DISTINCT ON drops rows per key.
169        if matches!(&select.distinct, Some(Distinct::On(_))) {
170            return Ok(None);
171        }
172        let [SelectItem::UnnamedExpr(proj) | SelectItem::ExprWithAlias { expr: proj, .. }] =
173            select.projection.as_slice()
174        else {
175            return Ok(None);
176        };
177
178        let left_key = parse_sql_expr(lhs, self, Some(outer_schema))?
179            .meta()
180            .undo_aliases();
181        if has_expr(&left_key, |e| matches!(e, Expr::SubPlan(_, _)))
182            || !expr_to_leaf_column_names_iter(&left_key)
183                .all(|name| outer_schema.contains(name.as_str()))
184        {
185            return Ok(None);
186        }
187
188        let mut ctx = self.isolated();
189        let Some((inner_names, inner_lf, inner_schema)) =
190            ctx.resolve_subquery_from(&select.from[0])?
191        else {
192            return Ok(None);
193        };
194        // The membership key must be a plain expression over the inner relation;
195        // any alias it carries is cosmetic and not allowed in a join key.
196        let Some(right_key) = ctx.try_parse_inner_only_expr(proj, &inner_schema, outer_schema)?
197        else {
198            return Ok(None);
199        };
200        let right_key = right_key.meta().undo_aliases();
201
202        let SubqueryConjuncts {
203            mut left_on,
204            mut right_on,
205            local_filters,
206        } = match &select.selection {
207            Some(selection) => {
208                let Some(split) = ctx.split_subquery_conjuncts(
209                    selection,
210                    &inner_names,
211                    &inner_schema,
212                    outer_schema,
213                )?
214                else {
215                    return Ok(None);
216                };
217                split
218            },
219            None => SubqueryConjuncts::default(),
220        };
221
222        // Correlation keys for the "NOT IN" 3VL correction
223        let corr_outer = left_on.clone();
224        let corr_inner = right_on.clone();
225        left_on.insert(0, left_key.clone());
226        right_on.insert(0, right_key.clone());
227
228        // Inline, so filtered inner frame can be reused for the correction
229        let inner_lf = local_filters.into_iter().fold(inner_lf, LazyFrame::filter);
230        inner_lf.set_cached_arena(ctx.lp_arena, ctx.expr_arena);
231        let joined = build_semi_anti_join(lf, inner_lf.clone(), left_on, right_on, anti);
232
233        // Only `KeepTrue` "NOT IN" needs 3VL correction.
234        if !(anti && filter_mode == FilterMode::KeepTrue) {
235            return Ok(Some(joined));
236        }
237        Ok(Some(refine_not_in_anti_join(
238            joined,
239            inner_lf,
240            &left_key,
241            &right_key,
242            &corr_outer,
243            &corr_inner,
244        )))
245    }
246
247    // Apply the local filters to the inner relation, hand this (isolated, now
248    // finished) context's arenas to it, and build the semi / anti join against
249    // the outer frame. Consumes the context: nothing may be parsed in the
250    // subquery's scope after the join is built.
251    #[cfg(feature = "semi_anti_join")]
252    fn finish_decorrelated_join(
253        self,
254        lf: &LazyFrame,
255        inner_lf: LazyFrame,
256        left_on: Vec<Expr>,
257        right_on: Vec<Expr>,
258        local_filters: Vec<Expr>,
259        anti: bool,
260    ) -> LazyFrame {
261        let inner_lf = local_filters.into_iter().fold(inner_lf, LazyFrame::filter);
262        inner_lf.set_cached_arena(self.lp_arena, self.expr_arena);
263        build_semi_anti_join(lf, inner_lf, left_on, right_on, anti)
264    }
265
266    // Resolve the subquery's FROM (a single relation, possibly with joins) into
267    // the inner LazyFrame, its schema, and the set of relation names/aliases
268    // used to classify qualified correlation columns.
269    #[cfg(feature = "semi_anti_join")]
270    fn resolve_subquery_from(
271        &mut self,
272        tbl_expr: &TableWithJoins,
273    ) -> PolarsResult<Option<(PlHashSet<String>, LazyFrame, SchemaRef)>> {
274        let Some(inner_names) = std::iter::once(&tbl_expr.relation)
275            .chain(tbl_expr.joins.iter().map(|j| &j.relation))
276            .map(get_table_name)
277            .collect::<Option<PlHashSet<_>>>()
278        else {
279            return Ok(None);
280        };
281        let mut inner_lf = self.execute_from_statement(tbl_expr)?;
282        let inner_schema = self.get_frame_schema(&mut inner_lf)?;
283        Ok(Some((inner_names, inner_lf, inner_schema)))
284    }
285
286    // Split a subquery's WHERE conjuncts into a `SubqueryConjuncts`, or `None`
287    // when a conjunct is neither a correlation key pair nor an inner-only
288    // filter (an outer column in a non-equi shape, an unresolvable name).
289    #[cfg(feature = "semi_anti_join")]
290    fn split_subquery_conjuncts(
291        &mut self,
292        selection: &SQLExpr,
293        inner_names: &PlHashSet<String>,
294        inner_schema: &Schema,
295        outer_schema: &Schema,
296    ) -> PolarsResult<Option<SubqueryConjuncts>> {
297        let mut left_on = Vec::new();
298        let mut right_on = Vec::new();
299        let mut local_filters = Vec::new();
300        for conj in MintermIter::new(selection) {
301            if let Some((outer_key, inner_key)) =
302                correlation_key_pair(conj, inner_names, inner_schema, outer_schema)
303            {
304                left_on.push(col(outer_key));
305                right_on.push(col(inner_key));
306                continue;
307            }
308            let Some(filter) = self.try_parse_inner_only_expr(conj, inner_schema, outer_schema)?
309            else {
310                return Ok(None);
311            };
312            local_filters.push(filter);
313        }
314        Ok(Some(SubqueryConjuncts {
315            left_on,
316            right_on,
317            local_filters,
318        }))
319    }

321    // Parse a subquery expression as one over the inner relation only, or `None`
322    // if it references any outer column (a correlation shape we don't handle) or
323    // contains a nested subquery.
324    #[cfg(feature = "semi_anti_join")]
325    fn try_parse_inner_only_expr(
326        &mut self,
327        sql_expr: &SQLExpr,
328        inner_schema: &Schema,
329        outer_schema: &Schema,
330    ) -> PolarsResult<Option<Expr>> {
331        let expr = parse_sql_expr(sql_expr, self, Some(inner_schema))?;
332        // A nested subquery parses to `Expr::SubPlan`, which is only valid after
333        // `process_subqueries` lowering; it can't be used as a plain expression.
334        if has_expr(&expr, |e| matches!(e, Expr::SubPlan(_, _))) {
335            return Ok(None);
336        }
337        let only_inner = expr_to_leaf_column_names_iter(&expr).all(|name| {
338            inner_schema.contains(name.as_str()) && !outer_schema.contains(name.as_str())
339        });
340        Ok(only_inner.then_some(expr))
341    }
342
343    #[cfg(not(feature = "semi_anti_join"))]
344    fn try_rewrite_exists_as_join(
345        &mut self,
346        _lf: &LazyFrame,
347        _subquery: &Query,
348        _negated: bool,
349        _outer_schema: &Schema,
350    ) -> PolarsResult<Option<LazyFrame>> {
351        Ok(None)
352    }
353
354    #[cfg(not(feature = "semi_anti_join"))]
355    #[expect(clippy::too_many_arguments)]
356    fn try_rewrite_in_subquery_as_join(
357        &mut self,
358        _lf: &LazyFrame,
359        _lhs: &SQLExpr,
360        _subquery: &Query,
361        _anti: bool,
362        _filter_mode: FilterMode,
363        _outer_schema: &Schema,
364    ) -> PolarsResult<Option<LazyFrame>> {
365        Ok(None)
366    }

369// Semi/anti join the outer frame against the (filtered, arena-cached) inner.
370#[cfg(feature = "semi_anti_join")]
371fn build_semi_anti_join(
372    lf: &LazyFrame,
373    inner_lf: LazyFrame,
374    left_on: Vec<Expr>,
375    right_on: Vec<Expr>,
376    anti: bool,
377) -> LazyFrame {
378    let join_type = if anti { JoinType::Anti } else { JoinType::Semi };
379    lf.clone()
380        .join_builder()
381        .with(inner_lf)
382        .left_on(left_on)
383        .right_on(right_on)
384        .how(join_type)
385        .finish()
386}

388// Account for 3VL interaction with NULL values
389#[cfg(feature = "semi_anti_join")]
390fn refine_not_in_anti_join(
391    joined: LazyFrame,
392    inner_lf: LazyFrame,
393    left_key: &Expr,
394    right_key: &Expr,
395    corr_outer: &[Expr],
396    corr_inner: &[Expr],
397) -> LazyFrame {
398    if corr_inner.is_empty() {
399        // Uncorrelated
400        let flag_name = unique_column_name();
401        let flag = when(len().eq(lit(0u32)))
402            .then(lit(NULL).cast(DataType::Boolean))
403            .otherwise(right_key.clone().is_null().any(true))
404            .alias(flag_name.clone());
405        let keep = when(col(flag_name.clone()).is_null())
406            .then(lit(true)) // empty set
407            .when(col(flag_name.clone())) // set has a NULL
408            .then(lit(false))
409            .otherwise(left_key.clone().is_not_null());
410
411        return joined
412            .join_builder()
413            .with(inner_lf.select([flag]))
414            .how(JoinType::Cross)
415            .finish()
416            .filter(keep)
417            .drop(Selector::ByName {
418                names: [flag_name].into(),
419                strict: true,
420            });
421    }
422
423    // Correlated
424    let corr_keys = |lf: LazyFrame| lf.select(corr_inner).unique(None, UniqueKeepStrategy::Any);
425    let exclude_groups = |rows: LazyFrame, groups: LazyFrame| {
426        rows.join_builder()
427            .with(groups)
428            .left_on(corr_outer)
429            .right_on(corr_inner)
430            .how(JoinType::Anti)
431            .finish()
432    };
433    let kept_non_null = exclude_groups(
434        joined.clone().filter(left_key.clone().is_not_null()),
435        corr_keys(inner_lf.clone().filter(right_key.clone().is_null())),
436    );
437    let kept_null = exclude_groups(
438        joined.filter(left_key.clone().is_null()),
439        corr_keys(inner_lf),
440    );
441
442    concat(
443        [kept_non_null, kept_null],
444        UnionArgs {
445            rechunk: false,
446            parallel: true,
447            ..Default::default()
448        },
449    )
450    .expect("'NOT IN' 3VL union has identical schemas")
451}

453/// An iterator over all the minterms in an SQL boolean expression: the terms
454/// that `AND` together to form it, descending through parenthesized `Nested`
455/// expressions. The SQL-AST analogue of the `AExpr`-level
456/// `polars_plan::plans::aexpr::MintermIter`.
457struct MintermIter<'a> {
458    stack: Vec<&'a SQLExpr>,
459}
460
461impl<'a> Iterator for MintermIter<'a> {
462    type Item = &'a SQLExpr;
463
464    fn next(&mut self) -> Option<Self::Item> {
465        let mut top = self.stack.pop()?;
466        loop {
467            match top {
468                SQLExpr::Nested(inner) => top = inner,
469                SQLExpr::BinaryOp {
470                    left,
471                    op: SQLBinaryOperator::And,
472                    right,
473                } => {
474                    self.stack.push(right);
475                    top = left;
476                },
477                _ => return Some(top),
478            }
479        }
480    }
481}
482
483impl<'a> MintermIter<'a> {
484    fn new(root: &'a SQLExpr) -> Self {
485        Self { stack: vec![root] }
486    }
487}

489#[cfg(feature = "semi_anti_join")]
490enum CorrelationSide {
491    Inner,
492    Outer,
493}
494
495// A subquery WHERE split into equi-correlation join keys (outer side in
496// `left_on`, inner side in `right_on`) and filters over inner columns only.
497#[cfg(feature = "semi_anti_join")]
498#[derive(Default)]
499struct SubqueryConjuncts {
500    left_on: Vec<Expr>,
501    right_on: Vec<Expr>,
502    local_filters: Vec<Expr>,
503}
504
505// An equi-correlation conjunct `inner.col = outer.col` (either way round) as a
506// `(outer key, inner key)` column-name pair, or `None` when the conjunct is
507// anything else (non-equality, unresolvable names, both columns on the same
508// side).
509#[cfg(feature = "semi_anti_join")]
510fn correlation_key_pair(
511    conj: &SQLExpr,
512    inner_names: &PlHashSet<String>,
513    inner_schema: &Schema,
514    outer_schema: &Schema,
515) -> Option<(PlSmallStr, PlSmallStr)> {
516    let SQLExpr::BinaryOp {
517        left,
518        op: SQLBinaryOperator::Eq,
519        right,
520    } = conj
521    else {
522        return None;
523    };
524    let (lside, lname) =
525        classify_correlation_column(left, inner_names, inner_schema, outer_schema)?;
526    let (rside, rname) =
527        classify_correlation_column(right, inner_names, inner_schema, outer_schema)?;
528    match (lside, rside) {
529        (CorrelationSide::Outer, CorrelationSide::Inner) => Some((lname, rname)),
530        (CorrelationSide::Inner, CorrelationSide::Outer) => Some((rname, lname)),
531        _ => None,
532    }
533}

535// Classify a correlation operand as an inner- or outer-query column and return
536// its bare name. A qualified identifier (`tbl.col`) resolves by its qualifier:
537// an inner relation's name/alias means inner, anything else means outer (so
538// same-named columns like `o.id = c.id` resolve). An unqualified identifier
539// resolves by schema membership. `None` for non-identifiers or names that can't
540// be placed (in neither schema, or ambiguous).
541#[cfg(feature = "semi_anti_join")]
542fn classify_correlation_column(
543    expr: &SQLExpr,
544    inner_names: &PlHashSet<String>,
545    inner_schema: &Schema,
546    outer_schema: &Schema,
547) -> Option<(CorrelationSide, PlSmallStr)> {
548    let (qualifier, name): (Option<&str>, PlSmallStr) = match expr {
549        SQLExpr::Identifier(ident) => (None, ident.value.as_str().into()),
550        SQLExpr::CompoundIdentifier(parts) => {
551            let (last, init) = parts.split_last()?;
552            // Only the table part: catalog/schema prefixes are dropped, just
553            // as `get_table_name` drops them when building `inner_names`.
554            (
555                init.last().map(|q| q.value.as_str()),
556                last.value.as_str().into(),
557            )
558        },
559        _ => return None,
560    };
561    match qualifier {
562        Some(q) if inner_names.contains(q) => inner_schema
563            .contains(name.as_str())
564            .then_some((CorrelationSide::Inner, name)),
565        // Any other qualifier is taken as outer: the outer query may span
566        // several relations and their names/aliases aren't visible here, so
567        // only the column's schema membership can be checked.
568        Some(_) => outer_schema
569            .contains(name.as_str())
570            .then_some((CorrelationSide::Outer, name)),
571        None => match (
572            inner_schema.contains(name.as_str()),
573            outer_schema.contains(name.as_str()),
574        ) {
575            (true, false) => Some((CorrelationSide::Inner, name)),
576            (false, true) => Some((CorrelationSide::Outer, name)),
577            _ => None,
578        },
579    }
580}

582// Shared eligibility gate for the rewrites: bail on any clause that changes
583// which rows the subquery yields. Exhaustive destructuring (no `..`) is on
584// purpose: a new sqlparser clause must not compile until it gets an explicit
585// keep-or-bail decision here.
586#[cfg(feature = "semi_anti_join")]
587fn eligible_subquery_select(subquery: &Query) -> Option<&Select> {
588    let Query {
589        with, // CTEs aren't resolved inside the rewrite: bail
590        body,
591        order_by: _,      // row order can't affect existence/membership
592        limit_clause,     // LIMIT/OFFSET change the yielded rows: bail
593        fetch,            // FETCH FIRST is LIMIT spelled differently: bail
594        locks: _,         // row locking doesn't change the rows
595        for_clause,       // FOR XML/JSON reshape the result: bail
596        settings,         // ClickHouse SETTINGS can change results: bail
597        format_clause: _, // output serialization only
598        pipe_operators,   // `|>` operators transform the rows: bail
599    } = subquery;
600    if with.is_some()
601        || limit_clause.is_some()
602        || fetch.is_some()
603        || for_clause.is_some()
604        || settings.is_some()
605        || !pipe_operators.is_empty()
606    {
607        return None;
608    }
609    let SetExpr::Select(select) = body.as_ref() else {
610        return None;
611    };
612    let Select {
613        select_token: _,
614        // Deduplication is existence/membership-invariant; the IN rewrite
615        // separately bails on `DISTINCT ON`, which is not.
616        distinct: _,
617        top,                    // TOP is LIMIT spelled differently: bail
618        top_before_distinct: _, // only meaningful with `top`
619        // The projection is validated by the callers: EXISTS ignores it, IN
620        // requires a single plain expression (which also rules out wildcards
621        // and the `exclude` modifier).
622        projection: _,
623        exclude: _,
624        into,          // SELECT INTO is not a pure subquery: bail
625        from,          // must be one (possibly joined) relation
626        lateral_views, // row-multiplying: bail
627        prewhere,      // an extra filter we don't fold in: bail
628        selection: _,  // split into join keys/filters by callers
629        group_by,      // aggregation changes the yielded rows: bail
630        cluster_by: _, // layout/order hints: row-set preserving
631        distribute_by: _,
632        sort_by: _,
633        having,                   // aggregation filter: bail
634        named_window: _,          // definitions only; uses are parsed later
635        qualify,                  // post-window filter changes the rows: bail
636        window_before_qualify: _, // only meaningful with `qualify`
637        value_table_mode,         // changes what a row is: bail
638        connect_by,               // hierarchical recursion: bail
639        optimizer_hints,          // unsupported: bail
640        select_modifiers,         // unsupported: bail
641        flavor: _,                // surface syntax only
642    } = select.as_ref();
643    let no_group_by = matches!(
644        group_by,
645        GroupByExpr::Expressions(e, m) if e.is_empty() && m.is_empty()
646    );
647    if from.len() != 1
648        || !no_group_by
649        || top.is_some()
650        || into.is_some()
651        || having.is_some()
652        || qualify.is_some()
653        || prewhere.is_some()
654        || !connect_by.is_empty()
655        || value_table_mode.is_some()
656        || !lateral_views.is_empty()
657        || !optimizer_hints.is_empty()
658        || select_modifiers.is_some()
659    {
660        return None;
661    }
662    Some(select)
663}
