1//! Rewrites of `[NOT] EXISTS` / `[NOT] IN (subquery)` predicates into semi /
2//! anti joins, decorrelating equi-correlation predicates into join keys.
3//! Subquery shapes the rewrites can't soundly express return `None` so the
4//! caller falls back to the generic filter path.

22impl SQLContext {
23    // Entry point: offer each WHERE conjunct to the rewrite, returning the
24    // (possibly join-extended) frame together with the conjuncts left for the
25    // ordinary filter path. In `KeepTrue` mode each top-level AND-conjunct is
26    // offered independently. In `RemoveTrue` mode conjuncts can't be split
27    // (`NOT (a AND b)` is a disjunction), so only a sole (possibly
28    // parenthesized) subquery predicate rewrites, and anything else is
29    // returned whole as the residual.
30    pub(crate) fn rewrite_subquery_conjuncts<'a>(
31        &mut self,
32        mut lf: LazyFrame,
33        expr: &'a SQLExpr,
34        filter_mode: FilterMode,
35        schema: &Schema,
36    ) -> PolarsResult<(LazyFrame, Vec<&'a SQLExpr>)> {
37        let residual = match filter_mode {
38            FilterMode::RemoveTrue => {
39                let mut unwrapped = expr;
40                while let SQLExpr::Nested(inner) = unwrapped {
41                    unwrapped = inner;
42                }
43                match self.try_rewrite_subquery_conjunct(&lf, unwrapped, filter_mode, schema)? {
44                    Some(new_lf) => {
45                        lf = new_lf;
46                        Vec::new()
47                    },
48                    None => vec![expr],
49                }
50            },
51            FilterMode::KeepTrue => {
52                let mut residual = Vec::new();
53                for conj in MintermIter::new(expr) {
54                    if let Some(new_lf) =
55                        self.try_rewrite_subquery_conjunct(&lf, conj, filter_mode, schema)?
56                    {
57                        lf = new_lf;
58                    } else {
59                        residual.push(conj);
60                    }
61                }
62                residual
63            },
64        };
65        Ok((lf, residual))
66    }
67
68    // Dispatch one conjunct to the matching rewrite. `RemoveTrue` mode (DELETE)
69    // flips the join polarity. Removing `IN` rows keeps
70    // rows whose membership is false or NULL, exactly what an anti-join
71    // produces; removing `NOT IN` rows would additionally keep NULL keys, which
72    // a semi join can't express, so it stays on the filter path.
73    fn try_rewrite_subquery_conjunct(
74        &mut self,
75        lf: &LazyFrame,
76        conj: &SQLExpr,
77        filter_mode: FilterMode,
78        schema: &Schema,
79    ) -> PolarsResult<Option<LazyFrame>> {
80        let removing = filter_mode == FilterMode::RemoveTrue;
81        match conj {
82            SQLExpr::Exists { subquery, negated } => {
83                self.try_rewrite_exists_as_join(lf, subquery, *negated != removing, schema)
84            },
85            SQLExpr::InSubquery {
86                expr: lhs,
87                subquery,
88                negated,
89            } if !(*negated && removing) => self.try_rewrite_in_subquery_as_join(
90                lf,
91                lhs,
92                subquery,
93                *negated != removing,
94                filter_mode,
95                schema,
96            ),
97            _ => Ok(None),
98        }
99    }
100
101    // Lower `[NOT] EXISTS (SELECT ... FROM rel WHERE rel.k = outer.k ...)` to a
102    // semi / anti join by decorrelating the equi-correlation predicate(s) into
103    // join keys. DISTINCT is ignored: existence is invariant under
104    // deduplication.
105    #[cfg(feature = "semi_anti_join")]
106    fn try_rewrite_exists_as_join(
107        &mut self,
108        lf: &LazyFrame,
109        subquery: &Query,
110        negated: bool,
111        outer_schema: &Schema,
112    ) -> PolarsResult<Option<LazyFrame>> {
113        let Some(select) = eligible_subquery_select(subquery) else {
114            return Ok(None);
115        };
116        let Some(selection) = &select.selection else {
117            return Ok(None);
118        };
119        // Resolve and parse the inner relation in an isolated context so its
120        // table/alias registrations don't leak into the outer query's scope.
121        let mut ctx = self.isolated();
122        let Some((inner_names, inner_lf, inner_schema)) =
123            ctx.resolve_subquery_from(&select.from[0])?
124        else {
125            return Ok(None);
126        };
127        let Some(SubqueryConjuncts {
128            left_on,
129            right_on,
130            local_filters,
131        }) = ctx.split_subquery_conjuncts(selection, &inner_names, &inner_schema, outer_schema)?
132        else {
133            return Ok(None);
134        };
135        // An uncorrelated EXISTS (no correlation key found) has no join key to
136        // build from, so leave it to the existing path.
137        if left_on.is_empty() {
138            return Ok(None);
139        }
140        Ok(Some(ctx.finish_decorrelated_join(
141            lf,
142            inner_lf,
143            left_on,
144            right_on,
145            local_filters,
146            negated,
147        )))
148    }

150    // Lower `lhs [NOT] IN (SELECT col FROM rel ...)` to a semi / anti join: the
151    // projected column is the membership key and any equi-correlations in the
152    // subquery WHERE become additional join keys.
153    #[cfg(feature = "semi_anti_join")]
154    fn try_rewrite_in_subquery_as_join(
155        &mut self,
156        lf: &LazyFrame,
157        lhs: &SQLExpr,
158        subquery: &Query,
159        anti: bool,
160        filter_mode: FilterMode,
161        outer_schema: &Schema,
162    ) -> PolarsResult<Option<LazyFrame>> {
163        let Some(select) = eligible_subquery_select(subquery) else {
164            return Ok(None);
165        };
166        // DISTINCT is membership-invariant, but DISTINCT ON drops rows per key.
167        if matches!(&select.distinct, Some(Distinct::On(_))) {
168            return Ok(None);
169        }
170        let [SelectItem::UnnamedExpr(proj) | SelectItem::ExprWithAlias { expr: proj, .. }] =
171            select.projection.as_slice()
172        else {
173            return Ok(None);
174        };
175
176        let left_key = parse_sql_expr(lhs, self, Some(outer_schema))?
177            .meta()
178            .undo_aliases();
179        if has_expr(&left_key, |e| matches!(e, Expr::SubPlan(_, _)))
180            || !expr_to_leaf_column_names_iter(&left_key)
181                .all(|name| outer_schema.contains(name.as_str()))
182        {
183            return Ok(None);
184        }
185
186        let mut ctx = self.isolated();
187        let Some((inner_names, inner_lf, inner_schema)) =
188            ctx.resolve_subquery_from(&select.from[0])?
189        else {
190            return Ok(None);
191        };
192        // The membership key must be a plain expression over the inner relation;
193        // any alias it carries is cosmetic and not allowed in a join key.
194        let Some(right_key) = ctx.try_parse_inner_only_expr(proj, &inner_schema, outer_schema)?
195        else {
196            return Ok(None);
197        };
198        let right_key = right_key.meta().undo_aliases();
199
200        let SubqueryConjuncts {
201            mut left_on,
202            mut right_on,
203            local_filters,
204        } = match &select.selection {
205            Some(selection) => {
206                let Some(split) = ctx.split_subquery_conjuncts(
207                    selection,
208                    &inner_names,
209                    &inner_schema,
210                    outer_schema,
211                )?
212                else {
213                    return Ok(None);
214                };
215                split
216            },
217            None => SubqueryConjuncts::default(),
218        };
219        left_on.insert(0, left_key.clone());
220        right_on.insert(0, right_key);
221        let joined =
222            ctx.finish_decorrelated_join(lf, inner_lf, left_on, right_on, local_filters, anti);
223        // `lhs NOT IN (...)` is NULL (and so excludes the row) when `lhs` is
224        // NULL, but an anti-join keeps null keys since they match nothing; drop
225        // them. In `RemoveTrue` mode a NULL membership test keeps the row, so the
226        // anti join is already exact.
227        Ok(Some(if anti && filter_mode == FilterMode::KeepTrue {
228            joined.filter(left_key.is_not_null())
229        } else {
230            joined
231        }))
232    }
233
234    // Apply the local filters to the inner relation, hand this (isolated, now
235    // finished) context's arenas to it, and build the semi / anti join against
236    // the outer frame. Consumes the context: nothing may be parsed in the
237    // subquery's scope after the join is built.
238    #[cfg(feature = "semi_anti_join")]
239    fn finish_decorrelated_join(
240        self,
241        lf: &LazyFrame,
242        inner_lf: LazyFrame,
243        left_on: Vec<Expr>,
244        right_on: Vec<Expr>,
245        local_filters: Vec<Expr>,
246        anti: bool,
247    ) -> LazyFrame {
248        let inner_lf = local_filters.into_iter().fold(inner_lf, LazyFrame::filter);
249        inner_lf.set_cached_arena(self.lp_arena, self.expr_arena);
250
251        let join_type = if anti { JoinType::Anti } else { JoinType::Semi };
252        lf.clone()
253            .join_builder()
254            .with(inner_lf)
255            .left_on(left_on)
256            .right_on(right_on)
257            .how(join_type)
258            .finish()
259    }
260
261    // Resolve the subquery's FROM (a single relation, possibly with joins) into
262    // the inner LazyFrame, its schema, and the set of relation names/aliases
263    // used to classify qualified correlation columns.
264    #[cfg(feature = "semi_anti_join")]
265    fn resolve_subquery_from(
266        &mut self,
267        tbl_expr: &TableWithJoins,
268    ) -> PolarsResult<Option<(PlHashSet<String>, LazyFrame, SchemaRef)>> {
269        let Some(inner_names) = std::iter::once(&tbl_expr.relation)
270            .chain(tbl_expr.joins.iter().map(|j| &j.relation))
271            .map(get_table_name)
272            .collect::<Option<PlHashSet<_>>>()
273        else {
274            return Ok(None);
275        };
276        let mut inner_lf = self.execute_from_statement(tbl_expr)?;
277        let inner_schema = self.get_frame_schema(&mut inner_lf)?;
278        Ok(Some((inner_names, inner_lf, inner_schema)))
279    }

281    // Split a subquery's WHERE conjuncts into a `SubqueryConjuncts`, or `None`
282    // when a conjunct is neither a correlation key pair nor an inner-only
283    // filter (an outer column in a non-equi shape, an unresolvable name).
284    #[cfg(feature = "semi_anti_join")]
285    fn split_subquery_conjuncts(
286        &mut self,
287        selection: &SQLExpr,
288        inner_names: &PlHashSet<String>,
289        inner_schema: &Schema,
290        outer_schema: &Schema,
291    ) -> PolarsResult<Option<SubqueryConjuncts>> {
292        let mut left_on = Vec::new();
293        let mut right_on = Vec::new();
294        let mut local_filters = Vec::new();
295        for conj in MintermIter::new(selection) {
296            if let Some((outer_key, inner_key)) =
297                correlation_key_pair(conj, inner_names, inner_schema, outer_schema)
298            {
299                left_on.push(col(outer_key));
300                right_on.push(col(inner_key));
301                continue;
302            }
303            let Some(filter) = self.try_parse_inner_only_expr(conj, inner_schema, outer_schema)?
304            else {
305                return Ok(None);
306            };
307            local_filters.push(filter);
308        }
309        Ok(Some(SubqueryConjuncts {
310            left_on,
311            right_on,
312            local_filters,
313        }))
314    }

316    // Parse a subquery expression as one over the inner relation only, or `None`
317    // if it references any outer column (a correlation shape we don't handle) or
318    // contains a nested subquery.
319    #[cfg(feature = "semi_anti_join")]
320    fn try_parse_inner_only_expr(
321        &mut self,
322        sql_expr: &SQLExpr,
323        inner_schema: &Schema,
324        outer_schema: &Schema,
325    ) -> PolarsResult<Option<Expr>> {
326        let expr = parse_sql_expr(sql_expr, self, Some(inner_schema))?;
327        // A nested subquery parses to `Expr::SubPlan`, which is only valid after
328        // `process_subqueries` lowering; it can't be used as a plain expression.
329        if has_expr(&expr, |e| matches!(e, Expr::SubPlan(_, _))) {
330            return Ok(None);
331        }
332        let only_inner = expr_to_leaf_column_names_iter(&expr).all(|name| {
333            inner_schema.contains(name.as_str()) && !outer_schema.contains(name.as_str())
334        });
335        Ok(only_inner.then_some(expr))
336    }
337
338    #[cfg(not(feature = "semi_anti_join"))]
339    fn try_rewrite_exists_as_join(
340        &mut self,
341        _lf: &LazyFrame,
342        _subquery: &Query,
343        _negated: bool,
344        _outer_schema: &Schema,
345    ) -> PolarsResult<Option<LazyFrame>> {
346        Ok(None)
347    }

349    #[cfg(not(feature = "semi_anti_join"))]
350    #[expect(clippy::too_many_arguments)]
351    fn try_rewrite_in_subquery_as_join(
352        &mut self,
353        _lf: &LazyFrame,
354        _lhs: &SQLExpr,
355        _subquery: &Query,
356        _anti: bool,
357        _filter_mode: FilterMode,
358        _outer_schema: &Schema,
359    ) -> PolarsResult<Option<LazyFrame>> {
360        Ok(None)
361    }

364/// An iterator over all the minterms in an SQL boolean expression: the terms
365/// that `AND` together to form it, descending through parenthesized `Nested`
366/// expressions. The SQL-AST analogue of the `AExpr`-level
367/// `polars_plan::plans::aexpr::MintermIter`.
368struct MintermIter<'a> {
369    stack: Vec<&'a SQLExpr>,
370}
371
372impl<'a> Iterator for MintermIter<'a> {
373    type Item = &'a SQLExpr;
374
375    fn next(&mut self) -> Option<Self::Item> {
376        let mut top = self.stack.pop()?;
377        loop {
378            match top {
379                SQLExpr::Nested(inner) => top = inner,
380                SQLExpr::BinaryOp {
381                    left,
382                    op: SQLBinaryOperator::And,
383                    right,
384                } => {
385                    self.stack.push(right);
386                    top = left;
387                },
388                _ => return Some(top),
389            }
390        }
391    }
392}
393
394impl<'a> MintermIter<'a> {
395    fn new(root: &'a SQLExpr) -> Self {
396        Self { stack: vec![root] }
397    }
398}

400#[cfg(feature = "semi_anti_join")]
401enum CorrelationSide {
402    Inner,
403    Outer,
404}
405
406// A subquery WHERE split into equi-correlation join keys (outer side in
407// `left_on`, inner side in `right_on`) and filters over inner columns only.
408#[cfg(feature = "semi_anti_join")]
409#[derive(Default)]
410struct SubqueryConjuncts {
411    left_on: Vec<Expr>,
412    right_on: Vec<Expr>,
413    local_filters: Vec<Expr>,
414}

416// An equi-correlation conjunct `inner.col = outer.col` (either way round) as a
417// `(outer key, inner key)` column-name pair, or `None` when the conjunct is
418// anything else (non-equality, unresolvable names, both columns on the same
419// side).
420#[cfg(feature = "semi_anti_join")]
421fn correlation_key_pair(
422    conj: &SQLExpr,
423    inner_names: &PlHashSet<String>,
424    inner_schema: &Schema,
425    outer_schema: &Schema,
426) -> Option<(PlSmallStr, PlSmallStr)> {
427    let SQLExpr::BinaryOp {
428        left,
429        op: SQLBinaryOperator::Eq,
430        right,
431    } = conj
432    else {
433        return None;
434    };
435    let (lside, lname) =
436        classify_correlation_column(left, inner_names, inner_schema, outer_schema)?;
437    let (rside, rname) =
438        classify_correlation_column(right, inner_names, inner_schema, outer_schema)?;
439    match (lside, rside) {
440        (CorrelationSide::Outer, CorrelationSide::Inner) => Some((lname, rname)),
441        (CorrelationSide::Inner, CorrelationSide::Outer) => Some((rname, lname)),
442        _ => None,
443    }
444}

446// Classify a correlation operand as an inner- or outer-query column and return
447// its bare name. A qualified identifier (`tbl.col`) resolves by its qualifier:
448// an inner relation's name/alias means inner, anything else means outer (so
449// same-named columns like `o.id = c.id` resolve). An unqualified identifier
450// resolves by schema membership. `None` for non-identifiers or names that can't
451// be placed (in neither schema, or ambiguous).
452#[cfg(feature = "semi_anti_join")]
453fn classify_correlation_column(
454    expr: &SQLExpr,
455    inner_names: &PlHashSet<String>,
456    inner_schema: &Schema,
457    outer_schema: &Schema,
458) -> Option<(CorrelationSide, PlSmallStr)> {
459    let (qualifier, name): (Option<&str>, PlSmallStr) = match expr {
460        SQLExpr::Identifier(ident) => (None, ident.value.as_str().into()),
461        SQLExpr::CompoundIdentifier(parts) => {
462            let (last, init) = parts.split_last()?;
463            // Only the table part: catalog/schema prefixes are dropped, just
464            // as `get_table_name` drops them when building `inner_names`.
465            (
466                init.last().map(|q| q.value.as_str()),
467                last.value.as_str().into(),
468            )
469        },
470        _ => return None,
471    };
472    match qualifier {
473        Some(q) if inner_names.contains(q) => inner_schema
474            .contains(name.as_str())
475            .then_some((CorrelationSide::Inner, name)),
476        // Any other qualifier is taken as outer: the outer query may span
477        // several relations and their names/aliases aren't visible here, so
478        // only the column's schema membership can be checked.
479        Some(_) => outer_schema
480            .contains(name.as_str())
481            .then_some((CorrelationSide::Outer, name)),
482        None => match (
483            inner_schema.contains(name.as_str()),
484            outer_schema.contains(name.as_str()),
485        ) {
486            (true, false) => Some((CorrelationSide::Inner, name)),
487            (false, true) => Some((CorrelationSide::Outer, name)),
488            _ => None,
489        },
490    }
491}

493// Shared eligibility gate for the rewrites: bail on any clause that changes
494// which rows the subquery yields. Exhaustive destructuring (no `..`) is on
495// purpose: a new sqlparser clause must not compile until it gets an explicit
496// keep-or-bail decision here.
497#[cfg(feature = "semi_anti_join")]
498fn eligible_subquery_select(subquery: &Query) -> Option<&Select> {
499    let Query {
500        with, // CTEs aren't resolved inside the rewrite: bail
501        body,
502        order_by: _,      // row order can't affect existence/membership
503        limit_clause,     // LIMIT/OFFSET change the yielded rows: bail
504        fetch,            // FETCH FIRST is LIMIT spelled differently: bail
505        locks: _,         // row locking doesn't change the rows
506        for_clause,       // FOR XML/JSON reshape the result: bail
507        settings,         // ClickHouse SETTINGS can change results: bail
508        format_clause: _, // output serialization only
509        pipe_operators,   // `|>` operators transform the rows: bail
510    } = subquery;
511    if with.is_some()
512        || limit_clause.is_some()
513        || fetch.is_some()
514        || for_clause.is_some()
515        || settings.is_some()
516        || !pipe_operators.is_empty()
517    {
518        return None;
519    }
520    let SetExpr::Select(select) = body.as_ref() else {
521        return None;
522    };
523    let Select {
524        select_token: _,
525        // Deduplication is existence/membership-invariant; the IN rewrite
526        // separately bails on `DISTINCT ON`, which is not.
527        distinct: _,
528        top,                    // TOP is LIMIT spelled differently: bail
529        top_before_distinct: _, // only meaningful with `top`
530        // The projection is validated by the callers: EXISTS ignores it, IN
531        // requires a single plain expression (which also rules out wildcards
532        // and the `exclude` modifier).
533        projection: _,
534        exclude: _,
535        into,          // SELECT INTO is not a pure subquery: bail
536        from,          // must be one (possibly joined) relation
537        lateral_views, // row-multiplying: bail
538        prewhere,      // an extra filter we don't fold in: bail
539        selection: _,  // split into join keys/filters by callers
540        group_by,      // aggregation changes the yielded rows: bail
541        cluster_by: _, // layout/order hints: row-set preserving
542        distribute_by: _,
543        sort_by: _,
544        having,                   // aggregation filter: bail
545        named_window: _,          // definitions only; uses are parsed later
546        qualify,                  // post-window filter changes the rows: bail
547        window_before_qualify: _, // only meaningful with `qualify`
548        value_table_mode,         // changes what a row is: bail
549        connect_by,               // hierarchical recursion: bail
550        optimizer_hints,          // unsupported: bail
551        select_modifiers,         // unsupported: bail
552        flavor: _,                // surface syntax only
553    } = select.as_ref();
554    let no_group_by = matches!(
555        group_by,
556        GroupByExpr::Expressions(e, m) if e.is_empty() && m.is_empty()
557    );
558    if from.len() != 1
559        || !no_group_by
560        || top.is_some()
561        || into.is_some()
562        || having.is_some()
563        || qualify.is_some()
564        || prewhere.is_some()
565        || !connect_by.is_empty()
566        || value_table_mode.is_some()
567        || !lateral_views.is_empty()
568        || !optimizer_hints.is_empty()
569        || select_modifiers.is_some()
570    {
571        return None;
572    }
573    Some(select)
574}
