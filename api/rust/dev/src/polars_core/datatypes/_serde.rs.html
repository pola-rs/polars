1//! Having `Object<&;static> in [`DataType`] make serde tag the `Deserialize` trait bound 'static
2//! even though we skip serializing `Object`.
3//!
4//! We could use [serde_1712](https://github.com/serde-rs/serde/issues/1712), but that gave problems caused by
5//! [rust_96956](https://github.com/rust-lang/rust/issues/96956), so we make a dummy type without static
6

14impl<'a> Deserialize<'a> for DataType {
15    fn deserialize<D>(deserializer: D) -> std::result::Result<Self, D::Error>
16    where
17        D: Deserializer<'a>,
18    {
19        Ok(SerializableDataType::deserialize(deserializer)?.into())
20    }
21}
22
23impl Serialize for DataType {
24    fn serialize<S>(&self, serializer: S) -> std::result::Result<S::Ok, S::Error>
25    where
26        S: Serializer,
27    {
28        let dt: SerializableDataType = self.into();
29        dt.serialize(serializer)
30    }
31}

51#[cfg(feature = "dtype-categorical")]
52impl serde::Serialize for Wrap<Utf8ViewArray> {
53    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
54    where
55        S: Serializer,
56    {
57        serializer.collect_seq(self.0.values_iter())
58    }
59}

61#[cfg(feature = "dtype-categorical")]
62impl<'de> serde::Deserialize<'de> for Wrap<Utf8ViewArray> {
63    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
64    where
65        D: Deserializer<'de>,
66    {
67        struct Utf8Visitor;
68
69        impl<'de> Visitor<'de> for Utf8Visitor {
70            type Value = Wrap<Utf8ViewArray>;
71
72            fn expecting(&self, formatter: &mut Formatter) -> std::fmt::Result {
73                formatter.write_str("Utf8Visitor string sequence.")
74            }
75
76            fn visit_seq<A>(self, mut seq: A) -> Result<Self::Value, A::Error>
77            where
78                A: SeqAccess<'de>,
79            {
80                let mut utf8array = MutablePlString::with_capacity(seq.size_hint().unwrap_or(10));
81                while let Some(key) = seq.next_element()? {
82                    let key: Option<String> = key;
83                    utf8array.push(key)
84                }
85                Ok(Wrap(utf8array.into()))
86            }
87        }
88
89        deserializer.deserialize_seq(Utf8Visitor)
90    }
91}

96enum SerializableDataType {
97    Boolean,
98    UInt8,
99    UInt16,
100    UInt32,
101    UInt64,
102    Int8,
103    Int16,
104    Int32,
105    Int64,
106    Int128,
107    Float32,
108    Float64,
109    String,
110    Binary,
111    BinaryOffset,
112    /// A 32-bit date representing the elapsed time since UNIX epoch (1970-01-01)
113    /// in days (32 bits).
114    Date,
115    /// A 64-bit date representing the elapsed time since UNIX epoch (1970-01-01)
116    /// in the given ms/us/ns TimeUnit (64 bits).
117    Datetime(TimeUnit, Option<TimeZone>),
118    // 64-bit integer representing difference between times in milli|micro|nano seconds
119    Duration(TimeUnit),
120    /// A 64-bit time representing elapsed time since midnight in the given TimeUnit.
121    Time,
122    List(Box<SerializableDataType>),
123    #[cfg(feature = "dtype-array")]
124    Array(Box<SerializableDataType>, usize),
125    Null,
126    #[cfg(feature = "dtype-struct")]
127    Struct(Vec<Field>),
128    // some logical types we cannot know statically, e.g. Datetime
129    Unknown(UnknownKind),
130    #[cfg(feature = "dtype-categorical")]
131    Categorical {
132        name: String,
133        namespace: String,
134        physical: CategoricalPhysical,
135    },
136    #[cfg(feature = "dtype-categorical")]
137    Enum {
138        strings: Series,
139    },
140    #[cfg(feature = "dtype-decimal")]
141    Decimal(Option<usize>, Option<usize>),
142    #[cfg(feature = "object")]
143    Object(String),
144}

146impl From<&DataType> for SerializableDataType {
147    fn from(dt: &DataType) -> Self {
148        use DataType::*;
149        match dt {
150            Boolean => Self::Boolean,
151            UInt8 => Self::UInt8,
152            UInt16 => Self::UInt16,
153            UInt32 => Self::UInt32,
154            UInt64 => Self::UInt64,
155            Int8 => Self::Int8,
156            Int16 => Self::Int16,
157            Int32 => Self::Int32,
158            Int64 => Self::Int64,
159            Int128 => Self::Int128,
160            Float32 => Self::Float32,
161            Float64 => Self::Float64,
162            String => Self::String,
163            Binary => Self::Binary,
164            BinaryOffset => Self::BinaryOffset,
165            Date => Self::Date,
166            Datetime(tu, tz) => Self::Datetime(*tu, tz.clone()),
167            Duration(tu) => Self::Duration(*tu),
168            Time => Self::Time,
169            List(dt) => Self::List(Box::new(dt.as_ref().into())),
170            #[cfg(feature = "dtype-array")]
171            Array(dt, width) => Self::Array(Box::new(dt.as_ref().into()), *width),
172            Null => Self::Null,
173            Unknown(kind) => Self::Unknown(*kind),
174            #[cfg(feature = "dtype-struct")]
175            Struct(flds) => Self::Struct(flds.clone()),
176            #[cfg(feature = "dtype-categorical")]
177            Categorical(cats, _) => Self::Categorical {
178                name: cats.name().to_string(),
179                namespace: cats.namespace().to_string(),
180                physical: cats.physical(),
181            },
182            #[cfg(feature = "dtype-categorical")]
183            Enum(fcats, _) => Self::Enum {
184                strings: StringChunked::with_chunk(
185                    PlSmallStr::from_static("categories"),
186                    fcats.categories().clone(),
187                )
188                .into_series(),
189            },
190            #[cfg(feature = "dtype-decimal")]
191            Decimal(precision, scale) => Self::Decimal(*precision, *scale),
192            #[cfg(feature = "object")]
193            Object(name) => Self::Object(name.to_string()),
194        }
195    }
196}
197impl From<SerializableDataType> for DataType {
198    fn from(dt: SerializableDataType) -> Self {
199        use SerializableDataType::*;
200        match dt {
201            Boolean => Self::Boolean,
202            UInt8 => Self::UInt8,
203            UInt16 => Self::UInt16,
204            UInt32 => Self::UInt32,
205            UInt64 => Self::UInt64,
206            Int8 => Self::Int8,
207            Int16 => Self::Int16,
208            Int32 => Self::Int32,
209            Int64 => Self::Int64,
210            Int128 => Self::Int128,
211            Float32 => Self::Float32,
212            Float64 => Self::Float64,
213            String => Self::String,
214            Binary => Self::Binary,
215            BinaryOffset => Self::BinaryOffset,
216            Date => Self::Date,
217            Datetime(tu, tz) => Self::Datetime(tu, tz),
218            Duration(tu) => Self::Duration(tu),
219            Time => Self::Time,
220            List(dt) => Self::List(Box::new((*dt).into())),
221            #[cfg(feature = "dtype-array")]
222            Array(dt, width) => Self::Array(Box::new((*dt).into()), width),
223            Null => Self::Null,
224            Unknown(kind) => Self::Unknown(kind),
225            #[cfg(feature = "dtype-struct")]
226            Struct(flds) => Self::Struct(flds),
227            #[cfg(feature = "dtype-categorical")]
228            Categorical {
229                name,
230                namespace,
231                physical,
232            } => {
233                let cats = Categories::new(
234                    PlSmallStr::from(name),
235                    PlSmallStr::from(namespace),
236                    physical,
237                );
238                let mapping = cats.mapping();
239                Self::Categorical(cats, mapping)
240            },
241            #[cfg(feature = "dtype-categorical")]
242            Enum { strings } => {
243                let ca = strings.str().unwrap();
244                let fcats = FrozenCategories::new(ca.iter().flatten()).unwrap();
245                let mapping = fcats.mapping().clone();
246                Self::Enum(fcats, mapping)
247            },
248            #[cfg(feature = "dtype-decimal")]
249            Decimal(precision, scale) => Self::Decimal(precision, scale),
250            #[cfg(feature = "object")]
251            Object(_) => Self::Object("unknown"),
252        }
253    }
