10/// Characterizes the name and the [`DataType`] of a column.
11#[derive(Clone, Debug, PartialEq, Eq, Hash)]
12#[cfg_attr(
13    any(feature = "serde", feature = "serde-lazy"),
14    derive(Serialize, Deserialize)
15)]
16#[cfg_attr(feature = "dsl-schema", derive(schemars::JsonSchema))]
17pub struct Field {
18    pub name: PlSmallStr,
19    pub dtype: DataType,
20}
21
22impl From<Field> for (PlSmallStr, DataType) {
23    fn from(value: Field) -> Self {
24        (value.name, value.dtype)
25    }
26}
27
28pub type FieldRef = Arc<Field>;

30impl Field {
31    /// Creates a new `Field`.
32    ///
33    /// # Example
34    ///
35    /// ```rust
36    /// # use polars_core::prelude::*;
37    /// let f1 = Field::new("Fruit name".into(), DataType::String);
38    /// let f2 = Field::new("Lawful".into(), DataType::Boolean);
39    /// let f2 = Field::new("Departure".into(), DataType::Time);
40    /// ```
41    #[inline]
42    pub fn new(name: PlSmallStr, dtype: DataType) -> Self {
43        Field { name, dtype }
44    }
45
46    /// Returns a reference to the `Field` name.
47    ///
48    /// # Example
49    ///
50    /// ```rust
51    /// # use polars_core::prelude::*;
52    /// let f = Field::new("Year".into(), DataType::Int32);
53    ///
54    /// assert_eq!(f.name(), "Year");
55    /// ```
56    #[inline]
57    pub fn name(&self) -> &PlSmallStr {
58        &self.name
59    }
60
61    /// Returns a reference to the `Field` datatype.
62    ///
63    /// # Example
64    ///
65    /// ```rust
66    /// # use polars_core::prelude::*;
67    /// let f = Field::new("Birthday".into(), DataType::Date);
68    ///
69    /// assert_eq!(f.dtype(), &DataType::Date);
70    /// ```
71    #[inline]
72    pub fn dtype(&self) -> &DataType {
73        &self.dtype
74    }
75
76    /// Sets the `Field` datatype.
77    ///
78    /// # Example
79    ///
80    /// ```rust
81    /// # use polars_core::prelude::*;
82    /// let mut f = Field::new("Temperature".into(), DataType::Int32);
83    /// f.coerce(DataType::Float32);
84    ///
85    /// assert_eq!(f, Field::new("Temperature".into(), DataType::Float32));
86    /// ```
87    pub fn coerce(&mut self, dtype: DataType) {
88        self.dtype = dtype;
89    }
90
91    /// Sets the `Field` name.
92    ///
93    /// # Example
94    ///
95    /// ```rust
96    /// # use polars_core::prelude::*;
97    /// let mut f = Field::new("Atomic number".into(), DataType::UInt32);
98    /// f.set_name("Proton".into());
99    ///
100    /// assert_eq!(f, Field::new("Proton".into(), DataType::UInt32));
101    /// ```
102    pub fn set_name(&mut self, name: PlSmallStr) {
103        self.name = name;
104    }
105
106    /// Returns this `Field`, renamed.
107    pub fn with_name(mut self, name: PlSmallStr) -> Self {
108        self.name = name;
109        self
110    }
111
112    // Returns this `Field`, with a different datatype.
113    pub fn with_dtype(mut self, dtype: DataType) -> Self {
114        self.dtype = dtype;
115        self
116    }
117
118    /// Converts the `Field` to an `arrow::datatypes::Field`.
119    ///
120    /// # Example
121    ///
122    /// ```rust
123    /// # use polars_core::prelude::*;
124    /// let f = Field::new("Value".into(), DataType::Int64);
125    /// let af = arrow::datatypes::Field::new("Value".into(), arrow::datatypes::ArrowDataType::Int64, true);
126    ///
127    /// assert_eq!(f.to_arrow(CompatLevel::newest()), af);
128    /// ```
129    pub fn to_arrow(&self, compat_level: CompatLevel) -> ArrowField {
130        self.dtype.to_arrow_field(self.name.clone(), compat_level)
131    }
132
133    pub fn to_physical(&self) -> Field {
134        Self {
135            name: self.name.clone(),
136            dtype: self.dtype().to_physical(),
137        }
138    }

166    pub fn from_arrow(dt: &ArrowDataType, md: Option<&Metadata>) -> DataType {
167        match dt {
168            ArrowDataType::Null => DataType::Null,
169            ArrowDataType::UInt8 => DataType::UInt8,
170            ArrowDataType::UInt16 => DataType::UInt16,
171            ArrowDataType::UInt32 => DataType::UInt32,
172            ArrowDataType::UInt64 => DataType::UInt64,
173            #[cfg(feature = "dtype-u128")]
174            ArrowDataType::UInt128 => DataType::UInt128,
175            ArrowDataType::Int8 => DataType::Int8,
176            ArrowDataType::Int16 => DataType::Int16,
177            ArrowDataType::Int32 => DataType::Int32,
178            ArrowDataType::Int64 => DataType::Int64,
179            #[cfg(feature = "dtype-i128")]
180            ArrowDataType::Int128 => DataType::Int128,
181            ArrowDataType::Boolean => DataType::Boolean,
182            #[cfg(feature = "dtype-f16")]
183            ArrowDataType::Float16 => DataType::Float16,
184            ArrowDataType::Float32 => DataType::Float32,
185            ArrowDataType::Float64 => DataType::Float64,
186            #[cfg(feature = "dtype-array")]
187            ArrowDataType::FixedSizeList(f, size) => {
188                DataType::Array(DataType::from_arrow_field(f).boxed(), *size)
189            },
190            ArrowDataType::LargeList(f) | ArrowDataType::List(f) => {
191                DataType::List(DataType::from_arrow_field(f).boxed())
192            },
193            ArrowDataType::Date32 => DataType::Date,
194            ArrowDataType::Timestamp(tu, tz) => {
195                DataType::Datetime(tu.into(), TimeZone::opt_try_new(tz.clone()).unwrap())
196            },
197            ArrowDataType::Duration(tu) => DataType::Duration(tu.into()),
198            ArrowDataType::Date64 => DataType::Datetime(TimeUnit::Milliseconds, None),
199            ArrowDataType::Time64(_) | ArrowDataType::Time32(_) => DataType::Time,
200
201            #[cfg(feature = "dtype-categorical")]
202            ArrowDataType::Dictionary(_, value_type, _) => {
203                // The metadata encoding here must match DataType::to_arrow_field.
204                if let Some(mut enum_md) = md.and_then(|md| md.pl_enum_metadata()) {
205                    let cats = move || {
206                        if enum_md.is_empty() {
207                            return None;
208                        }
209
210                        let len;
211                        (len, enum_md) = enum_md.split_once(';').unwrap();
212                        let len = len.parse::<usize>().unwrap();
213                        let cat;
214                        (cat, enum_md) = enum_md.split_at(len);
215                        Some(cat)
216                    };
217
218                    let fcats = FrozenCategories::new(std::iter::from_fn(cats)).unwrap();
219                    DataType::from_frozen_categories(fcats)
220                } else if let Some(mut cat_md) = md.and_then(|md| md.pl_categorical_metadata()) {
221                    let name_len;
222                    (name_len, cat_md) = cat_md.split_once(';').unwrap();
223                    let name_len = name_len.parse::<usize>().unwrap();
224                    let name;
225                    (name, cat_md) = cat_md.split_at(name_len);
226
227                    let namespace_len;
228                    (namespace_len, cat_md) = cat_md.split_once(';').unwrap();
229                    let namespace_len = namespace_len.parse::<usize>().unwrap();
230                    let namespace;
231                    (namespace, cat_md) = cat_md.split_at(namespace_len);
232
233                    let (physical, _rest) = cat_md.split_once(';').unwrap();
234
235                    let physical: CategoricalPhysical = physical.parse().ok().unwrap();
236                    let cats = Categories::new(
237                        PlSmallStr::from_str(name),
238                        PlSmallStr::from_str(namespace),
239                        physical,
240                    );
241                    DataType::from_categories(cats)
242                } else if matches!(
243                    value_type.as_ref(),
244                    ArrowDataType::Utf8 | ArrowDataType::LargeUtf8 | ArrowDataType::Utf8View
245                ) {
246                    DataType::from_categories(Categories::global())
247                } else {
248                    Self::from_arrow(value_type, None)
249                }
250            },
251
252            #[cfg(feature = "dtype-struct")]
253            ArrowDataType::Struct(fields) => {
254                DataType::Struct(fields.iter().map(|fld| fld.into()).collect())
255            },
256            #[cfg(not(feature = "dtype-struct"))]
257            ArrowDataType::Struct(_) => {
258                panic!("activate the 'dtype-struct' feature to handle struct data types")
259            },
260            ArrowDataType::Extension(ext) if ext.name.as_str() == POLARS_OBJECT_EXTENSION_NAME => {
261                #[cfg(feature = "object")]
262                {
263                    DataType::Object("object")
264                }
265                #[cfg(not(feature = "object"))]
266                {
267                    panic!("activate the 'object' feature to be able to load POLARS_EXTENSION_TYPE")
268                }
269            },
270            #[cfg(feature = "dtype-extension")]
271            ArrowDataType::Extension(ext) => {
272                use crate::prelude::extension::get_extension_type_or_storage;
273                let storage = DataType::from_arrow(&ext.inner, md);
274                match get_extension_type_or_storage(&ext.name, &storage, ext.metadata.as_deref()) {
275                    Some(typ) => DataType::Extension(typ, Box::new(storage)),
276                    None => storage,
277                }
278            },
279            #[cfg(feature = "dtype-decimal")]
280            ArrowDataType::Decimal(precision, scale) => DataType::Decimal(*precision, *scale),
281            ArrowDataType::Utf8View | ArrowDataType::LargeUtf8 | ArrowDataType::Utf8 => {
282                DataType::String
283            },
284            ArrowDataType::BinaryView => DataType::Binary,
285            ArrowDataType::LargeBinary if md.is_some() => {
286                let md = md.unwrap();
287                if md.maintain_type() {
288                    DataType::BinaryOffset
289                } else {
290                    DataType::Binary
291                }
292            },
293            ArrowDataType::LargeBinary | ArrowDataType::Binary => DataType::Binary,
294            ArrowDataType::FixedSizeBinary(_) => DataType::Binary,
295            ArrowDataType::Map(inner, _is_sorted) => {
296                DataType::List(Self::from_arrow_field(inner).boxed())
297            },
298            ArrowDataType::Interval(IntervalUnit::MonthDayNano) => {
299                check_allow_importing_interval_as_struct("month_day_nano_interval").unwrap();
300                feature_gated!("dtype-struct", DataType::_month_days_ns_struct_type())
301            },
302            ArrowDataType::Interval(IntervalUnit::MonthDayMillis) => {
303                check_allow_importing_interval_as_struct("month_day_millisecond_interval").unwrap();
304                feature_gated!("dtype-struct", DataType::_month_days_ns_struct_type())
305            },
306            dt => panic!(
307                "Arrow datatype {dt:?} not supported by Polars. \
308                You probably need to activate that data-type feature."
309            ),
310        }
311    }
312}
