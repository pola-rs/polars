17impl<'a, T: PolarsObject> ArrayFromIter<Option<&'a T>> for ObjectArray<T> {
18    fn arr_from_iter<I: IntoIterator<Item = Option<&'a T>>>(iter: I) -> Self {
19        Self::try_arr_from_iter(iter.into_iter().map(|o| -> Result<_, ()> { Ok(o) })).unwrap()
20    }
21
22    fn try_arr_from_iter<E, I: IntoIterator<Item = Result<Option<&'a T>, E>>>(
23        iter: I,
24    ) -> Result<Self, E> {
25        let iter = iter.into_iter();
26        let size = iter.size_hint().0;
27
28        let mut null_mask_builder = BitmapBuilder::with_capacity(size);
29        let values: Vec<T> = iter
30            .map(|value| match value? {
31                Some(value) => {
32                    null_mask_builder.push(true);
33                    Ok(value.clone())
34                },
35                None => {
36                    null_mask_builder.push(false);
37                    Ok(T::default())
38                },
39            })
40            .collect::<Result<Vec<T>, E>>()?;
41
42        Ok(ObjectArray::from(values).with_validity(null_mask_builder.into_opt_validity()))
43    }
44}
