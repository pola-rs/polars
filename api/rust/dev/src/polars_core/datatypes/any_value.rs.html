117        match dtype {
118            DataType::String => AnyValue::StringOwned(PlSmallStr::EMPTY),
119            DataType::Binary => AnyValue::BinaryOwned(Vec::new()),

140    /// Generate a default dummy value for a given datatype.
141    pub fn default_value(
142        dtype: &DataType,
143        numeric_to_one: bool,
144        num_list_values: usize,
145    ) -> AnyValue<'static> {
146        use {AnyValue as AV, DataType as DT};
147        match dtype {
148            DT::Boolean => AV::Boolean(false),
149            DT::UInt8 => AV::UInt8(numeric_to_one.into()),
150            DT::UInt16 => AV::UInt16(numeric_to_one.into()),
151            DT::UInt32 => AV::UInt32(numeric_to_one.into()),
152            DT::UInt64 => AV::UInt64(numeric_to_one.into()),
153            DT::UInt128 => AV::UInt128(numeric_to_one.into()),
154            DT::Int8 => AV::Int8(numeric_to_one.into()),
155            DT::Int16 => AV::Int16(numeric_to_one.into()),
156            DT::Int32 => AV::Int32(numeric_to_one.into()),
157            DT::Int64 => AV::Int64(numeric_to_one.into()),
158            DT::Int128 => AV::Int128(numeric_to_one.into()),
159            DT::Float32 => AV::Float32(numeric_to_one.into()),
160            DT::Float64 => AV::Float64(numeric_to_one.into()),
161            #[cfg(feature = "dtype-decimal")]
162            DT::Decimal(p, s) => AV::Decimal(0, *p, *s),
163            DT::String => AV::String(""),
164            DT::Binary => AV::Binary(&[]),
165            DT::BinaryOffset => AV::Binary(&[]),
166            DT::Date => feature_gated!("dtype-date", AV::Date(0)),
167            DT::Datetime(time_unit, time_zone) => feature_gated!(
168                "dtype-datetime",
169                AV::DatetimeOwned(0, *time_unit, time_zone.clone().map(Arc::new))
170            ),
171            DT::Duration(time_unit) => {
172                feature_gated!("dtype-duration", AV::Duration(0, *time_unit))
173            },
174            DT::Time => feature_gated!("dtype-time", AV::Time(0)),
175            #[cfg(feature = "dtype-array")]
176            DT::Array(inner_dtype, width) => {
177                let inner_value =
178                    AnyValue::default_value(inner_dtype, numeric_to_one, num_list_values);
179                AV::Array(
180                    Scalar::new(inner_dtype.as_ref().clone(), inner_value)
181                        .into_series(PlSmallStr::EMPTY)
182                        .new_from_index(0, *width),
183                    *width,
184                )
185            },
186            DT::List(inner_dtype) => AV::List(if num_list_values == 0 {
187                Series::new_empty(PlSmallStr::EMPTY, inner_dtype.as_ref())
188            } else {
189                let inner_value =
190                    AnyValue::default_value(inner_dtype, numeric_to_one, num_list_values);
191
192                Scalar::new(inner_dtype.as_ref().clone(), inner_value)
193                    .into_series(PlSmallStr::EMPTY)
194                    .new_from_index(0, num_list_values)
195            }),
196            #[cfg(feature = "object")]
197            DT::Object(_) => AV::Null,
198            DT::Null => AV::Null,
199            #[cfg(feature = "dtype-categorical")]
200            DT::Categorical(_, _) => AV::Null,
201            #[cfg(feature = "dtype-categorical")]
202            DT::Enum(categories, mapping) => match categories.categories().is_empty() {
203                true => AV::Null,
204                false => AV::EnumOwned(0, mapping.clone()),
205            },
206            #[cfg(feature = "dtype-struct")]
207            DT::Struct(fields) => AV::StructOwned(Box::new((
208                fields
209                    .iter()
210                    .map(|f| AnyValue::default_value(f.dtype(), numeric_to_one, num_list_values))
211                    .collect(),
212                fields.clone(),
213            ))),
214            DT::Unknown(_) => unreachable!(),
215        }
216    }

222    /// Note: For `Categorical` and `Enum` values, the exact mapping information

224    pub fn dtype(&self) -> DataType {
225        use AnyValue::*;
226        match self {
227            Null => DataType::Null,
228            Boolean(_) => DataType::Boolean,
229            Int8(_) => DataType::Int8,
230            Int16(_) => DataType::Int16,
231            Int32(_) => DataType::Int32,
232            Int64(_) => DataType::Int64,
233            Int128(_) => DataType::Int128,
234            UInt8(_) => DataType::UInt8,
235            UInt16(_) => DataType::UInt16,
236            UInt32(_) => DataType::UInt32,
237            UInt64(_) => DataType::UInt64,
238            UInt128(_) => DataType::UInt128,
239            Float32(_) => DataType::Float32,
240            Float64(_) => DataType::Float64,
241            String(_) | StringOwned(_) => DataType::String,
242            Binary(_) | BinaryOwned(_) => DataType::Binary,
243            #[cfg(feature = "dtype-date")]
244            Date(_) => DataType::Date,
245            #[cfg(feature = "dtype-time")]
246            Time(_) => DataType::Time,
247            #[cfg(feature = "dtype-datetime")]
248            Datetime(_, tu, tz) => DataType::Datetime(*tu, (*tz).cloned()),
249            #[cfg(feature = "dtype-datetime")]
250            DatetimeOwned(_, tu, tz) => {
251                DataType::Datetime(*tu, tz.as_ref().map(|v| v.as_ref().clone()))
252            },
253            #[cfg(feature = "dtype-duration")]
254            Duration(_, tu) => DataType::Duration(*tu),
255            #[cfg(feature = "dtype-categorical")]

257                unimplemented!("can not get dtype of Categorical AnyValue")
258            },
259            #[cfg(feature = "dtype-categorical")]
260            Enum(_, _) | EnumOwned(_, _) => unimplemented!("can not get dtype of Enum AnyValue"),
261            List(s) => DataType::List(Box::new(s.dtype().clone())),
262            #[cfg(feature = "dtype-array")]
263            Array(s, size) => DataType::Array(Box::new(s.dtype().clone()), *size),
264            #[cfg(feature = "dtype-struct")]
265            Struct(_, _, fields) => DataType::Struct(fields.to_vec()),
266            #[cfg(feature = "dtype-struct")]
267            StructOwned(payload) => DataType::Struct(payload.1.clone()),
268            #[cfg(feature = "dtype-decimal")]
269            Decimal(_, p, s) => DataType::Decimal(*p, *s),
270            #[cfg(feature = "object")]
271            Object(o) => DataType::Object(o.type_name()),
272            #[cfg(feature = "object")]
273            ObjectOwned(o) => DataType::Object(o.0.type_name()),
274        }

277    /// Extract a numerical value from the AnyValue
278    #[doc(hidden)]
279    #[inline]
280    pub fn extract<T: NumCast + IsFloat>(&self) -> Option<T> {
281        use AnyValue::*;
282        match self {
283            Int8(v) => NumCast::from(*v),
284            Int16(v) => NumCast::from(*v),
285            Int32(v) => NumCast::from(*v),
286            Int64(v) => NumCast::from(*v),
287            Int128(v) => NumCast::from(*v),
288            UInt8(v) => NumCast::from(*v),
289            UInt16(v) => NumCast::from(*v),
290            UInt32(v) => NumCast::from(*v),
291            UInt64(v) => NumCast::from(*v),
292            UInt128(v) => NumCast::from(*v),
293            Float32(v) => NumCast::from(*v),
294            Float64(v) => NumCast::from(*v),
295            #[cfg(feature = "dtype-date")]
296            Date(v) => NumCast::from(*v),
297            #[cfg(feature = "dtype-datetime")]
298            Datetime(v, _, _) | DatetimeOwned(v, _, _) => NumCast::from(*v),
299            #[cfg(feature = "dtype-time")]
300            Time(v) => NumCast::from(*v),
301            #[cfg(feature = "dtype-duration")]
302            Duration(v, _) => NumCast::from(*v),
303            #[cfg(feature = "dtype-decimal")]
304            Decimal(v, _p, s) => {
305                if T::is_float() {
306                    NumCast::from(dec128_to_f64(*v, *s))
307                } else {
308                    NumCast::from(dec128_to_i128(*v, *s))
309                }
310            },
311            Boolean(v) => NumCast::from(if *v { 1 } else { 0 }),
312            String(v) => {
313                if let Ok(val) = (*v).parse::<i128>() {
314                    NumCast::from(val)
315                } else {
316                    NumCast::from((*v).parse::<f64>().ok()?)
317                }
318            },
319            StringOwned(v) => String(v.as_str()).extract(),
320            _ => None,
321        }
322    }

396    /// Cast `AnyValue` to the provided data type and return a new `AnyValue` with type `dtype`,
397    /// if possible.
398    pub fn strict_cast(&self, dtype: &'a DataType) -> Option<AnyValue<'a>> {
399        let new_av = match (self, dtype) {
400            // to numeric
401            (av, DataType::UInt8) => AnyValue::UInt8(av.extract::<u8>()?),
402            (av, DataType::UInt16) => AnyValue::UInt16(av.extract::<u16>()?),
403            (av, DataType::UInt32) => AnyValue::UInt32(av.extract::<u32>()?),
404            (av, DataType::UInt64) => AnyValue::UInt64(av.extract::<u64>()?),
405            (av, DataType::UInt128) => AnyValue::UInt128(av.extract::<u128>()?),
406            (av, DataType::Int8) => AnyValue::Int8(av.extract::<i8>()?),
407            (av, DataType::Int16) => AnyValue::Int16(av.extract::<i16>()?),
408            (av, DataType::Int32) => AnyValue::Int32(av.extract::<i32>()?),
409            (av, DataType::Int64) => AnyValue::Int64(av.extract::<i64>()?),
410            (av, DataType::Int128) => AnyValue::Int128(av.extract::<i128>()?),
411            (av, DataType::Float32) => AnyValue::Float32(av.extract::<f32>()?),
412            (av, DataType::Float64) => AnyValue::Float64(av.extract::<f64>()?),
413
414            // to boolean
415            (AnyValue::UInt8(v), DataType::Boolean) => AnyValue::Boolean(*v != u8::default()),
416            (AnyValue::UInt16(v), DataType::Boolean) => AnyValue::Boolean(*v != u16::default()),
417            (AnyValue::UInt32(v), DataType::Boolean) => AnyValue::Boolean(*v != u32::default()),
418            (AnyValue::UInt64(v), DataType::Boolean) => AnyValue::Boolean(*v != u64::default()),
419            (AnyValue::UInt128(v), DataType::Boolean) => AnyValue::Boolean(*v != u128::default()),
420            (AnyValue::Int8(v), DataType::Boolean) => AnyValue::Boolean(*v != i8::default()),
421            (AnyValue::Int16(v), DataType::Boolean) => AnyValue::Boolean(*v != i16::default()),
422            (AnyValue::Int32(v), DataType::Boolean) => AnyValue::Boolean(*v != i32::default()),
423            (AnyValue::Int64(v), DataType::Boolean) => AnyValue::Boolean(*v != i64::default()),
424            (AnyValue::Int128(v), DataType::Boolean) => AnyValue::Boolean(*v != i128::default()),
425            (AnyValue::Float32(v), DataType::Boolean) => AnyValue::Boolean(*v != f32::default()),
426            (AnyValue::Float64(v), DataType::Boolean) => AnyValue::Boolean(*v != f64::default()),
427
428            // Categorical casts.
429            #[cfg(feature = "dtype-categorical")]
430            (
431                &AnyValue::Categorical(cat, &ref lmap) | &AnyValue::CategoricalOwned(cat, ref lmap),
432                DataType::Categorical(_, rmap),
433            ) => {
434                if Arc::ptr_eq(lmap, rmap) {
435                    self.clone()
436                } else {
437                    let s = unsafe { lmap.cat_to_str_unchecked(cat) };
438                    let new_cat = rmap.insert_cat(s).unwrap();
439                    AnyValue::CategoricalOwned(new_cat, rmap.clone())
440                }
441            },
442
443            #[cfg(feature = "dtype-categorical")]
444            (
445                &AnyValue::Enum(cat, &ref lmap) | &AnyValue::EnumOwned(cat, ref lmap),
446                DataType::Enum(_, rmap),
447            ) => {
448                if Arc::ptr_eq(lmap, rmap) {
449                    self.clone()
450                } else {
451                    let s = unsafe { lmap.cat_to_str_unchecked(cat) };
452                    let new_cat = rmap.get_cat(s)?;
453                    AnyValue::EnumOwned(new_cat, rmap.clone())
454                }
455            },
456
457            #[cfg(feature = "dtype-categorical")]
458            (
459                &AnyValue::Categorical(cat, &ref map)
460                | &AnyValue::CategoricalOwned(cat, ref map)
461                | &AnyValue::Enum(cat, &ref map)
462                | &AnyValue::EnumOwned(cat, ref map),
463                DataType::String,
464            ) => {
465                let s = unsafe { map.cat_to_str_unchecked(cat) };
466                AnyValue::StringOwned(PlSmallStr::from(s))
467            },
468
469            #[cfg(feature = "dtype-categorical")]
470            (AnyValue::String(s), DataType::Categorical(_, map)) => {
471                AnyValue::CategoricalOwned(map.insert_cat(s).unwrap(), map.clone())
472            },
473
474            #[cfg(feature = "dtype-categorical")]
475            (AnyValue::StringOwned(s), DataType::Categorical(_, map)) => {
476                AnyValue::CategoricalOwned(map.insert_cat(s).unwrap(), map.clone())
477            },
478
479            #[cfg(feature = "dtype-categorical")]
480            (AnyValue::String(s), DataType::Enum(_, map)) => {
481                AnyValue::CategoricalOwned(map.get_cat(s)?, map.clone())
482            },
483
484            #[cfg(feature = "dtype-categorical")]
485            (AnyValue::StringOwned(s), DataType::Enum(_, map)) => {
486                AnyValue::CategoricalOwned(map.get_cat(s)?, map.clone())
487            },
488
489            // to string
490            (AnyValue::String(v), DataType::String) => AnyValue::String(v),
491            (AnyValue::StringOwned(v), DataType::String) => AnyValue::StringOwned(v.clone()),
492
493            (av, DataType::String) => {
494                let mut tmp = vec![];
495                if av.is_unsigned_integer() {
496                    let val = av.extract::<u64>()?;
497                    SerPrimitive::write(&mut tmp, val);
498                } else if av.is_float() {
499                    let val = av.extract::<f64>()?;
500                    SerPrimitive::write(&mut tmp, val);
501                } else {
502                    let val = av.extract::<i64>()?;
503                    SerPrimitive::write(&mut tmp, val);
504                }
505                AnyValue::StringOwned(PlSmallStr::from_str(std::str::from_utf8(&tmp).unwrap()))
506            },
507
508            // to binary
509            (AnyValue::String(v), DataType::Binary) => AnyValue::Binary(v.as_bytes()),
510
511            // to datetime
512            #[cfg(feature = "dtype-datetime")]
513            (av, DataType::Datetime(tu, tz)) if av.is_primitive_numeric() => {
514                AnyValue::Datetime(av.extract::<i64>()?, *tu, tz.as_ref())
515            },
516            #[cfg(all(feature = "dtype-datetime", feature = "dtype-date"))]
517            (AnyValue::Date(v), DataType::Datetime(tu, _)) => AnyValue::Datetime(
518                match tu {
519                    TimeUnit::Nanoseconds => (*v as i64) * NS_IN_DAY,
520                    TimeUnit::Microseconds => (*v as i64) * US_IN_DAY,
521                    TimeUnit::Milliseconds => (*v as i64) * MS_IN_DAY,
522                },
523                *tu,
524                None,
525            ),
526            #[cfg(feature = "dtype-datetime")]
527            (
528                AnyValue::Datetime(v, tu, _) | AnyValue::DatetimeOwned(v, tu, _),
529                DataType::Datetime(tu_r, tz_r),
530            ) => AnyValue::Datetime(
531                match (tu, tu_r) {
532                    (TimeUnit::Nanoseconds, TimeUnit::Microseconds) => *v / 1_000i64,
533                    (TimeUnit::Nanoseconds, TimeUnit::Milliseconds) => *v / 1_000_000i64,
534                    (TimeUnit::Microseconds, TimeUnit::Nanoseconds) => *v * 1_000i64,
535                    (TimeUnit::Microseconds, TimeUnit::Milliseconds) => *v / 1_000i64,
536                    (TimeUnit::Milliseconds, TimeUnit::Microseconds) => *v * 1_000i64,
537                    (TimeUnit::Milliseconds, TimeUnit::Nanoseconds) => *v * 1_000_000i64,
538                    _ => *v,
539                },
540                *tu_r,
541                tz_r.as_ref(),
542            ),
543
544            // to date
545            #[cfg(feature = "dtype-date")]
546            (av, DataType::Date) if av.is_primitive_numeric() => {
547                AnyValue::Date(av.extract::<i32>()?)
548            },
549            #[cfg(all(feature = "dtype-date", feature = "dtype-datetime"))]
550            (AnyValue::Datetime(v, tu, _) | AnyValue::DatetimeOwned(v, tu, _), DataType::Date) => {
551                AnyValue::Date(match tu {
552                    TimeUnit::Nanoseconds => *v / NS_IN_DAY,
553                    TimeUnit::Microseconds => *v / US_IN_DAY,
554                    TimeUnit::Milliseconds => *v / MS_IN_DAY,
555                } as i32)
556            },
557
558            // to time
559            #[cfg(feature = "dtype-time")]
560            (av, DataType::Time) if av.is_primitive_numeric() => {
561                AnyValue::Time(av.extract::<i64>()?)
562            },
563            #[cfg(all(feature = "dtype-time", feature = "dtype-datetime"))]
564            (AnyValue::Datetime(v, tu, _) | AnyValue::DatetimeOwned(v, tu, _), DataType::Time) => {
565                AnyValue::Time(match tu {
566                    TimeUnit::Nanoseconds => *v % NS_IN_DAY,
567                    TimeUnit::Microseconds => (*v % US_IN_DAY) * 1_000i64,
568                    TimeUnit::Milliseconds => (*v % MS_IN_DAY) * 1_000_000i64,
569                })
570            },
571
572            // to duration
573            #[cfg(feature = "dtype-duration")]
574            (av, DataType::Duration(tu)) if av.is_primitive_numeric() => {
575                AnyValue::Duration(av.extract::<i64>()?, *tu)
576            },
577            #[cfg(all(feature = "dtype-duration", feature = "dtype-time"))]
578            (AnyValue::Time(v), DataType::Duration(tu)) => AnyValue::Duration(
579                match *tu {
580                    TimeUnit::Nanoseconds => *v,
581                    TimeUnit::Microseconds => *v / 1_000i64,
582                    TimeUnit::Milliseconds => *v / 1_000_000i64,
583                },
584                *tu,
585            ),
586            #[cfg(feature = "dtype-duration")]
587            (AnyValue::Duration(v, tu), DataType::Duration(tu_r)) => AnyValue::Duration(
588                match (tu, tu_r) {
589                    (_, _) if tu == tu_r => *v,
590                    (TimeUnit::Nanoseconds, TimeUnit::Microseconds) => *v / 1_000i64,
591                    (TimeUnit::Nanoseconds, TimeUnit::Milliseconds) => *v / 1_000_000i64,
592                    (TimeUnit::Microseconds, TimeUnit::Nanoseconds) => *v * 1_000i64,
593                    (TimeUnit::Microseconds, TimeUnit::Milliseconds) => *v / 1_000i64,
594                    (TimeUnit::Milliseconds, TimeUnit::Microseconds) => *v * 1_000i64,
595                    (TimeUnit::Milliseconds, TimeUnit::Nanoseconds) => *v * 1_000_000i64,
596                    _ => *v,
597                },
598                *tu_r,
599            ),
600
601            #[cfg(feature = "dtype-decimal")]
602            (av, DataType::Decimal(p, s)) if av.is_integer() => {
603                let int = av.try_extract::<i128>().ok()?;
604                let dec = i128_to_dec128(int, *p, *s)?;
605                AnyValue::Decimal(dec, *p, *s)
606            },
607
608            #[cfg(feature = "dtype-decimal")]
609            (av, DataType::Decimal(p, s)) if av.is_float() => {
610                let f = av.try_extract::<f64>().unwrap();
611                let dec = f64_to_dec128(f, *p, *s)?;
612                AnyValue::Decimal(dec, *p, *s)
613            },
614
615            #[cfg(feature = "dtype-decimal")]
616            (AnyValue::Decimal(value, _old_p, old_s), DataType::Decimal(p, s)) => {
617                let converted = dec128_rescale(*value, *old_s, *p, *s)?;
618                AnyValue::Decimal(converted, *p, *s)
619            },
620
621            // to self
622            (av, dtype) if av.dtype() == *dtype => self.clone(),
623
624            _ => return None,
625        };
626        Some(new_av)
627    }

777    pub fn hash_impl<H: Hasher>(&self, state: &mut H, cheap: bool) {
778        use AnyValue::*;
779        std::mem::discriminant(self).hash(state);
780        match self {
781            Int8(v) => v.hash(state),
782            Int16(v) => v.hash(state),
783            Int32(v) => v.hash(state),
784            Int64(v) => v.hash(state),
785            Int128(v) => feature_gated!("dtype-i128", v.hash(state)),
786            UInt8(v) => v.hash(state),
787            UInt16(v) => v.hash(state),
788            UInt32(v) => v.hash(state),
789            UInt64(v) => v.hash(state),
790            UInt128(v) => feature_gated!("dtype-u128", v.hash(state)),
791            String(v) => v.hash(state),
792            StringOwned(v) => v.hash(state),
793            Float32(v) => v.to_ne_bytes().hash(state),
794            Float64(v) => v.to_ne_bytes().hash(state),
795            Binary(v) => v.hash(state),
796            BinaryOwned(v) => v.hash(state),
797            Boolean(v) => v.hash(state),
798            List(v) => {
799                if !cheap || v.len() < CHEAP_SERIES_HASH_LIMIT {
800                    Hash::hash(&Wrap(v.clone()), state)
801                }
802            },
803            #[cfg(feature = "dtype-array")]
804            Array(v, width) => {
805                if !cheap || v.len() < CHEAP_SERIES_HASH_LIMIT {
806                    Hash::hash(&Wrap(v.clone()), state)
807                }
808                width.hash(state)
809            },
810            #[cfg(feature = "dtype-date")]
811            Date(v) => v.hash(state),
812            #[cfg(feature = "dtype-datetime")]
813            Datetime(v, tu, tz) => {
814                v.hash(state);
815                tu.hash(state);
816                tz.hash(state);
817            },
818            #[cfg(feature = "dtype-datetime")]
819            DatetimeOwned(v, tu, tz) => {
820                v.hash(state);
821                tu.hash(state);
822                tz.hash(state);
823            },
824            #[cfg(feature = "dtype-duration")]
825            Duration(v, tz) => {
826                v.hash(state);
827                tz.hash(state);
828            },
829            #[cfg(feature = "dtype-time")]
830            Time(v) => v.hash(state),
831            #[cfg(feature = "dtype-categorical")]
832            Categorical(v, _) | CategoricalOwned(v, _) | Enum(v, _) | EnumOwned(v, _) => {
833                v.hash(state)
834            },
835            #[cfg(feature = "object")]
836            Object(_) => {},
837            #[cfg(feature = "object")]
838            ObjectOwned(_) => {},
839            #[cfg(feature = "dtype-struct")]
840            Struct(_, _, _) => {
841                if !cheap {
842                    let mut buf = vec![];
843                    self._materialize_struct_av(&mut buf);
844                    buf.hash(state)
845                }
846            },
847            #[cfg(feature = "dtype-struct")]
848            StructOwned(v) => v.0.hash(state),
849            #[cfg(feature = "dtype-decimal")]
850            Decimal(v, s, p) => {
851                v.hash(state);
852                s.hash(state);
853                p.hash(state);
854            },
855            Null => {},
856        }
857    }

1094    pub fn eq_missing(&self, other: &Self, null_equal: bool) -> bool {
1095        fn struct_owned_value_iter<'a>(
1096            v: &'a (Vec<AnyValue<'_>>, Vec<Field>),
1097        ) -> impl ExactSizeIterator<Item = AnyValue<'a>> {
1098            v.0.iter().map(|v| v.as_borrowed())
1099        }
1100        fn struct_value_iter(
1101            idx: usize,
1102            arr: &StructArray,
1103        ) -> impl ExactSizeIterator<Item = AnyValue<'_>> {
1104            assert!(idx < arr.len());
1105
1106            arr.values().iter().map(move |field_arr| unsafe {
1107                // SAFETY: We asserted before that idx is smaller than the array length. Since it
1108                // is an invariant of StructArray that all fields have the same length this is fine
1109                // to do.
1110                field_arr.get_unchecked(idx)
1111            })
1112        }
1113
1114        fn struct_eq_missing<'a>(
1115            l: impl ExactSizeIterator<Item = AnyValue<'a>>,
1116            r: impl ExactSizeIterator<Item = AnyValue<'a>>,
1117            null_equal: bool,
1118        ) -> bool {
1119            if l.len() != r.len() {
1120                return false;
1121            }
1122
1123            l.zip(r).all(|(lv, rv)| lv.eq_missing(&rv, null_equal))
1124        }
1125
1126        use AnyValue::*;
1127        match (self, other) {
1128            // Map to borrowed.
1129            (StringOwned(l), r) => AnyValue::String(l.as_str()) == *r,
1130            (BinaryOwned(l), r) => AnyValue::Binary(l.as_slice()) == *r,
1131            #[cfg(feature = "object")]

1133            (l, StringOwned(r)) => *l == AnyValue::String(r.as_str()),
1134            (l, BinaryOwned(r)) => *l == AnyValue::Binary(r.as_slice()),
1135            #[cfg(feature = "object")]
1136            (l, ObjectOwned(r)) => *l == AnyValue::Object(&*r.0),
1137            #[cfg(feature = "dtype-datetime")]
1138            (DatetimeOwned(lv, ltu, ltz), r) => {
1139                Datetime(*lv, *ltu, ltz.as_ref().map(|v| v.as_ref())) == *r
1140            },
1141            #[cfg(feature = "dtype-datetime")]
1142            (l, DatetimeOwned(rv, rtu, rtz)) => {
1143                *l == Datetime(*rv, *rtu, rtz.as_ref().map(|v| v.as_ref()))
1144            },
1145            #[cfg(feature = "dtype-categorical")]
1146            (CategoricalOwned(cat, map), r) => Categorical(*cat, map) == *r,
1147            #[cfg(feature = "dtype-categorical")]
1148            (l, CategoricalOwned(cat, map)) => *l == Categorical(*cat, map),
1149            #[cfg(feature = "dtype-categorical")]
1150            (EnumOwned(cat, map), r) => Enum(*cat, map) == *r,
1151            #[cfg(feature = "dtype-categorical")]
1152            (l, EnumOwned(cat, map)) => *l == Enum(*cat, map),
1153
1154            // Comparison with null.
1155            (Null, Null) => null_equal,
1156            (Null, _) => false,
1157            (_, Null) => false,
1158
1159            // Equality between equal types.
1160            (Boolean(l), Boolean(r)) => *l == *r,
1161            (UInt8(l), UInt8(r)) => *l == *r,
1162            (UInt16(l), UInt16(r)) => *l == *r,
1163            (UInt32(l), UInt32(r)) => *l == *r,
1164            (UInt64(l), UInt64(r)) => *l == *r,
1165            (UInt128(l), UInt128(r)) => *l == *r,
1166            (Int8(l), Int8(r)) => *l == *r,
1167            (Int16(l), Int16(r)) => *l == *r,
1168            (Int32(l), Int32(r)) => *l == *r,
1169            (Int64(l), Int64(r)) => *l == *r,
1170            (Int128(l), Int128(r)) => *l == *r,
1171            (Float32(l), Float32(r)) => l.to_total_ord() == r.to_total_ord(),
1172            (Float64(l), Float64(r)) => l.to_total_ord() == r.to_total_ord(),
1173            (String(l), String(r)) => l == r,
1174            (Binary(l), Binary(r)) => l == r,
1175            #[cfg(feature = "dtype-time")]
1176            (Time(l), Time(r)) => *l == *r,
1177            #[cfg(all(feature = "dtype-datetime", feature = "dtype-date"))]
1178            (Date(l), Date(r)) => *l == *r,
1179            #[cfg(all(feature = "dtype-datetime", feature = "dtype-date"))]
1180            (Datetime(l, tul, tzl), Datetime(r, tur, tzr)) => {
1181                *l == *r && *tul == *tur && tzl == tzr
1182            },
1183            (List(l), List(r)) => l == r,
1184            #[cfg(feature = "dtype-categorical")]
1185            (Categorical(cat_l, map_l), Categorical(cat_r, map_r)) => {
1186                if !Arc::ptr_eq(map_l, map_r) {
1187                    // We can't support this because our Hash impl directly hashes the index. If you
1188                    // add support for this we must change the Hash impl.
1189                    unimplemented!(
1190                        "comparing categoricals with different Categories is not supported through AnyValue"
1191                    );
1192                }
1193
1194                cat_l == cat_r
1195            },
1196            #[cfg(feature = "dtype-categorical")]
1197            (Enum(cat_l, map_l), Enum(cat_r, map_r)) => {
1198                if !Arc::ptr_eq(map_l, map_r) {
1199                    // We can't support this because our Hash impl directly hashes the index. If you
1200                    // add support for this we must change the Hash impl.
1201                    unimplemented!(
1202                        "comparing enums with different FrozenCategories is not supported through AnyValue"
1203                    );
1204                }
1205
1206                cat_l == cat_r
1207            },
1208            #[cfg(feature = "dtype-duration")]
1209            (Duration(l, tu_l), Duration(r, tu_r)) => l == r && tu_l == tu_r,
1210
1211            #[cfg(feature = "dtype-struct")]
1212            (StructOwned(l), StructOwned(r)) => struct_eq_missing(
1213                struct_owned_value_iter(l.as_ref()),
1214                struct_owned_value_iter(r.as_ref()),
1215                null_equal,
1216            ),
1217            #[cfg(feature = "dtype-struct")]
1218            (StructOwned(l), Struct(idx, arr, _)) => struct_eq_missing(
1219                struct_owned_value_iter(l.as_ref()),
1220                struct_value_iter(*idx, arr),
1221                null_equal,
1222            ),
1223            #[cfg(feature = "dtype-struct")]
1224            (Struct(idx, arr, _), StructOwned(r)) => struct_eq_missing(
1225                struct_value_iter(*idx, arr),
1226                struct_owned_value_iter(r.as_ref()),
1227                null_equal,
1228            ),
1229            #[cfg(feature = "dtype-struct")]
1230            (Struct(l_idx, l_arr, _), Struct(r_idx, r_arr, _)) => struct_eq_missing(
1231                struct_value_iter(*l_idx, l_arr),
1232                struct_value_iter(*r_idx, r_arr),
1233                null_equal,
1234            ),
1235            #[cfg(feature = "dtype-decimal")]
1236            (Decimal(lv, _lp, ls), Decimal(rv, _rp, rs)) => dec128_eq(*lv, *ls, *rv, *rs),
1237            #[cfg(feature = "object")]
1238            (Object(l), Object(r)) => l == r,
1239            #[cfg(feature = "dtype-array")]
1240            (Array(l_values, l_size), Array(r_values, r_size)) => {
1241                if l_size != r_size {
1242                    return false;
1243                }
1244
1245                debug_assert_eq!(l_values.len(), *l_size);
1246                debug_assert_eq!(r_values.len(), *r_size);
1247
1248                let mut is_equal = true;
1249                for i in 0..*l_size {
1250                    let l = unsafe { l_values.get_unchecked(i) };
1251                    let r = unsafe { r_values.get_unchecked(i) };
1252
1253                    is_equal &= l.eq_missing(&r, null_equal);
1254                }
1255                is_equal
1256            },
1257
1258            (l, r) if l.to_i128().is_some() && r.to_i128().is_some() => l.to_i128() == r.to_i128(),
1259            (l, r) if l.to_f64().is_some() && r.to_f64().is_some() => {
1260                l.to_f64().unwrap().to_total_ord() == r.to_f64().unwrap().to_total_ord()
1261            },
1262
1263            (_, _) => {
1264                unimplemented!(
1265                    "scalar eq_missing for mixed dtypes {self:?} and {other:?} is not supported"
1266                )
1267            },
1268        }
1269    }
1270}

1295                Datetime(*lv, *ltu, ltz.as_ref().map(|v| v.as_ref())).partial_cmp(r)

1297            #[cfg(feature = "dtype-datetime")]
1298            (l, DatetimeOwned(rv, rtu, rtz)) => {

1305            #[cfg(feature = "dtype-categorical")]
1306            (EnumOwned(cat, map), r) => Enum(*cat, map).partial_cmp(r),
1307            #[cfg(feature = "dtype-categorical")]
1308            (l, EnumOwned(cat, map)) => l.partial_cmp(&Enum(*cat, map)),
1309
1310            // Comparison with null.
1311            (Null, Null) => Some(Ordering::Equal),
1312            (Null, _) => Some(Ordering::Less),
1313            (_, Null) => Some(Ordering::Greater),
1314
1315            // Comparison between equal types.
1316            (Boolean(l), Boolean(r)) => l.partial_cmp(r),
1317            (UInt8(l), UInt8(r)) => l.partial_cmp(r),
1318            (UInt16(l), UInt16(r)) => l.partial_cmp(r),
1319            (UInt32(l), UInt32(r)) => l.partial_cmp(r),

1325            (Int64(l), Int64(r)) => l.partial_cmp(r),
1326            (Int128(l), Int128(r)) => l.partial_cmp(r),

1328            (Float64(l), Float64(r)) => Some(l.tot_cmp(r)),
1329            (String(l), String(r)) => l.partial_cmp(r),
1330            (Binary(l), Binary(r)) => l.partial_cmp(r),

1332            (Date(l), Date(r)) => l.partial_cmp(r),
1333            #[cfg(feature = "dtype-datetime")]
1334            (Datetime(lt, lu, lz), Datetime(rt, ru, rz)) => {

1336                    unimplemented!(
1337                        "comparing datetimes with different units or timezones is not supported"

1342            },
1343            #[cfg(feature = "dtype-duration")]

1353            #[cfg(feature = "dtype-categorical")]
1354            (Categorical(l_cat, l_map), Categorical(r_cat, r_map)) => unsafe {
1355                let l_str = l_map.cat_to_str_unchecked(*l_cat);

1359            #[cfg(feature = "dtype-categorical")]
1360            (Enum(l_cat, l_map), Enum(r_cat, r_map)) => {

1384            #[cfg(feature = "dtype-decimal")]

1386
1387            (_, _) => {
1388                unimplemented!(
1389                    "scalar ordering for mixed dtypes {self:?} and {other:?} is not supported"
1390                )
1391            },
1392        }
1393    }
