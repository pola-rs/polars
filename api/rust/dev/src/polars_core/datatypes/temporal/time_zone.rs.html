38    /// Converts timezones to canonical form.
39    ///
40    /// If the "timezones" feature is enabled, additionally performs validation and converts to
41    /// Etc/GMT form where applicable.
42    #[inline]
43    pub fn opt_try_new(zone_str: Option<impl Into<PlSmallStr>>) -> PolarsResult<Option<Self>> {
44        Self::new_impl(zone_str.map(|x| x.into()))
45    }
46
47    fn new_impl(zone_str: Option<PlSmallStr>) -> PolarsResult<Option<Self>> {
48        // Needed for selectors https://github.com/pola-rs/polars/pull/9641
49        if zone_str.as_deref() == Some("*") {
50            return Ok(Some(Self {
51                inner: PlSmallStr::from_static("*"),
52            }));
53        }
54
55        let mut canonical_tz = Self::_canonical_timezone_impl(zone_str);
56
57        #[cfg(feature = "timezones")]
58        if let Some(tz) = canonical_tz.as_mut() {
59            if Self::validate_time_zone(tz).is_err() {
60                match parse_fixed_offset(tz) {
61                    Ok(v) => *tz = v,
62                    Err(err) => {
63                        // This can be used if there are externally created arrow buffers / dtypes
64                        // with unknown timezones.
65                        if std::env::var("POLARS_IGNORE_TIMEZONE_PARSE_ERROR").as_deref() == Ok("1")
66                        {
67                            if config::verbose() {
68                                eprintln!("WARN: {err}")
69                            }
70                        } else {
71                            return Err(err.wrap_msg(|s| {
72                                format!(
73                                    "{s}. If you would like to forcibly disable \
74                                    timezone validation, set \
75                                    POLARS_IGNORE_TIMEZONE_PARSE_ERROR=1."
76                                )
77                            }));
78                        }
79                    },
80                }
81            }
82        }
83
84        Ok(canonical_tz.map(|inner| Self { inner }))
85    }
86
87    /// Equality where `None` is treated as UTC.
88    pub fn eq_none_as_utc(this: Option<&TimeZone>, other: Option<&TimeZone>) -> bool {
89        this.unwrap_or(&Self::UTC) == other.unwrap_or(&Self::UTC)
90    }

92    pub fn _canonical_timezone_impl(tz: Option<PlSmallStr>) -> Option<PlSmallStr> {
93        match tz.as_deref() {
94            Some("") | None => None,
95            #[cfg(feature = "timezones")]
96            Some("+00:00") | Some("00:00") | Some("utc") => Some(PlSmallStr::from_static("UTC")),
97            Some(_) => tz,
98        }
99    }

141#[cfg(feature = "timezones")]
142static FIXED_OFFSET_PATTERN: &str = r#"(?x)
143    ^
144    (?P<sign>[-+])?            # optional sign
145    (?P<hour>0[0-9]|1[0-4])    # hour (between 0 and 14)
146    :?                         # optional separator
147    00                         # minute
148    $
149    "#;
150
151#[cfg(feature = "timezones")]
152polars_utils::regex_cache::cached_regex! {
153    static FIXED_OFFSET_RE = FIXED_OFFSET_PATTERN;
154}

156/// Parse a time zone string to [`chrono_tz::Tz`]
157#[cfg(feature = "timezones")]
158pub fn parse_time_zone(tz: &str) -> PolarsResult<chrono_tz::Tz> {
159    match tz.parse::<chrono_tz::Tz>() {
160        Ok(tz) => Ok(tz),
161        Err(_) => unable_to_parse_err(tz),
162    }
163}
164
165/// Convert fixed offset to Etc/GMT one from time zone database
166///
167/// E.g. +01:00 -> Etc/GMT-1
168///
169/// Note: the sign appears reversed, but is correct, see <https://en.wikipedia.org/wiki/Tz_database#Area>:
170/// > In order to conform with the POSIX style, those zone names beginning with
171/// > "Etc/GMT" have their sign reversed from the standard ISO 8601 convention.
172/// > In the "Etc" area, zones west of GMT have a positive sign and those east
173/// > have a negative sign in their name (e.g "Etc/GMT-14" is 14 hours ahead of GMT).
174#[cfg(feature = "timezones")]
175pub fn parse_fixed_offset(tz: &str) -> PolarsResult<PlSmallStr> {
176    use polars_utils::format_pl_smallstr;
177
178    if let Some(caps) = FIXED_OFFSET_RE.captures(tz) {
179        let sign = match caps.name("sign").map(|s| s.as_str()) {
180            Some("-") => "+",
181            _ => "-",
182        };
183        let hour = caps.name("hour").unwrap().as_str().parse::<i32>().unwrap();
184        let etc_tz = format_pl_smallstr!("Etc/GMT{}{}", sign, hour);
185        if etc_tz.parse::<chrono_tz::Tz>().is_ok() {
186            return Ok(etc_tz);
187        }
188    }
189
190    unable_to_parse_err(tz)
191}

193#[cfg(feature = "timezones")]
194fn unable_to_parse_err<T>(tz: &str) -> PolarsResult<T> {
195    use polars_error::polars_bail;
196
197    polars_bail!(
198        ComputeError:
199        "unable to parse time zone: '{}'. Please check the \
200        Time Zone Database for a list of available time zones.",
201        tz
202    )
