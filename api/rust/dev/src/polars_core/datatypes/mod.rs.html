74/// # Safety
75///
76/// The StaticArray and dtype return must be correct.
77pub unsafe trait PolarsDataType: Send + Sync + Sized + 'static {
78    type Physical<'a>: std::fmt::Debug + Clone;
79    type OwnedPhysical: std::fmt::Debug + Send + Sync + Clone + PartialEq;
80    type ZeroablePhysical<'a>: Zeroable + From<Self::Physical<'a>>;
81    type Array: for<'a> StaticArray<
82            ValueT<'a> = Self::Physical<'a>,
83            ZeroableValueT<'a> = Self::ZeroablePhysical<'a>,
84        >;
85    type IsNested;
86    type HasViews;
87    type IsStruct;
88    type IsObject;
89
90    /// Returns the DataType variant associated with this PolarsDataType.
91    /// Not implemented for types whose DataTypes have parameters.
92    fn get_static_dtype() -> DataType
93    where
94        Self: Sized;
95}

97pub trait PolarsPhysicalType: PolarsDataType {
98    // A physical type is one backed by a ChunkedArray directly, as opposed to
99    // logical types which wrap physical data.
100    fn ca_into_series(ca: ChunkedArray<Self>) -> Series;
101}

103pub trait PolarsNumericType: PolarsPhysicalType + 'static
104where
105    Self: for<'a> PolarsDataType<
106            OwnedPhysical = Self::Native,
107            Physical<'a> = Self::Native,
108            ZeroablePhysical<'a> = Self::Native,
109            Array = PrimitiveArray<Self::Native>,
110            IsNested = FalseT,
111            HasViews = FalseT,
112            IsStruct = FalseT,
113            IsObject = FalseT,
114        >,
115{
116    type Native: NumericNative;
117}

119pub trait PolarsIntegerType: PolarsNumericType {}
120pub trait PolarsFloatType: PolarsNumericType {}

122/// # Safety
123/// The physical() return type must be correct for Native.
124#[cfg(feature = "dtype-categorical")]
125pub unsafe trait PolarsCategoricalType: PolarsDataType {
126    type Native: NumericNative + CatNative + DictionaryKey + PartialEq + Eq + Hash;
127    type PolarsPhysical: PolarsIntegerType<Native = Self::Native>;

132macro_rules! impl_polars_num_datatype {
133    ($trait: ident, $pdt:ident, $variant:ident, $physical:ty, $owned_phys:ty) => {
134        #[derive(Clone, Copy)]
135        pub struct $pdt {}
136
137        unsafe impl PolarsDataType for $pdt {
138            type Physical<'a> = $physical;
139            type OwnedPhysical = $owned_phys;
140            type ZeroablePhysical<'a> = $physical;
141            type Array = PrimitiveArray<$physical>;
142            type IsNested = FalseT;
143            type HasViews = FalseT;
144            type IsStruct = FalseT;
145            type IsObject = FalseT;
146
147            #[inline]
148            fn get_static_dtype() -> DataType {
149                DataType::$variant
150            }
151        }
152
153        impl PolarsNumericType for $pdt {
154            type Native = $physical;
155        }
156
157        impl $trait for $pdt {}
158    };
159}

184macro_rules! impl_polars_categorical_datatype {
185    ($pdt:ident, $phys:ty, $native:ty, $phys_variant:ident) => {
186        impl_polars_datatype!(
187            $pdt,
188            unimplemented!(),
189            PrimitiveArray<$native>,
190            'a,
191            $native,
192            $native,
193            $native,
194            FalseT
195        );
196
197        #[cfg(feature = "dtype-categorical")]
198        unsafe impl PolarsCategoricalType for $pdt {
199            type Native = $native;
200            type PolarsPhysical = $phys;
201
202            fn physical() -> CategoricalPhysical {
203                CategoricalPhysical::$phys_variant
204            }
205        }
206    }
207}

278    type IsNested = TrueT;
279    type HasViews = FalseT;

281    type IsObject = FalseT;
282
283    fn get_static_dtype() -> DataType
284    where
285        Self: Sized,
