19pub trait MetaDataExt: IntoMetadata {
20    fn pl_enum_metadata(&self) -> Option<&str> {
21        let md = self.into_metadata_ref();
22        let values = md
23            .get(DTYPE_ENUM_VALUES_NEW)
24            .or_else(|| md.get(DTYPE_ENUM_VALUES_LEGACY));
25        Some(values?.as_str())
26    }
27
28    fn pl_categorical_metadata(&self) -> Option<&str> {
29        // We ignore DTYPE_CATEGORICAL_LEGACY here, as we already map all
30        // string-typed arrow dictionaries to the global Categories, and the
31        // legacy metadata format only specifies the now-removed physical
32        // ordering parameter.
33        Some(
34            self.into_metadata_ref()
35                .get(DTYPE_CATEGORICAL_NEW)?
36                .as_str(),
37        )
38    }
39
40    fn maintain_type(&self) -> bool {
41        let metadata = self.into_metadata_ref();
42        metadata.get(PL_KEY).map(|s| s.as_str()) == Some(MAINTAIN_PL_TYPE)
43    }
44}

74impl UnknownKind {
75    pub fn materialize(&self) -> Option<DataType> {
76        let dtype = match self {
77            UnknownKind::Int(v) => materialize_dyn_int(*v).dtype(),
78            UnknownKind::Float => DataType::Float64,
79            UnknownKind::Str => DataType::String,
80            UnknownKind::Any => return None,
81        };
82        Some(dtype)
83    }

87pub enum DataType {
88    Boolean,
89    UInt8,
90    UInt16,
91    UInt32,
92    UInt64,
93    UInt128,
94    Int8,
95    Int16,
96    Int32,
97    Int64,
98    Int128,
99    Float32,
100    Float64,
101    /// Fixed point decimal type optional precision and non-negative scale.
102    /// This is backed by a signed 128-bit integer which allows for up to 38 significant digits.
103    /// Meaning max precision is 38.
104    #[cfg(feature = "dtype-decimal")]
105    Decimal(usize, usize), // (precision, scale), invariant: 1 <= precision <= 38.
106    /// String data
107    String,
108    Binary,
109    BinaryOffset,
110    /// A 32-bit date representing the elapsed time since UNIX epoch (1970-01-01)
111    /// in days (32 bits).
112    Date,
113    /// A 64-bit date representing the elapsed time since UNIX epoch (1970-01-01)
114    /// in the given timeunit (64 bits).
115    Datetime(TimeUnit, Option<TimeZone>),
116    /// 64-bit integer representing difference between times in milliseconds or nanoseconds
117    Duration(TimeUnit),
118    /// A 64-bit time representing the elapsed time since midnight in nanoseconds
119    Time,
120    /// A nested list with a fixed size in each row
121    #[cfg(feature = "dtype-array")]
122    Array(Box<DataType>, usize),
123    /// A nested list with a variable size in each row
124    List(Box<DataType>),
125    /// A generic type that can be used in a `Series`
126    /// &'static str can be used to determine/set inner type
127    #[cfg(feature = "object")]
128    Object(&'static str),
129    Null,
130    #[cfg(feature = "dtype-categorical")]
131    Categorical(Arc<Categories>, Arc<CategoricalMapping>),
132    // It is an Option, so that matching Enum/Categoricals can take the same guards.
133    #[cfg(feature = "dtype-categorical")]
134    Enum(Arc<FrozenCategories>, Arc<CategoricalMapping>),
135    #[cfg(feature = "dtype-struct")]
136    Struct(Vec<Field>),
137    // some logical types we cannot know statically, e.g. Datetime
138    Unknown(UnknownKind),
139}

157impl PartialEq for DataType {
158    fn eq(&self, other: &Self) -> bool {
159        use DataType::*;
160        {
161            match (self, other) {
162                #[cfg(feature = "dtype-categorical")]
163                (Categorical(cats_l, _), Categorical(cats_r, _)) => Arc::ptr_eq(cats_l, cats_r),
164                #[cfg(feature = "dtype-categorical")]
165                (Enum(fcats_l, _), Enum(fcats_r, _)) => Arc::ptr_eq(fcats_l, fcats_r),
166                (Datetime(tu_l, tz_l), Datetime(tu_r, tz_r)) => tu_l == tu_r && tz_l == tz_r,
167                (List(left_inner), List(right_inner)) => left_inner == right_inner,
168                #[cfg(feature = "dtype-duration")]
169                (Duration(tu_l), Duration(tu_r)) => tu_l == tu_r,
170                #[cfg(feature = "dtype-decimal")]
171                (Decimal(p1, s1), Decimal(p2, s2)) => (p1, s1) == (p2, s2),
172                #[cfg(feature = "object")]
173                (Object(lhs), Object(rhs)) => lhs == rhs,
174                #[cfg(feature = "dtype-struct")]
175                (Struct(lhs), Struct(rhs)) => {
176                    std::ptr::eq(Vec::as_ptr(lhs), Vec::as_ptr(rhs)) || lhs == rhs
177                },
178                #[cfg(feature = "dtype-array")]
179                (Array(left_inner, left_width), Array(right_inner, right_width)) => {
180                    left_width == right_width && left_inner == right_inner
181                },
182                (Unknown(l), Unknown(r)) => match (l, r) {
183                    (UnknownKind::Int(_), UnknownKind::Int(_)) => true,
184                    _ => l == r,
185                },
186                _ => std::mem::discriminant(self) == std::mem::discriminant(other),
187            }
188        }
189    }
190}

254    /// Materialize this datatype if it is unknown. All other datatypes
255    /// are left unchanged.
256    pub fn materialize_unknown(self, allow_unknown: bool) -> PolarsResult<DataType> {
257        match self {
258            DataType::Unknown(u) => match u.materialize() {
259                Some(known) => Ok(known),
260                None => {
261                    if allow_unknown {
262                        Ok(DataType::Unknown(u))
263                    } else {
264                        polars_bail!(SchemaMismatch: "failed to materialize unknown type")
265                    }
266                },
267            },
268            DataType::List(inner) => Ok(DataType::List(Box::new(
269                inner.materialize_unknown(allow_unknown)?,
270            ))),
271            #[cfg(feature = "dtype-array")]
272            DataType::Array(inner, size) => Ok(DataType::Array(
273                Box::new(inner.materialize_unknown(allow_unknown)?),
274                size,
275            )),
276            #[cfg(feature = "dtype-struct")]
277            DataType::Struct(fields) => Ok(DataType::Struct(
278                fields
279                    .into_iter()
280                    .map(|f| {
281                        PolarsResult::Ok(Field::new(
282                            f.name,
283                            f.dtype.materialize_unknown(allow_unknown)?,
284                        ))
285                    })
286                    .try_collect_vec()?,
287            )),
288            _ => Ok(self),
289        }
290    }

294    pub fn get_shape(&self) -> Option<Vec<usize>> {
295        fn get_shape_impl(dt: &DataType, shape: &mut Vec<usize>) {
296            if let DataType::Array(inner, size) = dt {
297                shape.push(*size);
298                get_shape_impl(inner, shape);
299            }

301
302        if let DataType::Array(inner, size) = self {
303            let mut shape = vec![*size];
304            get_shape_impl(inner, &mut shape);
305            Some(shape)
306        } else {
307            None

342    pub fn leaf_dtype(&self) -> &DataType {
343        let mut prev = self;
344        while let Some(dtype) = prev.inner_dtype() {
345            prev = dtype

376    /// Return whether the cast to `to` makes sense.
377    ///
378    /// If it `None`, we are not sure.
379    pub fn can_cast_to(&self, to: &DataType) -> Option<bool> {
380        if self == to {
381            return Some(true);
382        }
383        if self.is_primitive_numeric() && to.is_primitive_numeric() {
384            return Some(true);
385        }
386
387        if self.is_null() {
388            return Some(true);
389        }
390
391        use DataType as D;
392        Some(match (self, to) {
393            #[cfg(feature = "dtype-categorical")]
394            (D::Categorical(_, _) | D::Enum(_, _), D::Binary)
395            | (D::Binary, D::Categorical(_, _) | D::Enum(_, _)) => false, // TODO @ cat-rework: why can we not cast to Binary?
396
397            #[cfg(feature = "object")]
398            (D::Object(_), D::Object(_)) => true,
399            #[cfg(feature = "object")]
400            (D::Object(_), _) | (_, D::Object(_)) => false,
401
402            (D::Boolean, dt) | (dt, D::Boolean) => match dt {
403                dt if dt.is_primitive_numeric() => true,
404                #[cfg(feature = "dtype-decimal")]
405                D::Decimal(_, _) => true,
406                D::String | D::Binary => true,
407                _ => false,
408            },
409
410            (D::List(from), D::List(to)) => from.can_cast_to(to)?,
411            #[cfg(feature = "dtype-array")]
412            (D::Array(from, l_width), D::Array(to, r_width)) => {
413                l_width == r_width && from.can_cast_to(to)?
414            },
415            #[cfg(feature = "dtype-struct")]
416            (D::Struct(l_fields), D::Struct(r_fields)) => {
417                if l_fields.is_empty() {
418                    return Some(true);
419                }
420
421                if l_fields.len() != r_fields.len() {
422                    return Some(false);
423                }
424
425                for (l, r) in l_fields.iter().zip(r_fields) {
426                    if !l.dtype().can_cast_to(r.dtype())? {
427                        return Some(false);
428                    }
429                }
430
431                true
432            },
433
434            // @NOTE: we are being conversative
435            _ => return None,
436        })
437    }

443    /// Convert to the physical data type
444    #[must_use]
445    pub fn to_physical(&self) -> DataType {
446        use DataType::*;
447        match self {
448            Date => Int32,
449            Datetime(_, _) => Int64,
450            Duration(_) => Int64,
451            Time => Int64,
452            #[cfg(feature = "dtype-decimal")]
453            Decimal(_, _) => Int128,
454            #[cfg(feature = "dtype-categorical")]
455            Categorical(cats, _) => cats.physical().dtype(),
456            #[cfg(feature = "dtype-categorical")]
457            Enum(fcats, _) => fcats.physical().dtype(),
458            #[cfg(feature = "dtype-array")]
459            Array(dt, width) => Array(Box::new(dt.to_physical()), *width),
460            List(dt) => List(Box::new(dt.to_physical())),
461            #[cfg(feature = "dtype-struct")]
462            Struct(fields) => {
463                let new_fields = fields
464                    .iter()
465                    .map(|s| Field::new(s.name().clone(), s.dtype().to_physical()))
466                    .collect();
467                Struct(new_fields)
468            },
469            _ => self.clone(),
470        }
471    }

733    /// Convert to an Arrow Field.
734    pub fn to_arrow_field(&self, name: PlSmallStr, compat_level: CompatLevel) -> ArrowField {
735        let metadata = match self {
736            #[cfg(feature = "dtype-categorical")]
737            DataType::Enum(fcats, _map) => {
738                let cats = fcats.categories();
739                let strings_size: usize = cats
740                    .values_iter()
741                    .map(|s| (s.len() + 1).ilog10() as usize + 1 + s.len())
742                    .sum();
743                let mut encoded = String::with_capacity(strings_size);
744                for cat in cats.values_iter() {
745                    encoded.push_str(itoa::Buffer::new().format(cat.len()));
746                    encoded.push(';');
747                    encoded.push_str(cat);
748                }
749                Some(BTreeMap::from([(
750                    PlSmallStr::from_static(DTYPE_ENUM_VALUES_NEW),
751                    PlSmallStr::from_string(encoded),
752                )]))
753            },
754            #[cfg(feature = "dtype-categorical")]
755            DataType::Categorical(cats, _) => {
756                let mut encoded = String::new();
757                encoded.push_str(itoa::Buffer::new().format(cats.name().len()));
758                encoded.push(';');
759                encoded.push_str(cats.name());
760                encoded.push_str(itoa::Buffer::new().format(cats.namespace().len()));
761                encoded.push(';');
762                encoded.push_str(cats.namespace());
763                encoded.push_str(cats.physical().as_str());
764                encoded.push(';');
765
766                Some(BTreeMap::from([(
767                    PlSmallStr::from_static(DTYPE_CATEGORICAL_NEW),
768                    PlSmallStr::from_string(encoded),
769                )]))
770            },
771            DataType::BinaryOffset => Some(BTreeMap::from([(
772                PlSmallStr::from_static(PL_KEY),
773                PlSmallStr::from_static(MAINTAIN_PL_TYPE),
774            )])),
775            _ => None,
776        };
777
778        let field = ArrowField::new(name, self.to_arrow(compat_level), true);
779
780        if let Some(metadata) = metadata {
781            field.with_metadata(metadata)
782        } else {
783            field
784        }
785    }

839    #[inline]
840    pub fn try_to_arrow(&self, compat_level: CompatLevel) -> PolarsResult<ArrowDataType> {
841        use DataType::*;
842        match self {
843            Boolean => Ok(ArrowDataType::Boolean),
844            UInt8 => Ok(ArrowDataType::UInt8),
845            UInt16 => Ok(ArrowDataType::UInt16),
846            UInt32 => Ok(ArrowDataType::UInt32),
847            UInt64 => Ok(ArrowDataType::UInt64),
848            UInt128 => Ok(ArrowDataType::UInt128),
849            Int8 => Ok(ArrowDataType::Int8),
850            Int16 => Ok(ArrowDataType::Int16),
851            Int32 => Ok(ArrowDataType::Int32),
852            Int64 => Ok(ArrowDataType::Int64),
853            Int128 => Ok(ArrowDataType::Int128),
854            Float32 => Ok(ArrowDataType::Float32),
855            Float64 => Ok(ArrowDataType::Float64),
856            #[cfg(feature = "dtype-decimal")]
857            Decimal(precision, scale) => {
858                assert!(*precision >= 1 && *precision <= 38);
859                Ok(ArrowDataType::Decimal(*precision, *scale))
860            },
861            String => {
862                let dt = if compat_level.0 >= 1 {
863                    ArrowDataType::Utf8View
864                } else {
865                    ArrowDataType::LargeUtf8
866                };
867                Ok(dt)
868            },
869            Binary => {
870                let dt = if compat_level.0 >= 1 {
871                    ArrowDataType::BinaryView
872                } else {
873                    ArrowDataType::LargeBinary
874                };
875                Ok(dt)
876            },
877            Date => Ok(ArrowDataType::Date32),
878            Datetime(unit, tz) => Ok(ArrowDataType::Timestamp(
879                unit.to_arrow(),
880                tz.as_deref().cloned(),
881            )),
882            Duration(unit) => Ok(ArrowDataType::Duration(unit.to_arrow())),
883            Time => Ok(ArrowDataType::Time64(ArrowTimeUnit::Nanosecond)),
884            #[cfg(feature = "dtype-array")]
885            Array(dt, size) => Ok(dt
886                .try_to_arrow(compat_level)?
887                .to_fixed_size_list(*size, true)),
888            List(dt) => Ok(ArrowDataType::LargeList(Box::new(
889                dt.to_arrow_field(LIST_VALUES_NAME, compat_level),
890            ))),
891            Null => Ok(ArrowDataType::Null),
892            #[cfg(feature = "object")]
893            Object(_) => Ok(get_object_physical_type()),
894            #[cfg(feature = "dtype-categorical")]
895            Categorical(_, _) | Enum(_, _) => {
896                let arrow_phys = match self.cat_physical().unwrap() {
897                    CategoricalPhysical::U8 => IntegerType::UInt8,
898                    CategoricalPhysical::U16 => IntegerType::UInt16,
899                    CategoricalPhysical::U32 => IntegerType::UInt32,
900                };
901
902                let values = if compat_level.0 >= 1 {
903                    ArrowDataType::Utf8View
904                } else {
905                    ArrowDataType::LargeUtf8
906                };
907
908                Ok(ArrowDataType::Dictionary(
909                    arrow_phys,
910                    Box::new(values),
911                    false,
912                ))
913            },
914            #[cfg(feature = "dtype-struct")]
915            Struct(fields) => {
916                let fields = fields
917                    .iter()
918                    .map(|fld| fld.to_arrow(compat_level))
919                    .collect();
920                Ok(ArrowDataType::Struct(fields))
921            },
922            BinaryOffset => Ok(ArrowDataType::LargeBinary),
923            Unknown(kind) => {
924                let dt = match kind {
925                    UnknownKind::Any => ArrowDataType::Unknown,
926                    UnknownKind::Float => ArrowDataType::Float64,
927                    UnknownKind::Str => ArrowDataType::Utf8View,
928                    UnknownKind::Int(v) => {
929                        return materialize_dyn_int(*v).dtype().try_to_arrow(compat_level);
930                    },
931                };
932                Ok(dt)
933            },
934        }
935    }

950    /// Answers if this type matches the given type of a schema.
951    ///
952    /// Allows (nested) Null types in this type to match any type in the schema,
953    /// but not vice versa. In such a case Ok(true) is returned, because a cast
954    /// is necessary. If no cast is necessary Ok(false) is returned, and an
955    /// error is returned if the types are incompatible.
956    pub fn matches_schema_type(&self, schema_type: &DataType) -> PolarsResult<bool> {
957        match (self, schema_type) {
958            (DataType::List(l), DataType::List(r)) => l.matches_schema_type(r),
959            #[cfg(feature = "dtype-array")]
960            (DataType::Array(l, sl), DataType::Array(r, sr)) => {
961                Ok(l.matches_schema_type(r)? && sl == sr)
962            },
963            #[cfg(feature = "dtype-struct")]
964            (DataType::Struct(l), DataType::Struct(r)) => {
965                if l.len() != r.len() {
966                    polars_bail!(SchemaMismatch: "structs have different number of fields: {} vs {}", l.len(), r.len());
967                }
968                let mut must_cast = false;
969                for (l, r) in l.iter().zip(r.iter()) {
970                    must_cast |= l.dtype.matches_schema_type(&r.dtype)?;
971                }
972                Ok(must_cast)
973            },
974            (DataType::Null, DataType::Null) => Ok(false),
975            #[cfg(feature = "dtype-decimal")]
976            (DataType::Decimal(p1, s1), DataType::Decimal(p2, s2)) => Ok((p1, s1) != (p2, s2)),
977            // We don't allow the other way around, only if our current type is
978            // null and the schema isn't we allow it.
979            (DataType::Null, _) => Ok(true),
980            #[cfg(feature = "dtype-categorical")]
981            (DataType::Categorical(l, _), DataType::Categorical(r, _)) => {
982                ensure_same_categories(l, r)?;
983                Ok(false)
984            },
985            #[cfg(feature = "dtype-categorical")]
986            (DataType::Enum(l, _), DataType::Enum(r, _)) => {
987                ensure_same_frozen_categories(l, r)?;
988                Ok(false)
989            },
990
991            (l, r) if l == r => Ok(false),
992            (l, r) => {
993                polars_bail!(SchemaMismatch: "type {:?} is incompatible with expected type {:?}", l, r)
994            },
995        }
996    }

1004        let mut level = 0;

1008            slf = inner_dtype;

1187pub fn merge_dtypes(left: &DataType, right: &DataType) -> PolarsResult<DataType> {
1188    use DataType::*;
1189    Ok(match (left, right) {
1190        #[cfg(feature = "dtype-categorical")]
1191        (Categorical(cats_l, map), Categorical(cats_r, _)) => {
1192            ensure_same_categories(cats_l, cats_r)?;
1193            Categorical(cats_l.clone(), map.clone())
1194        },
1195        #[cfg(feature = "dtype-categorical")]
1196        (Enum(fcats_l, map), Enum(fcats_r, _)) => {
1197            ensure_same_frozen_categories(fcats_l, fcats_r)?;
1198            Enum(fcats_l.clone(), map.clone())
1199        },
1200        (List(inner_l), List(inner_r)) => {
1201            let merged = merge_dtypes(inner_l, inner_r)?;
1202            List(Box::new(merged))
1203        },
1204        #[cfg(feature = "dtype-struct")]
1205        (Struct(inner_l), Struct(inner_r)) => {
1206            polars_ensure!(inner_l.len() == inner_r.len(), ComputeError: "cannot combine structs with differing amounts of fields ({} != {})", inner_l.len(), inner_r.len());
1207            let fields = inner_l.iter().zip(inner_r.iter()).map(|(l, r)| {
1208                polars_ensure!(l.name() == r.name(), ComputeError: "cannot combine structs with different fields ({} != {})", l.name(), r.name());
1209                let merged = merge_dtypes(l.dtype(), r.dtype())?;
1210                Ok(Field::new(l.name().clone(), merged))
1211            }).collect::<PolarsResult<Vec<_>>>()?;
1212            Struct(fields)
1213        },
1214        #[cfg(feature = "dtype-array")]
1215        (Array(inner_l, width_l), Array(inner_r, width_r)) => {
1216            polars_ensure!(width_l == width_r, ComputeError: "widths of FixedSizeWidth Series are not equal");
1217            let merged = merge_dtypes(inner_l, inner_r)?;
1218            Array(Box::new(merged), *width_l)
1219        },
1220        (left, right) if left == right => left.clone(),
1221        _ => polars_bail!(ComputeError: "unable to merge datatypes"),
1222    })
1223}

1225fn collect_nested_types(
1226    dtype: &DataType,
1227    result: &mut PlHashSet<DataType>,
1228    include_compound_types: bool,
1229) {
1230    match dtype {
1231        DataType::List(inner) => {
1232            if include_compound_types {
1233                result.insert(dtype.clone());
1234            }
1235            collect_nested_types(inner, result, include_compound_types);
1236        },
1237        #[cfg(feature = "dtype-array")]
1238        DataType::Array(inner, _) => {
1239            if include_compound_types {
1240                result.insert(dtype.clone());
1241            }
1242            collect_nested_types(inner, result, include_compound_types);
1243        },
1244        #[cfg(feature = "dtype-struct")]
1245        DataType::Struct(fields) => {
1246            if include_compound_types {
1247                result.insert(dtype.clone());
1248            }
1249            for field in fields {
1250                collect_nested_types(field.dtype(), result, include_compound_types);
1251            }
1252        },
1253        _ => {
1254            result.insert(dtype.clone());
1255        },
1256    }
1257}
1258
1259pub fn unpack_dtypes(dtype: &DataType, include_compound_types: bool) -> PlHashSet<DataType> {
1260    let mut result = PlHashSet::new();
1261    collect_nested_types(dtype, &mut result, include_compound_types);
1262    result
1263}

1270impl CompatLevel {

1272        CompatLevel(1)
1273    }
1274
1275    pub const fn oldest() -> CompatLevel {
1276        CompatLevel(0)
1277    }
1278
1279    // The following methods are only used internally
1280
1281    #[doc(hidden)]
1282    pub fn with_level(level: u16) -> PolarsResult<CompatLevel> {
1283        if level > CompatLevel::newest().0 {
1284            polars_bail!(InvalidOperation: "invalid compat level");
1285        }
1286        Ok(CompatLevel(level))
1287    }
1288
1289    #[doc(hidden)]
1290    pub fn get_level(&self) -> u16 {
1291        self.0
1292    }
1293}
