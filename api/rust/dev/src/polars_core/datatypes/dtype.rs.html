21pub trait MetaDataExt: IntoMetadata {
22    fn pl_enum_metadata(&self) -> Option<&str> {
23        let md = self.into_metadata_ref();
24        let values = md
25            .get(DTYPE_ENUM_VALUES_NEW)
26            .or_else(|| md.get(DTYPE_ENUM_VALUES_LEGACY));
27        Some(values?.as_str())
28    }
29
30    fn pl_categorical_metadata(&self) -> Option<&str> {
31        // We ignore DTYPE_CATEGORICAL_LEGACY here, as we already map all
32        // string-typed arrow dictionaries to the global Categories, and the
33        // legacy metadata format only specifies the now-removed physical
34        // ordering parameter.
35        Some(
36            self.into_metadata_ref()
37                .get(DTYPE_CATEGORICAL_NEW)?
38                .as_str(),
39        )
40    }
41
42    fn maintain_type(&self) -> bool {
43        let metadata = self.into_metadata_ref();
44        metadata.get(PL_KEY).map(|s| s.as_str()) == Some(MAINTAIN_PL_TYPE)
45    }
46}

76impl UnknownKind {
77    pub fn materialize(&self) -> Option<DataType> {
78        let dtype = match self {
79            UnknownKind::Int(v) => materialize_dyn_int(*v).dtype(),
80            UnknownKind::Float => DataType::Float64,
81            UnknownKind::Str => DataType::String,
82            UnknownKind::Any => return None,
83        };
84        Some(dtype)
85    }

104    /// Fixed point decimal type optional precision and non-negative scale.
105    /// This is backed by a signed 128-bit integer which allows for up to 38 significant digits.
106    /// Meaning max precision is 38.
107    #[cfg(feature = "dtype-decimal")]
108    Decimal(usize, usize), // (precision, scale), invariant: 1 <= precision <= 38.
109    /// String data
110    String,
111    Binary,
112    BinaryOffset,
113    /// A 32-bit date representing the elapsed time since UNIX epoch (1970-01-01)
114    /// in days (32 bits).
115    Date,
116    /// A 64-bit date representing the elapsed time since UNIX epoch (1970-01-01)
117    /// in the given timeunit (64 bits).
118    Datetime(TimeUnit, Option<TimeZone>),
119    /// 64-bit integer representing difference between times in milliseconds or nanoseconds
120    Duration(TimeUnit),
121    /// A 64-bit time representing the elapsed time since midnight in nanoseconds
122    Time,
123    /// A nested list with a fixed size in each row
124    #[cfg(feature = "dtype-array")]
125    Array(Box<DataType>, usize),
126    /// A nested list with a variable size in each row
127    List(Box<DataType>),
128    /// A generic type that can be used in a `Series`
129    /// &'static str can be used to determine/set inner type
130    #[cfg(feature = "object")]
131    Object(&'static str),
132    Null,
133    #[cfg(feature = "dtype-categorical")]
134    Categorical(Arc<Categories>, Arc<CategoricalMapping>),
135    // It is an Option, so that matching Enum/Categoricals can take the same guards.
136    #[cfg(feature = "dtype-categorical")]
137    Enum(Arc<FrozenCategories>, Arc<CategoricalMapping>),
138    #[cfg(feature = "dtype-struct")]
139    Struct(Vec<Field>),
140    #[cfg(feature = "dtype-extension")]
141    Extension(ExtensionTypeInstance, Box<DataType>),

156impl PartialEq for DataType {
157    fn eq(&self, other: &Self) -> bool {
158        use DataType::*;
159        {
160            match (self, other) {
161                #[cfg(feature = "dtype-categorical")]
162                (Categorical(cats_l, _), Categorical(cats_r, _)) => Arc::ptr_eq(cats_l, cats_r),
163                #[cfg(feature = "dtype-categorical")]
164                (Enum(fcats_l, _), Enum(fcats_r, _)) => Arc::ptr_eq(fcats_l, fcats_r),
165                (Datetime(tu_l, tz_l), Datetime(tu_r, tz_r)) => tu_l == tu_r && tz_l == tz_r,
166                (List(left_inner), List(right_inner)) => left_inner == right_inner,
167                #[cfg(feature = "dtype-duration")]
168                (Duration(tu_l), Duration(tu_r)) => tu_l == tu_r,
169                #[cfg(feature = "dtype-decimal")]
170                (Decimal(p1, s1), Decimal(p2, s2)) => (p1, s1) == (p2, s2),
171                #[cfg(feature = "object")]
172                (Object(lhs), Object(rhs)) => lhs == rhs,
173                #[cfg(feature = "dtype-struct")]
174                (Struct(lhs), Struct(rhs)) => {
175                    std::ptr::eq(Vec::as_ptr(lhs), Vec::as_ptr(rhs)) || lhs == rhs
176                },
177                #[cfg(feature = "dtype-array")]
178                (Array(left_inner, left_width), Array(right_inner, right_width)) => {
179                    left_width == right_width && left_inner == right_inner
180                },
181                (Unknown(l), Unknown(r)) => match (l, r) {
182                    (UnknownKind::Int(_), UnknownKind::Int(_)) => true,
183                    _ => l == r,
184                },
185                _ => std::mem::discriminant(self) == std::mem::discriminant(other),
186            }
187        }
188    }
189}

205    pub fn pretty_format(&self) -> String {
206        match self {
207            #[cfg(feature = "dtype-struct")]
208            Self::Struct(fields) => {
209                let formatted_fields = fields
210                    .iter()
211                    .map(|field| format!("{}: {}", field.name, field.dtype.pretty_format()))
212                    .collect::<Vec<String>>()
213                    .join(", ");
214                format!("struct {{{}}}", formatted_fields)
215            },
216            Self::List(inner_dtype) => {
217                let formatted_dtype = inner_dtype.pretty_format();
218                format!("list[{}]", formatted_dtype)
219            },
220            #[cfg(feature = "dtype-array")]
221            Self::Array(inner_dtype, size) => {
222                let formatted_dtype = inner_dtype.pretty_format();
223                format!("array[{}, {}]", formatted_dtype, size)
224            },
225            _ => {
226                format!("{}", self)
227            },
228        }
229    }

279    /// Materialize this datatype if it is unknown. All other datatypes
280    /// are left unchanged.
281    pub fn materialize_unknown(self, allow_unknown: bool) -> PolarsResult<DataType> {
282        match self {
283            DataType::Unknown(u) => match u.materialize() {
284                Some(known) => Ok(known),
285                None => {
286                    if allow_unknown {
287                        Ok(DataType::Unknown(u))
288                    } else {
289                        polars_bail!(SchemaMismatch: "failed to materialize unknown type")
290                    }
291                },
292            },
293            DataType::List(inner) => Ok(DataType::List(Box::new(
294                inner.materialize_unknown(allow_unknown)?,
295            ))),
296            #[cfg(feature = "dtype-array")]
297            DataType::Array(inner, size) => Ok(DataType::Array(
298                Box::new(inner.materialize_unknown(allow_unknown)?),
299                size,
300            )),
301            #[cfg(feature = "dtype-struct")]
302            DataType::Struct(fields) => Ok(DataType::Struct(
303                fields
304                    .into_iter()
305                    .map(|f| {
306                        PolarsResult::Ok(Field::new(
307                            f.name,
308                            f.dtype.materialize_unknown(allow_unknown)?,
309                        ))
310                    })
311                    .try_collect_vec()?,
312            )),
313            _ => Ok(self),
314        }
315    }

318    /// Get the full shape of a multidimensional array.
319    pub fn get_shape(&self) -> Option<Vec<usize>> {
320        fn get_shape_impl(dt: &DataType, shape: &mut Vec<usize>) {
321            if let DataType::Array(inner, size) = dt {
322                shape.push(*size);
323                get_shape_impl(inner, shape);
324            }
325        }
326
327        if let DataType::Array(inner, size) = self {
328            let mut shape = vec![*size];
329            get_shape_impl(inner, &mut shape);
330            Some(shape)
331        } else {
332            None

367    pub fn leaf_dtype(&self) -> &DataType {
368        let mut prev = self;
369        while let Some(dtype) = prev.inner_dtype() {
370            prev = dtype

401    /// Map all leaf types of nested dtypes (list, array, struct) using the
402    /// supplied function.
403    pub fn map_leaves<F: FnMut(DataType) -> DataType>(self, f: &mut F) -> DataType {
404        use DataType::*;
405        match self {
406            List(inner) => List(Box::new(inner.map_leaves(f))),
407            #[cfg(feature = "dtype-array")]
408            Array(inner, size) => Array(Box::new(inner.map_leaves(f)), size),
409            #[cfg(feature = "dtype-struct")]
410            Struct(fields) => {
411                let new_fields = fields
412                    .into_iter()
413                    .map(|fld| Field::new(fld.name, fld.dtype.map_leaves(f)))
414                    .collect();
415                Struct(new_fields)
416            },
417            #[cfg(feature = "dtype-extension")]
418            Extension(ext, storage) => Extension(ext, Box::new(storage.map_leaves(f))),
419            _ => f(self),
420        }
421    }

423    /// Return whether the cast to `to` makes sense.
424    ///
425    /// If it `None`, we are not sure.
426    pub fn can_cast_to(&self, to: &DataType) -> Option<bool> {
427        if self == to {
428            return Some(true);
429        }
430        if self.is_primitive_numeric() && to.is_primitive_numeric() {
431            return Some(true);
432        }
433
434        if self.is_null() {
435            return Some(true);
436        }
437
438        use DataType as D;
439        Some(match (self, to) {
440            #[cfg(feature = "dtype-categorical")]
441            (D::Categorical(_, _) | D::Enum(_, _), D::Binary)
442            | (D::Binary, D::Categorical(_, _) | D::Enum(_, _)) => false, // TODO @ cat-rework: why can we not cast to Binary?
443
444            #[cfg(feature = "object")]
445            (D::Object(_), D::Object(_)) => true,
446            #[cfg(feature = "object")]
447            (D::Object(_), _) | (_, D::Object(_)) => false,
448
449            (D::Boolean, dt) | (dt, D::Boolean) => match dt {
450                dt if dt.is_primitive_numeric() => true,
451                #[cfg(feature = "dtype-decimal")]
452                D::Decimal(_, _) => true,
453                D::String | D::Binary => true,
454                _ => false,
455            },
456
457            (D::List(from), D::List(to)) => from.can_cast_to(to)?,
458            #[cfg(feature = "dtype-array")]
459            (D::Array(from, l_width), D::Array(to, r_width)) => {
460                l_width == r_width && from.can_cast_to(to)?
461            },
462            #[cfg(feature = "dtype-struct")]
463            (D::Struct(l_fields), D::Struct(r_fields)) => {
464                if l_fields.is_empty() {
465                    return Some(true);
466                }
467
468                if l_fields.len() != r_fields.len() {
469                    return Some(false);
470                }
471
472                for (l, r) in l_fields.iter().zip(r_fields) {
473                    if !l.dtype().can_cast_to(r.dtype())? {
474                        return Some(false);
475                    }
476                }
477
478                true
479            },
480
481            // @NOTE: we are being conversative
482            _ => return None,
483        })
484    }

490    /// Convert to the physical data type
491    #[must_use]
492    pub fn to_physical(&self) -> DataType {
493        use DataType::*;
494        match self {
495            Date => Int32,
496            Datetime(_, _) => Int64,
497            Duration(_) => Int64,
498            Time => Int64,
499            #[cfg(feature = "dtype-decimal")]
500            Decimal(_, _) => Int128,
501            #[cfg(feature = "dtype-categorical")]
502            Categorical(cats, _) => cats.physical().dtype(),
503            #[cfg(feature = "dtype-categorical")]
504            Enum(fcats, _) => fcats.physical().dtype(),
505            #[cfg(feature = "dtype-array")]
506            Array(dt, width) => Array(Box::new(dt.to_physical()), *width),
507            List(dt) => List(Box::new(dt.to_physical())),
508            #[cfg(feature = "dtype-struct")]
509            Struct(fields) => {
510                let new_fields = fields
511                    .iter()
512                    .map(|s| Field::new(s.name().clone(), s.dtype().to_physical()))
513                    .collect();
514                Struct(new_fields)
515            },
516            #[cfg(feature = "dtype-extension")]
517            Extension(_, storage) => storage.to_physical(),
518            _ => self.clone(),
519        }
520    }

829    /// Convert to an Arrow Field.
830    pub fn to_arrow_field(&self, name: PlSmallStr, compat_level: CompatLevel) -> ArrowField {
831        let metadata = match self {
832            #[cfg(feature = "dtype-categorical")]
833            DataType::Enum(fcats, _map) => {
834                let cats = fcats.categories();
835                let strings_size: usize = cats
836                    .values_iter()
837                    .map(|s| (s.len() + 1).ilog10() as usize + 1 + s.len())
838                    .sum();
839                let mut encoded = String::with_capacity(strings_size);
840                for cat in cats.values_iter() {
841                    encoded.push_str(itoa::Buffer::new().format(cat.len()));
842                    encoded.push(';');
843                    encoded.push_str(cat);
844                }
845                Some(BTreeMap::from([(
846                    PlSmallStr::from_static(DTYPE_ENUM_VALUES_NEW),
847                    PlSmallStr::from_string(encoded),
848                )]))
849            },
850            #[cfg(feature = "dtype-categorical")]
851            DataType::Categorical(cats, _) => {
852                let mut encoded = String::new();
853                encoded.push_str(itoa::Buffer::new().format(cats.name().len()));
854                encoded.push(';');
855                encoded.push_str(cats.name());
856                encoded.push_str(itoa::Buffer::new().format(cats.namespace().len()));
857                encoded.push(';');
858                encoded.push_str(cats.namespace());
859                encoded.push_str(cats.physical().as_str());
860                encoded.push(';');
861
862                Some(BTreeMap::from([(
863                    PlSmallStr::from_static(DTYPE_CATEGORICAL_NEW),
864                    PlSmallStr::from_string(encoded),
865                )]))
866            },
867            DataType::BinaryOffset => Some(BTreeMap::from([(
868                PlSmallStr::from_static(PL_KEY),
869                PlSmallStr::from_static(MAINTAIN_PL_TYPE),
870            )])),
871            _ => None,
872        };
873
874        let field = ArrowField::new(name, self.to_arrow(compat_level), true);
875
876        if let Some(metadata) = metadata {
877            field.with_metadata(metadata)
878        } else {
879            field
880        }
881    }

937    #[inline]
938    pub fn try_to_arrow(&self, compat_level: CompatLevel) -> PolarsResult<ArrowDataType> {
939        use DataType::*;
940        match self {
941            Boolean => Ok(ArrowDataType::Boolean),
942            UInt8 => Ok(ArrowDataType::UInt8),
943            UInt16 => Ok(ArrowDataType::UInt16),
944            UInt32 => Ok(ArrowDataType::UInt32),
945            UInt64 => Ok(ArrowDataType::UInt64),
946            UInt128 => Ok(ArrowDataType::UInt128),
947            Int8 => Ok(ArrowDataType::Int8),
948            Int16 => Ok(ArrowDataType::Int16),
949            Int32 => Ok(ArrowDataType::Int32),
950            Int64 => Ok(ArrowDataType::Int64),
951            Int128 => Ok(ArrowDataType::Int128),
952            Float16 => Ok(ArrowDataType::Float16),
953            Float32 => Ok(ArrowDataType::Float32),
954            Float64 => Ok(ArrowDataType::Float64),
955            #[cfg(feature = "dtype-decimal")]
956            Decimal(precision, scale) => {
957                assert!(*precision >= 1 && *precision <= 38);
958                Ok(ArrowDataType::Decimal(*precision, *scale))
959            },
960            String => {
961                let dt = if compat_level.0 >= 1 {
962                    ArrowDataType::Utf8View
963                } else {
964                    ArrowDataType::LargeUtf8
965                };
966                Ok(dt)
967            },
968            Binary => {
969                let dt = if compat_level.0 >= 1 {
970                    ArrowDataType::BinaryView
971                } else {
972                    ArrowDataType::LargeBinary
973                };
974                Ok(dt)
975            },
976            Date => Ok(ArrowDataType::Date32),
977            Datetime(unit, tz) => Ok(ArrowDataType::Timestamp(
978                unit.to_arrow(),
979                tz.as_deref().cloned(),
980            )),
981            Duration(unit) => Ok(ArrowDataType::Duration(unit.to_arrow())),
982            Time => Ok(ArrowDataType::Time64(ArrowTimeUnit::Nanosecond)),
983            #[cfg(feature = "dtype-array")]
984            Array(dt, width) => Ok(ArrowDataType::FixedSizeList(
985                Box::new(dt.to_arrow_field(LIST_VALUES_NAME, compat_level)),
986                *width,
987            )),
988            List(dt) => Ok(ArrowDataType::LargeList(Box::new(
989                dt.to_arrow_field(LIST_VALUES_NAME, compat_level),
990            ))),
991            Null => Ok(ArrowDataType::Null),
992            #[cfg(feature = "object")]
993            Object(_) => Ok(get_object_physical_type()),
994            #[cfg(feature = "dtype-categorical")]
995            Categorical(_, _) | Enum(_, _) => {
996                let arrow_phys = match self.cat_physical().unwrap() {
997                    CategoricalPhysical::U8 => IntegerType::UInt8,
998                    CategoricalPhysical::U16 => IntegerType::UInt16,
999                    CategoricalPhysical::U32 => IntegerType::UInt32,
1000                };
1001
1002                let values = if compat_level.0 >= 1 {
1003                    ArrowDataType::Utf8View
1004                } else {
1005                    ArrowDataType::LargeUtf8
1006                };
1007
1008                Ok(ArrowDataType::Dictionary(
1009                    arrow_phys,
1010                    Box::new(values),
1011                    false,
1012                ))
1013            },
1014            #[cfg(feature = "dtype-struct")]
1015            Struct(fields) => {
1016                let fields = fields
1017                    .iter()
1018                    .map(|fld| fld.to_arrow(compat_level))
1019                    .collect();
1020                Ok(ArrowDataType::Struct(fields))
1021            },
1022            BinaryOffset => Ok(ArrowDataType::LargeBinary),
1023            #[cfg(feature = "dtype-extension")]
1024            Extension(typ, inner) => Ok(ArrowDataType::Extension(Box::new(
1025                arrow::datatypes::ExtensionType {
1026                    name: typ.name().into(),
1027                    inner: inner.try_to_arrow(compat_level)?,
1028                    metadata: typ.serialize_metadata().map(|m| m.into()),
1029                },
1030            ))),
1031            Unknown(kind) => {
1032                let dt = match kind {
1033                    UnknownKind::Any => ArrowDataType::Unknown,
1034                    UnknownKind::Float => ArrowDataType::Float64,
1035                    UnknownKind::Str => ArrowDataType::Utf8View,
1036                    UnknownKind::Int(v) => {
1037                        return materialize_dyn_int(*v).dtype().try_to_arrow(compat_level);
1038                    },
1039                };
1040                Ok(dt)
1041            },
1042        }
1043    }

1058    /// Answers if this type matches the given type of a schema.
1059    ///
1060    /// Allows (nested) Null types in this type to match any type in the schema,
1061    /// but not vice versa. In such a case Ok(true) is returned, because a cast
1062    /// is necessary. If no cast is necessary Ok(false) is returned, and an
1063    /// error is returned if the types are incompatible.
1064    pub fn matches_schema_type(&self, schema_type: &DataType) -> PolarsResult<bool> {
1065        match (self, schema_type) {
1066            (DataType::List(l), DataType::List(r)) => l.matches_schema_type(r),
1067            #[cfg(feature = "dtype-array")]
1068            (DataType::Array(l, sl), DataType::Array(r, sr)) => {
1069                Ok(l.matches_schema_type(r)? && sl == sr)
1070            },
1071            #[cfg(feature = "dtype-struct")]
1072            (DataType::Struct(l), DataType::Struct(r)) => {
1073                if l.len() != r.len() {
1074                    polars_bail!(SchemaMismatch: "structs have different number of fields: {} vs {}", l.len(), r.len());
1075                }
1076                let mut must_cast = false;
1077                for (l, r) in l.iter().zip(r.iter()) {
1078                    must_cast |= l.dtype.matches_schema_type(&r.dtype)?;
1079                }
1080                Ok(must_cast)
1081            },
1082            (DataType::Null, DataType::Null) => Ok(false),
1083            #[cfg(feature = "dtype-decimal")]
1084            (DataType::Decimal(p1, s1), DataType::Decimal(p2, s2)) => Ok((p1, s1) != (p2, s2)),
1085            // We don't allow the other way around, only if our current type is
1086            // null and the schema isn't we allow it.
1087            (DataType::Null, _) => Ok(true),
1088            #[cfg(feature = "dtype-categorical")]
1089            (DataType::Categorical(l, _), DataType::Categorical(r, _)) => {
1090                ensure_same_categories(l, r)?;
1091                Ok(false)
1092            },
1093            #[cfg(feature = "dtype-categorical")]
1094            (DataType::Enum(l, _), DataType::Enum(r, _)) => {
1095                ensure_same_frozen_categories(l, r)?;
1096                Ok(false)
1097            },
1098
1099            (l, r) if l == r => Ok(false),
1100            (l, r) => {
1101                polars_bail!(SchemaMismatch: "type {:?} is incompatible with expected type {:?}", l, r)
1102            },
1103        }
1104    }

1273            Categorical(cats, _) => {
1274                if cats.is_global() {
1275                    write!(f, "Categorical")
1276                } else if cats.namespace().is_empty() && cats.physical() == CategoricalPhysical::U32
1277                {
1278                    write!(f, "Categorical('{}')", cats.name())
1279                } else {
1280                    write!(
1281                        f,
1282                        "Categorical('{}', '{}', {:?})",
1283                        cats.name(),
1284                        cats.namespace(),
1285                        cats.physical()
1286                    )
1287                }
1288            },
1289            #[cfg(feature = "dtype-categorical")]
1290            Enum(_, _) => write!(f, "Enum([...])"),
1291            #[cfg(feature = "object")]
1292            Object(_) => write!(f, "Object"),
1293            Null => write!(f, "Null"),
1294            #[cfg(feature = "dtype-extension")]
1295            Extension(typ, inner) => write!(f, "Extension({}, {inner:?})", typ.0.dyn_debug()),
1296            Unknown(kind) => write!(f, "Unknown({kind:?})"),
1297        }
1298    }
1299}

1301pub fn merge_dtypes(left: &DataType, right: &DataType) -> PolarsResult<DataType> {
1302    use DataType::*;
1303    Ok(match (left, right) {
1304        #[cfg(feature = "dtype-categorical")]
1305        (Categorical(cats_l, map), Categorical(cats_r, _)) => {
1306            ensure_same_categories(cats_l, cats_r)?;
1307            Categorical(cats_l.clone(), map.clone())
1308        },
1309        #[cfg(feature = "dtype-categorical")]
1310        (Enum(fcats_l, map), Enum(fcats_r, _)) => {
1311            ensure_same_frozen_categories(fcats_l, fcats_r)?;
1312            Enum(fcats_l.clone(), map.clone())
1313        },
1314        (List(inner_l), List(inner_r)) => {
1315            let merged = merge_dtypes(inner_l, inner_r)?;
1316            List(Box::new(merged))
1317        },
1318        #[cfg(feature = "dtype-struct")]
1319        (Struct(inner_l), Struct(inner_r)) => {
1320            polars_ensure!(inner_l.len() == inner_r.len(), ComputeError: "cannot combine structs with differing amounts of fields ({} != {})", inner_l.len(), inner_r.len());
1321            let fields = inner_l.iter().zip(inner_r.iter()).map(|(l, r)| {
1322                polars_ensure!(l.name() == r.name(), ComputeError: "cannot combine structs with different fields ({} != {})", l.name(), r.name());
1323                let merged = merge_dtypes(l.dtype(), r.dtype())?;
1324                Ok(Field::new(l.name().clone(), merged))
1325            }).collect::<PolarsResult<Vec<_>>>()?;
1326            Struct(fields)
1327        },
1328        #[cfg(feature = "dtype-array")]
1329        (Array(inner_l, width_l), Array(inner_r, width_r)) => {
1330            polars_ensure!(width_l == width_r, ComputeError: "widths of FixedSizeWidth Series are not equal");
1331            let merged = merge_dtypes(inner_l, inner_r)?;
1332            Array(Box::new(merged), *width_l)
1333        },
1334        (left, right) if left == right => left.clone(),
1335        _ => polars_bail!(ComputeError: "unable to merge datatypes"),
1336    })
1337}
1338
1339fn collect_nested_types(
1340    dtype: &DataType,
1341    result: &mut PlHashSet<DataType>,
1342    include_compound_types: bool,
1343) {
1344    match dtype {
1345        DataType::List(inner) => {
1346            if include_compound_types {
1347                result.insert(dtype.clone());
1348            }
1349            collect_nested_types(inner, result, include_compound_types);
1350        },
1351        #[cfg(feature = "dtype-array")]
1352        DataType::Array(inner, _) => {
1353            if include_compound_types {
1354                result.insert(dtype.clone());
1355            }
1356            collect_nested_types(inner, result, include_compound_types);
1357        },
1358        #[cfg(feature = "dtype-struct")]
1359        DataType::Struct(fields) => {
1360            if include_compound_types {
1361                result.insert(dtype.clone());
1362            }
1363            for field in fields {
1364                collect_nested_types(field.dtype(), result, include_compound_types);
1365            }
1366        },
1367        _ => {
1368            result.insert(dtype.clone());
1369        },
1370    }
1371}
1372
1373pub fn unpack_dtypes(dtype: &DataType, include_compound_types: bool) -> PlHashSet<DataType> {
1374    let mut result = PlHashSet::new();
1375    collect_nested_types(dtype, &mut result, include_compound_types);
1376    result
1377}
