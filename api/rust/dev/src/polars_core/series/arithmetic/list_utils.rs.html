14#[derive(Debug, Clone)]
15pub(super) enum NumericOp {

18    Mul,

20    Rem,

24impl NumericOp {
25    pub(super) fn name(&self) -> &'static str {
26        match self {
27            Self::Add => "add",
28            Self::Sub => "sub",
29            Self::Mul => "mul",
30            Self::Div => "div",
31            Self::Rem => "rem",
32            Self::FloorDiv => "floor_div",
33        }
34    }

36    pub(super) fn try_get_leaf_supertype(
37        &self,
38        prim_dtype_lhs: &DataType,
39        prim_dtype_rhs: &DataType,
40    ) -> PolarsResult<DataType> {
41        let dtype = try_get_supertype(prim_dtype_lhs, prim_dtype_rhs)?;
42
43        Ok(if matches!(self, Self::Div) {
44            if dtype.is_float() {
45                dtype
46            } else {
47                DataType::Float64
48            }
49        } else {
50            dtype
51        })
52    }

54    /// For operations that perform divisions on integers, sets the validity to NULL on rows where
55    /// the denominator is 0.
56    pub(super) fn prepare_numeric_op_side_validities<T: PolarsNumericType>(
57        &self,
58        lhs: &mut PrimitiveArray<T::Native>,
59        rhs: &mut PrimitiveArray<T::Native>,
60        swapped: bool,
61    ) where
62        PrimitiveArray<T::Native>: polars_compute::comparisons::TotalEqKernel<Scalar = T::Native>,
63        T::Native: Zero + IsFloat,
64    {
65        if !T::Native::is_float() {
66            match self {
67                Self::Div | Self::Rem | Self::FloorDiv => {
68                    let target = if swapped { lhs } else { rhs };
69                    let ne_0 = target.tot_ne_kernel_broadcast(&T::Native::zero());
70                    let validity = combine_validities_and(target.validity(), Some(&ne_0));
71                    target.set_validity(validity);
72                },
73                _ => {},
74            }
75        }
76    }

78    /// # Panics
79    /// Panics if:
80    /// * lhs.len() != rhs.len()
81    /// * dtype is not numeric.
82    pub(super) fn apply_series(&self, lhs: &Series, rhs: &Series) -> Box<dyn Array> {
83        assert_eq!(lhs.len(), rhs.len());
84        debug_assert_eq!(lhs.dtype(), rhs.dtype());
85
86        let lhs = lhs.rechunk();
87        let rhs = rhs.rechunk();
88
89        with_match_physical_numeric_polars_type!(lhs.dtype(), |$T| {
90            let lhs: &ChunkedArray<$T> = lhs.as_ref().as_ref().as_ref();
91            let rhs: &ChunkedArray<$T> = rhs.as_ref().as_ref().as_ref();
92
93            let lhs = lhs.downcast_get(0).unwrap();
94            let rhs = rhs.downcast_get(0).unwrap();
95
96            Box::new(self.apply_arithmetic_kernel::<$T>(lhs.clone(), rhs.clone()))
97        })
98    }
99
100    fn apply_arithmetic_kernel<T: PolarsNumericType>(
101        &self,
102        lhs: PrimitiveArray<T::Native>,
103        rhs: PrimitiveArray<T::Native>,
104    ) -> PrimitiveArray<T::Native> {
105        match self {
106            Self::Add => ArithmeticKernel::wrapping_add(lhs, rhs),
107            Self::Sub => ArithmeticKernel::wrapping_sub(lhs, rhs),
108            Self::Mul => ArithmeticKernel::wrapping_mul(lhs, rhs),
109            Self::Div => ArithmeticKernel::legacy_div(lhs, rhs),
110            Self::Rem => ArithmeticKernel::wrapping_mod(lhs, rhs),
111            Self::FloorDiv => ArithmeticKernel::wrapping_floor_div(lhs, rhs),
112        }
113    }
114
115    /// For list<->primitive where the primitive is broadcasted, we can dispatch to
116    /// `ArithmeticKernel`, which can have optimized codepaths for when one side is
117    /// a scalar.
118    pub(super) fn apply_array_to_scalar<T: PolarsNumericType>(
119        &self,
120        arr_lhs: PrimitiveArray<T::Native>,
121        r: T::Native,
122        swapped: bool,
123    ) -> PrimitiveArray<T::Native> {
124        match self {
125            Self::Add => ArithmeticKernel::wrapping_add_scalar(arr_lhs, r),
126            Self::Sub => {
127                if swapped {
128                    ArithmeticKernel::wrapping_sub_scalar_lhs(r, arr_lhs)
129                } else {
130                    ArithmeticKernel::wrapping_sub_scalar(arr_lhs, r)
131                }
132            },
133            Self::Mul => ArithmeticKernel::wrapping_mul_scalar(arr_lhs, r),
134            Self::Div => {
135                if swapped {
136                    ArithmeticKernel::legacy_div_scalar_lhs(r, arr_lhs)
137                } else {
138                    ArithmeticKernel::legacy_div_scalar(arr_lhs, r)
139                }
140            },
141            Self::Rem => {
142                if swapped {
143                    ArithmeticKernel::wrapping_mod_scalar_lhs(r, arr_lhs)
144                } else {
145                    ArithmeticKernel::wrapping_mod_scalar(arr_lhs, r)
146                }
147            },
148            Self::FloorDiv => {
149                if swapped {
150                    ArithmeticKernel::wrapping_floor_div_scalar_lhs(r, arr_lhs)
151                } else {
152                    ArithmeticKernel::wrapping_floor_div_scalar(arr_lhs, r)
153                }
154            },
155        }
156    }
157}

159macro_rules! with_match_pl_num_arith {
160    ($op:expr, $swapped:expr, | $_:tt $OP:tt | $($body:tt)* ) => ({
161        macro_rules! __with_func__ {( $_ $OP:tt ) => ( $($body)* )}
162
163        match $op {
164            NumericOp::Add => __with_func__! { (PlNumArithmetic::wrapping_add) },
165            NumericOp::Sub => {
166                if $swapped {
167                    __with_func__! { (|b, a| PlNumArithmetic::wrapping_sub(a, b)) }
168                } else {
169                    __with_func__! { (PlNumArithmetic::wrapping_sub) }
170                }
171            },
172            NumericOp::Mul => __with_func__! { (PlNumArithmetic::wrapping_mul) },
173            NumericOp::Div => {
174                if $swapped {
175                    __with_func__! { (|b, a| PlNumArithmetic::legacy_div(a, b)) }
176                } else {
177                    __with_func__! { (PlNumArithmetic::legacy_div) }
178                }
179            },
180            NumericOp::Rem => {
181                if $swapped {
182                    __with_func__! { (|b, a| PlNumArithmetic::wrapping_mod(a, b)) }
183                } else {
184                    __with_func__! { (PlNumArithmetic::wrapping_mod) }
185                }
186            },
187            NumericOp::FloorDiv => {
188                if $swapped {
189                    __with_func__! { (|b, a| PlNumArithmetic::wrapping_floor_div(a, b)) }
190                } else {
191                    __with_func__! { (PlNumArithmetic::wrapping_floor_div) }
192                }
193            },
194        }
195    })
