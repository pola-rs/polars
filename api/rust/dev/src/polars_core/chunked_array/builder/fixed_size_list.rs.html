6pub(crate) struct FixedSizeListNumericBuilder<T: NativeType> {
7    inner: Option<MutableFixedSizeListArray<MutablePrimitiveArray<T>>>,
8    width: usize,
9    name: PlSmallStr,
10    logical_dtype: DataType,
11}

13impl<T: NativeType> FixedSizeListNumericBuilder<T> {
14    /// # Safety
15    ///
16    /// The caller must ensure that the physical numerical type match logical type.
17    pub(crate) unsafe fn new(
18        name: PlSmallStr,
19        width: usize,
20        capacity: usize,
21        logical_dtype: DataType,
22    ) -> Self {
23        let mp = MutablePrimitiveArray::<T>::with_capacity(capacity * width);
24        let inner = Some(MutableFixedSizeListArray::new(mp, width));
25        Self {
26            inner,
27            width,
28            name,
29            logical_dtype,
30        }
31    }
32}

34pub(crate) trait FixedSizeListBuilder {
35    unsafe fn push_unchecked(&mut self, arr: &dyn Array, offset: usize);
36    unsafe fn push_null(&mut self);
37    fn finish(&mut self) -> ArrayChunked;
38}
39
40impl<T: NativeType> FixedSizeListBuilder for FixedSizeListNumericBuilder<T> {
41    #[inline]
42    unsafe fn push_unchecked(&mut self, arr: &dyn Array, offset: usize) {
43        let start = offset * self.width;
44        let end = start + self.width;
45        let arr = arr
46            .as_any()
47            .downcast_ref::<PrimitiveArray<T>>()
48            .unwrap_unchecked();
49        let inner = self.inner.as_mut().unwrap_unchecked();
50
51        let values = arr.values().as_slice();
52        let validity = arr.validity();
53        if let Some(validity) = validity {
54            let iter = (start..end).map(|i| {
55                if validity.get_bit_unchecked(i) {
56                    Some(*values.get_unchecked(i))
57                } else {
58                    None
59                }
60            });
61            inner.push_unchecked(Some(iter))
62        } else {
63            let iter = (start..end).map(|i| Some(*values.get_unchecked(i)));
64            inner.push_unchecked(Some(iter))
65        }
66    }
67
68    #[inline]
69    unsafe fn push_null(&mut self) {
70        let inner = self.inner.as_mut().unwrap_unchecked();
71        inner.push_null()
72    }
73
74    fn finish(&mut self) -> ArrayChunked {
75        let arr: FixedSizeListArray = self.inner.take().unwrap().into();
76        // SAFETY: physical type matches the logical
77        unsafe {
78            ChunkedArray::from_chunks_and_dtype(
79                self.name.clone(),
80                vec![Box::new(arr)],
81                DataType::Array(Box::new(self.logical_dtype.clone()), self.width),
82            )
83        }
84    }
85}

87pub(crate) struct AnonymousOwnedFixedSizeListBuilder {
88    inner: fixed_size_list::AnonymousBuilder,
89    name: PlSmallStr,

93impl AnonymousOwnedFixedSizeListBuilder {
94    pub(crate) fn new(
95        name: PlSmallStr,
96        width: usize,
97        capacity: usize,
98        inner_dtype: Option<DataType>,
99    ) -> Self {
100        let inner = fixed_size_list::AnonymousBuilder::new(capacity, width);
101        Self {
102            inner,
103            name,
104            inner_dtype,
105        }
106    }
107}

109impl FixedSizeListBuilder for AnonymousOwnedFixedSizeListBuilder {
110    #[inline]
111    unsafe fn push_unchecked(&mut self, arr: &dyn Array, offset: usize) {
112        let arr = arr.sliced_unchecked(offset * self.inner.width, self.inner.width);
113        self.inner.push(arr)
114    }
115
116    #[inline]
117    unsafe fn push_null(&mut self) {
118        self.inner.push_null()
119    }
120
121    fn finish(&mut self) -> ArrayChunked {
122        let arr = std::mem::take(&mut self.inner)
123            .finish(
124                self.inner_dtype
125                    .as_ref()
126                    .map(|dt| dt.to_arrow(CompatLevel::newest()))
127                    .as_ref(),
128            )
129            .unwrap();
130        ChunkedArray::with_chunk(self.name.clone(), arr)
131    }
132}

134pub(crate) fn get_fixed_size_list_builder(
135    inner_type_logical: &DataType,
136    capacity: usize,
137    width: usize,
138    name: PlSmallStr,
139) -> PolarsResult<Box<dyn FixedSizeListBuilder>> {
140    let phys_dtype = inner_type_logical.to_physical();
141
142    let builder = if phys_dtype.is_primitive_numeric() {
143        with_match_physical_numeric_type!(phys_dtype, |$T| {
144        // SAFETY: physical type match logical type
145        unsafe {
146            Box::new(FixedSizeListNumericBuilder::<$T>::new(name, width, capacity,inner_type_logical.clone())) as Box<dyn FixedSizeListBuilder>
147        }
148        })
149    } else {
150        Box::new(AnonymousOwnedFixedSizeListBuilder::new(
151            name,
152            width,
153            capacity,
154            Some(inner_type_logical.clone()),
155        ))
156    };
157    Ok(builder)
