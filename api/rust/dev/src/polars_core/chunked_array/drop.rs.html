8    // guarded by the type system
9    // the transmute only convinces the type system that we are a list
10    #[allow(clippy::transmute_undefined_repr)]

12        drop_list(std::mem::transmute::<&mut ChunkedArray<T>, &ListChunked>(
13            ca,
14        ))
15    }
16}
