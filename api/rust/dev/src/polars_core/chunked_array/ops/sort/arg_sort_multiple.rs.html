7pub(crate) fn args_validate<T: PolarsDataType>(
8    ca: &ChunkedArray<T>,
9    other: &[Column],
10    param_value: &[bool],
11    param_name: &str,
12) -> PolarsResult<()> {
13    for s in other {
14        assert_eq!(ca.len(), s.len());
15    }
16    polars_ensure!(other.len() == (param_value.len() - 1),
17        ComputeError:
18        "the length of `{}` ({}) does not match the number of series ({})",
19        param_name, param_value.len(), other.len() + 1,
20    );
21    Ok(())
22}

24pub(crate) fn arg_sort_multiple_impl<T: NullOrderCmp + Send + Copy>(
25    mut vals: Vec<(IdxSize, T)>,
26    by: &[Column],
27    options: &SortMultipleOptions,
28) -> PolarsResult<IdxCa> {
29    let nulls_last = &options.nulls_last;
30    let descending = &options.descending;
31
32    debug_assert_eq!(descending.len() - 1, by.len());
33    debug_assert_eq!(nulls_last.len() - 1, by.len());
34
35    let compare_inner: Vec<_> = by
36        .iter()
37        .map(|c| c.into_total_ord_inner())
38        .collect_trusted();
39
40    let first_descending = descending[0];
41    let first_nulls_last = nulls_last[0];
42
43    let compare = |tpl_a: &(_, T), tpl_b: &(_, T)| -> Ordering {
44        match (
45            first_descending,
46            tpl_a
47                .1
48                .null_order_cmp(&tpl_b.1, first_nulls_last ^ first_descending),
49        ) {
50            // if ordering is equal, we check the other arrays until we find a non-equal ordering
51            // if we have exhausted all arrays, we keep the equal ordering.
52            (_, Ordering::Equal) => {
53                let idx_a = tpl_a.0 as usize;
54                let idx_b = tpl_b.0 as usize;
55                unsafe {
56                    ordering_other_columns(
57                        &compare_inner,
58                        descending.get_unchecked(1..),
59                        nulls_last.get_unchecked(1..),
60                        idx_a,
61                        idx_b,
62                    )
63                }
64            },
65            (true, Ordering::Less) => Ordering::Greater,
66            (true, Ordering::Greater) => Ordering::Less,
67            (_, ord) => ord,
68        }
69    };
70
71    match (options.multithreaded, options.maintain_order) {
72        (true, true) => POOL.install(|| {
73            vals.par_sort_by(compare);
74        }),
75        (true, false) => POOL.install(|| {
76            vals.par_sort_unstable_by(compare);
77        }),
78        (false, true) => vals.sort_by(compare),
79        (false, false) => vals.sort_unstable_by(compare),
80    }
81
82    let ca: NoNull<IdxCa> = vals.into_iter().map(|(idx, _v)| idx).collect_trusted();
83    // Don't set to sorted. Argsort indices are not sorted.
84    Ok(ca.into_inner())
85}

87pub(crate) fn argsort_multiple_row_fmt(
88    by: &[Column],
89    mut descending: Vec<bool>,
90    mut nulls_last: Vec<bool>,
91    parallel: bool,
92) -> PolarsResult<IdxCa> {
93    _broadcast_bools(by.len(), &mut descending);
94    _broadcast_bools(by.len(), &mut nulls_last);
95
96    let rows_encoded = _get_rows_encoded(by, &descending, &nulls_last)?;
97    let mut items: Vec<_> = rows_encoded.iter().enumerate_idx().collect();
98
99    if parallel {
100        POOL.install(|| items.par_sort_by_key(|i| i.1));
101    } else {
102        items.sort_by_key(|i| i.1);
103    }
104
105    let ca: NoNull<IdxCa> = items.into_iter().map(|tpl| tpl.0).collect();
106    Ok(ca.into_inner())
107}
