71#[inline]
72pub fn unary_elementwise<'a, T, V, F>(ca: &'a ChunkedArray<T>, mut op: F) -> ChunkedArray<V>
73where
74    T: PolarsDataType,
75    V: PolarsDataType,
76    F: UnaryFnMut<Option<T::Physical<'a>>>,
77    V::Array: ArrayFromIter<<F as UnaryFnMut<Option<T::Physical<'a>>>>::Ret>,
78{
79    if ca.has_nulls() {
80        let iter = ca
81            .downcast_iter()
82            .map(|arr| arr.iter().map(&mut op).collect_arr());
83        ChunkedArray::from_chunk_iter(ca.name().clone(), iter)
84    } else {
85        let iter = ca
86            .downcast_iter()
87            .map(|arr| arr.values_iter().map(|x| op(Some(x))).collect_arr());
88        ChunkedArray::from_chunk_iter(ca.name().clone(), iter)
89    }
90}

205#[inline]
206pub fn binary_elementwise<T, U, V, F>(
207    lhs: &ChunkedArray<T>,
208    rhs: &ChunkedArray<U>,
209    mut op: F,
210) -> ChunkedArray<V>
211where
212    T: PolarsDataType,
213    U: PolarsDataType,
214    V: PolarsDataType,
215    F: for<'a> BinaryFnMut<Option<T::Physical<'a>>, Option<U::Physical<'a>>>,
216    V::Array: for<'a> ArrayFromIter<
217        <F as BinaryFnMut<Option<T::Physical<'a>>, Option<U::Physical<'a>>>>::Ret,
218    >,
219{
220    let (lhs, rhs) = align_chunks_binary(lhs, rhs);
221    let iter = lhs
222        .downcast_iter()
223        .zip(rhs.downcast_iter())
224        .map(|(lhs_arr, rhs_arr)| {
225            let element_iter = lhs_arr
226                .iter()
227                .zip(rhs_arr.iter())
228                .map(|(lhs_opt_val, rhs_opt_val)| op(lhs_opt_val, rhs_opt_val));
229            element_iter.collect_arr()
230        });
231    ChunkedArray::from_chunk_iter(lhs.name().clone(), iter)
232}

234#[inline]
235pub fn binary_elementwise_for_each<'a, 'b, T, U, F>(
236    lhs: &'a ChunkedArray<T>,
237    rhs: &'b ChunkedArray<U>,
238    mut op: F,
239) where
240    T: PolarsDataType,
241    U: PolarsDataType,
242    F: FnMut(Option<T::Physical<'a>>, Option<U::Physical<'b>>),
243{
244    let mut lhs_arr_iter = lhs.downcast_iter();
245    let mut rhs_arr_iter = rhs.downcast_iter();
246
247    let lhs_arr = lhs_arr_iter.next().unwrap();
248    let rhs_arr = rhs_arr_iter.next().unwrap();
249
250    let mut lhs_remaining = lhs_arr.len();
251    let mut rhs_remaining = rhs_arr.len();
252    let mut lhs_iter = lhs_arr.iter();
253    let mut rhs_iter = rhs_arr.iter();
254
255    loop {
256        let range = std::cmp::min(lhs_remaining, rhs_remaining);
257
258        for _ in 0..range {
259            // SAFETY: we loop until the smaller iter is exhausted.
260            let lhs_opt_val = unsafe { lhs_iter.next().unwrap_unchecked() };
261            let rhs_opt_val = unsafe { rhs_iter.next().unwrap_unchecked() };
262            op(lhs_opt_val, rhs_opt_val)
263        }
264        lhs_remaining -= range;
265        rhs_remaining -= range;
266
267        if lhs_remaining == 0 {
268            let Some(new_arr) = lhs_arr_iter.next() else {
269                return;
270            };
271            lhs_remaining = new_arr.len();
272            lhs_iter = new_arr.iter();
273        }
274        if rhs_remaining == 0 {
275            let Some(new_arr) = rhs_arr_iter.next() else {
276                return;
277            };
278            rhs_remaining = new_arr.len();
279            rhs_iter = new_arr.iter();
280        }
281    }
282}

311#[inline]
312pub fn binary_elementwise_values<T, U, V, F, K>(
313    lhs: &ChunkedArray<T>,
314    rhs: &ChunkedArray<U>,
315    mut op: F,
316) -> ChunkedArray<V>
317where
318    T: PolarsDataType,
319    U: PolarsDataType,
320    V: PolarsDataType,
321    F: for<'a> FnMut(T::Physical<'a>, U::Physical<'a>) -> K,
322    V::Array: ArrayFromIter<K>,
323{
324    if lhs.null_count() == lhs.len() || rhs.null_count() == rhs.len() {
325        let len = lhs.len().min(rhs.len());
326        let arr = V::Array::full_null(len, V::get_static_dtype().to_arrow(CompatLevel::newest()));
327
328        return ChunkedArray::with_chunk(lhs.name().clone(), arr);
329    }
330
331    let (lhs, rhs) = align_chunks_binary(lhs, rhs);
332
333    let iter = lhs
334        .downcast_iter()
335        .zip(rhs.downcast_iter())
336        .map(|(lhs_arr, rhs_arr)| {
337            let validity = combine_validities_and(lhs_arr.validity(), rhs_arr.validity());
338
339            let element_iter = lhs_arr
340                .values_iter()
341                .zip(rhs_arr.values_iter())
342                .map(|(lhs_val, rhs_val)| op(lhs_val, rhs_val));
343
344            let array: V::Array = element_iter.collect_arr();
345            array.with_validity_typed(validity)
346        });
347    ChunkedArray::from_chunk_iter(lhs.name().clone(), iter)
348}

350/// Apply elementwise binary function which produces string, amortising allocations.
351///
352/// Currently unused within Polars itself, but it's a useful utility for plugin authors.
353#[inline]
354pub fn binary_elementwise_into_string_amortized<T, U, F>(
355    lhs: &ChunkedArray<T>,
356    rhs: &ChunkedArray<U>,
357    mut op: F,
358) -> StringChunked
359where
360    T: PolarsDataType,
361    U: PolarsDataType,
362    F: for<'a> FnMut(T::Physical<'a>, U::Physical<'a>, &mut String),
363{
364    let (lhs, rhs) = align_chunks_binary(lhs, rhs);
365    let mut buf = String::new();
366    let iter = lhs
367        .downcast_iter()
368        .zip(rhs.downcast_iter())
369        .map(|(lhs_arr, rhs_arr)| {
370            let mut mutarr = MutablePlString::with_capacity(lhs_arr.len());
371            lhs_arr
372                .iter()
373                .zip(rhs_arr.iter())
374                .for_each(|(lhs_opt, rhs_opt)| match (lhs_opt, rhs_opt) {
375                    (None, _) | (_, None) => mutarr.push_null(),
376                    (Some(lhs_val), Some(rhs_val)) => {
377                        buf.clear();
378                        op(lhs_val, rhs_val, &mut buf);
379                        mutarr.push_value(&buf)
380                    },
381                });
382            mutarr.freeze()
383        });
384    ChunkedArray::from_chunk_iter(lhs.name().clone(), iter)
385}

387/// Applies a kernel that produces `Array` types.
388///
389/// Intended for kernels that apply on values, this function will filter out any
390/// results which do not have two non-null inputs.
391#[inline]
392pub fn binary_mut_values<T, U, V, F, Arr>(
393    lhs: &ChunkedArray<T>,
394    rhs: &ChunkedArray<U>,
395    mut op: F,
396    name: PlSmallStr,
397) -> ChunkedArray<V>
398where
399    T: PolarsDataType,
400    U: PolarsDataType,
401    V: PolarsDataType<Array = Arr>,
402    Arr: Array + StaticArray,
403    F: FnMut(&T::Array, &U::Array) -> Arr,
404{
405    let (lhs, rhs) = align_chunks_binary(lhs, rhs);
406    let iter = lhs
407        .downcast_iter()
408        .zip(rhs.downcast_iter())
409        .map(|(lhs_arr, rhs_arr)| {
410            let ret = op(lhs_arr, rhs_arr);
411            let inp_val = combine_validities_and(lhs_arr.validity(), rhs_arr.validity());

413            ret.with_validity_typed(val)
414        });
415    ChunkedArray::from_chunk_iter(name, iter)
416}

524/// Applies a kernel that produces `ArrayRef` of the same type.
525///
526/// # Safety
527/// Caller must ensure that the returned `ArrayRef` belongs to `T: PolarsDataType`.
528#[inline]
529pub unsafe fn binary_unchecked_same_type<T, U, F>(
530    lhs: &ChunkedArray<T>,
531    rhs: &ChunkedArray<U>,
532    mut op: F,
533    keep_sorted: bool,
534    keep_fast_explode: bool,
535) -> ChunkedArray<T>
536where
537    T: PolarsDataType,
538    U: PolarsDataType,
539    F: FnMut(&T::Array, &U::Array) -> Box<dyn Array>,
540{
541    let (lhs, rhs) = align_chunks_binary(lhs, rhs);
542    let chunks = lhs
543        .downcast_iter()
544        .zip(rhs.downcast_iter())
545        .map(|(lhs_arr, rhs_arr)| op(lhs_arr, rhs_arr))
546        .collect();
547
548    let mut ca = lhs.copy_with_chunks(chunks);
549
550    let mut retain_flags = StatisticsFlags::empty();
551    use StatisticsFlags as F;
552    retain_flags.set(F::IS_SORTED_ANY, keep_sorted);
553    retain_flags.set(F::CAN_FAST_EXPLODE_LIST, keep_fast_explode);
554    ca.retain_flags_from(lhs.as_ref(), retain_flags);
555
556    ca
557}

611/// # Safety
612/// Caller must ensure that the returned `ArrayRef` belongs to `T: PolarsDataType`.
613#[inline]
614pub unsafe fn try_binary_unchecked_same_type<T, U, F, E>(
615    lhs: &ChunkedArray<T>,
616    rhs: &ChunkedArray<U>,
617    mut op: F,
618    keep_sorted: bool,
619    keep_fast_explode: bool,
620) -> Result<ChunkedArray<T>, E>
621where
622    T: PolarsDataType,
623    U: PolarsDataType,
624    F: FnMut(&T::Array, &U::Array) -> Result<Box<dyn Array>, E>,
625    E: Error,
626{
627    let (lhs, rhs) = align_chunks_binary(lhs, rhs);
628    let chunks = lhs
629        .downcast_iter()
630        .zip(rhs.downcast_iter())
631        .map(|(lhs_arr, rhs_arr)| op(lhs_arr, rhs_arr))
632        .collect::<Result<Vec<_>, E>>()?;
633    let mut ca = lhs.copy_with_chunks(chunks);
634
635    let mut retain_flags = StatisticsFlags::empty();
636    use StatisticsFlags as F;
637    retain_flags.set(F::IS_SORTED_ANY, keep_sorted);
638    retain_flags.set(F::CAN_FAST_EXPLODE_LIST, keep_fast_explode);
639    ca.retain_flags_from(lhs.as_ref(), retain_flags);
640
641    Ok(ca)
642}

644#[inline]
645pub fn try_ternary_elementwise<T, U, V, G, F, K, E>(
646    ca1: &ChunkedArray<T>,
647    ca2: &ChunkedArray<U>,
648    ca3: &ChunkedArray<G>,
649    mut op: F,
650) -> Result<ChunkedArray<V>, E>
651where
652    T: PolarsDataType,
653    U: PolarsDataType,
654    V: PolarsDataType,
655    G: PolarsDataType,
656    F: for<'a> FnMut(
657        Option<T::Physical<'a>>,
658        Option<U::Physical<'a>>,
659        Option<G::Physical<'a>>,
660    ) -> Result<Option<K>, E>,
661    V::Array: ArrayFromIter<Option<K>>,
662{
663    let (ca1, ca2, ca3) = align_chunks_ternary(ca1, ca2, ca3);
664    let iter = ca1
665        .downcast_iter()
666        .zip(ca2.downcast_iter())
667        .zip(ca3.downcast_iter())
668        .map(|((ca1_arr, ca2_arr), ca3_arr)| {
669            let element_iter = ca1_arr.iter().zip(ca2_arr.iter()).zip(ca3_arr.iter()).map(
670                |((ca1_opt_val, ca2_opt_val), ca3_opt_val)| {
671                    op(ca1_opt_val, ca2_opt_val, ca3_opt_val)
672                },
673            );
674            element_iter.try_collect_arr()
675        });
676    ChunkedArray::try_from_chunk_iter(ca1.name().clone(), iter)
677}

720pub fn broadcast_binary_elementwise<T, U, V, F>(
721    lhs: &ChunkedArray<T>,
722    rhs: &ChunkedArray<U>,
723    mut op: F,
724) -> ChunkedArray<V>
725where
726    T: PolarsDataType,
727    U: PolarsDataType,
728    V: PolarsDataType,
729    F: for<'a> BinaryFnMut<Option<T::Physical<'a>>, Option<U::Physical<'a>>>,
730    V::Array: for<'a> ArrayFromIter<
731        <F as BinaryFnMut<Option<T::Physical<'a>>, Option<U::Physical<'a>>>>::Ret,
732    >,
733{
734    match (lhs.len(), rhs.len()) {
735        (1, _) => {
736            let a = unsafe { lhs.get_unchecked(0) };
737            unary_elementwise(rhs, |b| op(a.clone(), b)).with_name(lhs.name().clone())
738        },
739        (_, 1) => {
740            let b = unsafe { rhs.get_unchecked(0) };
741            unary_elementwise(lhs, |a| op(a, b.clone()))
742        },
743        _ => binary_elementwise(lhs, rhs, op),
744    }
745}

747pub fn broadcast_try_binary_elementwise<T, U, V, F, K, E>(
748    lhs: &ChunkedArray<T>,
749    rhs: &ChunkedArray<U>,
750    mut op: F,
751) -> Result<ChunkedArray<V>, E>
752where
753    T: PolarsDataType,
754    U: PolarsDataType,
755    V: PolarsDataType,
756    F: for<'a> FnMut(Option<T::Physical<'a>>, Option<U::Physical<'a>>) -> Result<Option<K>, E>,
757    V::Array: ArrayFromIter<Option<K>>,
758{
759    match (lhs.len(), rhs.len()) {
760        (1, _) => {
761            let a = unsafe { lhs.get_unchecked(0) };
762            Ok(try_unary_elementwise(rhs, |b| op(a.clone(), b))?.with_name(lhs.name().clone()))
763        },
764        (_, 1) => {
765            let b = unsafe { rhs.get_unchecked(0) };
766            try_unary_elementwise(lhs, |a| op(a, b.clone()))
767        },
768        _ => try_binary_elementwise(lhs, rhs, op),
769    }
770}

772pub fn broadcast_binary_elementwise_values<T, U, V, F, K>(
773    lhs: &ChunkedArray<T>,
774    rhs: &ChunkedArray<U>,
775    mut op: F,
776) -> ChunkedArray<V>
777where
778    T: PolarsDataType,
779    U: PolarsDataType,
780    V: PolarsDataType,
781    F: for<'a> FnMut(T::Physical<'a>, U::Physical<'a>) -> K,
782    V::Array: ArrayFromIter<K>,
783{
784    if lhs.null_count() == lhs.len() || rhs.null_count() == rhs.len() {
785        let min = lhs.len().min(rhs.len());
786        let max = lhs.len().max(rhs.len());
787        let len = if min == 1 { max } else { min };
788        let arr = V::Array::full_null(len, V::get_static_dtype().to_arrow(CompatLevel::newest()));
789
790        return ChunkedArray::with_chunk(lhs.name().clone(), arr);
791    }
792
793    match (lhs.len(), rhs.len()) {
794        (1, _) => {
795            let a = unsafe { lhs.value_unchecked(0) };
796            unary_elementwise_values(rhs, |b| op(a.clone(), b)).with_name(lhs.name().clone())
797        },
798        (_, 1) => {
799            let b = unsafe { rhs.value_unchecked(0) };
800            unary_elementwise_values(lhs, |a| op(a, b.clone()))
801        },
802        _ => binary_elementwise_values(lhs, rhs, op),
803    }
804}

806pub fn apply_binary_kernel_broadcast<'l, 'r, L, R, O, K, LK, RK>(
807    lhs: &'l ChunkedArray<L>,
808    rhs: &'r ChunkedArray<R>,
809    kernel: K,
810    lhs_broadcast_kernel: LK,
811    rhs_broadcast_kernel: RK,
812) -> ChunkedArray<O>
813where
814    L: PolarsDataType,
815    R: PolarsDataType,
816    O: PolarsDataType,
817    K: Fn(&L::Array, &R::Array) -> O::Array,
818    LK: Fn(L::Physical<'l>, &R::Array) -> O::Array,
819    RK: Fn(&L::Array, R::Physical<'r>) -> O::Array,
820{
821    let name = lhs.name();
822    let out = match (lhs.len(), rhs.len()) {
823        (a, b) if a == b => binary(lhs, rhs, |lhs, rhs| kernel(lhs, rhs)),
824        // broadcast right path
825        (_, 1) => {
826            let opt_rhs = rhs.get(0);
827            match opt_rhs {
828                None => {
829                    let arr = O::Array::full_null(
830                        lhs.len(),
831                        O::get_static_dtype().to_arrow(CompatLevel::newest()),
832                    );
833                    ChunkedArray::<O>::with_chunk(lhs.name().clone(), arr)
834                },
835                Some(rhs) => unary_kernel(lhs, |arr| rhs_broadcast_kernel(arr, rhs.clone())),
836            }
837        },
838        (1, _) => {
839            let opt_lhs = lhs.get(0);
840            match opt_lhs {
841                None => {
842                    let arr = O::Array::full_null(
843                        rhs.len(),
844                        O::get_static_dtype().to_arrow(CompatLevel::newest()),
845                    );
846                    ChunkedArray::<O>::with_chunk(lhs.name().clone(), arr)
847                },
848                Some(lhs) => unary_kernel(rhs, |arr| lhs_broadcast_kernel(lhs.clone(), arr)),
849            }
850        },
851        _ => panic!("Cannot apply operation on arrays of different lengths"),
852    };
853    out.with_name(name.clone())
854}

856pub fn apply_binary_kernel_broadcast_owned<L, R, O, K, LK, RK>(
857    lhs: ChunkedArray<L>,
858    rhs: ChunkedArray<R>,
859    kernel: K,
860    lhs_broadcast_kernel: LK,
861    rhs_broadcast_kernel: RK,
862) -> ChunkedArray<O>
863where
864    L: PolarsDataType,
865    R: PolarsDataType,
866    O: PolarsDataType,
867    K: Fn(L::Array, R::Array) -> O::Array,
868    for<'a> LK: Fn(L::Physical<'a>, R::Array) -> O::Array,
869    for<'a> RK: Fn(L::Array, R::Physical<'a>) -> O::Array,
870{
871    let name = lhs.name().to_owned();
872    let out = match (lhs.len(), rhs.len()) {
873        (a, b) if a == b => binary_owned(lhs, rhs, kernel),
874        // broadcast right path
875        (_, 1) => {
876            let opt_rhs = rhs.get(0);
877            match opt_rhs {
878                None => {
879                    let arr = O::Array::full_null(
880                        lhs.len(),
881                        O::get_static_dtype().to_arrow(CompatLevel::newest()),
882                    );
883                    ChunkedArray::<O>::with_chunk(lhs.name().clone(), arr)
884                },
885                Some(rhs) => unary_kernel_owned(lhs, |arr| rhs_broadcast_kernel(arr, rhs.clone())),
886            }
887        },
888        (1, _) => {
889            let opt_lhs = lhs.get(0);
890            match opt_lhs {
891                None => {
892                    let arr = O::Array::full_null(
893                        rhs.len(),
894                        O::get_static_dtype().to_arrow(CompatLevel::newest()),
895                    );
896                    ChunkedArray::<O>::with_chunk(lhs.name().clone(), arr)
897                },
898                Some(lhs) => unary_kernel_owned(rhs, |arr| lhs_broadcast_kernel(lhs.clone(), arr)),
899            }
900        },
901        _ => panic!("Cannot apply operation on arrays of different lengths"),
902    };
903    out.with_name(name)
