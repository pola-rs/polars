1use std::mem::ManuallyDrop;
2
3use super::*;
4use crate::prelude::*;
5
6pub struct PolarsExtension {
7    array: Option<FixedSizeBinaryArray>,
8}
9
10impl PolarsExtension {
11    pub(crate) unsafe fn new(array: FixedSizeBinaryArray) -> Self {
12        Self { array: Some(array) }
13    }
14
15    /// Take the Array hold by [`PolarsExtension`] and forget polars extension,
16    /// so that drop is not called
17    pub(crate) fn take_and_forget(self) -> FixedSizeBinaryArray {
18        let mut md = ManuallyDrop::new(self);
19        md.array.take().unwrap()
20    }
21
22    /// Apply a function with the sentinel, without the sentinels drop being called
23    unsafe fn with_sentinel<T, F: FnOnce(&mut ExtensionSentinel) -> T>(&self, fun: F) -> T {
24        let mut sentinel = self.get_sentinel();
25        let out = fun(&mut sentinel);
26        std::mem::forget(sentinel);
27        out
28    }
29
30    /// Load the sentinel from the heap.
31    /// be very careful, this dereferences a raw pointer on the heap,
32    unsafe fn get_sentinel(&self) -> Box<ExtensionSentinel> {
33        if let ArrowDataType::Extension(ext) = self.array.as_ref().unwrap().dtype() {
34            let metadata = ext
35                .metadata
36                .as_ref()
37                .expect("should have metadata in extension type");
38            let mut iter = metadata.split(';');
39
40            let pid = iter.next().unwrap().parse::<u128>().unwrap();
41            let ptr = iter.next().unwrap().parse::<usize>().unwrap();
42            if pid == *PROCESS_ID {
43                Box::from_raw(ptr as *const ExtensionSentinel as *mut ExtensionSentinel)
44            } else {
45                panic!("pid did not mach process id")
46            }
47        } else {
48            panic!("should be extension type")
49        }
50    }
51
52    /// Calls the heap allocated function in the [`ExtensionSentinel`] that knows
53    /// how to convert the [`FixedSizeBinaryArray`] to a `Series` of type [`ObjectChunked<T>`]
54    pub(crate) unsafe fn get_series(&self, name: &PlSmallStr) -> Series {
55        self.with_sentinel(|sent| {
56            (sent.to_series_fn.as_ref().unwrap())(self.array.as_ref().unwrap(), name)
57        })
58    }
59
60    // heap allocates a function that converts the binary array to a Series of [`ObjectChunked<T>`]
61    // the `name` will be the `name` of the output `Series` when this function is called (later).
62    pub(crate) unsafe fn set_to_series_fn<T: PolarsObject>(&mut self) {
63        let f = Box::new(move |arr: &FixedSizeBinaryArray, name: &PlSmallStr| {
64            let iter = arr.iter().map(|opt| {
65                opt.map(|bytes| {
66                    let t = std::ptr::read_unaligned(bytes.as_ptr() as *const T);
67
68                    let ret = t.clone();
69                    std::mem::forget(t);
70                    ret
71                })
72            });
73
74            let ca = ObjectChunked::<T>::from_iter_options(name.clone(), iter);
75            ca.into_series()
76        });
77        self.with_sentinel(move |sent| {
78            sent.to_series_fn = Some(f);
79        });
80    }
81}
82
83impl Drop for PolarsExtension {
84    fn drop(&mut self) {
85        // implicitly drop by taking ownership
86        unsafe { self.get_sentinel() };
87    }
