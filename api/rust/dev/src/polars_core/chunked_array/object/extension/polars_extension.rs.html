1use std::mem::ManuallyDrop;

3use super::*;
4use crate::prelude::*;

6pub struct PolarsExtension {
7    array: Option<FixedSizeBinaryArray>,
8}
9
10impl PolarsExtension {
11    /// This is very expensive
12    pub(crate) unsafe fn arr_to_av(arr: &FixedSizeBinaryArray, i: usize) -> AnyValue<'_> {
13        let arr = arr.slice_typed_unchecked(i, 1);
14        let pe = Self::new(arr);
15        let pe = ManuallyDrop::new(pe);
16        pe.get_series(&PlSmallStr::EMPTY)
17            .get(0)
18            .unwrap()
19            .into_static()
20    }
21
22    pub(crate) unsafe fn new(array: FixedSizeBinaryArray) -> Self {
23        Self { array: Some(array) }
24    }
25
26    /// Take the Array hold by [`PolarsExtension`] and forget polars extension,
27    /// so that drop is not called
28    pub(crate) fn take_and_forget(self) -> FixedSizeBinaryArray {
29        let mut md = ManuallyDrop::new(self);
30        md.array.take().unwrap()
31    }
32
33    /// Apply a function with the sentinel, without the sentinels drop being called
34    unsafe fn with_sentinel<T, F: FnOnce(&mut ExtensionSentinel) -> T>(&self, fun: F) -> T {
35        let mut sentinel = self.get_sentinel();
36        let out = fun(&mut sentinel);
37        std::mem::forget(sentinel);
38        out
39    }
40
41    /// Load the sentinel from the heap.
42    /// be very careful, this dereferences a raw pointer on the heap,
43    unsafe fn get_sentinel(&self) -> Box<ExtensionSentinel> {
44        if let ArrowDataType::Extension(ext) = self.array.as_ref().unwrap().dtype() {
45            let metadata = ext
46                .metadata
47                .as_ref()
48                .expect("should have metadata in extension type");
49            let mut iter = metadata.split(';');
50
51            let pid = iter.next().unwrap().parse::<u128>().unwrap();
52            let ptr = iter.next().unwrap().parse::<usize>().unwrap();
53            if pid == *PROCESS_ID {
54                Box::from_raw(ptr as *const ExtensionSentinel as *mut ExtensionSentinel)
55            } else {
56                panic!("pid did not mach process id")
57            }
58        } else {
59            panic!("should be extension type")
60        }
61    }
62
63    /// Calls the heap allocated function in the [`ExtensionSentinel`] that knows
64    /// how to convert the [`FixedSizeBinaryArray`] to a `Series` of type [`ObjectChunked<T>`]
65    pub(crate) unsafe fn get_series(&self, name: &PlSmallStr) -> Series {
66        self.with_sentinel(|sent| {
67            (sent.to_series_fn.as_ref().unwrap())(self.array.as_ref().unwrap(), name)
68        })
69    }
70
71    // heap allocates a function that converts the binary array to a Series of [`ObjectChunked<T>`]
72    // the `name` will be the `name` of the output `Series` when this function is called (later).
73    pub(crate) unsafe fn set_to_series_fn<T: PolarsObject>(&mut self) {
74        let f = Box::new(move |arr: &FixedSizeBinaryArray, name: &PlSmallStr| {
75            let iter = arr.iter().map(|opt| {
76                opt.map(|bytes| {
77                    let t = std::ptr::read_unaligned(bytes.as_ptr() as *const T);
78
79                    let ret = t.clone();
80                    std::mem::forget(t);
81                    ret
82                })
83            });
84
85            let ca = ObjectChunked::<T>::from_iter_options(name.clone(), iter);
86            ca.into_series()
87        });
88        self.with_sentinel(move |sent| {
89            sent.to_series_fn = Some(f);
90        });
91    }
92}
93
94impl Drop for PolarsExtension {
95    fn drop(&mut self) {
96        // implicitly drop by taking ownership
97        unsafe { self.get_sentinel() };
98    }
