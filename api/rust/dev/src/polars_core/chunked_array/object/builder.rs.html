8pub struct ObjectChunkedBuilder<T> {
9    field: Field,
10    bitmask_builder: BitmapBuilder,
11    values: Vec<T>,
12}
13
14impl<T> ObjectChunkedBuilder<T>
15where
16    T: PolarsObject,
17{
18    pub fn field(&self) -> &Field {
19        &self.field
20    }
21    pub fn new(name: PlSmallStr, capacity: usize) -> Self {
22        ObjectChunkedBuilder {
23            field: Field::new(name, DataType::Object(T::type_name())),
24            values: Vec::with_capacity(capacity),
25            bitmask_builder: BitmapBuilder::with_capacity(capacity),
26        }
27    }
28
29    /// Appends a value of type `T` into the builder
30    #[inline]
31    pub fn append_value(&mut self, v: T) {
32        self.values.push(v);
33        self.bitmask_builder.push(true);
34    }
35
36    /// Appends a null slot into the builder
37    #[inline]
38    pub fn append_null(&mut self) {
39        self.values.push(T::default());
40        self.bitmask_builder.push(false);
41    }
42
43    #[inline]
44    pub fn append_value_from_any(&mut self, v: &dyn Any) -> PolarsResult<()> {
45        let Some(v) = v.downcast_ref::<T>() else {
46            polars_bail!(SchemaMismatch: "cannot downcast any in ObjectBuilder");
47        };
48        self.append_value(v.clone());
49        Ok(())
50    }
51
52    #[inline]
53    pub fn append_option(&mut self, opt: Option<T>) {
54        match opt {
55            Some(s) => self.append_value(s),
56            None => self.append_null(),
57        }
58    }
59
60    pub fn finish(mut self) -> ObjectChunked<T> {
61        let null_bitmap: Option<Bitmap> = self.bitmask_builder.into_opt_validity();
62
63        let len = self.values.len();
64        let null_count = null_bitmap
65            .as_ref()
66            .map(|validity| validity.unset_bits())
67            .unwrap_or(0);
68
69        let arr = Box::new(ObjectArray {
70            dtype: ArrowDataType::FixedSizeBinary(size_of::<T>()),
71            values: self.values.into(),
72            validity: null_bitmap,
73        });
74
75        self.field.dtype = get_object_type::<T>();
76
77        unsafe { ChunkedArray::new_with_dims(Arc::new(self.field), vec![arr], len, null_count) }
78    }
79}

100    fn from_slice(name: PlSmallStr, v: &[T]) -> Self {

102    }

104    fn from_slice_options(name: PlSmallStr, opt_v: &[Option<T>]) -> Self {
105        let mut builder = ObjectChunkedBuilder::<T>::new(name, opt_v.len());
106        opt_v
107            .iter()
108            .cloned()
109            .for_each(|opt| builder.append_option(opt));
110        builder.finish()
111    }
112
113    fn from_iter_options(
114        name: PlSmallStr,
115        it: impl Iterator<Item = Option<T>>,
116    ) -> ObjectChunked<T> {
117        let mut builder = ObjectChunkedBuilder::new(name, get_iter_capacity(&it));
118        it.for_each(|opt| builder.append_option(opt));
119        builder.finish()
120    }
121
122    /// Create a new ChunkedArray from an iterator.
123    fn from_iter_values(name: PlSmallStr, it: impl Iterator<Item = T>) -> ObjectChunked<T> {
124        let mut builder = ObjectChunkedBuilder::new(name, get_iter_capacity(&it));
125        it.for_each(|v| builder.append_value(v));
126        builder.finish()
127    }

168/// Convert a Series of dtype object to an Arrow Array of FixedSizeBinary
169pub(crate) fn object_series_to_arrow_array(s: &Series) -> ArrayRef {
170    // The list builder knows how to create an arrow array
171    // we simply piggy back on that code.
172
173    // SAFETY: 0..len is in bounds
174    let list_s = unsafe {
175        let groups = vec![[0, s.len() as IdxSize]];
176        s.agg_list(&GroupsType::new_slice(groups, false, true))
177    };
178    let arr = &list_s.chunks()[0];
179    let arr = arr.as_any().downcast_ref::<ListArray<i64>>().unwrap();
180    arr.values().to_boxed()
181}

183impl<T: PolarsObject> ArrayBuilder for ObjectChunkedBuilder<T> {
184    fn dtype(&self) -> &ArrowDataType {
185        &ArrowDataType::FixedSizeBinary(size_of::<T>())
186    }
187
188    fn reserve(&mut self, additional: usize) {
189        self.bitmask_builder.reserve(additional);
190        self.values.reserve(additional);
191    }
192
193    fn freeze(self) -> Box<dyn Array> {
194        Box::new(ObjectArray {
195            dtype: ArrowDataType::FixedSizeBinary(size_of::<T>()),
196            values: self.values.into(),
197            validity: self.bitmask_builder.into_opt_validity(),
198        })
199    }
200
201    fn freeze_reset(&mut self) -> Box<dyn Array> {
202        Box::new(ObjectArray {
203            dtype: ArrowDataType::FixedSizeBinary(size_of::<T>()),
204            values: core::mem::take(&mut self.values).into(),
205            validity: core::mem::take(&mut self.bitmask_builder).into_opt_validity(),
206        })
207    }
208
209    fn len(&self) -> usize {
210        self.values.len()
211    }
212
213    fn extend_nulls(&mut self, length: usize) {
214        self.values.resize(self.values.len() + length, T::default());
215        self.bitmask_builder.extend_constant(length, false);
216    }
217
218    fn subslice_extend(
219        &mut self,
220        other: &dyn Array,
221        start: usize,
222        length: usize,
223        _share: ShareStrategy,
224    ) {
225        let other: &ObjectArray<T> = other.as_any().downcast_ref().unwrap();
226        self.values
227            .extend_from_slice(&other.values[start..start + length]);
228        self.bitmask_builder
229            .subslice_extend_from_opt_validity(other.validity(), start, length);
230    }
231
232    fn subslice_extend_repeated(
233        &mut self,
234        other: &dyn Array,
235        start: usize,
236        length: usize,
237        repeats: usize,
238        share: ShareStrategy,
239    ) {
240        for _ in 0..repeats {
241            self.subslice_extend(other, start, length, share)
242        }
243    }
244
245    fn subslice_extend_each_repeated(
246        &mut self,
247        other: &dyn Array,
248        start: usize,
249        length: usize,
250        repeats: usize,
251        _share: ShareStrategy,
252    ) {
253        let other: &ObjectArray<T> = other.as_any().downcast_ref().unwrap();
254
255        self.values.reserve(length * repeats);
256        for value in other.values[start..start + length].iter() {
257            unsafe {
258                for _ in 0..repeats {
259                    self.values.push_unchecked(value.clone());
260                }
261            }
262        }
263        self.bitmask_builder
264            .subslice_extend_each_repeated_from_opt_validity(
265                other.validity(),
266                start,
267                length,
268                repeats,
269            );
270    }
271
272    unsafe fn gather_extend(&mut self, other: &dyn Array, idxs: &[IdxSize], _share: ShareStrategy) {
273        let other: &ObjectArray<T> = other.as_any().downcast_ref().unwrap();
274        let other_values_slice = other.values.as_slice();
275        self.values.extend(
276            idxs.iter()
277                .map(|idx| other_values_slice.get_unchecked(*idx as usize).clone()),
278        );
279        self.bitmask_builder
280            .gather_extend_from_opt_validity(other.validity(), idxs, other.len());
281    }
282
283    fn opt_gather_extend(&mut self, other: &dyn Array, idxs: &[IdxSize], _share: ShareStrategy) {
284        let other: &ObjectArray<T> = other.as_any().downcast_ref().unwrap();
285        let other_values_slice = other.values.as_slice();
286        self.values.reserve(idxs.len());
287        unsafe {
288            for idx in idxs {
289                let val = if (*idx as usize) < other.len() {
290                    other_values_slice.get_unchecked(*idx as usize).clone()
291                } else {
292                    T::default()
293                };
294                self.values.push_unchecked(val);
295            }
296        }
297        self.bitmask_builder.opt_gather_extend_from_opt_validity(
298            other.validity(),
299            idxs,
300            other.len(),
301        );
302    }
303}
