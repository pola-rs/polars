35    fn get_cat_idx(&mut self, s: &str, h: u64) -> (u32, bool) {
36        let len = self.local_mapping.len() as u32;
37
38        // SAFETY: index in hashmap are within bounds of categories
39        unsafe {
40            let r = self.local_mapping.entry(
41                h,
42                |k| self.categories.value_unchecked(*k as usize) == s,
43                |k| {
44                    self.local_hasher
45                        .hash_one(self.categories.value_unchecked(*k as usize))
46                },
47            );
48
49            match r {
50                HTEntry::Occupied(v) => (*v.get(), false),
51                HTEntry::Vacant(slot) => {
52                    self.categories.push(Some(s));
53                    slot.insert(len);
54                    (len, true)
55                },
56            }
57        }
58    }

60    fn try_get_cat_idx(&mut self, s: &str, h: u64) -> Option<u32> {
61        // SAFETY: index in hashmap are within bounds of categories
62        unsafe {
63            let r = self.local_mapping.entry(
64                h,
65                |k| self.categories.value_unchecked(*k as usize) == s,
66                |k| {
67                    self.local_hasher
68                        .hash_one(self.categories.value_unchecked(*k as usize))
69                },
70            );
71
72            match r {
73                HTEntry::Occupied(v) => Some(*v.get()),
74                HTEntry::Vacant(_) => None,
75            }
76        }
77    }

79    /// Append a new category, but fail if it didn't exist yet in the category state.
80    /// You can register categories up front with `register_value`, or via `append`.

82    pub fn try_append_value(&mut self, s: &str) -> PolarsResult<()> {
83        let h = self.local_hasher.hash_one(s);
84        let idx = self.try_get_cat_idx(s, h).ok_or_else(
85            || polars_err!(ComputeError: "category {} doesn't exist in Enum dtype", s),
86        )?;
87        self.cat_builder.push(Some(idx));
88        Ok(())

139    /// Fast path for global categorical which preserves hashes and saves an allocation by
140    /// altering the keys in place.
141    fn drain_iter_global_and_finish<'a, I>(&mut self, i: I) -> CategoricalChunked
142    where
143        I: IntoIterator<Item = Option<&'a str>>,
144    {
145        let iter = i.into_iter();
146        // Save hashes for later when inserting into the global hashmap.
147        let mut hashes = Vec::with_capacity(_HASHMAP_INIT_SIZE);
148        for s in self.categories.values_iter() {
149            hashes.push(self.local_hasher.hash_one(s));
150        }
151
152        for opt_s in iter {
153            match opt_s {
154                None => self.append_null(),
155                Some(s) => {
156                    let hash = self.local_hasher.hash_one(s);
157                    let (cat_idx, new) = self.get_cat_idx(s, hash);
158                    self.cat_builder.push(Some(cat_idx));
159                    if new {
160                        // We appended a value to the map.
161                        hashes.push(hash);
162                    }
163                },
164            }
165        }
166
167        let categories = std::mem::take(&mut self.categories).freeze();
168
169        // We will create a mapping from our local categoricals to global categoricals
170        // and a mapping from global categoricals to our local categoricals.
171        let mut local_to_global: Vec<u32> = Vec::with_capacity(categories.len());
172        let (id, local_to_global) = crate::STRING_CACHE.apply(|cache| {
173            for (s, h) in categories.values_iter().zip(hashes) {
174                // SAFETY: we allocated enough.
175                unsafe { local_to_global.push_unchecked(cache.insert_from_hash(h, s)) }
176            }
177            local_to_global
178        });
179
180        // Change local indices inplace to their global counterparts.
181        let update_cats = || {
182            if !local_to_global.is_empty() {
183                // when all categorical are null, `local_to_global` is empty and all cats physical values are 0.
184                self.cat_builder.apply_values(|cats| {
185                    for cat in cats {
186                        debug_assert!((*cat as usize) < local_to_global.len());
187                        *cat = *unsafe { local_to_global.get_unchecked(*cat as usize) };
188                    }
189                })
190            }
191        };
192
193        let mut global_to_local = PlHashMap::with_capacity(local_to_global.len());
194        POOL.join(
195            || fill_global_to_local(&local_to_global, &mut global_to_local),
196            update_cats,
197        );
198
199        let indices = std::mem::take(&mut self.cat_builder).into();
200        let indices = UInt32Chunked::with_chunk(self.name.clone(), indices);
201
202        // SAFETY: indices are in bounds of new rev_map
203        unsafe {
204            CategoricalChunked::from_cats_and_rev_map_unchecked(
205                indices,
206                Arc::new(RevMapping::Global(global_to_local, categories, id)),
207                false,
208                self.ordering,
209            )
210            .with_fast_unique(true)
211        }
212    }
213
214    pub fn drain_iter_and_finish<'a, I>(mut self, i: I) -> CategoricalChunked
215    where
216        I: IntoIterator<Item = Option<&'a str>>,
217    {
218        if using_string_cache() {
219            self.drain_iter_global_and_finish(i)
220        } else {
221            self.drain_iter(i);
222            self.finish()
223        }
224    }
225
226    pub fn finish(self) -> CategoricalChunked {
227        // SAFETY: keys and values are in bounds
228        unsafe {
229            CategoricalChunked::from_keys_and_values(
230                self.name.clone(),
231                &self.cat_builder.into(),
232                &self.categories.into(),
233                self.ordering,
234            )
235            .with_fast_unique(true)
236        }
237    }

240fn fill_global_to_local(local_to_global: &[u32], global_to_local: &mut PlHashMap<u32, u32>) {
241    let mut local_idx = 0;
242    #[allow(clippy::explicit_counter_loop)]
243    for global_idx in local_to_global {
244        // we know the keys are unique so this is much faster
245        unsafe {
246            global_to_local.insert_unique_unchecked(*global_idx, local_idx);
247        }
248        local_idx += 1;
249    }

253    /// Create a [`CategoricalChunked`] from a categorical indices. The indices will
254    /// probe the global string cache.
255    pub(crate) fn from_global_indices(
256        cats: UInt32Chunked,
257        ordering: CategoricalOrdering,
258    ) -> PolarsResult<CategoricalChunked> {
259        let len = crate::STRING_CACHE.read_map().len() as u32;
260        let oob = cats.into_iter().flatten().any(|cat| cat >= len);
261        polars_ensure!(
262            !oob,
263            ComputeError:
264            "cannot construct Categorical from these categories; at least one of them is out of bounds"
265        );
266        Ok(unsafe { Self::from_global_indices_unchecked(cats, ordering) })
267    }

269    /// Create a [`CategoricalChunked`] from a categorical indices. The indices will
270    /// probe the global string cache.
271    ///
272    /// # Safety
273    /// This does not do any bound checks
274    pub unsafe fn from_global_indices_unchecked(
275        cats: UInt32Chunked,
276        ordering: CategoricalOrdering,
277    ) -> CategoricalChunked {
278        let cache = crate::STRING_CACHE.read_map();
279
280        let cap = std::cmp::min(std::cmp::min(cats.len(), cache.len()), _HASHMAP_INIT_SIZE);
281        let mut rev_map = PlHashMap::with_capacity(cap);
282        let mut str_values = MutablePlString::with_capacity(cap);
283
284        for arr in cats.downcast_iter() {
285            for cat in arr.into_iter().flatten().copied() {
286                let offset = str_values.len() as u32;
287
288                if let Entry::Vacant(entry) = rev_map.entry(cat) {
289                    entry.insert(offset);
290                    let str_val = cache.get_unchecked(cat);
291                    str_values.push(Some(str_val))
292                }
293            }
294        }
295
296        let rev_map = RevMapping::Global(rev_map, str_values.into(), cache.uuid);
297
298        CategoricalChunked::from_cats_and_rev_map_unchecked(
299            cats,
300            Arc::new(rev_map),
301            false,
302            ordering,
303        )
304    }

306    pub(crate) unsafe fn from_keys_and_values_global(
307        name: PlSmallStr,
308        keys: impl IntoIterator<Item = Option<u32>> + Send,
309        capacity: usize,
310        values: &Utf8ViewArray,
311        ordering: CategoricalOrdering,
312    ) -> Self {
313        // Vec<u32> where the index is local and the value is the global index
314        let mut local_to_global: Vec<u32> = Vec::with_capacity(values.len());
315        let (id, local_to_global) = crate::STRING_CACHE.apply(|cache| {
316            // locally we don't need a hashmap because we all categories are 1 integer apart
317            // so the index is local, and the values is global
318            for s in values.values_iter() {
319                // SAFETY: we allocated enough
320                unsafe { local_to_global.push_unchecked(cache.insert(s)) }
321            }
322            local_to_global
323        });
324
325        let compute_cats = || {
326            let mut result = UInt32Vec::with_capacity(capacity);
327
328            for opt_value in keys.into_iter() {
329                result.push(opt_value.map(|cat| {
330                    debug_assert!((cat as usize) < local_to_global.len());
331                    *unsafe { local_to_global.get_unchecked(cat as usize) }
332                }));
333            }
334            result
335        };
336
337        let mut global_to_local = PlHashMap::with_capacity(local_to_global.len());
338        let (_, cats) = POOL.join(
339            || fill_global_to_local(&local_to_global, &mut global_to_local),
340            compute_cats,
341        );
342        unsafe {
343            CategoricalChunked::from_cats_and_rev_map_unchecked(
344                UInt32Chunked::with_chunk(name, cats.into()),
345                Arc::new(RevMapping::Global(global_to_local, values.clone(), id)),
346                false,
347                ordering,
348            )
349        }
350    }

366    /// # Safety
367    /// The caller must ensure that index values in the `keys` are in within bounds of the `values` length.
368    pub(crate) unsafe fn from_keys_and_values(
369        name: PlSmallStr,
370        keys: &PrimitiveArray<u32>,
371        values: &Utf8ViewArray,
372        ordering: CategoricalOrdering,
373    ) -> Self {
374        if !using_string_cache() {
375            CategoricalChunked::from_keys_and_values_local(name, keys, values, ordering)
376        } else {
377            CategoricalChunked::from_keys_and_values_global(
378                name,
379                keys.into_iter().map(|c| c.copied()),
380                keys.len(),
381                values,
382                ordering,
383            )
384        }
385    }

387    /// Create a [`CategoricalChunked`] from a fixed list of categories and a List of strings.
388    /// This will error if a string is not in the fixed list of categories
389    pub fn from_string_to_enum(
390        values: &StringChunked,
391        categories: &Utf8ViewArray,
392        ordering: CategoricalOrdering,
393    ) -> PolarsResult<CategoricalChunked> {
394        polars_ensure!(categories.null_count()  == 0, ComputeError: "categories can not contain null values");
395
396        // Build a mapping string -> idx
397        let mut map = PlHashMap::with_capacity(categories.len());
398        for (idx, cat) in categories.values_iter().enumerate_idx() {
399            #[allow(clippy::unnecessary_cast)]
400            map.insert(cat, idx as u32);
401        }
402        // Find idx of every value in the map
403        let iter = values.downcast_iter().map(|arr| {
404            arr.iter()
405                .map(|opt_s: Option<&str>| opt_s.and_then(|s| map.get(s).copied()))
406                .collect_arr()
407        });
408        let mut keys: UInt32Chunked = ChunkedArray::from_chunk_iter(values.name().clone(), iter);
409        keys.rename(values.name().clone());
410        let rev_map = RevMapping::build_local(categories.clone());
411        unsafe {
412            Ok(CategoricalChunked::from_cats_and_rev_map_unchecked(
413                keys,
414                Arc::new(rev_map),
415                true,
416                ordering,
417            )
418            .with_fast_unique(false))
419        }
420    }

428    #[test]
429    fn test_categorical_rev() -> PolarsResult<()> {
430        let _lock = SINGLE_LOCK.lock();
431        disable_string_cache();
432        let slice = &[
433            Some("foo"),
434            None,
435            Some("bar"),
436            Some("foo"),
437            Some("foo"),
438            Some("bar"),
439        ];
440        let ca = StringChunked::new(PlSmallStr::from_static("a"), slice);
441        let out = ca.cast(&DataType::Categorical(None, Default::default()))?;
442        let out = out.categorical().unwrap().clone();
443        assert_eq!(out.get_rev_map().len(), 2);
444
445        // test the global branch
446        enable_string_cache();
447        // empty global cache
448        let out = ca.cast(&DataType::Categorical(None, Default::default()))?;
449        let out = out.categorical().unwrap().clone();
450        assert_eq!(out.get_rev_map().len(), 2);
451        // full global cache
452        let out = ca.cast(&DataType::Categorical(None, Default::default()))?;
453        let out = out.categorical().unwrap().clone();
454        assert_eq!(out.get_rev_map().len(), 2);
455
456        // Check that we don't panic if we append two categorical arrays
457        // build under the same string cache
458        // https://github.com/pola-rs/polars/issues/1115
459        let ca1 = StringChunked::new(PlSmallStr::from_static("a"), slice)
460            .cast(&DataType::Categorical(None, Default::default()))?;
461        let mut ca1 = ca1.categorical().unwrap().clone();
462        let ca2 = StringChunked::new(PlSmallStr::from_static("a"), slice)
463            .cast(&DataType::Categorical(None, Default::default()))?;
464        let ca2 = ca2.categorical().unwrap();
465        ca1.append(ca2).unwrap();
466
467        Ok(())
468    }

470    #[test]
471    fn test_categorical_builder() {
472        use crate::{disable_string_cache, enable_string_cache};
473        let _lock = crate::SINGLE_LOCK.lock();
474        for use_string_cache in [false, true] {
475            disable_string_cache();
476            if use_string_cache {
477                enable_string_cache();
478            }
479
480            // Use 2 builders to check if the global string cache
481            // does not interfere with the index mapping
482            let builder1 = CategoricalChunkedBuilder::new(
483                PlSmallStr::from_static("foo"),
484                10,
485                Default::default(),
486            );
487            let builder2 = CategoricalChunkedBuilder::new(
488                PlSmallStr::from_static("foo"),
489                10,
490                Default::default(),
491            );
492            let s = builder1
493                .drain_iter_and_finish(vec![None, Some("hello"), Some("vietnam")])
494                .into_series();
495            assert_eq!(s.str_value(0).unwrap(), "null");
496            assert_eq!(s.str_value(1).unwrap(), "hello");
497            assert_eq!(s.str_value(2).unwrap(), "vietnam");
498
499            let s = builder2
500                .drain_iter_and_finish(vec![Some("hello"), None, Some("world")])
501                .into_series();
502            assert_eq!(s.str_value(0).unwrap(), "hello");
503            assert_eq!(s.str_value(1).unwrap(), "null");
504            assert_eq!(s.str_value(2).unwrap(), "world");
505        }
