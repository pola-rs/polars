18    pub unsafe fn agg_valid_count(&self, groups: &GroupsType) -> Series {
19        // Prevent a rechunk for every individual group.
20        let s = if groups.len() > 1 && self.null_count() > 0 {
21            self.rechunk()
22        } else {
23            self.clone()
24        };
25
26        match groups {
27            GroupsType::Idx(groups) => agg_helper_idx_on_all::<IdxType, _>(groups, |idx| {
28                debug_assert!(idx.len() <= s.len());
29                if idx.is_empty() {
30                    None
31                } else if s.null_count() == 0 {
32                    Some(idx.len() as IdxSize)
33                } else {
34                    let take = unsafe { s.take_slice_unchecked(idx) };
35                    Some((take.len() - take.null_count()) as IdxSize)
36                }
37            }),
38            GroupsType::Slice { groups, .. } => {
39                _agg_helper_slice::<IdxType, _>(groups, |[first, len]| {
40                    debug_assert!(len <= s.len() as IdxSize);
41                    if len == 0 {
42                        None
43                    } else if s.null_count() == 0 {
44                        Some(len)
45                    } else {
46                        let take = s.slice_from_offsets(first, len);
47                        Some((take.len() - take.null_count()) as IdxSize)
48                    }
49                })
50            },
51        }
52    }

54    #[doc(hidden)]
55    pub unsafe fn agg_first(&self, groups: &GroupsType) -> Series {
56        // Prevent a rechunk for every individual group.
57        let s = if groups.len() > 1 {
58            self.rechunk()
59        } else {
60            self.clone()
61        };
62
63        let mut out = match groups {
64            GroupsType::Idx(groups) => {
65                let indices = groups
66                    .iter()
67                    .map(
68                        |(first, idx)| {
69                            if idx.is_empty() { None } else { Some(first) }
70                        },
71                    )
72                    .collect_ca(PlSmallStr::EMPTY);
73                // SAFETY: groups are always in bounds.
74                s.take_unchecked(&indices)
75            },
76            GroupsType::Slice { groups, .. } => {
77                let indices = groups
78                    .iter()
79                    .map(|&[first, len]| if len == 0 { None } else { Some(first) })
80                    .collect_ca(PlSmallStr::EMPTY);
81                // SAFETY: groups are always in bounds.
82                s.take_unchecked(&indices)
83            },
84        };
85        if groups.is_sorted_flag() {
86            out.set_sorted_flag(s.is_sorted_flag())
87        }
88        s.restore_logical(out)
89    }

115                    } else {
116                        let take = s.slice_from_offsets(first, len);
117                        take.n_unique().unwrap() as IdxSize
118                    }
119                })
120            },
121        }

125    pub unsafe fn agg_mean(&self, groups: &GroupsType) -> Series {
126        // Prevent a rechunk for every individual group.
127        let s = if groups.len() > 1 {
128            self.rechunk()
129        } else {
130            self.clone()
131        };
132
133        use DataType::*;
134        match s.dtype() {
135            Boolean => s.cast(&Float64).unwrap().agg_mean(groups),
136            Float32 => SeriesWrap(s.f32().unwrap().clone()).agg_mean(groups),
137            Float64 => SeriesWrap(s.f64().unwrap().clone()).agg_mean(groups),
138            dt if dt.is_primitive_numeric() => apply_method_physical_integer!(s, agg_mean, groups),
139            #[cfg(feature = "dtype-decimal")]
140            Decimal(_, _) => self.cast(&Float64).unwrap().agg_mean(groups),
141            #[cfg(feature = "dtype-datetime")]
142            dt @ Datetime(_, _) => self
143                .to_physical_repr()
144                .agg_mean(groups)
145                .cast(&Int64)
146                .unwrap()
147                .cast(dt)
148                .unwrap(),
149            #[cfg(feature = "dtype-duration")]
150            dt @ Duration(_) => self
151                .to_physical_repr()
152                .agg_mean(groups)
153                .cast(&Int64)
154                .unwrap()
155                .cast(dt)
156                .unwrap(),
157            #[cfg(feature = "dtype-time")]
158            Time => self
159                .to_physical_repr()
160                .agg_mean(groups)
161                .cast(&Int64)
162                .unwrap()
163                .cast(&Time)
164                .unwrap(),
165            #[cfg(feature = "dtype-date")]
166            Date => (self
167                .to_physical_repr()
168                .agg_mean(groups)
169                .cast(&Float64)
170                .unwrap()
171                * (US_IN_DAY as f64))
172                .cast(&Datetime(TimeUnit::Microseconds, None))
173                .unwrap(),
174            _ => Series::full_null(PlSmallStr::EMPTY, groups.len(), s.dtype()),
175        }
176    }

234    #[doc(hidden)]
235    pub unsafe fn agg_quantile(
236        &self,
237        groups: &GroupsType,
238        quantile: f64,
239        method: QuantileMethod,
240    ) -> Series {
241        // Prevent a rechunk for every individual group.
242        let s = if groups.len() > 1 {
243            self.rechunk()
244        } else {
245            self.clone()
246        };
247
248        use DataType::*;
249        match s.dtype() {
250            Float32 => s.f32().unwrap().agg_quantile(groups, quantile, method),
251            Float64 => s.f64().unwrap().agg_quantile(groups, quantile, method),
252            #[cfg(feature = "dtype-decimal")]
253            Decimal(_, _) => s
254                .cast(&DataType::Float64)
255                .unwrap()
256                .agg_quantile(groups, quantile, method),
257            dt if dt.is_primitive_numeric() || dt.is_temporal() => {
258                let ca = s.to_physical_repr();
259                let physical_type = ca.dtype();
260                let s = apply_method_physical_integer!(ca, agg_quantile, groups, quantile, method);
261                if dt.is_logical() {
262                    // back to physical and then
263                    // back to logical type
264                    s.cast(physical_type).unwrap().cast(dt).unwrap()
265                } else {
266                    s
267                }
268            },
269            _ => Series::full_null(PlSmallStr::EMPTY, groups.len(), s.dtype()),
270        }
271    }

273    #[doc(hidden)]
274    pub unsafe fn agg_last(&self, groups: &GroupsType) -> Series {
275        // Prevent a rechunk for every individual group.
276        let s = if groups.len() > 1 {
277            self.rechunk()
278        } else {
279            self.clone()
280        };
281
282        let out = match groups {
283            GroupsType::Idx(groups) => {
284                let indices = groups
285                    .all()
286                    .iter()
287                    .map(|idx| {
288                        if idx.is_empty() {
289                            None
290                        } else {
291                            Some(idx[idx.len() - 1])
292                        }
293                    })
294                    .collect_ca(PlSmallStr::EMPTY);
295                s.take_unchecked(&indices)
296            },
297            GroupsType::Slice { groups, .. } => {
298                let indices = groups
299                    .iter()
300                    .map(|&[first, len]| {
301                        if len == 0 {
302                            None
303                        } else {
304                            Some(first + len - 1)
305                        }
306                    })
307                    .collect_ca(PlSmallStr::EMPTY);
308                s.take_unchecked(&indices)
309            },
310        };
311        s.restore_logical(out)
312    }
