22    pub unsafe fn agg_valid_count(&self, groups: &GroupsType) -> Series {
23        // Prevent a rechunk for every individual group.
24        let s = if groups.len() > 1 && self.null_count() > 0 {
25            self.rechunk()
26        } else {
27            self.clone()
28        };
29
30        match groups {
31            GroupsType::Idx(groups) => agg_helper_idx_on_all::<IdxType, _>(groups, |idx| {
32                debug_assert!(idx.len() <= s.len());
33                if idx.is_empty() {
34                    None
35                } else if s.null_count() == 0 {
36                    Some(idx.len() as IdxSize)
37                } else {
38                    let take = unsafe { s.take_slice_unchecked(idx) };
39                    Some((take.len() - take.null_count()) as IdxSize)
40                }
41            }),
42            GroupsType::Slice { groups, .. } => {
43                _agg_helper_slice::<IdxType, _>(groups, |[first, len]| {
44                    debug_assert!(len <= s.len() as IdxSize);
45                    if len == 0 {
46                        None
47                    } else if s.null_count() == 0 {
48                        Some(len)
49                    } else {
50                        let take = s.slice_from_offsets(first, len);
51                        Some((take.len() - take.null_count()) as IdxSize)
52                    }
53                })
54            },
55        }
56    }

59    pub unsafe fn agg_first(&self, groups: &GroupsType) -> Series {
60        // Prevent a rechunk for every individual group.
61        let s = if groups.len() > 1 {
62            self.rechunk()
63        } else {
64            self.clone()
65        };
66
67        let mut out = match groups {
68            GroupsType::Idx(groups) => {
69                let indices = groups
70                    .iter()
71                    .map(
72                        |(first, idx)| {
73                            if idx.is_empty() { None } else { Some(first) }
74                        },
75                    )
76                    .collect_ca(PlSmallStr::EMPTY);
77                // SAFETY: groups are always in bounds.
78                s.take_unchecked(&indices)
79            },
80            GroupsType::Slice { groups, .. } => {
81                let indices = groups
82                    .iter()
83                    .map(|&[first, len]| if len == 0 { None } else { Some(first) })
84                    .collect_ca(PlSmallStr::EMPTY);
85                // SAFETY: groups are always in bounds.
86                s.take_unchecked(&indices)
87            },
88        };
89        if groups.is_sorted_flag() {
90            out.set_sorted_flag(s.is_sorted_flag())
91        }
92        s.restore_logical(out)
93    }

96    pub unsafe fn agg_first_non_null(&self, groups: &GroupsType) -> Series {
97        if !self.has_nulls() {
98            return self.agg_first(groups);
99        }
100
101        // Prevent a rechunk for every individual group.
102        let s = if groups.len() > 1 {
103            self.rechunk()
104        } else {
105            self.clone()
106        };
107
108        let validity = s.rechunk_validity().unwrap();
109        let indices = match groups {
110            GroupsType::Idx(groups) => {
111                groups
112                    .iter()
113                    .map(|(_, idx)| {
114                        let mut this_idx = None;
115                        for &ii in idx.iter() {
116                            // SAFETY: null_values has no null values
117                            if validity.get_bit_unchecked(ii as usize) {
118                                this_idx = Some(ii);
119                                break;
120                            }
121                        }
122                        this_idx
123                    })
124                    .collect_ca(PlSmallStr::EMPTY)
125            },
126            GroupsType::Slice { groups, .. } => {
127                let mask = BitMask::from_bitmap(&validity);
128                groups
129                    .iter()
130                    .map(|&[first, len]| {
131                        // SAFETY: group slice is valid.
132                        let validity = mask.sliced_unchecked(first as usize, len as usize);
133                        let leading_zeros = validity.leading_zeros() as IdxSize;
134                        if leading_zeros == len {
135                            // All values are null, we have no first non-null.
136                            None
137                        } else {
138                            Some(first + leading_zeros)
139                        }
140                    })
141                    .collect_ca(PlSmallStr::EMPTY)
142            },
143        };
144        // SAFETY: groups are always in bounds.
145        let mut out = s.take_unchecked(&indices);
146        if groups.is_sorted_flag() {
147            out.set_sorted_flag(s.is_sorted_flag())
148        }
149        s.restore_logical(out)
150    }

287    pub unsafe fn agg_n_unique(&self, groups: &GroupsType) -> Series {
288        let values = self.to_physical_repr();
289        let dtype = values.dtype();
290        let values = if dtype.contains_objects() {
291            panic!("{}", polars_err!(opq = unique, dtype));
292        } else if let Some(ca) = values.try_str() {
293            ca.as_binary().into_column()
294        } else if dtype.is_nested() {
295            encode_rows_unordered(&[values.into_owned().into_column()])
296                .unwrap()
297                .into_column()
298        } else {
299            values.into_owned().into_column()
300        };
301
302        let values = values.rechunk_to_arrow(CompatLevel::newest());
303        let values = values.as_ref();
304        let state = amortized_unique_from_dtype(values.dtype());
305
306        struct CloneWrapper(Box<dyn AmortizedUnique>);
307        impl Clone for CloneWrapper {
308            fn clone(&self) -> Self {
309                Self(self.0.new_empty())
310            }
311        }
312
313        POOL.install(|| match groups {
314            GroupsType::Idx(idx) => idx
315                .all()
316                .into_par_iter()
317                .map_with(CloneWrapper(state), |state, idxs| unsafe {
318                    state.0.n_unique_idx(values, idxs.as_slice())
319                })
320                .collect::<NoNull<IdxCa>>(),
321            GroupsType::Slice {
322                groups,
323                overlapping: _,
324                monotonic: _,
325            } => groups
326                .into_par_iter()
327                .map_with(CloneWrapper(state), |state, [start, len]| {
328                    state.0.n_unique_slice(values, *start, *len)
329                })
330                .collect::<NoNull<IdxCa>>(),
331        })
332        .into_inner()
333        .into_series()
334    }

348            Float32 => SeriesWrap(s.f32().unwrap().clone()).agg_mean(groups),
349            Float64 => SeriesWrap(s.f64().unwrap().clone()).agg_mean(groups),
350            dt if dt.is_primitive_numeric() => apply_method_physical_integer!(s, agg_mean, groups),
351            #[cfg(feature = "dtype-decimal")]
352            Decimal(_, _) => self.cast(&Float64).unwrap().agg_mean(groups),
353            #[cfg(feature = "dtype-datetime")]
354            dt @ Datetime(_, _) => self
355                .to_physical_repr()
356                .agg_mean(groups)
357                .cast(&Int64)
358                .unwrap()
359                .cast(dt)

361            #[cfg(feature = "dtype-duration")]
362            dt @ Duration(_) => self
363                .to_physical_repr()
364                .agg_mean(groups)
365                .cast(&Int64)
366                .unwrap()
367                .cast(dt)
368                .unwrap(),

373                .cast(&Int64)
374                .unwrap()
375                .cast(&Time)
376                .unwrap(),

378            Date => (self

469            #[cfg(feature = "dtype-datetime")]
470            Datetime(tu, tz) => self

478                .to_physical_repr()
479                .agg_quantile(groups, quantile, method)
480                .cast(&Int64)
481                .unwrap()
482                .into_duration(*tu),
483            #[cfg(feature = "dtype-time")]
484            Time => self
485                .to_physical_repr()
486                .agg_quantile(groups, quantile, method)
487                .cast(&Int64)
488                .unwrap()
489                .into_time(),
490            #[cfg(feature = "dtype-date")]
491            Date => (self
492                .to_physical_repr()
493                .agg_quantile(groups, quantile, method)
494                .cast(&Float64)
495                .unwrap()
496                * (US_IN_DAY as f64))
497                .cast(&DataType::Int64)
498                .unwrap()
499                .into_datetime(TimeUnit::Microseconds, None),

509        // Prevent a rechunk for every individual group.
510        let s = if groups.len() > 1 {
511            self.rechunk()

523                            None
524                        } else {

529                s.take_unchecked(&indices)
530            },

532                let indices = groups

535                        if len == 0 {
536                            None
537                        } else {
538                            Some(first + len - 1)
539                        }
540                    })
541                    .collect_ca(PlSmallStr::EMPTY);
542                s.take_unchecked(&indices)
543            },
544        };
545        s.restore_logical(out)

549    pub unsafe fn agg_last_non_null(&self, groups: &GroupsType) -> Series {
550        if !self.has_nulls() {
551            return self.agg_last(groups);
552        }
553
554        // Prevent a rechunk for every individual group.
555        let s = if groups.len() > 1 {
556            self.rechunk()
557        } else {
558            self.clone()
559        };
560
561        let validity = s.rechunk_validity().unwrap();
562        let indices = match groups {
563            GroupsType::Idx(groups) => {
564                groups
565                    .iter()
566                    .map(|(_, idx)| {
567                        // We may or may not find a valid value.
568                        let mut opt_idx = None;
569                        for &ii in idx.iter().rev() {
570                            // SAFETY: index is always in range.
571                            if validity.get_bit_unchecked(ii as usize) {
572                                opt_idx = Some(ii);
573                                break;
574                            }
575                        }
576                        opt_idx
577                    })
578                    .collect_ca(PlSmallStr::EMPTY)
579            },
580            GroupsType::Slice { groups, .. } => {
581                let mask = BitMask::from_bitmap(&validity);
582                groups
583                    .iter()
584                    .map(|&[first, len]| {
585                        // SAFETY: group slice is valid.
586                        let validity = mask.sliced_unchecked(first as usize, len as usize);
587                        let trailing_zeros = validity.trailing_zeros() as IdxSize;
588                        if trailing_zeros == len {
589                            // All values are null, we have no last non-null.
590                            None
591                        } else {
592                            Some(first + len - trailing_zeros - 1)
593                        }
594                    })
595                    .collect_ca(PlSmallStr::EMPTY)
596            },
597        };
598        // SAFETY: groups are always in bounds.
599        let mut out = s.take_unchecked(&indices);
600        if groups.is_sorted_flag() {
601            out.set_sorted_flag(s.is_sorted_flag())
602        }
603        s.restore_logical(out)
604    }
