27/// Returns a numeric supertype that `l` and `r` can be safely upcasted to if it exists.
28pub fn get_numeric_upcast_supertype_lossless(l: &DataType, r: &DataType) -> Option<DataType> {
29    use DataType::*;
30
31    if l == r || matches!(l, Unknown(_)) || matches!(r, Unknown(_)) {
32        None
33    } else if l.is_float() && r.is_float() {
34        match (l, r) {
35            (Float64, _) | (_, Float64) => Some(Float64),
36            (Float32, _) | (_, Float32) => Some(Float32),
37            v => {
38                // Did we add a new float type?
39                if cfg!(debug_assertions) {
40                    panic!("{v:?}")
41                } else {
42                    None
43                }
44            },
45        }
46    } else if l.is_signed_integer() && r.is_signed_integer() {
47        match (l, r) {
48            (Int128, _) | (_, Int128) => Some(Int128),
49            (Int64, _) | (_, Int64) => Some(Int64),
50            (Int32, _) | (_, Int32) => Some(Int32),
51            (Int16, _) | (_, Int16) => Some(Int16),
52            (Int8, _) | (_, Int8) => Some(Int8),
53            v => {
54                if cfg!(debug_assertions) {
55                    panic!("{v:?}")
56                } else {
57                    None
58                }
59            },
60        }
61    } else if l.is_unsigned_integer() && r.is_unsigned_integer() {
62        match (l, r) {
63            (UInt128, _) | (_, UInt128) => Some(UInt128),
64            (UInt64, _) | (_, UInt64) => Some(UInt64),
65            (UInt32, _) | (_, UInt32) => Some(UInt32),
66            (UInt16, _) | (_, UInt16) => Some(UInt16),
67            (UInt8, _) | (_, UInt8) => Some(UInt8),
68            v => {
69                if cfg!(debug_assertions) {
70                    panic!("{v:?}")
71                } else {
72                    None
73                }
74            },
75        }
76    } else if l.is_integer() && r.is_integer() {
77        // One side is signed, the other is unsigned. We just need to upcast the
78        // unsigned side to a signed integer with the next-largest bit width.
79        match (l, r) {
80            (UInt128, _) | (_, UInt128) => None, // No lossless cast possible
81            (UInt64, _) | (_, UInt64) | (Int128, _) | (_, Int128) => Some(Int128),
82            (UInt32, _) | (_, UInt32) | (Int64, _) | (_, Int64) => Some(Int64),
83            (UInt16, _) | (_, UInt16) | (Int32, _) | (_, Int32) => Some(Int32),
84            (UInt8, _) | (_, UInt8) | (Int16, _) | (_, Int16) => Some(Int16),
85            v => {
86                // One side was UInt and we should have already matched against
87                // all the UInt types
88                if cfg!(debug_assertions) {
89                    panic!("{v:?}")
90                } else {
91                    None
92                }
93            },
94        }
95    } else {
96        None
97    }
98}

100bitflags! {
101    #[repr(transparent)]
102    #[derive(Copy, Clone, PartialEq, Eq, Debug, Hash)]
103    pub struct SuperTypeFlags: u8 {
104        /// Implode lists to match nesting types.
105        const ALLOW_IMPLODE_LIST = 1 << 0;
106        /// Allow casting of primitive types (numeric, bools) to strings
107        const ALLOW_PRIMITIVE_TO_STRING = 1 << 1;
108    }
109}
110
111impl Default for SuperTypeFlags {
112    fn default() -> Self {
113        SuperTypeFlags::from_bits_truncate(0) | SuperTypeFlags::ALLOW_PRIMITIVE_TO_STRING
114    }
115}

117#[derive(Clone, Copy, PartialEq, Eq, Debug, Hash, Default)]
118pub struct SuperTypeOptions {
119    pub flags: SuperTypeFlags,
120}
121
122impl From<SuperTypeFlags> for SuperTypeOptions {
123    fn from(flags: SuperTypeFlags) -> Self {
124        SuperTypeOptions { flags }
125    }
126}
127
128impl SuperTypeOptions {
129    pub fn allow_implode_list(&self) -> bool {
130        self.flags.contains(SuperTypeFlags::ALLOW_IMPLODE_LIST)
131    }
132
133    pub fn allow_primitive_to_string(&self) -> bool {
134        self.flags
135            .contains(SuperTypeFlags::ALLOW_PRIMITIVE_TO_STRING)
136    }
137}

519                let fits = |v| { i128_to_dec128(v, *prec, *scale).is_some() };
520                let fits_orig_prec_scale = match dt {
521                    UInt8 => fits(u8::MAX as i128),
522                    UInt16 => fits(u16::MAX as i128),
523                    UInt32 => fits(u32::MAX as i128),
524                    UInt64 => fits(u64::MAX as i128),
525                    UInt128 => false,
526                    Int8 => fits(i8::MAX as i128),
527                    Int16 => fits(i16::MAX as i128),
528                    Int32 => fits(i32::MAX as i128),
529                    Int64 => fits(i64::MAX as i128),
530                    Int128 => false,
531                    _ => unreachable!(),
532                };
533                if fits_orig_prec_scale {
534                    Some(Decimal(*prec, *scale))
535                } else {
536                    Some(Decimal(DEC128_MAX_PREC, *scale))
537                }
538            }
539            _ => None,
540        }
541    }
542
543    inner(l, r, options).or_else(|| inner(r, l, options))
544}

546/// Given multiple data types, determine the data type that all types can safely be cast to.
547///
548/// Returns [`DataType::Null`] if no data types were passed.
549pub fn dtypes_to_supertype<'a, I>(dtypes: I) -> PolarsResult<DataType>
550where
551    I: IntoIterator<Item = &'a DataType>,
552{
553    dtypes
554        .into_iter()
555        .try_fold(DataType::Null, |supertype, dtype| {
556            try_get_supertype(&supertype, dtype)
557        })
558}
559
560#[cfg(feature = "dtype-struct")]
561fn union_struct_fields(fields_a: &[Field], fields_b: &[Field]) -> Option<DataType> {
562    let (longest, shortest) = {
563        // if equal length we also take the lhs
564        // so that the lhs determines the order of the fields
565        if fields_a.len() >= fields_b.len() {
566            (fields_a, fields_b)
567        } else {
568            (fields_b, fields_a)
569        }
570    };
571    let mut longest_map =
572        PlIndexMap::from_iter(longest.iter().map(|fld| (&fld.name, fld.dtype.clone())));
573    for field in shortest {
574        let dtype_longest = longest_map
575            .entry(&field.name)
576            .or_insert_with(|| field.dtype.clone());
577        if &field.dtype != dtype_longest {
578            let st = get_supertype(&field.dtype, dtype_longest)?;
579            *dtype_longest = st
580        }
581    }
582    let new_fields = longest_map
583        .into_iter()
584        .map(|(name, dtype)| Field::new(name.clone(), dtype))
585        .collect::<Vec<_>>();
586    Some(DataType::Struct(new_fields))
587}
588
589#[cfg(feature = "dtype-struct")]
590fn super_type_structs(fields_a: &[Field], fields_b: &[Field]) -> Option<DataType> {
591    if fields_a.len() != fields_b.len() {
592        union_struct_fields(fields_a, fields_b)
593    } else {
594        let mut new_fields = Vec::with_capacity(fields_a.len());
595        for (a, b) in fields_a.iter().zip(fields_b) {
596            if a.name != b.name {
597                return union_struct_fields(fields_a, fields_b);
598            }
599            let st = get_supertype(&a.dtype, &b.dtype)?;
600            new_fields.push(Field::new(a.name.clone(), st))
601        }
602        Some(DataType::Struct(new_fields))
603    }
604}

606pub fn materialize_dyn_int(v: i128) -> AnyValue<'static> {
607    // Try to get the "smallest" fitting value.
608    // TODO! next breaking go to true smallest.
609    if let Ok(v) = i32::try_from(v) {
610        return AnyValue::Int32(v);
611    }
612    if let Ok(v) = i64::try_from(v) {
613        return AnyValue::Int64(v);
614    }
615    if let Ok(v) = u64::try_from(v) {
616        return AnyValue::UInt64(v);
617    }
618    #[cfg(feature = "dtype-i128")]
619    {
620        AnyValue::Int128(v)
621    }
622
623    #[cfg(not(feature = "dtype-i128"))]
624    AnyValue::Null
625}

627fn materialize_dyn_int_pos(v: i128) -> AnyValue<'static> {
628    // Try to get the "smallest" fitting value.
629    // TODO! next breaking go to true smallest.
630    #[cfg(feature = "dtype-u8")]
631    if let Ok(v) = u8::try_from(v) {
632        return AnyValue::UInt8(v);
633    }
634    #[cfg(feature = "dtype-u16")]
635    if let Ok(v) = u16::try_from(v) {
636        return AnyValue::UInt16(v);
637    }
638    match u32::try_from(v).ok() {
639        Some(v) => AnyValue::UInt32(v),
640        None => match u64::try_from(v).ok() {
641            Some(v) => AnyValue::UInt64(v),
642            None => AnyValue::Null,
643        },
644    }
645}

648    #[cfg(feature = "dtype-i8")]

651    }
652    #[cfg(feature = "dtype-i16")]
653    if let Ok(v) = i16::try_from(v) {
654        return AnyValue::Int16(v);
655    }
656    match i32::try_from(v).ok() {
657        Some(v) => AnyValue::Int32(v),
658        None => match i64::try_from(v).ok() {
659            Some(v) => AnyValue::Int64(v),
660            None => match u64::try_from(v).ok() {
661                Some(v) => AnyValue::UInt64(v),
662                None => AnyValue::Null,
663            },
664        },
665    }

668pub fn merge_dtypes_many<I: IntoIterator<Item = D> + Clone, D: AsRef<DataType>>(
669    into_iter: I,
670) -> PolarsResult<DataType> {
671    let mut iter = into_iter.into_iter();
672
673    let mut st = iter
674        .next()
675        .ok_or_else(|| polars_err!(ComputeError: "expect at least 1 dtype"))
676        .map(|d| d.as_ref().clone())?;
677
678    for d in iter {
679        st = try_get_supertype(d.as_ref(), &st)?;
680    }
681
682    Ok(st)
683}
