27/// Returns a numeric supertype that `l` and `r` can be safely upcasted to if it exists.
28pub fn get_numeric_upcast_supertype_lossless(l: &DataType, r: &DataType) -> Option<DataType> {
29    use DataType::*;
30
31    if l == r || matches!(l, Unknown(_)) || matches!(r, Unknown(_)) {
32        None
33    } else if l.is_float() && r.is_float() {
34        match (l, r) {
35            (Float64, _) | (_, Float64) => Some(Float64),
36            v => {
37                // Did we add a new float type?
38                if cfg!(debug_assertions) {
39                    panic!("{v:?}")
40                } else {
41                    None
42                }
43            },
44        }
45    } else if l.is_signed_integer() && r.is_signed_integer() {
46        match (l, r) {
47            (Int128, _) | (_, Int128) => Some(Int128),
48            (Int64, _) | (_, Int64) => Some(Int64),
49            (Int32, _) | (_, Int32) => Some(Int32),
50            (Int16, _) | (_, Int16) => Some(Int16),
51            (Int8, _) | (_, Int8) => Some(Int8),
52            v => {
53                if cfg!(debug_assertions) {
54                    panic!("{v:?}")
55                } else {
56                    None
57                }
58            },
59        }
60    } else if l.is_unsigned_integer() && r.is_unsigned_integer() {
61        match (l, r) {
62            (UInt128, _) | (_, UInt128) => Some(UInt128),
63            (UInt64, _) | (_, UInt64) => Some(UInt64),
64            (UInt32, _) | (_, UInt32) => Some(UInt32),
65            (UInt16, _) | (_, UInt16) => Some(UInt16),
66            (UInt8, _) | (_, UInt8) => Some(UInt8),
67            v => {
68                if cfg!(debug_assertions) {
69                    panic!("{v:?}")
70                } else {
71                    None
72                }
73            },
74        }
75    } else if l.is_integer() && r.is_integer() {
76        // One side is signed, the other is unsigned. We just need to upcast the
77        // unsigned side to a signed integer with the next-largest bit width.
78        match (l, r) {
79            (UInt128, _) | (_, UInt128) => None, // No lossless cast possible
80            (UInt64, _) | (_, UInt64) | (Int128, _) | (_, Int128) => Some(Int128),
81            (UInt32, _) | (_, UInt32) | (Int64, _) | (_, Int64) => Some(Int64),
82            (UInt16, _) | (_, UInt16) | (Int32, _) | (_, Int32) => Some(Int32),
83            (UInt8, _) | (_, UInt8) | (Int16, _) | (_, Int16) => Some(Int16),
84            v => {
85                // One side was UInt and we should have already matched against
86                // all the UInt types
87                if cfg!(debug_assertions) {
88                    panic!("{v:?}")
89                } else {
90                    None
91                }
92            },
93        }
94    } else {
95        None
96    }
97}

99bitflags! {
100    #[repr(transparent)]
101    #[derive(Copy, Clone, PartialEq, Eq, Debug, Hash)]
102    pub struct SuperTypeFlags: u8 {
103        /// Implode lists to match nesting types.
104        const ALLOW_IMPLODE_LIST = 1 << 0;
105        /// Allow casting of primitive types (numeric, bools) to strings
106        const ALLOW_PRIMITIVE_TO_STRING = 1 << 1;
107    }
108}
109
110impl Default for SuperTypeFlags {
111    fn default() -> Self {
112        SuperTypeFlags::from_bits_truncate(0) | SuperTypeFlags::ALLOW_PRIMITIVE_TO_STRING
113    }

429                    UnknownKind::Int(v) if dt.is_primitive_numeric() => {
430                        // Both dyn int

432                            // Take the maximum value to ensure we bubble up the required minimal size.
433                            Some(Unknown(UnknownKind::Int(std::cmp::max(*v, *v_other))))
434                        }
435                        // dyn int vs number
436                        else {
437                            let smallest_fitting_dtype = if dt.is_unsigned_integer() && !v.is_negative() {
438                                materialize_dyn_int_pos(*v).dtype()
439                            } else {
440                                materialize_smallest_dyn_int(*v).dtype()
441                            };
442                            match dt {
443                                UInt64 if smallest_fitting_dtype.is_signed_integer() => {
444                                    // Ensure we don't cast to float when dealing with dynamic literals
445                                    Some(Int64)
446                                },
447                                _ => {
448                                    get_supertype(dt, &smallest_fitting_dtype)
449                                }
450                            }
451                        }
452                    }
453                    #[cfg(feature = "dtype-decimal")]
454                    UnknownKind::Int(_) if dt.is_decimal() => {
455                        let DataType::Decimal(_prec, scale) = dt else { unreachable!() };
456                        Some(DataType::Decimal(DEC128_MAX_PREC, *scale))
457                    }
458                    _ => Some(Unknown(UnknownKind::Any))
459                }
460            },
461            #[cfg(feature = "dtype-struct")]
462            (Struct(fields_a), Struct(fields_b)) => {
463                super_type_structs(fields_a, fields_b)
464            }
465            #[cfg(feature = "dtype-struct")]
466            (Struct(fields_a), rhs) if rhs.is_primitive_numeric() => {
467                let mut new_fields = Vec::with_capacity(fields_a.len());
468                for a in fields_a {
469                    let st = get_supertype(&a.dtype, rhs)?;
470                    new_fields.push(Field::new(a.name.clone(), st))
471                }
472                Some(Struct(new_fields))
473            }
474            #[cfg(feature = "dtype-decimal")]
475            (Decimal(p1, s1), Decimal(p2, s2)) => {
476                Some(Decimal((*p1).max(*p2), (*s1).max(*s2)))
477            },
478            #[cfg(feature = "dtype-decimal")]
479            (Decimal(_, _), Float32 | Float64) => Some(Float64),
480            #[cfg(feature = "dtype-decimal")]
481            (Decimal(prec, scale), dt) if dt.is_signed_integer() || dt.is_unsigned_integer() => {
482                let fits = |v| { i128_to_dec128(v, *prec, *scale).is_some() };
483                let fits_orig_prec_scale = match dt {
484                    UInt8 => fits(u8::MAX as i128),
485                    UInt16 => fits(u16::MAX as i128),
486                    UInt32 => fits(u32::MAX as i128),
487                    UInt64 => fits(u64::MAX as i128),
488                    UInt128 => false,
489                    Int8 => fits(i8::MAX as i128),
490                    Int16 => fits(i16::MAX as i128),
491                    Int32 => fits(i32::MAX as i128),
492                    Int64 => fits(i64::MAX as i128),
493                    Int128 => false,
494                    _ => unreachable!(),
495                };
496                if fits_orig_prec_scale {
497                    Some(Decimal(*prec, *scale))
498                } else {
499                    Some(Decimal(DEC128_MAX_PREC, *scale))
500                }
501            }
502            _ => None,
503        }
504    }
505
506    inner(l, r, options).or_else(|| inner(r, l, options))
507}

509/// Given multiple data types, determine the data type that all types can safely be cast to.
510///
511/// Returns [`DataType::Null`] if no data types were passed.
512pub fn dtypes_to_supertype<'a, I>(dtypes: I) -> PolarsResult<DataType>
513where
514    I: IntoIterator<Item = &'a DataType>,
515{
516    dtypes
517        .into_iter()
518        .try_fold(DataType::Null, |supertype, dtype| {
519            try_get_supertype(&supertype, dtype)
520        })
521}

523#[cfg(feature = "dtype-struct")]
524fn union_struct_fields(fields_a: &[Field], fields_b: &[Field]) -> Option<DataType> {
525    let (longest, shortest) = {
526        // if equal length we also take the lhs
527        // so that the lhs determines the order of the fields
528        if fields_a.len() >= fields_b.len() {
529            (fields_a, fields_b)
530        } else {
531            (fields_b, fields_a)
532        }
533    };
534    let mut longest_map =
535        PlIndexMap::from_iter(longest.iter().map(|fld| (&fld.name, fld.dtype.clone())));
536    for field in shortest {
537        let dtype_longest = longest_map
538            .entry(&field.name)
539            .or_insert_with(|| field.dtype.clone());
540        if &field.dtype != dtype_longest {
541            let st = get_supertype(&field.dtype, dtype_longest)?;
542            *dtype_longest = st
543        }
544    }
545    let new_fields = longest_map
546        .into_iter()
547        .map(|(name, dtype)| Field::new(name.clone(), dtype))
548        .collect::<Vec<_>>();
549    Some(DataType::Struct(new_fields))
550}
551
552#[cfg(feature = "dtype-struct")]
553fn super_type_structs(fields_a: &[Field], fields_b: &[Field]) -> Option<DataType> {
554    if fields_a.len() != fields_b.len() {
555        union_struct_fields(fields_a, fields_b)
556    } else {
557        let mut new_fields = Vec::with_capacity(fields_a.len());
558        for (a, b) in fields_a.iter().zip(fields_b) {
559            if a.name != b.name {
560                return union_struct_fields(fields_a, fields_b);
561            }
562            let st = get_supertype(&a.dtype, &b.dtype)?;
563            new_fields.push(Field::new(a.name.clone(), st))
564        }
565        Some(DataType::Struct(new_fields))
566    }
567}

569pub fn materialize_dyn_int(v: i128) -> AnyValue<'static> {
570    // Try to get the "smallest" fitting value.
571    // TODO! next breaking go to true smallest.
572    if let Ok(v) = i32::try_from(v) {
573        return AnyValue::Int32(v);
574    }
575    if let Ok(v) = i64::try_from(v) {
576        return AnyValue::Int64(v);
577    }
578    if let Ok(v) = u64::try_from(v) {
579        return AnyValue::UInt64(v);
580    }
581    #[cfg(feature = "dtype-i128")]
582    {
583        AnyValue::Int128(v)
584    }
585
586    #[cfg(not(feature = "dtype-i128"))]
587    AnyValue::Null
588}

590fn materialize_dyn_int_pos(v: i128) -> AnyValue<'static> {
591    // Try to get the "smallest" fitting value.
592    // TODO! next breaking go to true smallest.
593    #[cfg(feature = "dtype-u8")]
594    if let Ok(v) = u8::try_from(v) {
595        return AnyValue::UInt8(v);
596    }
597    #[cfg(feature = "dtype-u16")]
598    if let Ok(v) = u16::try_from(v) {
599        return AnyValue::UInt16(v);
600    }
601    match u32::try_from(v).ok() {
602        Some(v) => AnyValue::UInt32(v),
603        None => match u64::try_from(v).ok() {
604            Some(v) => AnyValue::UInt64(v),
605            None => AnyValue::Null,
606        },
607    }
608}

631pub fn merge_dtypes_many<I: IntoIterator<Item = D> + Clone, D: AsRef<DataType>>(
632    into_iter: I,
633) -> PolarsResult<DataType> {
634    let mut iter = into_iter.into_iter();
635
636    let mut st = iter
637        .next()
638        .ok_or_else(|| polars_err!(ComputeError: "expect at least 1 dtype"))
639        .map(|d| d.as_ref().clone())?;
640
641    for d in iter {
642        st = try_get_supertype(d.as_ref(), &st)?;
643    }
644
645    Ok(st)
646}
