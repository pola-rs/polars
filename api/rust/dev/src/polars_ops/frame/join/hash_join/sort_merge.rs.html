10#[cfg(feature = "performant")]
11fn par_sorted_merge_left_impl<T>(
12    s_left: &ChunkedArray<T>,
13    s_right: &ChunkedArray<T>,
14) -> (Vec<IdxSize>, Vec<NullableIdxSize>)
15where
16    T: PolarsNumericType,
17{
18    let offsets = _split_offsets(s_left.len(), POOL.current_num_threads());
19    let s_left = s_left.rechunk();
20    let s_right = s_right.rechunk();
21
22    // we can unwrap because we should not have nulls
23    let slice_left = s_left.cont_slice().unwrap();
24    let slice_right = s_right.cont_slice().unwrap();
25
26    let indexes = offsets.into_par_iter().map(|(offset, len)| {
27        let slice_left = &slice_left[offset..offset + len];
28        sorted_join::left::join(slice_left, slice_right, offset as IdxSize)
29    });
30    let indexes = POOL.install(|| indexes.collect::<Vec<_>>());
31
32    let lefts = indexes.iter().map(|t| &t.0).collect::<Vec<_>>();
33    let rights = indexes.iter().map(|t| &t.1).collect::<Vec<_>>();
34
35    (flatten_par(&lefts), flatten_par(&rights))
36}

39pub(super) fn par_sorted_merge_left(
40    s_left: &Series,
41    s_right: &Series,
42) -> (Vec<IdxSize>, Vec<NullableIdxSize>) {
43    // Don't use bit_repr here. It messes up sortedness.
44    debug_assert_eq!(s_left.dtype(), s_right.dtype());
45    let s_left = s_left.to_physical_repr();
46    let s_right = s_right.to_physical_repr();
47
48    match s_left.dtype() {
49        #[cfg(feature = "dtype-i8")]
50        DataType::Int8 => par_sorted_merge_left_impl(s_left.i8().unwrap(), s_right.i8().unwrap()),
51        #[cfg(feature = "dtype-u8")]
52        DataType::UInt8 => par_sorted_merge_left_impl(s_left.u8().unwrap(), s_right.u8().unwrap()),
53        #[cfg(feature = "dtype-u16")]
54        DataType::UInt16 => {
55            par_sorted_merge_left_impl(s_left.u16().unwrap(), s_right.u16().unwrap())
56        },
57        #[cfg(feature = "dtype-i16")]
58        DataType::Int16 => {
59            par_sorted_merge_left_impl(s_left.i16().unwrap(), s_right.i16().unwrap())
60        },
61        DataType::UInt32 => {
62            par_sorted_merge_left_impl(s_left.u32().unwrap(), s_right.u32().unwrap())
63        },
64        DataType::Int32 => {
65            par_sorted_merge_left_impl(s_left.i32().unwrap(), s_right.i32().unwrap())
66        },
67        DataType::UInt64 => {
68            par_sorted_merge_left_impl(s_left.u64().unwrap(), s_right.u64().unwrap())
69        },
70        DataType::Int64 => {
71            par_sorted_merge_left_impl(s_left.i64().unwrap(), s_right.i64().unwrap())
72        },
73        #[cfg(feature = "dtype-u128")]
74        DataType::UInt128 => {
75            par_sorted_merge_left_impl(s_left.u128().unwrap(), s_right.u128().unwrap())
76        },
77        #[cfg(feature = "dtype-i128")]
78        DataType::Int128 => {
79            par_sorted_merge_left_impl(s_left.i128().unwrap(), s_right.i128().unwrap())
80        },
81        DataType::Float32 => {
82            par_sorted_merge_left_impl(s_left.f32().unwrap(), s_right.f32().unwrap())

203#[cfg(feature = "performant")]
204pub(crate) fn _sort_or_hash_inner(
205    s_left: &Series,
206    s_right: &Series,
207    verbose: bool,
208    validate: JoinValidation,
209    nulls_equal: bool,
210) -> PolarsResult<(InnerJoinIds, bool)> {
211    // We check if keys are sorted.
212    // - If they are we can do a sorted merge join
213    // If one of the keys is not, it can still be faster to sort that key and use
214    // the `arg_sort` indices to revert the sort once the join keys are determined.
215    let size_factor_rhs = s_right.len() as f32 / s_left.len() as f32;
216    let size_factor_lhs = s_left.len() as f32 / s_right.len() as f32;
217    let size_factor_acceptable = std::env::var("POLARS_JOIN_SORT_FACTOR")
218        .map(|s| s.parse::<f32>().unwrap())
219        .unwrap_or(1.0);
220    let is_numeric = s_left.dtype().to_physical().is_primitive_numeric();
221
222    if validate.needs_checks() {
223        return s_left.hash_join_inner(s_right, validate, nulls_equal);
224    }
225
226    let no_nulls = s_left.null_count() == 0 && s_right.null_count() == 0;
227    match (s_left.is_sorted_flag(), s_right.is_sorted_flag(), no_nulls) {
228        (IsSorted::Ascending, IsSorted::Ascending, true) if is_numeric => {
229            if verbose {
230                eprintln!("inner join: keys are sorted: use sorted merge join");
231            }
232            Ok((par_sorted_merge_inner_no_nulls(s_left, s_right), true))
233        },
234        (IsSorted::Ascending, _, true)
235            if is_numeric && size_factor_rhs < size_factor_acceptable =>
236        {
237            if verbose {
238                eprintln!("right key will be descending sorted in inner join operation.")
239            }
240
241            let sort_idx = s_right.arg_sort(SortOptions {
242                descending: false,
243                nulls_last: false,
244                multithreaded: true,
245                maintain_order: false,
246                limit: None,
247            });
248            let s_right = unsafe { s_right.take_unchecked(&sort_idx) };
249            let ids = par_sorted_merge_inner_no_nulls(s_left, &s_right);
250            let reverse_idx_map = create_reverse_map_from_arg_sort(sort_idx);
251
252            let (left, mut right) = ids;
253
254            POOL.install(|| {
255                right.par_iter_mut().for_each(|idx| {
256                    *idx = unsafe { *reverse_idx_map.get_unchecked(*idx as usize) };
257                });
258            });
259
260            Ok(((left, right), true))
261        },
262        (_, IsSorted::Ascending, true)
263            if is_numeric && size_factor_lhs < size_factor_acceptable =>
264        {
265            if verbose {
266                eprintln!("left key will be descending sorted in inner join operation.")
267            }
268
269            let sort_idx = s_left.arg_sort(SortOptions {
270                descending: false,
271                nulls_last: false,
272                multithreaded: true,
273                maintain_order: false,
274                limit: None,
275            });
276            let s_left = unsafe { s_left.take_unchecked(&sort_idx) };
277            let ids = par_sorted_merge_inner_no_nulls(&s_left, s_right);
278            let reverse_idx_map = create_reverse_map_from_arg_sort(sort_idx);
279
280            let (mut left, right) = ids;
281
282            POOL.install(|| {
283                left.par_iter_mut().for_each(|idx| {
284                    *idx = unsafe { *reverse_idx_map.get_unchecked(*idx as usize) };
285                });
286            });
287
288            // set sorted to `false` as we descending sorted the left key.
289            Ok(((left, right), false))
290        },
291        _ => s_left.hash_join_inner(s_right, validate, nulls_equal),
292    }
293}

306#[cfg(feature = "performant")]
307pub(crate) fn sort_or_hash_left(
308    s_left: &Series,
309    s_right: &Series,
310    verbose: bool,
311    validate: JoinValidation,
312    nulls_equal: bool,
313) -> PolarsResult<LeftJoinIds> {
314    if validate.needs_checks() {
315        return s_left.hash_join_left(s_right, validate, nulls_equal);
316    }
317
318    let size_factor_rhs = s_right.len() as f32 / s_left.len() as f32;
319    let size_factor_acceptable = std::env::var("POLARS_JOIN_SORT_FACTOR")
320        .map(|s| s.parse::<f32>().unwrap())
321        .unwrap_or(1.0);
322    let is_numeric = s_left.dtype().to_physical().is_primitive_numeric();
323
324    let no_nulls = s_left.null_count() == 0 && s_right.null_count() == 0;
325
326    match (s_left.is_sorted_flag(), s_right.is_sorted_flag(), no_nulls) {
327        (IsSorted::Ascending, IsSorted::Ascending, true) if is_numeric => {
328            if verbose {
329                eprintln!("left join: keys are sorted: use sorted merge join");
330            }
331            let (left_idx, right_idx) = par_sorted_merge_left(s_left, s_right);
332            Ok(to_left_join_ids(left_idx, right_idx))
333        },
334        (IsSorted::Ascending, _, true)
335            if is_numeric && size_factor_rhs < size_factor_acceptable =>
336        {
337            if verbose {
338                eprintln!("right key will be reverse sorted in left join operation.")
339            }
340
341            let sort_idx = s_right.arg_sort(SortOptions {
342                descending: false,
343                nulls_last: false,
344                multithreaded: true,
345                maintain_order: false,
346                limit: None,
347            });
348            let s_right = unsafe { s_right.take_unchecked(&sort_idx) };
349
350            let ids = par_sorted_merge_left(s_left, &s_right);
351            let reverse_idx_map = create_reverse_map_from_arg_sort(sort_idx);
352            let (left, mut right) = ids;
353
354            POOL.install(|| {
355                right.par_iter_mut().for_each(|opt_idx| {
356                    if !opt_idx.is_null_idx() {
357                        *opt_idx =
358                            unsafe { *reverse_idx_map.get_unchecked(opt_idx.idx() as usize) }
359                                .into();
360                    }
361                });
362            });
363
364            Ok(to_left_join_ids(left, right))
365        },
366        // don't reverse sort a left join key yet. Have to figure out how to set sorted flag
367        _ => s_left.hash_join_left(s_right, validate, nulls_equal),
368    }
