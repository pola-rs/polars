13fn sum_between_offsets<T, S>(values: &[T], offset: &[i64]) -> Vec<S>
14where
15    T: NativeType + ToPrimitive,
16    S: NumCast + std::iter::Sum,
17{
18    offset
19        .windows(2)
20        .map(|w| {
21            values
22                .get(w[0] as usize..w[1] as usize)
23                .map(sum_slice)
24                .unwrap_or(S::from(0).unwrap())
25        })
26        .collect()
27}

29fn dispatch_sum<T, S>(arr: &dyn Array, offsets: &[i64], validity: Option<&Bitmap>) -> ArrayRef
30where
31    T: NativeType + ToPrimitive,
32    S: NativeType + NumCast + std::iter::Sum,
33{
34    let values = arr.as_any().downcast_ref::<PrimitiveArray<T>>().unwrap();
35    let values = values.values().as_slice();
36    Box::new(PrimitiveArray::from_data_default(
37        sum_between_offsets::<_, S>(values, offsets).into(),
38        validity.cloned(),
39    )) as ArrayRef
40}

42pub(super) fn sum_list_numerical(ca: &ListChunked, inner_type: &DataType) -> Series {
43    use DataType::*;
44    let chunks = ca
45        .downcast_iter()
46        .map(|arr| {
47            let offsets = arr.offsets().as_slice();
48            let values = arr.values().as_ref();
49
50            match inner_type {
51                Int8 => dispatch_sum::<i8, i64>(values, offsets, arr.validity()),
52                Int16 => dispatch_sum::<i16, i64>(values, offsets, arr.validity()),
53                Int32 => dispatch_sum::<i32, i32>(values, offsets, arr.validity()),
54                Int64 => dispatch_sum::<i64, i64>(values, offsets, arr.validity()),
55                Int128 => dispatch_sum::<i128, i128>(values, offsets, arr.validity()),
56                UInt8 => dispatch_sum::<u8, i64>(values, offsets, arr.validity()),
57                UInt16 => dispatch_sum::<u16, i64>(values, offsets, arr.validity()),
58                UInt32 => dispatch_sum::<u32, u32>(values, offsets, arr.validity()),
59                UInt64 => dispatch_sum::<u64, u64>(values, offsets, arr.validity()),
60                UInt128 => dispatch_sum::<u128, u128>(values, offsets, arr.validity()),
61                Float32 => dispatch_sum::<f32, f32>(values, offsets, arr.validity()),
62                Float64 => dispatch_sum::<f64, f64>(values, offsets, arr.validity()),
63                _ => unimplemented!(),
64            }
65        })
66        .collect::<Vec<_>>();
67
68    Series::try_from((ca.name().clone(), chunks)).unwrap()
69}

71pub(super) fn sum_with_nulls(ca: &ListChunked, inner_dtype: &DataType) -> PolarsResult<Series> {
72    use DataType::*;
73    let mut out = match inner_dtype {
74        Boolean => {
75            let out: IdxCa =
76                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<IdxSize>().unwrap()));
77            out.into_series()
78        },
79        UInt8 => {
80            let out: Int64Chunked =
81                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i64>().unwrap()));
82            out.into_series()
83        },
84        UInt16 => {
85            let out: Int64Chunked =
86                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i64>().unwrap()));
87            out.into_series()
88        },
89        UInt32 => {
90            let out: UInt32Chunked =
91                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<u32>().unwrap()));
92            out.into_series()
93        },
94        UInt64 => {
95            let out: UInt64Chunked =
96                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<u64>().unwrap()));
97            out.into_series()
98        },
99        Int8 => {
100            let out: Int64Chunked =
101                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i64>().unwrap()));
102            out.into_series()
103        },
104        Int16 => {
105            let out: Int64Chunked =
106                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i64>().unwrap()));
107            out.into_series()
108        },
109        Int32 => {
110            let out: Int32Chunked =
111                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i32>().unwrap()));
112            out.into_series()
113        },
114        Int64 => {
115            let out: Int64Chunked =
116                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i64>().unwrap()));
117            out.into_series()
118        },
119        Float32 => {
120            let out: Float32Chunked =
121                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<f32>().unwrap()));
122            out.into_series()
123        },
124        Float64 => {
125            let out: Float64Chunked =
126                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<f64>().unwrap()));
127            out.into_series()
128        },
129        // slowest sum_as_series path
130        dt => ca
131            .try_apply_amortized(|s| {
132                s.as_ref()
133                    .sum_reduce()
134                    .map(|sc| sc.into_series(PlSmallStr::EMPTY))
135            })?
136            .explode(false)
137            .unwrap()
138            .into_series()
139            .cast(dt)?,
140    };
141    out.rename(ca.name().clone());
142    Ok(out)
143}

145fn mean_between_offsets<T, S>(values: &[T], offset: &[i64]) -> PrimitiveArray<S>
146where
147    T: NativeType + ToPrimitive,
148    S: NativeType + NumCast + std::iter::Sum + Div<Output = S>,
149{
150    offset
151        .windows(2)
152        .map(|w| {
153            values
154                .get(w[0] as usize..w[1] as usize)
155                .filter(|sl| !sl.is_empty())
156                .map(|sl| sum_slice::<_, S>(sl) / NumCast::from(sl.len()).unwrap())
157        })
158        .collect()
159}

161fn dispatch_mean<T, S>(arr: &dyn Array, offsets: &[i64], validity: Option<&Bitmap>) -> ArrayRef
162where
163    T: NativeType + ToPrimitive,
164    S: NativeType + NumCast + std::iter::Sum + Div<Output = S>,
165{
166    let values = arr.as_any().downcast_ref::<PrimitiveArray<T>>().unwrap();
167    let values = values.values().as_slice();
168    let out = mean_between_offsets::<_, S>(values, offsets);
169    let new_validity = combine_validities_and(out.validity(), validity);
170    out.with_validity(new_validity).to_boxed()
171}

202pub(super) fn mean_with_nulls(ca: &ListChunked) -> Series {
203    match ca.inner_dtype() {
204        DataType::Float32 => {
205            let out: Float32Chunked = ca
206                .apply_amortized_generic(|s| s.and_then(|s| s.as_ref().mean().map(|v| v as f32)))
207                .with_name(ca.name().clone());
208            out.into_series()
209        },
210        #[cfg(feature = "dtype-datetime")]
211        DataType::Date => {
212            let out: Int64Chunked = ca
213                .apply_amortized_generic(|s| {
214                    s.and_then(|s| s.as_ref().mean().map(|v| (v * (US_IN_DAY as f64)) as i64))
215                })
216                .with_name(ca.name().clone());
217            out.into_datetime(TimeUnit::Microseconds, None)
218                .into_series()
219        },
220        dt if dt.is_temporal() => {
221            let out: Int64Chunked = ca
222                .apply_amortized_generic(|s| s.and_then(|s| s.as_ref().mean().map(|v| v as i64)))
223                .with_name(ca.name().clone());
224            out.cast(dt).unwrap()
225        },
226        _ => {
227            let out: Float64Chunked = ca
228                .apply_amortized_generic(|s| s.and_then(|s| s.as_ref().mean()))
229                .with_name(ca.name().clone());
230            out.into_series()
231        },
232    }
