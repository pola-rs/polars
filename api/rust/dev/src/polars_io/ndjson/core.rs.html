195    reader_bytes: Option<ReaderBytes<'a>>,
196    n_rows: Option<usize>,
197    schema: SchemaRef,
198    n_threads: Option<usize>,

207impl<'a> CoreJsonReader<'a> {
208    #[allow(clippy::too_many_arguments)]
209    pub(crate) fn new(
210        reader_bytes: ReaderBytes<'a>,
211        n_rows: Option<usize>,
212        schema: Option<SchemaRef>,
213        schema_overwrite: Option<&Schema>,
214        n_threads: Option<usize>,
215        sample_size: usize,
216        chunk_size: NonZeroUsize,
217        low_memory: bool,
218        infer_schema_len: Option<NonZeroUsize>,
219        ignore_errors: bool,
220        row_index: Option<&'a mut RowIndex>,
221        predicate: Option<Arc<dyn PhysicalIoExpr>>,
222        projection: Option<Arc<[PlSmallStr]>>,
223    ) -> PolarsResult<CoreJsonReader<'a>> {
224        let reader_bytes = reader_bytes;
225
226        let mut schema = match schema {
227            Some(schema) => schema,
228            None => {
229                let bytes: &[u8] = &reader_bytes;
230                let mut cursor = Cursor::new(bytes);
231                Arc::new(crate::ndjson::infer_schema(&mut cursor, infer_schema_len)?)
232            },
233        };
234        if let Some(overwriting_schema) = schema_overwrite {
235            let schema = Arc::make_mut(&mut schema);
236            overwrite_schema(schema, overwriting_schema)?;
237        }
238
239        Ok(CoreJsonReader {
240            reader_bytes: Some(reader_bytes),
241            schema,
242            sample_size,
243            n_rows,
244            n_threads,
245            chunk_size,
246            low_memory,
247            ignore_errors,
248            row_index,
249            predicate,
250            projection,
251        })
252    }
253
254    fn count(mut self) -> PolarsResult<usize> {
255        let bytes = self.reader_bytes.take().unwrap();
256        Ok(super::count_rows_par(&bytes, self.n_threads))
257    }
258
259    fn parse_json(&mut self, mut n_threads: usize, bytes: &[u8]) -> PolarsResult<DataFrame> {
260        let mut bytes = bytes;
261        let mut total_rows = 128;
262
263        if let Some((mean, std)) = get_line_stats_json(bytes, self.sample_size) {
264            let line_length_upper_bound = mean + 1.1 * std;
265
266            total_rows = (bytes.len() as f32 / (mean - 0.01 * std)) as usize;
267            if let Some(n_rows) = self.n_rows {
268                total_rows = std::cmp::min(n_rows, total_rows);
269                // the guessed upper bound of  the no. of bytes in the file
270                let n_bytes = (line_length_upper_bound * (n_rows as f32)) as usize;
271
272                if n_bytes < bytes.len() {
273                    if let Some(pos) = next_line_position_naive_json(&bytes[n_bytes..]) {
274                        bytes = &bytes[..n_bytes + pos]
275                    }
276                }
277            }
278        }
279
280        if total_rows <= 128 {
281            n_threads = 1;
282        }
283
284        let rows_per_thread = total_rows / n_threads;
285
286        let max_proxy = bytes.len() / n_threads / 2;
287        let capacity = if self.low_memory {
288            usize::from(self.chunk_size)
289        } else {
290            std::cmp::min(rows_per_thread, max_proxy)
291        };
292        let file_chunks = get_file_chunks_json(bytes, n_threads);
293
294        let row_index = self.row_index.as_ref().map(|ri| ri as &RowIndex);
295        let (mut dfs, prepredicate_heights) = POOL.install(|| {
296            file_chunks
297                .into_par_iter()
298                .map(|(start_pos, stop_at_nbytes)| {
299                    let mut local_df = parse_ndjson(
300                        &bytes[start_pos..stop_at_nbytes],
301                        Some(capacity),
302                        &self.schema,
303                        self.ignore_errors,
304                    )?;
305
306                    let prepredicate_height = local_df.height() as IdxSize;
307                    if let Some(projection) = self.projection.as_deref() {
308                        local_df = local_df.select(projection.iter().cloned())?;
309                    }
310
311                    if let Some(row_index) = row_index {
312                        local_df = local_df
313                            .with_row_index(row_index.name.clone(), Some(row_index.offset))?;
314                    }
315
316                    if let Some(predicate) = &self.predicate {
317                        let s = predicate.evaluate_io(&local_df)?;
318                        let mask = s.bool()?;
319                        local_df = local_df.filter(mask)?;
320                    }
321
322                    Ok((local_df, prepredicate_height))
323                })
324                .collect::<PolarsResult<(Vec<_>, Vec<_>)>>()
325        })?;
326
327        if let Some(ref mut row_index) = self.row_index {
328            update_row_counts3(&mut dfs, &prepredicate_heights, 0);
329            row_index.offset += prepredicate_heights.iter().copied().sum::<IdxSize>();
330        }
331
332        accumulate_dataframes_vertical(dfs)
333    }
334
335    pub fn as_df(&mut self) -> PolarsResult<DataFrame> {
336        let n_threads = self.n_threads.unwrap_or_else(|| POOL.current_num_threads());
337
338        let reader_bytes = self.reader_bytes.take().unwrap();
339
340        let mut df = self.parse_json(n_threads, &reader_bytes)?;
341
342        // if multi-threaded the n_rows was probabilistically determined.
343        // Let's slice to correct number of rows if possible.
344        if let Some(n_rows) = self.n_rows {
345            if n_rows < df.height() {
346                df = df.slice(0, n_rows)
347            }
348        }
349        Ok(df)
350    }
351}

353#[inline(always)]
354fn parse_impl(
355    bytes: &[u8],
356    buffers: &mut PlIndexMap<BufferKey, Buffer>,
357    scratch: &mut Scratch,
358    ignore_errors: bool,
359) -> PolarsResult<usize> {
360    scratch.json.clear();
361    scratch.json.extend_from_slice(bytes);
362    let n = scratch.json.len();
363    let value = simd_json::to_borrowed_value_with_buffers(&mut scratch.json, &mut scratch.buffers)
364        .map_err(|e| polars_err!(ComputeError: "error parsing line: {}", e))?;
365    match value {
366        simd_json::BorrowedValue::Object(value) => {
367            buffers.iter_mut().try_for_each(|(s, inner)| {
368                match s.0.map_lookup(&value) {
369                    Some(v) => inner.add(v)?,
370                    None => inner.add_null(),
371                }
372                PolarsResult::Ok(())
373            })?;
374        },
375        _ if ignore_errors => {
376            buffers.iter_mut().for_each(|(_, inner)| inner.add_null());
377        },
378        v => {
379            polars_bail!(ComputeError: "NDJSON line expected to contain JSON object: {v}");
380        },
381    };
382    Ok(n)
383}

391pub fn json_lines(bytes: &[u8]) -> impl Iterator<Item = &[u8]> {
392    // This previously used `serde_json`'s `RawValue` to deserialize chunks without really deserializing them.
393    // However, this convenience comes at a cost. serde_json allocates and parses and does UTF-8 validation, all
394    // things we don't need since we use simd_json for them. Also, `serde_json::StreamDeserializer` has a more
395    // ambitious goal: it wants to parse potentially *non-delimited* sequences of JSON values, while we know
396    // our values are line-delimited. Turns out, custom splitting is very easy, and gives a very nice performance boost.
397    bytes.split(|&byte| byte == b'\n').filter(|&bytes| {
398        bytes
399            .iter()
400            .any(|&byte| !matches!(byte, b' ' | b'\t' | b'\r'))
401    })
402}
403
404fn parse_lines(
405    bytes: &[u8],
406    buffers: &mut PlIndexMap<BufferKey, Buffer>,
407    ignore_errors: bool,
408) -> PolarsResult<()> {
409    let mut scratch = Scratch::default();
410
411    let iter = json_lines(bytes);
412    for bytes in iter {
413        parse_impl(bytes, buffers, &mut scratch, ignore_errors)?;
414    }
415    Ok(())
416}

418pub fn parse_ndjson(
419    bytes: &[u8],
420    n_rows_hint: Option<usize>,
421    schema: &Schema,
422    ignore_errors: bool,
423) -> PolarsResult<DataFrame> {
424    let capacity = n_rows_hint.unwrap_or_else(|| estimate_n_lines_in_chunk(bytes));
425
426    let mut buffers = init_buffers(schema, capacity, ignore_errors)?;
427    parse_lines(bytes, &mut buffers, ignore_errors)?;
428
429    DataFrame::new(
430        buffers
431            .into_values()
432            .map(|buf| Ok(buf.into_series()?.into_column()))
433            .collect::<PolarsResult<_>>()
434            .map_err(|e| match e {
435                // Nested types raise SchemaMismatch instead of ComputeError, we map it back here to
436                // be consistent.
437                PolarsError::ComputeError(..) => e,
438                PolarsError::SchemaMismatch(e) => PolarsError::ComputeError(e),
439                e => e,
440            })?,
441    )
442}

444pub fn estimate_n_lines_in_file(file_bytes: &[u8], sample_size: usize) -> usize {
445    if let Some((mean, std)) = get_line_stats_json(file_bytes, sample_size) {
446        (file_bytes.len() as f32 / (mean - 0.01 * std)) as usize
447    } else {
448        estimate_n_lines_in_chunk(file_bytes)
449    }
450}

452/// Total len divided by max len of first and last non-empty lines. This is intended to be cheaper
453/// than `estimate_n_lines_in_file`.
454pub fn estimate_n_lines_in_chunk(chunk: &[u8]) -> usize {
455    chunk
456        .split(|&c| c == b'\n')
457        .find(|x| !x.is_empty())
458        .map_or(1, |x| {
459            chunk.len().div_ceil(
460                x.len().max(
461                    chunk
462                        .rsplit(|&c| c == b'\n')
463                        .find(|x| !x.is_empty())
464                        .unwrap()
465                        .len(),
466                ),
467            )
468        })
469}

471/// Find the nearest next line position.
472/// Does not check for new line characters embedded in String fields.
473/// This just looks for `}\n`
474pub(crate) fn next_line_position_naive_json(input: &[u8]) -> Option<usize> {
475    let pos = memchr::memchr(NEWLINE, input)?;
476    if pos == 0 {
477        return Some(1);
478    }
479
480    let is_closing_bracket = input.get(pos - 1) == Some(&CLOSING_BRACKET);
481    if is_closing_bracket {
482        Some(pos + 1)
483    } else {
484        None
485    }
486}

488/// Get the mean and standard deviation of length of lines in bytes
489pub(crate) fn get_line_stats_json(bytes: &[u8], n_lines: usize) -> Option<(f32, f32)> {
490    let mut lengths = Vec::with_capacity(n_lines);
491
492    let mut bytes_trunc;
493    let n_lines_per_iter = n_lines / 2;
494
495    let mut n_read = 0;
496
497    let bytes_len = bytes.len();
498
499    // sample from start and 75% in the file
500    for offset in [0, (bytes_len as f32 * 0.75) as usize] {
501        bytes_trunc = &bytes[offset..];
502        let pos = next_line_position_naive_json(bytes_trunc)?;
503        if pos >= bytes_len {
504            return None;
505        }
506        bytes_trunc = &bytes_trunc[pos + 1..];
507
508        for _ in offset..(offset + n_lines_per_iter) {
509            let pos = next_line_position_naive_json(bytes_trunc);
510            if let Some(pos) = pos {
511                lengths.push(pos);
512                let next_bytes = &bytes_trunc[pos..];
513                if next_bytes.is_empty() {
514                    return None;
515                }
516                bytes_trunc = next_bytes;
517                n_read += pos;
518            } else {
519                break;
520            }
521        }
522    }
523
524    let n_samples = lengths.len();
525    let mean = (n_read as f32) / (n_samples as f32);
526    let mut std = 0.0;
527    for &len in lengths.iter() {
528        std += (len as f32 - mean).pow(2.0)
529    }
530    std = (std / n_samples as f32).sqrt();
531    Some((mean, std))
532}

534pub(crate) fn get_file_chunks_json(bytes: &[u8], n_threads: usize) -> Vec<(usize, usize)> {
535    let mut last_pos = 0;
536    let total_len = bytes.len();
537    let chunk_size = total_len / n_threads;
538    let mut offsets = Vec::with_capacity(n_threads);
539    for _ in 0..n_threads {
540        let search_pos = last_pos + chunk_size;
541
542        if search_pos >= bytes.len() {
543            break;
544        }
545
546        let end_pos = match next_line_position_naive_json(&bytes[search_pos..]) {
547            Some(pos) => search_pos + pos,
548            None => {
549                break;
550            },
551        };
552        offsets.push((last_pos, end_pos));
553        last_pos = end_pos;
554    }
555    offsets.push((last_pos, total_len));
556    offsets
