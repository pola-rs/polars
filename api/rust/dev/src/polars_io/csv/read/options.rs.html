344#[derive(Clone, Debug, Eq, PartialEq, Hash)]
345#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
346#[cfg_attr(feature = "dsl-schema", derive(schemars::JsonSchema))]
347pub enum CommentPrefix {
348    /// A single byte character that indicates the start of a comment line.
349    Single(u8),
350    /// A string that indicates the start of a comment line.
351    /// This allows for multiple characters to be used as a comment identifier.
352    Multi(PlSmallStr),
353}
354
355impl CommentPrefix {
356    /// Creates a new `CommentPrefix` for the `Single` variant.
357    pub fn new_single(prefix: u8) -> Self {
358        CommentPrefix::Single(prefix)
359    }
360
361    /// Creates a new `CommentPrefix` for the `Multi` variant.
362    pub fn new_multi(prefix: PlSmallStr) -> Self {
363        CommentPrefix::Multi(prefix)
364    }
365
366    /// Creates a new `CommentPrefix` from a `&str`.
367    pub fn new_from_str(prefix: &str) -> Self {
368        assert!(!prefix.contains("\n"));
369        if prefix.len() == 1 && prefix.chars().next().unwrap().is_ascii() {
370            let c = prefix.as_bytes()[0];
371            CommentPrefix::Single(c)
372        } else {
373            CommentPrefix::Multi(PlSmallStr::from_str(prefix))
374        }
375    }
376}
377
378impl From<&str> for CommentPrefix {
379    fn from(value: &str) -> Self {
380        Self::new_from_str(value)
381    }
382}

384#[derive(Clone, Debug, Eq, PartialEq, Hash)]
385#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
386#[cfg_attr(feature = "dsl-schema", derive(schemars::JsonSchema))]
387pub enum NullValues {
388    /// A single value that's used for all columns
389    AllColumnsSingle(PlSmallStr),
390    /// Multiple values that are used for all columns
391    AllColumns(Vec<PlSmallStr>),
392    /// Tuples that map column names to null value of that column
393    Named(Vec<(PlSmallStr, PlSmallStr)>),
394}
395
396impl NullValues {
397    pub fn compile(self, schema: &Schema) -> PolarsResult<NullValuesCompiled> {
398        Ok(match self {
399            NullValues::AllColumnsSingle(v) => NullValuesCompiled::AllColumnsSingle(v),
400            NullValues::AllColumns(v) => NullValuesCompiled::AllColumns(v),
401            NullValues::Named(v) => {
402                let mut null_values = vec![PlSmallStr::from_static(""); schema.len()];
403                for (name, null_value) in v {
404                    let i = schema.try_index_of(&name)?;
405                    null_values[i] = null_value;
406                }
407                NullValuesCompiled::Columns(null_values)
408            },
409        })
410    }
411}
412
413#[derive(Debug, Clone)]
414pub enum NullValuesCompiled {
415    /// A single value that's used for all columns
416    AllColumnsSingle(PlSmallStr),
417    // Multiple null values that are null for all columns
418    AllColumns(Vec<PlSmallStr>),
419    /// A different null value per column, computed from `NullValues::Named`
420    Columns(Vec<PlSmallStr>),
421}
422
423impl NullValuesCompiled {
424    /// # Safety
425    ///
426    /// The caller must ensure that `index` is in bounds
427    pub(super) unsafe fn is_null(&self, field: &[u8], index: usize) -> bool {
428        use NullValuesCompiled::*;
429        match self {
430            AllColumnsSingle(v) => v.as_bytes() == field,
431            AllColumns(v) => v.iter().any(|v| v.as_bytes() == field),
432            Columns(v) => {
433                debug_assert!(index < v.len());
434                v.get_unchecked(index).as_bytes() == field
435            },
436        }
437    }
