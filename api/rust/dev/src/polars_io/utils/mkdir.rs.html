1use std::io;

3use polars_utils::pl_path::PlRefPath;

5pub fn mkdir_recursive(path: &PlRefPath) -> io::Result<()> {
6    if !path.has_scheme() {
7        std::fs::DirBuilder::new().recursive(true).create(
8            path.parent()
9                .ok_or(io::Error::other("path is not a file"))?,
10        )?;
11    }
12
13    Ok(())
14}
