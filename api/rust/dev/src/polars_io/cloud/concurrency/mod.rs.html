1//! Adaptive in-flight concurrency controller for cloud IO (Input only).
2//!
3//! Admission control for concurrency uses two budgets:
4//! - A (primary) bytes-based budget to model the bandwidth-delay product (BDP)
5//! - A (secondary) count-based budget to limit the number of in-flight requests
6//!
7//! The bytes-based budget models the BDP as
8//!   BDP = BW_max * TTFB_est
9//!
10//! Three components cooperate:
11//! - Model: records IO observations and models the network (BW_max, TTFB_est, BDP)
12//! - Regime: state machine driving the admission (Init / RampUp / Stable / ProbeUp)
13//! - Admission: admission control, enforces byte-based + request-based budgets
14
15// Loosely based on BBR: Congestion-Based Congestion Control
16// see https://queue.acm.org/detail.cfm?id=3022184

49#[derive(Debug, Clone)]
50pub struct ControllerConfig {
51    // Sliding window over which the most recent round-trip-time (RTT) and bandwidth (BW)
52    // will be calculated. Also acts as the retention window.
53    window: Duration,
54    // Byte-based budget during the Init phase, and as the base for the RampUp phase.
55    init_byte_budget: u64,
56    // Lower limit for the byte-based budget - needed to avoid deadlock.
57    floor_byte_budget: u64,
58    // Count-based request budget.
59    request_budget: u32,
60    // Controller regime update frequency.
61    control_interval: Duration,
62    // Total budget only resizes if the relative changes exceeds this threshold
63    budget_resize_threshold: f64,
64}
65
66impl Default for ControllerConfig {
67    fn default() -> Self {
68        // Only used for bytes-based budget.
69        let target_chunk_size = get_random_access_chunk_size() as u64;
70        Self {
71            window: Duration::from_millis(1000),
72
73            // Byte-based budget during the ramp-up phase.
74            // Starting too low results in lost opportunity (time) during ramp-up.
75            // Starting too high results in early congestion, delayed completion of the first chunk,
76            // and inflated bandwidth estimation.
77            //
78            // Some BDP numbers for reference:
79            //   1 Gbps x 20 ms = 2.5 MB
80            //   1 Gbps x 50 ms = 6.25 MB
81            //   10 Gbps x 50 ms = 62.5 MB
82            //   100 Gbps x 50 ms = 625 MB
83            init_byte_budget: get_init_byte_budget(target_chunk_size),
84
85            // Byte-based budget floor.
86            // Must be >=larger than target_chunk_size to avoid potential deadlock.
87            floor_byte_budget: target_chunk_size,
88
89            // Count-based budget, currently fixed
90            request_budget: get_request_budget(),
91            control_interval: Duration::from_millis(100),
92            budget_resize_threshold: 0.05,
93        }
94    }
95}

97/// Max number of bytes concurrently in flight during the init and start of rampup phase.
98fn get_init_byte_budget(target_chunk_size: u64) -> u64 {
99    let init_byte_budget = std::env::var("POLARS_INFLIGHT_INIT_BYTE_BUDGET")
100        .map(|x| {
101            x.parse::<NonZeroU64>()
102                .unwrap_or_else(|_| {
103                    panic!("invalid value for POLARS_INFLIGHT_INIT_BYTE_BUDGET: {x}")
104                })
105                .get()
106        })
107        .unwrap_or_else(|_| {
108            // This should be lower than the expected BDP so it can ramp-up, but
109            // too low a value delays the transition to stable.
110            // Heuristic: higher bandwidth is expected on larger instances.
111            let n = polars_config::config().max_threads() as u64;
112            n.div_ceil(8).max(4) * target_chunk_size
113        })
114        .max(1);
115
116    if init_byte_budget < target_chunk_size {
117        panic!("in-flight byte budget init must be larger than the target_chunk_size");
118    }
119
120    init_byte_budget
121}

123/// Maximum number of requests concurrently in flight.
124pub fn get_request_budget() -> u32 {
125    // Since object_store/reqwest use HTTP/1 with a connection pool, this value controls the
126    // max concurrent TCP sessions to S3 for the pipeline.
127    // When modifying this value, consider the max_thread count configuration(s), the OS limitations
128    // (e.g., ulimit -n), and any cloud infrastructure limitations.
129    std::env::var("POLARS_INFLIGHT_REQUEST_BUDGET")
130        .map(|x| {
131            x.parse::<NonZeroU32>()
132                .unwrap_or_else(|_| panic!("invalid value for POLARS_INFLIGHT_REQUEST_BUDGET: {x}"))
133                .get()
134        })
135        .unwrap_or(512)
136        .max(1)
137}

139#[derive(Debug)]
140pub struct ConcurrencyController {
141    config: ControllerConfig,
142    sample_queue: Arc<ArrayQueue<IoSample>>,
143    samples_dropped: Arc<RelaxedCell<u64>>,
144    inflight_budget: Arc<InFlightBudget>,
145    _control_task: tokio::task::JoinHandle<()>,
146}
147
148impl ConcurrencyController {
149    pub fn new(config: ControllerConfig) -> Self {
150        let sample_queue = Arc::new(ArrayQueue::new(SAMPLE_QUEUE_CAPACITY));
151        let samples_dropped = Arc::new(RelaxedCell::new_u64(0));
152
153        let inflight_budget = Arc::new(InFlightBudget::new(
154            config.init_byte_budget,
155            config.floor_byte_budget,
156            config.request_budget,
157        ));
158
159        let control_task = Self::spawn_control_loop(
160            sample_queue.clone(),
161            samples_dropped.clone(),
162            inflight_budget.clone(),
163            config.clone(),
164        );
165
166        Self {
167            config,
168            sample_queue,
169            samples_dropped,
170            inflight_budget,
171            _control_task: control_task,
172        }
173    }
174
175    pub fn config(&self) -> &ControllerConfig {
176        &self.config
177    }
178
179    /// Record a completed IO. Hot path.
180    pub fn record_io(&self, sample: IoSample) {
181        if self.sample_queue.push(sample).is_err() {
182            // Queue full: drop. Samples are statistics is considered acceptable.
183            self.samples_dropped.fetch_add(1);
184        }
185    }
186
187    pub fn inflight_budget(&self) -> &Arc<InFlightBudget> {
188        &self.inflight_budget
189    }
190
191    pub async fn acquire(&self, bytes: u64) -> InFlightPermit {
192        self.inflight_budget.acquire(bytes).await
193    }
194
195    fn spawn_control_loop(
196        sample_queue: Arc<ArrayQueue<IoSample>>,
197        samples_dropped: Arc<RelaxedCell<u64>>,
198        admission: Arc<InFlightBudget>,
199        config: ControllerConfig,
200    ) -> tokio::task::JoinHandle<()> {
201        if polars_config::config().verbose() {
202            eprintln!(
203                "[InFlightConcurrency]: spawn control loop: control_interval: {}ms",
204                config.control_interval.as_millis()
205            );
206        }
207        ASYNC.spawn(async move {
208            let mut model = Model::new(config.window);
209            let mut regime = Regime::new(Instant::now());
210
211            let mut ticker = tokio::time::interval(config.control_interval);
212            ticker.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
213
214            loop {
215                ticker.tick().await;
216                let now = Instant::now();
217
218                // Update model statistics and step regime.
219                let (state, signal, dropped, bw_hwm_held) = {
220                    for _ in 0..SAMPLE_QUEUE_CAPACITY {
221                        let Some(s) = sample_queue.pop() else { break };
222                        model.record(s);
223                    }
224                    let dropped = samples_dropped.swap(0);
225                    model.update(now);
226                    let signal = model.signal();
227                    let state = regime.step(signal, now);
228                    let bw_hwm_held = model.bw_hwm_bps();
229                    (state, signal, dropped, bw_hwm_held)
230                };
231
232                if !matches!(state, RegimeState::WarmIdle { .. }) {
233                    // Compute base BDP
234                    let base_budget = match (state, signal) {
235                        (RegimeState::Init, _) | (_, None) => config.init_byte_budget,
236                        (_, Some(signal)) => signal.bdp_bytes().max(config.init_byte_budget),
237                    };
238
239                    // Compute target BDP using the gain multiplier. This is similar to BBR cwnd_gain.
240                    let gain = match state {
241                        RegimeState::Init => 1.0,
242                        RegimeState::RampUp { .. } => 2.0,
243                        // NOTE: >> 1.0 for the purpose of absorbing environment noise.
244                        RegimeState::Stable => 2.0,
245                        RegimeState::ProbeUp { .. } => 3.0,
246                        // Unreachable.
247                        RegimeState::WarmIdle { .. } => 1.0,
248                    };
249                    let target_budget = (base_budget as f64 * gain) as u64;
250
251                    // Resize if needed.
252                    let current_byte_budget = admission.current_byte_budget();
253                    let threshold = config.budget_resize_threshold;
254                    let should_resize = match current_byte_budget {
255                        0 => target_budget > 0,
256                        current => {
257                            let ratio = target_budget as f64 / current as f64;
258                            ratio < (1.0 - threshold) || ratio > (1.0 + threshold)
259                        },
260                    };
261
262                    if should_resize {
263                        admission.resize_byte_budget(target_budget);
264                    }
265                }
266
267                // Log snapshot.
268                if std::env::var("POLARS_LOG_CONCURRENCY").is_ok() {
269                    let stats = admission.stats();
270                    eprintln!(
271                        "[InFlightConcurrency {}] regime={}, \
272                        bw_hwm={:.1} MB/s, \
273                        bw_avg={:.1} MB/s, \
274                        rtt_min={:.1} ms, \
275                        rtt_avg={:.1} ms, \
276                        bdp_obs={:.1} MB, \
277                        bytes_budget={:.1} MB, \
278                        bytes_in_use={:.1} MB, \
279                        bytes_sat={:.2}, \
280                        req_budget={}, \
281                        req_in_use={}, \
282                        req_sat={:.2}",
283                        chrono::Utc::now(),
284                        state.label(),
285                        signal.map(|s| s.bw_hwm_bps).or(bw_hwm_held).unwrap_or(0.0) / 1e6,
286                        signal.map_or(0.0, |s| s.bw_avg_bps) / 1e6,
287                        signal.map_or(0, |s| s.ttfb_min.as_millis()),
288                        signal.map_or(0, |s| s.ttfb_avg.as_millis()),
289                        signal.map_or(0, |s| s.bdp_bytes()) as f64 / 1e6,
290                        stats.bytes_budget as f64 / 1e6,
291                        stats.bytes_in_use as f64 / 1e6,
292                        stats.bytes_saturation,
293                        stats.request_budget,
294                        stats.requests_in_use,
295                        stats.requests_saturation
296                    );
297                    if dropped > 0 {
298                        eprintln!(
299                            "[InFlightConcurrency] WARN: {dropped} samples dropped (queue full)"
300                        );
301                    }
302                }
303            }
304        })
305    }
306}
