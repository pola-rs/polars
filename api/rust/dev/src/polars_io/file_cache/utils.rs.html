15pub static FILE_CACHE_PREFIX: LazyLock<Box<Path>> = LazyLock::new(|| {
16    let path = POLARS_TEMP_DIR_BASE_PATH
17        .join("file-cache/")
18        .into_boxed_path();
19
20    if let Err(err) = ensure_directory_init(path.as_ref()) {
21        panic!(
22            "failed to create file cache directory: path = {}, err = {}",
23            path.to_str().unwrap(),
24            err
25        );
26    }
27
28    path
29});

31pub(super) fn last_modified_u64(metadata: &std::fs::Metadata) -> u64 {
32    metadata
33        .modified()
34        .unwrap()
35        .duration_since(UNIX_EPOCH)
36        .unwrap()
37        .as_millis() as u64
38}

40pub(super) fn update_last_accessed(file: &std::fs::File) {
41    let file_metadata = file.metadata().unwrap();
42
43    if let Err(e) = file.set_times(
44        std::fs::FileTimes::new()
45            .set_modified(file_metadata.modified().unwrap())
46            .set_accessed(std::time::SystemTime::now()),
47    ) {
48        panic!("failed to update file last accessed time: {e}");
49    }
50}

52pub fn init_entries_from_uri_list(
53    uri_list: &[Arc<str>],
54    cloud_options: Option<&CloudOptions>,
55) -> PolarsResult<Vec<Arc<FileCacheEntry>>> {
56    if uri_list.is_empty() {
57        return Ok(Default::default());
58    }
59
60    let first_uri = uri_list.first().unwrap().as_ref();
61
62    let file_cache_ttl = cloud_options
63        .map(|x| x.file_cache_ttl)
64        .unwrap_or_else(get_env_file_cache_ttl);
65
66    if PlPathRef::new(first_uri).is_cloud_url() {
67        let object_stores = pl_async::get_runtime().block_in_place_on(async {
68            futures::future::try_join_all(
69                (0..if first_uri.starts_with("http") {
70                    // Object stores for http are tied to the path.
71                    uri_list.len()
72                } else {
73                    1
74                })
75                    .map(|i| async move {
76                        let (_, object_store) =
77                            build_object_store(&uri_list[i], cloud_options, false).await?;
78                        PolarsResult::Ok(object_store)
79                    }),
80            )
81            .await
82        })?;
83
84        uri_list
85            .iter()
86            .enumerate()
87            .map(|(i, uri)| {
88                FILE_CACHE.init_entry(
89                    uri.clone(),
90                    || {
91                        let CloudLocation { prefix, .. } =
92                            CloudLocation::new(uri.as_ref(), false).unwrap();
93                        let cloud_path = object_path_from_str(&prefix)?;
94
95                        let object_store =
96                            object_stores[std::cmp::min(i, object_stores.len() - 1)].clone();
97                        let uri = uri.clone();
98
99                        Ok(Arc::new(CloudFileFetcher {
100                            uri,
101                            object_store,
102                            cloud_path,
103                        }))
104                    },
105                    file_cache_ttl,
106                )
107            })
108            .collect::<PolarsResult<Vec<_>>>()
109    } else {
110        uri_list
111            .iter()
112            .map(|uri| {
113                let uri = std::fs::canonicalize(uri.as_ref()).map_err(|err| {
114                    let msg = Some(format!("{}: {}", err, uri.as_ref()).into());
115                    PolarsError::IO {
116                        error: err.into(),
117                        msg,
118                    }
119                })?;
120                let uri = Arc::<str>::from(uri.to_str().unwrap());
121
122                FILE_CACHE.init_entry(
123                    uri.clone(),
124                    || Ok(Arc::new(LocalFileFetcher::from_uri(uri.clone()))),
125                    file_cache_ttl,
126                )
127            })
128            .collect::<PolarsResult<Vec<_>>>()
129    }
130}
