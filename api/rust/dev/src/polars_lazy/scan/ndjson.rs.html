126impl LazyFileListReader for LazyJsonLineReader {
127    fn finish(self) -> PolarsResult<LazyFrame> {
128        let unified_scan_args = UnifiedScanArgs {
129            schema: None,
130            cloud_options: self.cloud_options,
131            hive_options: HiveOptions::new_disabled(),
132            rechunk: self.rechunk,
133            cache: false,
134            glob: true,
135            hidden_file_prefix: None,
136            projection: None,
137            column_mapping: None,
138            default_values: None,
139            row_index: self.row_index,
140            pre_slice: self.n_rows.map(|len| Slice::Positive { offset: 0, len }),
141            cast_columns_policy: CastColumnsPolicy::ERROR_ON_MISMATCH,
142            missing_columns_policy: MissingColumnsPolicy::Raise,
143            extra_columns_policy: ExtraColumnsPolicy::Raise,
144            include_file_paths: self.include_file_paths,
145            deletion_files: None,
146            table_statistics: None,
147            row_count: None,
148        };
149
150        let options = NDJsonReadOptions {
151            n_threads: None,
152            infer_schema_length: self.infer_schema_length,
153            chunk_size: NonZeroUsize::new(1 << 18).unwrap(),
154            low_memory: self.low_memory,
155            ignore_errors: self.ignore_errors,
156            schema: self.schema,
157            schema_overwrite: self.schema_overwrite,
158        };
159
160        let scan_type = Box::new(FileScanDsl::NDJson { options });
161
162        Ok(LazyFrame::from(DslPlan::Scan {
163            sources: self.sources,
164            unified_scan_args: Box::new(unified_scan_args),
165            scan_type,
166            cached_ir: Default::default(),
167        }))
168    }
169
170    fn finish_no_glob(self) -> PolarsResult<LazyFrame> {
171        unreachable!();
172    }
173
174    fn sources(&self) -> &ScanSources {
175        &self.sources
176    }
177
178    fn with_sources(mut self, sources: ScanSources) -> Self {
179        self.sources = sources;
180        self
181    }
182
183    fn with_n_rows(mut self, n_rows: impl Into<Option<usize>>) -> Self {

185        self
186    }
187
188    fn with_row_index(mut self, row_index: impl Into<Option<RowIndex>>) -> Self {

191    }

194        self.rechunk
195    }

197    /// Rechunk the memory to contiguous chunks when parsing is done.

199        self.rechunk = toggle;

201    }
202
203    /// Try to stop parsing when `n` rows are parsed. During multithreaded parsing the upper bound `n` cannot
204    /// be guaranteed.
205    fn n_rows(&self) -> Option<usize> {
206        self.n_rows
207    }
208
209    /// Add a row index column.

211        self.row_index.as_ref()
212    }
