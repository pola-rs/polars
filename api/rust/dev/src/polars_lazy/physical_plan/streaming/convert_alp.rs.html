8// The index of the pipeline tree we are building at this moment
9// if we have a node we cannot do streaming, we have finished that pipeline tree
10// and start a new one.
11type CurrentIdx = usize;
12
13// Frame in the stack of logical plans to process while inserting streaming nodes
14struct StackFrame {
15    node: Node, // LogicalPlan node
16    state: Branch,
17    current_idx: CurrentIdx,
18    insert_sink: bool,
19}
20
21impl StackFrame {
22    fn root(node: Node) -> StackFrame {
23        StackFrame {
24            node,
25            state: Branch::default(),
26            current_idx: 0,
27            insert_sink: false,
28        }
29    }
30
31    fn new(node: Node, state: Branch, current_idx: CurrentIdx) -> StackFrame {
32        StackFrame {
33            node,
34            state,
35            current_idx,
36            insert_sink: false,
37        }
38    }
39
40    // Create a new streaming subtree below a non-streaming node
41    fn new_subtree(node: Node, current_idx: CurrentIdx) -> StackFrame {
42        StackFrame {
43            node,
44            state: Branch::default(),
45            current_idx,
46            insert_sink: true,
47        }
48    }
49}
50
51fn process_non_streamable_node(
52    current_idx: &mut CurrentIdx,
53    state: &mut Branch,
54    stack: &mut Vec<StackFrame>,
55    scratch: &mut Vec<Node>,
56    pipeline_trees: &mut Vec<Vec<Branch>>,
57    lp: &IR,
58) {
59    lp.copy_inputs(scratch);
60    while let Some(input) = scratch.pop() {
61        if state.streamable {
62            *current_idx += 1;
63            // create a completely new streaming pipeline
64            // maybe we can stream a subsection of the plan
65            pipeline_trees.push(vec![]);
66        }
67        stack.push(StackFrame::new_subtree(input, *current_idx));
68    }
69    state.streamable = false;
70}

72fn insert_file_sink(mut root: Node, lp_arena: &mut Arena<IR>) -> Node {
73    // The pipelines need a final sink, we insert that here.
74    // this allows us to split at joins/unions and share a sink
75    if !matches!(lp_arena.get(root), IR::Sink { .. }) {
76        root = lp_arena.add(IR::Sink {
77            input: root,
78            payload: SinkType::Memory,
79        })
80    }
81    root
82}

84pub(crate) fn insert_streaming_nodes(
85    root: Node,
86    lp_arena: &mut Arena<IR>,
87    expr_arena: &mut Arena<AExpr>,
88    scratch: &mut Vec<Node>,
89    fmt: bool,
90    // whether the full plan needs to be translated
91    // to streaming
92    allow_partial: bool,
93    row_estimate: bool,
94) -> PolarsResult<bool> {
95    scratch.clear();
96
97    // This is needed to determine which side of the joins should be
98    // traversed first. As we want to keep the smallest table in the build phase as that keeps most
99    // data in memory.
100    if row_estimate {
101        set_estimated_row_counts(root, lp_arena, expr_arena, 0, scratch);
102    }
103
104    scratch.clear();
105
106    // The pipelines always need to end in a SINK, we insert that here.
107    // this allows us to split at joins/unions and share a sink
108    let root = insert_file_sink(root, lp_arena);
109
110    // We use a bool flag in the stack to communicate when we need to insert a file sink.
111    // This happens for instance when we
112    //
113    //     ________*non-streamable part of query
114    //   /\
115    //     ________*streamable below this line so we must insert
116    //    /\        a file sink here so the pipeline can be built
117    //     /\
118
119    let mut stack = Vec::with_capacity(16);
120
121    stack.push(StackFrame::root(root));
122
123    // A state holds a full pipeline until the breaker
124    //  1/\
125    //   2/\
126    //     3\
127    //
128    // so 1 and 2 are short pipelines and 3 goes all the way to the root.
129    // but 3 can only run if 1 and 2 have finished and set the join as operator in 3
130    // and state are filled with pipeline 1, 2, 3 in that order
131    //
132    //     / \
133    //  /\  3/\
134    //  1 2    4\
135    // or in this case 1, 2, 3, 4
136    // every inner vec contains a branch/pipeline of a complete pipeline tree
137    // the outer vec contains whole pipeline trees
138    //
139    // # Execution order
140    // Trees can have arbitrary splits via joins and unions
141    // the branches we have accumulated are flattened into a single Vec<Branch>
142    // this therefore has lost the information of the tree. To know in which
143    // order the branches need to be executed. For this reason we keep track of
144    // an `execution_id` which will be incremented on every stack operation.
145    // This way we know in which order the stack/tree was traversed and can
146    // use that info to determine the execution order of the single branch/pipelines
147    let mut pipeline_trees: Vec<Tree> = vec![vec![]];
148    // keep the counter global so that the order will match traversal order
149    let mut execution_id = 0;
150
151    use IR::*;
152    while let Some(StackFrame {
153        node: mut root,
154        mut state,
155        mut current_idx,
156        insert_sink,
157    }) = stack.pop()
158    {
159        if insert_sink {
160            root = insert_file_sink(root, lp_arena);
161        }
162        state.execution_id = execution_id;
163        execution_id += 1;
164        match lp_arena.get(root) {
165            Filter { input, predicate }
166                if is_elementwise_rec(expr_arena.get(predicate.node()), expr_arena) =>
167            {
168                state.streamable = true;
169                state.operators_sinks.push(PipelineNode::Operator(root));
170                stack.push(StackFrame::new(*input, state, current_idx))
171            },
172            HStack { input, exprs, .. } if all_elementwise(exprs, expr_arena) => {
173                state.streamable = true;
174                state.operators_sinks.push(PipelineNode::Operator(root));
175                stack.push(StackFrame::new(*input, state, current_idx))
176            },
177            Slice { input, offset, .. } if *offset >= 0 => {
178                state.streamable = true;
179                state.operators_sinks.push(PipelineNode::Sink(root));
180                stack.push(StackFrame::new(*input, state, current_idx))
181            },
182            Sink { input, .. } => {
183                state.streamable = true;
184                state.operators_sinks.push(PipelineNode::Sink(root));
185                stack.push(StackFrame::new(*input, state, current_idx))
186            },
187            Sort {
188                input,
189                by_column,
190                slice,
191                sort_options,
192            } if is_streamable_sort(slice, sort_options) && all_column(by_column, expr_arena) => {
193                state.streamable = true;
194                state.operators_sinks.push(PipelineNode::Sink(root));
195                stack.push(StackFrame::new(*input, state, current_idx))
196            },
197            Select { input, expr, .. } if all_elementwise(expr, expr_arena) => {
198                state.streamable = true;
199                state.operators_sinks.push(PipelineNode::Operator(root));
200                stack.push(StackFrame::new(*input, state, current_idx))
201            },
202            SimpleProjection { input, .. } => {
203                state.streamable = true;
204                state.operators_sinks.push(PipelineNode::Operator(root));
205                stack.push(StackFrame::new(*input, state, current_idx))
206            },
207            // Rechunks are ignored
208            MapFunction {
209                input,
210                function: FunctionIR::Rechunk,
211            } => {
212                state.streamable = true;
213                stack.push(StackFrame::new(*input, state, current_idx))
214            },
215            // Streamable functions will be converted
216            lp @ MapFunction { input, function } => {
217                if function.is_streamable() {
218                    state.streamable = true;
219                    state.operators_sinks.push(PipelineNode::Operator(root));
220                    stack.push(StackFrame::new(*input, state, current_idx))
221                } else {
222                    process_non_streamable_node(
223                        &mut current_idx,
224                        &mut state,
225                        &mut stack,
226                        scratch,
227                        &mut pipeline_trees,
228                        lp,
229                    )
230                }
231            },
232            Scan {
233                scan_type,
234                file_options,
235                ..
236            } if scan_type.streamable()
237                && file_options
238                    .pre_slice
239                    .map(|slice| slice.0 >= 0)
240                    .unwrap_or(true) =>
241            {
242                if state.streamable {
243                    state.sources.push(root);
244                    pipeline_trees[current_idx].push(state)
245                }
246            },
247            DataFrameScan { .. } => {
248                if state.streamable {
249                    state.sources.push(root);
250                    pipeline_trees[current_idx].push(state)
251                }
252            },
253            Join {
254                input_left,
255                input_right,
256                options,
257                ..
258            } if streamable_join(&options.args) => {
259                let input_left = *input_left;
260                let input_right = *input_right;
261                state.streamable = true;
262                state.join_count += 1;
263
264                // We swap so that the build phase contains the smallest table
265                // and then we stream the larger table
266                // *except* for a left join. In a left join we use the right
267                // table as build table and we stream the left table. This way
268                // we maintain order in the left join.
269                let (input_left, input_right) = if swap_join_order(options) {
270                    (input_right, input_left)
271                } else {
272                    (input_left, input_right)
273                };
274                let mut state_left = state.split();
275
276                // Rhs is second, so that is first on the stack.
277                let mut state_right = state;
278                state_right.join_count = 0;
279                state_right
280                    .operators_sinks
281                    .push(PipelineNode::RhsJoin(root));
282
283                // We want to traverse lhs last, so push it first on the stack
284                // rhs is a new pipeline.
285                state_left.operators_sinks.push(PipelineNode::Sink(root));
286                stack.push(StackFrame::new(input_left, state_left, current_idx));
287                stack.push(StackFrame::new(input_right, state_right, current_idx));
288            },
289            // add globbing patterns
290            #[cfg(any(feature = "csv", feature = "parquet"))]
291            Union { inputs, options }
292                if options.slice.is_none()
293                    && inputs.iter().all(|node| match lp_arena.get(*node) {
294                        Scan { .. } => true,
295                        MapFunction {
296                            input,
297                            function: FunctionIR::Rechunk,
298                        } => matches!(lp_arena.get(*input), Scan { .. }),
299                        _ => false,
300                    }) =>
301            {
302                state.sources.push(root);
303                pipeline_trees[current_idx].push(state);
304            },
305            Union { inputs, .. } => {
306                {
307                    state.streamable = true;
308                    for (i, input) in inputs.iter().enumerate() {
309                        let mut state = if i == 0 {
310                            // note the clone!
311                            let mut state = state.clone();
312                            state.join_count += inputs.len() as u32 - 1;
313                            state
314                        } else {
315                            let mut state = state.split_from_sink();
316                            state.join_count = 0;
317                            state
318                        };
319                        state.operators_sinks.push(PipelineNode::Union(root));
320                        stack.push(StackFrame::new(*input, state, current_idx));
321                    }
322                }
323            },
324            Distinct { input, options }
325                if !options.maintain_order
326                    && !matches!(options.keep_strategy, UniqueKeepStrategy::None) =>
327            {
328                state.streamable = true;
329                state.operators_sinks.push(PipelineNode::Sink(root));
330                stack.push(StackFrame::new(*input, state, current_idx))
331            },
332            #[allow(unused_variables)]
333            lp @ GroupBy {
334                input,
335                keys,
336                aggs,
337                maintain_order: false,
338                apply: None,
339                schema: output_schema,
340                options,
341                ..
342            } => {
343                #[cfg(feature = "dtype-categorical")]
344                let string_cache = polars_core::using_string_cache();
345                #[cfg(not(feature = "dtype-categorical"))]
346                let string_cache = true;
347
348                #[allow(unused_variables)]
349                fn allowed_dtype(dt: &DataType, string_cache: bool) -> bool {
350                    match dt {
351                        #[cfg(feature = "object")]
352                        DataType::Object(_) => false,
353                        #[cfg(feature = "dtype-categorical")]
354                        DataType::Categorical(_, _) => string_cache,
355                        DataType::List(inner) => allowed_dtype(inner, string_cache),
356                        #[cfg(feature = "dtype-struct")]
357                        DataType::Struct(fields) => fields
358                            .iter()
359                            .all(|fld| allowed_dtype(fld.dtype(), string_cache)),
360                        // We need to be able to sink to disk or produce the aggregate return dtype.
361                        DataType::Unknown(_) => false,
362                        #[cfg(feature = "dtype-decimal")]
363                        DataType::Decimal(_, _) => false,
364                        DataType::Int128 => false,
365                        _ => true,
366                    }
367                }
368                let input_schema = lp_arena.get(*input).schema(lp_arena);
369                #[allow(unused_mut)]
370                let mut can_stream = true;
371
372                #[cfg(feature = "dynamic_group_by")]
373                {
374                    if options.rolling.is_some() || options.dynamic.is_some() {
375                        can_stream = false
376                    }
377                }
378
379                let valid_agg = || {
380                    aggs.iter().all(|e| {
381                        polars_pipe::pipeline::can_convert_to_hash_agg(
382                            e.node(),
383                            expr_arena,
384                            &input_schema,
385                        )
386                    })
387                };
388
389                let valid_key = || {
390                    keys.iter().all(|e| {
391                        output_schema
392                            .get(e.output_name())
393                            .map(|dt| !matches!(dt, DataType::List(_)))
394                            .unwrap_or(false)
395                    })
396                };
397
398                let valid_types = || {
399                    output_schema
400                        .iter_values()
401                        .all(|dt| allowed_dtype(dt, string_cache))
402                };
403
404                if can_stream && valid_agg() && valid_key() && valid_types() {
405                    state.streamable = true;
406                    state.operators_sinks.push(PipelineNode::Sink(root));
407                    stack.push(StackFrame::new(*input, state, current_idx))
408                } else if allow_partial {
409                    process_non_streamable_node(
410                        &mut current_idx,
411                        &mut state,
412                        &mut stack,
413                        scratch,
414                        &mut pipeline_trees,
415                        lp,
416                    )
417                } else {
418                    return Ok(false);
419                }
420            },
421            lp => {
422                if allow_partial {
423                    process_non_streamable_node(
424                        &mut current_idx,
425                        &mut state,
426                        &mut stack,
427                        scratch,
428                        &mut pipeline_trees,
429                        lp,
430                    )
431                } else {
432                    return Ok(false);
433                }
434            },
435        }
436    }
437
438    let mut inserted = false;
439    for tree in pipeline_trees {
440        if is_valid_tree(&tree)
441            && super::construct_pipeline::construct(tree, lp_arena, expr_arena, fmt)?.is_some()
442        {
443            inserted = true;
444        }
445    }
446
447    Ok(inserted)
448}
