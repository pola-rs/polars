249    /// Add a sort operation to the logical plan.
250    ///
251    /// Sorts the LazyFrame by the column name specified using the provided options.
252    ///
253    /// # Example
254    ///
255    /// Sort DataFrame by 'sepal_width' column:
256    /// ```rust
257    /// # use polars_core::prelude::*;
258    /// # use polars_lazy::prelude::*;
259    /// fn sort_by_a(df: DataFrame) -> LazyFrame {
260    ///     df.lazy().sort(["sepal_width"], Default::default())
261    /// }
262    /// ```
263    /// Sort by a single column with specific order:
264    /// ```
265    /// # use polars_core::prelude::*;
266    /// # use polars_lazy::prelude::*;
267    /// fn sort_with_specific_order(df: DataFrame, descending: bool) -> LazyFrame {
268    ///     df.lazy().sort(
269    ///         ["sepal_width"],
270    ///         SortMultipleOptions::new()
271    ///             .with_order_descending(descending)
272    ///     )
273    /// }
274    /// ```
275    /// Sort by multiple columns with specifying order for each column:
276    /// ```
277    /// # use polars_core::prelude::*;
278    /// # use polars_lazy::prelude::*;
279    /// fn sort_by_multiple_columns_with_specific_order(df: DataFrame) -> LazyFrame {
280    ///     df.lazy().sort(
281    ///         ["sepal_width", "sepal_length"],
282    ///         SortMultipleOptions::new()
283    ///             .with_order_descending_multi([false, true])
284    ///     )
285    /// }
286    /// ```
287    /// See [`SortMultipleOptions`] for more options.
288    pub fn sort(self, by: impl IntoVec<PlSmallStr>, sort_options: SortMultipleOptions) -> Self {
289        let opt_state = self.get_opt_state();
290        let lp = self
291            .get_plan_builder()
292            .sort(by.into_vec().into_iter().map(col).collect(), sort_options)
293            .build();
294        Self::from_logical_plan(lp, opt_state)
295    }

298    ///
299    /// Sorts the LazyFrame by the provided list of expressions, which will be turned into
300    /// concrete columns before sorting.
301    ///
302    /// See [`SortMultipleOptions`] for more options.
303    ///
304    /// # Example
305    ///
306    /// ```rust

312    ///       df.lazy()

314    /// }

316    pub fn sort_by_exprs<E: AsRef<[Expr]>>(
317        self,
318        by_exprs: E,
319        sort_options: SortMultipleOptions,
320    ) -> Self {
321        let by_exprs = by_exprs.as_ref().to_vec();
322        if by_exprs.is_empty() {
323            self
324        } else {
325            let opt_state = self.get_opt_state();
326            let lp = self.get_plan_builder().sort(by_exprs, sort_options).build();
327            Self::from_logical_plan(lp, opt_state)
328        }
329    }

375    /// Rename columns in the DataFrame.
376    ///
377    /// `existing` and `new` are iterables of the same length containing the old and
378    /// corresponding new column names. Renaming happens to all `existing` columns
379    /// simultaneously, not iteratively. If `strict` is true, all columns in `existing`
380    /// must be present in the `LazyFrame` when `rename` is called; otherwise, only
381    /// those columns that are actually found will be renamed (others will be ignored).
382    pub fn rename<I, J, T, S>(self, existing: I, new: J, strict: bool) -> Self
383    where
384        I: IntoIterator<Item = T>,
385        J: IntoIterator<Item = S>,
386        T: AsRef<str>,
387        S: AsRef<str>,
388    {
389        let iter = existing.into_iter();
390        let cap = iter.size_hint().0;
391        let mut existing_vec: Vec<PlSmallStr> = Vec::with_capacity(cap);
392        let mut new_vec: Vec<PlSmallStr> = Vec::with_capacity(cap);
393
394        // TODO! should this error if `existing` and `new` have different lengths?
395        // Currently, the longer of the two is truncated.
396        for (existing, new) in iter.zip(new) {
397            let existing = existing.as_ref();
398            let new = new.as_ref();
399            if new != existing {
400                existing_vec.push(existing.into());
401                new_vec.push(new.into());
402            }
403        }
404
405        self.map_private(DslFunction::Rename {
406            existing: existing_vec.into(),
407            new: new_vec.into(),
408            strict,
409        })
410    }

523    pub(crate) fn optimize_with_scratch(
524        self,
525        lp_arena: &mut Arena<IR>,
526        expr_arena: &mut Arena<AExpr>,
527        scratch: &mut Vec<Node>,
528    ) -> PolarsResult<Node> {
529        #[allow(unused_mut)]
530        let mut opt_state = self.opt_state;
531        let new_streaming = self.opt_state.contains(OptFlags::NEW_STREAMING);
532
533        #[cfg(feature = "cse")]
534        if new_streaming {
535            // The new streaming engine can't deal with the way the common
536            // subexpression elimination adds length-incorrect with_columns.
537            opt_state &= !OptFlags::COMM_SUBEXPR_ELIM;
538        }
539
540        let lp_top = optimize(
541            self.logical_plan,
542            opt_state,
543            lp_arena,
544            expr_arena,
545            scratch,
546            Some(&|expr, expr_arena, schema| {
547                let phys_expr = create_physical_expr(
548                    expr,
549                    Context::Default,
550                    expr_arena,
551                    schema,
552                    &mut ExpressionConversionState::new(true),
553                )
554                .ok()?;
555                let io_expr = phys_expr_to_io_expr(phys_expr);
556                Some(io_expr)
557            }),
558        )?;
559
560        Ok(lp_top)
561    }

563    fn prepare_collect_post_opt<P>(
564        mut self,
565        check_sink: bool,
566        query_start: Option<std::time::Instant>,
567        post_opt: P,
568    ) -> PolarsResult<(ExecutionState, Box<dyn Executor>, bool)>
569    where
570        P: FnOnce(
571            Node,
572            &mut Arena<IR>,
573            &mut Arena<AExpr>,
574            Option<std::time::Duration>,
575        ) -> PolarsResult<()>,
576    {
577        let (mut lp_arena, mut expr_arena) = self.get_arenas();
578
579        let mut scratch = vec![];
580        let lp_top = self.optimize_with_scratch(&mut lp_arena, &mut expr_arena, &mut scratch)?;
581
582        post_opt(
583            lp_top,
584            &mut lp_arena,
585            &mut expr_arena,
586            // Post optimization callback gets the time since the
587            // query was started as its "base" timepoint.
588            query_start.map(|s| s.elapsed()),
589        )?;
590
591        // sink should be replaced
592        let no_file_sink = if check_sink {
593            !matches!(
594                lp_arena.get(lp_top),
595                IR::Sink {
596                    payload: SinkTypeIR::File { .. } | SinkTypeIR::Partition { .. },
597                    ..
598                }
599            )
600        } else {
601            true
602        };
603        let physical_plan = create_physical_plan(
604            lp_top,
605            &mut lp_arena,
606            &mut expr_arena,
607            BUILD_STREAMING_EXECUTOR,
608        )?;
609
610        let state = ExecutionState::new();
611        Ok((state, physical_plan, no_file_sink))
612    }

638    /// Execute all the lazy operations and collect them into a [`DataFrame`] using a specified
639    /// `engine`.
640    ///
641    /// The query is optimized prior to execution.
642    pub fn collect_with_engine(mut self, mut engine: Engine) -> PolarsResult<DataFrame> {
643        let payload = if let DslPlan::Sink { payload, .. } = &self.logical_plan {
644            payload.clone()
645        } else {
646            self.logical_plan = DslPlan::Sink {
647                input: Arc::new(self.logical_plan),
648                payload: SinkType::Memory,
649            };
650            SinkType::Memory
651        };
652
653        // Default engine for collect is InMemory, sink_* is Streaming
654        if engine == Engine::Auto {
655            engine = match payload {
656                #[cfg(feature = "new_streaming")]
657                SinkType::Callback { .. } | SinkType::File { .. } | SinkType::Partition { .. } => {
658                    Engine::Streaming
659                },
660                _ => Engine::InMemory,
661            };
662        }
663        // Gpu uses some hacks to dispatch.
664        if engine == Engine::Gpu {
665            engine = Engine::InMemory;
666        }
667
668        #[cfg(feature = "new_streaming")]
669        {
670            if let Some(result) = self.try_new_streaming_if_requested() {
671                return result.map(|v| v.unwrap_single());
672            }
673        }
674
675        match engine {
676            Engine::Auto => unreachable!(),
677            Engine::Streaming => {
678                feature_gated!("new_streaming", self = self.with_new_streaming(true))
679            },
680            _ => {},
681        }
682        let mut alp_plan = self.clone().to_alp_optimized()?;
683
684        match engine {
685            Engine::Auto | Engine::Streaming => feature_gated!("new_streaming", {
686                let result = polars_stream::run_query(
687                    alp_plan.lp_top,
688                    &mut alp_plan.lp_arena,
689                    &mut alp_plan.expr_arena,
690                );
691                result.map(|v| v.unwrap_single())
692            }),
693            Engine::Gpu => {
694                Err(polars_err!(InvalidOperation: "sink is not supported for the gpu engine"))
695            },
696            Engine::InMemory => {
697                let mut physical_plan = create_physical_plan(
698                    alp_plan.lp_top,
699                    &mut alp_plan.lp_arena,
700                    &mut alp_plan.expr_arena,
701                    BUILD_STREAMING_EXECUTOR,
702                )?;
703                let mut state = ExecutionState::new();
704                physical_plan.execute(&mut state)
705            },
706        }
707    }

718    pub fn collect_all_with_engine(
719        plans: Vec<DslPlan>,
720        mut engine: Engine,
721        opt_state: OptFlags,
722    ) -> PolarsResult<Vec<DataFrame>> {
723        if plans.is_empty() {
724            return Ok(Vec::new());
725        }
726
727        // Default engine for collect_all is InMemory
728        if engine == Engine::Auto {
729            engine = Engine::InMemory;
730        }
731        // Gpu uses some hacks to dispatch.
732        if engine == Engine::Gpu {
733            engine = Engine::InMemory;
734        }
735
736        let mut sink_multiple = LazyFrame {
737            logical_plan: DslPlan::SinkMultiple { inputs: plans },
738            opt_state,
739            cached_arena: Default::default(),
740        };
741
742        #[cfg(feature = "new_streaming")]
743        {
744            if let Some(result) = sink_multiple.try_new_streaming_if_requested() {
745                return result.map(|v| v.unwrap_multiple());
746            }
747        }
748
749        match engine {
750            Engine::Auto => unreachable!(),
751            Engine::Streaming => {
752                feature_gated!(
753                    "new_streaming",
754                    sink_multiple = sink_multiple.with_new_streaming(true)
755                )
756            },
757            _ => {},
758        }
759        let mut alp_plan = sink_multiple.to_alp_optimized()?;
760
761        if engine == Engine::Streaming {
762            feature_gated!("new_streaming", {
763                let result = polars_stream::run_query(
764                    alp_plan.lp_top,
765                    &mut alp_plan.lp_arena,
766                    &mut alp_plan.expr_arena,
767                );
768                return result.map(|v| v.unwrap_multiple());
769            });
770        }
771
772        let IR::SinkMultiple { inputs } = alp_plan.root() else {
773            unreachable!()
774        };
775
776        let mut multiplan = create_multiple_physical_plans(
777            inputs.clone().as_slice(),
778            &mut alp_plan.lp_arena,
779            &mut alp_plan.expr_arena,
780            BUILD_STREAMING_EXECUTOR,
781        )?;
782
783        match engine {
784            Engine::Gpu => polars_bail!(
785                InvalidOperation: "collect_all is not supported for the gpu engine"
786            ),
787            Engine::InMemory => {
788                // We don't use par_iter directly because the LP may also start threads for every LP (for instance scan_csv)
789                // this might then lead to a rayon SO. So we take a multitude of the threads to keep work stealing
790                // within bounds
791                let mut state = ExecutionState::new();
792                if let Some(mut cache_prefiller) = multiplan.cache_prefiller {
793                    cache_prefiller.execute(&mut state)?;
794                }
795                let out = POOL.install(|| {
796                    multiplan
797                        .physical_plans
798                        .chunks_mut(POOL.current_num_threads() * 3)
799                        .map(|chunk| {
800                            chunk
801                                .into_par_iter()
802                                .enumerate()
803                                .map(|(idx, input)| {
804                                    let mut input = std::mem::take(input);
805                                    let mut state = state.split();
806                                    state.branch_idx += idx;
807
808                                    let df = input.execute(&mut state)?;
809                                    Ok(df)
810                                })
811                                .collect::<PolarsResult<Vec<_>>>()
812                        })
813                        .collect::<PolarsResult<Vec<_>>>()
814                });
815                Ok(out?.into_iter().flatten().collect())
816            },
817            _ => unreachable!(),
818        }
819    }

1074    #[cfg(feature = "new_streaming")]
1075    pub fn try_new_streaming_if_requested(
1076        &mut self,
1077    ) -> Option<PolarsResult<polars_stream::QueryResult>> {
1078        let auto_new_streaming = std::env::var("POLARS_AUTO_NEW_STREAMING").as_deref() == Ok("1");
1079        let force_new_streaming = std::env::var("POLARS_FORCE_NEW_STREAMING").as_deref() == Ok("1");
1080
1081        if auto_new_streaming || force_new_streaming {
1082            // Try to run using the new streaming engine, falling back
1083            // if it fails in a todo!() error if auto_new_streaming is set.
1084            let mut new_stream_lazy = self.clone();
1085            new_stream_lazy.opt_state |= OptFlags::NEW_STREAMING;
1086            let mut alp_plan = match new_stream_lazy.to_alp_optimized() {
1087                Ok(v) => v,
1088                Err(e) => return Some(Err(e)),
1089            };
1090
1091            let f = || {
1092                polars_stream::run_query(
1093                    alp_plan.lp_top,
1094                    &mut alp_plan.lp_arena,
1095                    &mut alp_plan.expr_arena,
1096                )
1097            };
1098
1099            match std::panic::catch_unwind(std::panic::AssertUnwindSafe(f)) {
1100                Ok(v) => return Some(v),
1101                Err(e) => {
1102                    // Fallback to normal engine if error is due to not being implemented
1103                    // and auto_new_streaming is set, otherwise propagate error.
1104                    if !force_new_streaming
1105                        && auto_new_streaming
1106                        && e.downcast_ref::<&str>()
1107                            .map(|s| s.starts_with("not yet implemented"))
1108                            .unwrap_or(false)
1109                    {
1110                        if polars_core::config::verbose() {
1111                            eprintln!(
1112                                "caught unimplemented error in new streaming engine, falling back to normal engine"
1113                            );
1114                        }
1115                    } else {
1116                        std::panic::resume_unwind(e);
1117                    }
1118                },
1119            }
1120        }
1121
1122        None
1123    }

1288    ///
1289    /// Also works for index values of type UInt32, UInt64, Int32, or Int64.
1290    ///
1291    /// Different from a [`group_by_dynamic`][`Self::group_by_dynamic`], the windows are now determined by the
1292    /// individual values and are not of constant intervals. For constant intervals use
1293    /// *group_by_dynamic*
1294    #[cfg(feature = "dynamic_group_by")]
1295    pub fn rolling<E: AsRef<[Expr]>>(
1296        mut self,
1297        index_column: Expr,
1298        group_by: E,
1299        mut options: RollingGroupOptions,
1300    ) -> LazyGroupBy {

1302            options.index_column = name;

1305                .to_field(&self.collect_schema().unwrap())
1306                .unwrap();
1307            return self.with_column(index_column).rolling(
1308                Expr::Column(output_field.name().clone()),
1309                group_by,
1310                options,

1539    /// Generic function to join two LazyFrames.
1540    ///
1541    /// `join` can join on multiple columns, given as two list of expressions, and with a
1542    /// [`JoinType`] specified by `how`. Non-joined column names in the right DataFrame
1543    /// that already exist in this DataFrame are suffixed with `"_right"`. For control
1544    /// over how columns are renamed and parallelization options, use
1545    /// [`join_builder`](LazyFrame::join_builder).
1546    ///
1547    /// Any provided `args.slice` parameter is not considered, but set by the internal optimizer.
1548    ///
1549    /// # Example
1550    ///
1551    /// ```rust
1552    /// use polars_core::prelude::*;
1553    /// use polars_lazy::prelude::*;
1554    ///
1555    /// fn example(ldf: LazyFrame, other: LazyFrame) -> LazyFrame {
1556    ///         ldf
1557    ///         .join(other, [col("foo"), col("bar")], [col("foo"), col("bar")], JoinArgs::new(JoinType::Inner))
1558    /// }
1559    /// ```
1560    pub fn join<E: AsRef<[Expr]>>(
1561        self,
1562        other: LazyFrame,
1563        left_on: E,
1564        right_on: E,
1565        args: JoinArgs,
1566    ) -> LazyFrame {
1567        let left_on = left_on.as_ref().to_vec();
1568        let right_on = right_on.as_ref().to_vec();
1569
1570        self._join_impl(other, left_on, right_on, args)
1571    }
1572
1573    fn _join_impl(
1574        self,
1575        other: LazyFrame,
1576        left_on: Vec<Expr>,
1577        right_on: Vec<Expr>,
1578        args: JoinArgs,
1579    ) -> LazyFrame {
1580        let JoinArgs {
1581            how,
1582            validation,
1583            suffix,
1584            slice,
1585            nulls_equal,
1586            coalesce,
1587            maintain_order,
1588        } = args;
1589
1590        if slice.is_some() {
1591            panic!("impl error: slice is not handled")
1592        }
1593
1594        let mut builder = self
1595            .join_builder()
1596            .with(other)
1597            .left_on(left_on)
1598            .right_on(right_on)
1599            .how(how)
1600            .validate(validation)
1601            .join_nulls(nulls_equal)
1602            .coalesce(coalesce)
1603            .maintain_order(maintain_order);
1604
1605        if let Some(suffix) = suffix {
1606            builder = builder.suffix(suffix);
1607        }
1608
1609        // Note: args.slice is set by the optimizer
1610        builder.finish()
1611    }

2362    // Finish with join predicates
2363    pub fn join_where(self, predicates: Vec<Expr>) -> LazyFrame {
2364        let opt_state = self.lf.opt_state;
2365        let other = self.other.expect("with not set");
2366
2367        // Decompose `And` conjunctions into their component expressions
2368        fn decompose_and(predicate: Expr, expanded_predicates: &mut Vec<Expr>) {
2369            if let Expr::BinaryExpr {
2370                op: Operator::And,
2371                left,
2372                right,
2373            } = predicate
2374            {
2375                decompose_and((*left).clone(), expanded_predicates);
2376                decompose_and((*right).clone(), expanded_predicates);
2377            } else {
2378                expanded_predicates.push(predicate);
2379            }
2380        }
2381        let mut expanded_predicates = Vec::with_capacity(predicates.len() * 2);
2382        for predicate in predicates {
2383            decompose_and(predicate, &mut expanded_predicates);
2384        }
2385        let predicates: Vec<Expr> = expanded_predicates;
2386
2387        // Decompose `is_between` predicates to allow for cleaner expression of range joins
2388        #[cfg(feature = "is_between")]
2389        let predicates: Vec<Expr> = {
2390            let mut expanded_predicates = Vec::with_capacity(predicates.len() * 2);
2391            for predicate in predicates {
2392                if let Expr::Function {
2393                    function: FunctionExpr::Boolean(BooleanFunction::IsBetween { closed }),
2394                    input,
2395                    ..
2396                } = &predicate
2397                {
2398                    if let [expr, lower, upper] = input.as_slice() {
2399                        match closed {
2400                            ClosedInterval::Both => {
2401                                expanded_predicates.push(expr.clone().gt_eq(lower.clone()));
2402                                expanded_predicates.push(expr.clone().lt_eq(upper.clone()));
2403                            },
2404                            ClosedInterval::Right => {
2405                                expanded_predicates.push(expr.clone().gt(lower.clone()));
2406                                expanded_predicates.push(expr.clone().lt_eq(upper.clone()));
2407                            },
2408                            ClosedInterval::Left => {
2409                                expanded_predicates.push(expr.clone().gt_eq(lower.clone()));
2410                                expanded_predicates.push(expr.clone().lt(upper.clone()));
2411                            },
2412                            ClosedInterval::None => {
2413                                expanded_predicates.push(expr.clone().gt(lower.clone()));
2414                                expanded_predicates.push(expr.clone().lt(upper.clone()));
2415                            },
2416                        }
2417                        continue;
2418                    }
2419                }
2420                expanded_predicates.push(predicate);
2421            }
2422            expanded_predicates
2423        };
2424
2425        let args = JoinArgs {
2426            how: self.how,
2427            validation: self.validation,
2428            suffix: self.suffix,
2429            slice: None,
2430            nulls_equal: self.nulls_equal,
2431            coalesce: self.coalesce,
2432            maintain_order: self.maintain_order,
2433        };

2435            allow_parallel: self.allow_parallel,
2436            force_parallel: self.force_parallel,
2437            args,
2438        };
2439
2440        let lp = DslPlan::Join {
2441            input_left: Arc::new(self.lf.logical_plan),
2442            input_right: Arc::new(other.logical_plan),
2443            left_on: Default::default(),
2444            right_on: Default::default(),
2445            predicates,
2446            options: Arc::from(options),
2447        };
2448

2450    }
