fn to_dummies( &self, separator: Option<&str>, drop_first: bool, drop_nulls: bool, ) -> Result<DataFrame, PolarsError>

Create dummy variables.

§Example
ⓘ


 use polars_core::prelude::*;

 let df = df! {
      "id" => &[1, 2, 3, 1, 2, 3, 1, 1],
      "type" => &["A", "B", "B", "B", "C", "C", "C", "B"],
      "code" => &["X1", "X2", "X3", "X3", "X2", "X2", "X1", "X1"]
  }.unwrap();

  let dummies = df.to_dummies(None, false, false).unwrap();
  println!("{}", dummies);
